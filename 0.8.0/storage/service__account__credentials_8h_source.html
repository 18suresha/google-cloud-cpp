55 /**
56  * Wrapper class for Google OAuth 2.0 service account credentials.
57  *
58  * Takes a ServiceAccountCredentialsInfo and obtains access tokens from the
59  * Google Authorization Service as needed. Instances of this class should
60  * usually be created via the convenience methods declared in
61  * google_credentials.h.
62  *
63  * An HTTP Authorization header, with an access token as its value,
64  * can be obtained by calling the AuthorizationHeader() method; if the current
65  * access token is invalid or nearing expiration, this will class will first
66  * obtain a new access token before returning the Authorization header string.
67 
68  * @see https://developers.google.com/identity/protocols/OAuth2ServiceAccount
69  * for an overview of using service accounts with Google's OAuth 2.0 system.
70  *
71  * @see https://cloud.google.com/storage/docs/reference/libraries for details on
72  * how to obtain and get started with service account credentials.
73  *
74  * @tparam HttpRequestBuilderType a dependency injection point. It makes it
75  * possible to mock internal libcurl wrappers. This should generally not be
76  * overridden except for testing.
77  * @tparam ClockType a dependency injection point to fetch the current time.
78  * This should generally not be overridden except for testing.
79  */
80 template <typename HttpRequestBuilderType =
81  storage::internal::CurlRequestBuilder,
82  typename ClockType = std::chrono::system_clock>
83 class ServiceAccountCredentials : public Credentials {
84  public:
85  explicit ServiceAccountCredentials(ServiceAccountCredentialsInfo const& info)
86  : clock_() {
87  namespace nl = storage::internal::nl;
88 
89  HttpRequestBuilderType request_builder(
90  info.token_uri, storage::internal::GetDefaultCurlHandleFactory());
91  auto assertion_components = std::move(AssertionComponentsFromInfo(info));
92  // This is the value of grant_type for JSON-formatted service account
93  // keyfiles downloaded from Cloud Console.
94  std::string payload("grant_type=");
95  payload +=
96  request_builder
97  .MakeEscapedString("urn:ietf:params:oauth:grant-type:jwt-bearer")
98  .get();
99  payload += "&assertion=";
100  payload += MakeJWTAssertion(assertion_components.first,
101  assertion_components.second, info.private_key);
102  payload_ = std::move(payload);
103 
104  request_builder.AddHeader(
105  "Content-Type: application/x-www-form-urlencoded");
106  request_ = request_builder.BuildRequest();
107  info_ = info;
108  }

110  StatusOr<std::string> AuthorizationHeader() override {
111  std::unique_lock<std::mutex> lock(mu_);
112  return refreshing_creds_.AuthorizationHeader([this] { return Refresh(); });
113  }

115  /**
116  * Sign a blob using the credentials.
117  *
118  * Create a RSA SHA256 signature of the blob using the Credential object. If
119  * the credentials do not support signing blobs it returns an error status.
120  *
121  * @param text the bytes to sign.
122  * @return a Base64-encoded RSA SHA256 digest of @p blob using the current
123  * credentials.
124  */
125  std::pair<Status, std::string> SignString(std::string const& text) const {
126  return std::make_pair(
127  Status(), internal::Base64Encode(internal::SignStringWithPem(
128  text, info_.private_key, JwtSigningAlgorithms::RS256)));
129  }
130 
131  std::pair<Status, std::string> SignStringHex(std::string const& text) const {
132  return std::make_pair(
133  Status(), internal::HexEncode(internal::SignStringWithPem(
134  text, info_.private_key, JwtSigningAlgorithms::RS256)));
135  }

137  /// Return the client id of these credentials.
138  std::string const& client_id() const { return info_.client_email; }

140  private:
141  /**
142  * Returns the header and payload components needed to make a JWT assertion.
143  *
144  * @see
145  * https://cloud.google.com/endpoints/docs/frameworks/java/troubleshoot-jwt
146  *
147  * @see https://tools.ietf.org/html/rfc7523
148  */
149  std::pair<storage::internal::nl::json, storage::internal::nl::json>
150  AssertionComponentsFromInfo(ServiceAccountCredentialsInfo const& info) {
151  namespace nl = storage::internal::nl;
152  nl::json assertion_header = {
153  {"alg", "RS256"}, {"kid", info.private_key_id}, {"typ", "JWT"}};
154 
155  // Scopes must be specified in a comma-delimited string.
156  std::string scope_str;
157  if (!info.scopes) {
158  scope_str = GoogleOAuthScopeCloudPlatform();
159  } else {
160  std::string sep = "";
161  for (const auto& scope : *(info.scopes)) {
162  scope_str += sep + scope;
163  sep = ",";
164  }
165  }
166 
167  auto now = clock_.now();
168  auto expiration = now + GoogleOAuthAccessTokenLifetime();
169  // As much as possible, do the time arithmetic using the std::chrono types.
170  // Convert to longs only when we are dealing with timestamps since the
171  // epoch.
172  auto now_from_epoch =
173  static_cast<long>(std::chrono::system_clock::to_time_t(now));
174  auto expiration_from_epoch =
175  static_cast<long>(std::chrono::system_clock::to_time_t(expiration));
176  nl::json assertion_payload = {
177  {"iss", info.client_email},
178  {"scope", scope_str},
179  {"aud", info.token_uri},
180  {"iat", now_from_epoch},
181  // Resulting access token should be expire after one hour.
182  {"exp", expiration_from_epoch}};
183  if (info.subject) {
184  assertion_payload["sub"] = *(info.subject);
185  }
186 
187  return std::make_pair(std::move(assertion_header),
188  std::move(assertion_payload));
189  }
190 
191  /**
192  * Given a key and a JSON header and payload, creates a JWT assertion string.
193  *
194  * @see https://tools.ietf.org/html/rfc7519
195  */
196  std::string MakeJWTAssertion(storage::internal::nl::json const& header,
197  storage::internal::nl::json const& payload,
198  std::string const& pem_contents) {
199  std::string encoded_header = internal::UrlsafeBase64Encode(header.dump());
200  std::string encoded_payload = internal::UrlsafeBase64Encode(payload.dump());
201  std::string encoded_signature = internal::UrlsafeBase64Encode(
202  internal::SignStringWithPem(encoded_header + '.' + encoded_payload,
203  pem_contents, JwtSigningAlgorithms::RS256));
204  return encoded_header + '.' + encoded_payload + '.' + encoded_signature;
205  }

207  Status Refresh() {
208  namespace nl = storage::internal::nl;
209 
210  auto response = request_.MakeRequest(payload_);
211  if (!response) {
212  return std::move(response).status();
213  }
214  if (response->status_code >= 300) {
215  return AsStatus(*response);
216  }
217 
218  nl::json access_token = nl::json::parse(response->payload, nullptr, false);
219  if (access_token.is_discarded() ||
220  access_token.count("access_token") == 0U or
221  access_token.count("expires_in") == 0U or
222  access_token.count("token_type") == 0U) {
223  response->payload +=
224  "Could not find all required fields in response (access_token,"
225  " expires_in, token_type).";
226  return AsStatus(*response);
227  }
228  // Response should have the attributes "access_token", "expires_in", and
229  // "token_type".
230  std::string header =
231  "Authorization: " + access_token.value("token_type", "") + " " +
232  access_token.value("access_token", "");
233  auto expires_in =
234  std::chrono::seconds(access_token.value("expires_in", int(0)));
235  auto new_expiration = std::chrono::system_clock::now() + expires_in;
236  // Do not update any state until all potential exceptions are raised.
237  refreshing_creds_.authorization_header = std::move(header);
238  refreshing_creds_.expiration_time = new_expiration;
239  return Status();
240  }

247  ClockType clock_;
248 };
