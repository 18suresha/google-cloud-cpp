33 class RefreshingCredentialsWrapper {
34  public:
35  template <typename RefreshFunctor>
36  StatusOr<std::string> AuthorizationHeader(RefreshFunctor refresh_fn) {
37  if (IsValid()) {
38  return authorization_header;
39  }
40 
41  Status status = refresh_fn();
42  if (status.ok()) {
43  return authorization_header;
44  }
45  return status;
46  }

48  /**
49  * Returns whether the current access token should be considered expired.
50  *
51  * When determining if a Credentials object needs to be refreshed, the IsValid
52  * method should be used instead; there may be cases where a Credentials is
53  * not expired but should be considered invalid.
54  *
55  * If a Credentials is close to expiration but not quite expired, this method
56  * may still return false. This helps prevent the case where an access token
57  * expires between when it is obtained and when it is used.
58  */
59  bool IsExpired();

67  bool IsValid();
68
