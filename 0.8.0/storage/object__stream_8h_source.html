47 /**
48  * Defines a `std::basic_istream<char>` to read from a GCS Object.
49  */
50 class ObjectReadStream : public std::basic_istream<char> {
51  public:
52  /**
53  * Creates a stream not associated with any buffer.
54  *
55  * Attempts to use this stream will result in failures.
56  */
57  ObjectReadStream() : std::basic_istream<char>(nullptr), buf_() {}
58 
59  /**
60  * Creates a stream associated with the given `streambuf`.
61  */
62  explicit ObjectReadStream(std::unique_ptr<internal::ObjectReadStreambuf> buf)
63  : std::basic_istream<char>(nullptr), buf_(std::move(buf)) {
64  // Initialize the basic_ios<> class
65  init(buf_.get());
66  }
67 
68  ObjectReadStream(ObjectReadStream&& rhs) noexcept
69  : ObjectReadStream(std::move(rhs.buf_)) {
70  // We cannot use set_rdbuf() because older versions of libstdc++ do not
71  // implement this function. Unfortunately `move()` resets `rdbuf()`, and
72  // `rdbuf()` resets the state, so we have to manually copy the rest of
73  // the state.
74  setstate(rhs.rdstate());
75  copyfmt(rhs);
76  rhs.rdbuf(nullptr);
77  }
78 
79  ObjectReadStream& operator=(ObjectReadStream&& rhs) noexcept {
80  buf_ = std::move(rhs.buf_);
81  // Use rdbuf() (instead of set_rdbuf()) because older versions of libstdc++
82  // do not implement this function. Unfortunately `rdbuf()` resets the state,
83  // and `move()` resets `rdbuf()`, so we have to manually copy the rest of
84  // the state.
85  rdbuf(buf_.get());
86  setstate(rhs.rdstate());
87  copyfmt(rhs);
88  rhs.rdbuf(nullptr);
89  return *this;
90  }
91 
92  ObjectReadStream(ObjectReadStream const&) = delete;
93  ObjectReadStream& operator=(ObjectReadStream const&) = delete;
94 
95  /// Closes the stream (if necessary).
96  ~ObjectReadStream() override;
97 
98  bool IsOpen() const { return (bool)buf_ && buf_->IsOpen(); }
99 
100  /**
101  * Terminate the download, possibly before completing it.
102  */
103  void Close();
104 
105  //@{
106  /**
107  * Report any download errors.
108  *
109  * Note that errors may go undetected until the download completes.
110  */
111  Status const& status() const& { return buf_->status(); }
112 
113  /**
114  * The received CRC32C checksum and the MD5 hash values as reported by GCS.
115  *
116  * When the download is finalized (via `Close()` or the end of file) the GCS
117  * server reports the CRC32C checksum and, except for composite objects, the
118  * MD5 hash of the data. This class compares the locally computed and received
119  * hashes so applications can detect data download errors.
120  *
121  * The values are reported as comma separated `tag=value` pairs, e.g.
122  * `crc32c=AAAAAA==,md5=1B2M2Y8AsgTpgAmY7PhCfg==`. The format of this string
123  * is subject to change without notice, they are provided for informational
124  * purposes only.
125  *
126  * @see https://cloud.google.com/storage/docs/hashes-etags for more
127  * information on checksums and hashes in GCS.
128  */
129  std::string const& received_hash() const { return buf_->received_hash(); }
130 
131  /**
132  * The locally computed checksum and hashes, as a string.
133  *
134  * This object computes the CRC32C checksum and MD5 hash of the downloaded
135  * data. Note that there are several cases where these values may be empty or
136  * irrelevant, for example:
137  * - When reading only a portion of a blob the hash of that portion is
138  * irrelevant, note that GCS only reports the hashes for the full blob.
139  * - The application may disable the CRC32C and/or the MD5 hash computation.
140  *
141  * The string has the same format as the value returned by `received_hash()`.
142  * Note that the format of this string is also subject to change without
143  * notice.
144  *
145  * @see https://cloud.google.com/storage/docs/hashes-etags for more
146  * information on checksums and hashes in GCS.
147  */
148  std::string const& computed_hash() const { return buf_->computed_hash(); }
149 
150  /// The headers returned by the service, for debugging only.
151  std::multimap<std::string, std::string> const& headers() const {
152  return buf_->headers();
153  }
154  //@}
155 
156  private:
157  std::unique_ptr<internal::ObjectReadStreambuf> buf_;
158 };
