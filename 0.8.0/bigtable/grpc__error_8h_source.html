26 /**
27  * Wrap gRPC errors is a C++ exception.
28  *
29  * Applications that only wish to log errors can (and should) catch
30  * `std::exception` or `std::runtime_error`. The string returned by `what()`
31  * contains all the necessary information. If the application wants to handle
32  * errors raised by the gRPC library, they can catch this exception type, and
33  * use the `error_code()` member function to implement whatever error handling
34  * strategy they need.
35  *
36  * It is customary, though not required, for C++ libraries and applications
37  * to use classes in the `std::exception` hierarchy to report errors. Meanwhile,
38  * gRPC reports errors using the `grpc::Status` class. This class wraps the
39  * contents of a `grpc::Status` in an exception class. If the application is
40  * interested in the details of the exception it can catch it and examine all
41  * fields. If the application developers simply want to log all exceptions they
42  * can catch `std::exception` and log the contents of `what()`.
43  */
44 class GRpcError : public std::runtime_error {
45  public:
46  explicit GRpcError(char const* what, grpc::Status const& status);
47 
48  //@{
49  /// @name accessors.
50  grpc::StatusCode error_code() const { return error_code_; }
51  std::string const& error_message() const { return error_message_; }

53  //@}
