23 /**
24  * Wrap a gRPC timer into an `AsyncOperation`.
25  *
26  * Applications (or more likely, other components in the client library) will
27  * associate timers with a completion queue. gRPC timers require applications to
28  * create a unique `grpc::Alarm` object for each timer, and then to associate
29  * them with the completion queue using a `void*` tag.
30  *
31  * This class collaborates with our wrapper for `CompletionQueue` to associate
32  * a `future<AsyncTimerResult>` for each timer. This class takes care of
33  * allocating the `grpc::Alarm`, creating a unique `void*` associated with the
34  * timer, and satisfying the future when the timer expires.
35  *
36  * Note that this class is an implementation detail, hidden from the application
37  * developers.
38  */
39 class AsyncTimerFuture : public internal::AsyncGrpcOperation {
40  public:
41  AsyncTimerFuture() = default;
42 
43  future<std::chrono::system_clock::time_point> GetFuture() {
44  return promise_.get_future();
45  }
46 
47  void Set(grpc::CompletionQueue& cq,
48  std::chrono::system_clock::time_point deadline, void* tag) {
49  deadline_ = deadline;
50  alarm_.Set(&cq, deadline, tag);
51  }
52 
53  void Cancel() override { alarm_.Cancel(); }
54 
55  private:
56  bool Notify(CompletionQueue&, bool) override {
57  promise_.set_value(deadline_);
58  return true;
59  }
60 
61  promise<std::chrono::system_clock::time_point> promise_;
62  std::chrono::system_clock::time_point deadline_;
63  grpc::Alarm alarm_;
64 };

75 CompletionQueue::MakeDeadlineTimer(

77  auto op = std::make_shared<AsyncTimerFuture>();
