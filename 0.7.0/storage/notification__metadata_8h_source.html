31 /**

34  * Notifications send information about changes to objects in your buckets to

36  *

52  /// @name Accessors and modifiers to the custom attributes.
53  bool has_custom_attribute(std::string const& key) const {
54  return custom_attributes_.end() != custom_attributes_.find(key);

61  /// Delete a custom attribute. This is a no-op if the key does not exist.
62  NotificationMetadata& delete_custom_attribute(std::string const& key) {
63  auto i = custom_attributes_.find(key);
64  if (i == custom_attributes_.end()) {
65  return *this;
66  }
67  custom_attributes_.erase(i);
68  return *this;
69  }
70 
71  /// Insert or update the custom attribute.
72  NotificationMetadata& upsert_custom_attributes(std::string key,
73  std::string value) {
74  auto i = custom_attributes_.lower_bound(key);
75  if (i == custom_attributes_.end() || i->first != key) {
76  custom_attributes_.emplace_hint(i, std::move(key), std::move(value));
77  } else {
78  i->second = std::move(value);
79  }
80  return *this;
81  }
