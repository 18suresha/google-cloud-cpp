157  /**
158  * Set the grpclb fallback timeout with the timestamp @p fallback_timeout
159  * for the channel.
160  *
161  * For example:
162  *
163  * @code
164  * bigtable::ClientOptions::SetGrpclbFallbackTimeout(
165  * std::chrono::milliseconds(5000))
166  * bigtable::ClientOptions::SetGrpclbFallbackTimeout(
167  * std::chrono::seconds(5))
168  * @endcode
169  *
170  * @tparam Rep a placeholder to match the Rep tparam for @p fallback_timeout,
171  * the semantics of this template parameter are documented in
172  * `std::chrono::duration<>` (in brief, the underlying arithmetic type
173  * used to store the number of ticks), for our purposes it is simply a
174  * formal parameter.
175  * @tparam Period a placeholder to match the Period tparam for @p
176  * fallback_timeout, the semantics of this template parameter are
177  * documented in `std::chrono::duration<>` (in brief, the length of the
178  * tick in seconds, expressed as a `std::ratio<>`), for our purposes it
179  * is simply a formal parameter.
180  *
181  * @see
182  * [std::chrono::duration<>](http://en.cppreference.com/w/cpp/chrono/duration)
183  * for more details.
184  *
185  * Please see the docs for grpc::ChannelArguments::SetGrpclbFallbackTimeout()
186  * on https://grpc.io/grpc/cpp/classgrpc_1_1_channel_arguments.html
187  * for more details.
188  *
189  */
190  template <typename Rep, typename Period>
191  google::cloud::Status SetGrpclbFallbackTimeout(
192  std::chrono::duration<Rep, Period> fallback_timeout) {
193  std::chrono::milliseconds ft_ms =
194  std::chrono::duration_cast<std::chrono::milliseconds>(fallback_timeout);
195 
196  if (ft_ms.count() > std::numeric_limits<int>::max()) {
197  return google::cloud::Status(google::cloud::StatusCode::kOutOfRange,
198  "The supplied duration is larger than the "
199  "maximum value allowed by gRPC (INT_MAX)");
200  }
201  auto fallback_timeout_ms = static_cast<int>(ft_ms.count());
202  channel_arguments_.SetGrpclbFallbackTimeout(fallback_timeout_ms);
203  return google::cloud::Status();
204  }
