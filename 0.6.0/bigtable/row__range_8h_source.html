26 /**
27  * Define the interfaces to create row key ranges.
28  *
29  * Example:
30  * @code
31  * // Create a range for the keys starting with the given prefix.
32  * auto range = bigtable::RowRange("foo/");
33  * @endcode
34  */

45  /// Return the infinite range, i.e., a range including all possible keys.

48  /// Return the range starting at @p begin (included), with no upper limit.
49  static RowRange StartingAt(std::string begin) {

62  /// Return an empty range.
63  static RowRange Empty() {
64  RowRange result;
65  // Return an open interval that contains no key, using "\0" for the end key.
66  // We can't use "", because when appearing as the end it means 'infinity'.
67  result.row_range_.set_start_key_open("");
68  result.row_range_.set_end_key_open(std::string("\0", 1));
69  return result;
70  }

72  /// Return the range representing the interval [@p begin, @p end).
73  static RowRange Range(std::string begin, std::string end) {
74  return RightOpen(std::move(begin), std::move(end));
75  }

77  /// Return a range that contains all the keys starting with @p prefix.
