22 StatusOr<AuthorizedUserCredentialsInfo> ParseAuthorizedUserCredentials(
23  std::string const& content, std::string const& source,
24  std::string const& default_token_uri) {
25  auto credentials =
26  storage::internal::nl::json::parse(content, nullptr, false);
27  if (credentials.is_discarded()) {
28  return Status(
29  StatusCode::kInvalidArgument,
30  "Invalid AuthorizedUserCredentials, parsing failed on data from " +
31  source);
32  }
33 
34  char const client_id_key[] = "client_id";
35  char const client_secret_key[] = "client_secret";
36  char const refresh_token_key[] = "refresh_token";
37  char const token_uri_key[] = "token_uri"; // Not required; often not present.
38  for (auto const& key :
39  {client_id_key, client_secret_key, refresh_token_key}) {
40  if (credentials.count(key) == 0U) {
41  return Status(StatusCode::kInvalidArgument,
42  "Invalid AuthorizedUserCredentials, the " +
43  std::string(key) +
44  " field is missing on data loaded from " + source);
45  }
46  if (credentials.value(key, "").empty()) {
47  return Status(StatusCode::kInvalidArgument,
48  "Invalid AuthorizedUserCredentials, the " +
49  std::string(key) +
50  " field is empty on data loaded from " + source);
51  }
52  }
53  return AuthorizedUserCredentialsInfo{
54  credentials.value(client_id_key, ""),
55  credentials.value(client_secret_key, ""),
56  credentials.value(refresh_token_key, ""),
57  // Some credential formats (e.g. gcloud's ADC file) don't contain a
58  // "token_uri" attribute in the JSON object. In this case, we try using
59  // the default value.
60  credentials.value(token_uri_key, default_token_uri)};
61 }
