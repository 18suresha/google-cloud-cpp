62 std::string FieldPath::ToApiRepr() const {
63  // gcc-4.8 ships with a broken regex library (sigh), so don't use it.
64  auto is_simple_field_name = [](std::string const& part) {
65  if (part.empty()) {
66  return false;
67  }
68  if (part[0] != '_' && std::isalpha(part[0]) == 0) {
69  return false;
70  }
71  return std::all_of(part.begin(), part.end(),
72  [](char c) { return c == '_' || std::isalnum(c) != 0; });
73  };
74  std::string s;
75  if (valid_) {
76  for (auto part : parts_) {
77  auto const match = is_simple_field_name(part);
78  if (match) {
79  s += part + '.';
80  } else {
81  ReplaceAll(part, "\\", "\\\\");
82  ReplaceAll(part, "`", "\\`");
83  s += '`' + part + "`.";
84  }
85  }
86  s.resize(s.size() - 1); // cannot be empty and remove final period
87  }
88  return s; // let the server catch the empty string error for invalid
89 }

95 bool operator<(FieldPath const& lhs, FieldPath const& rhs) {
96  auto const lhs_size = lhs.parts_.size();
97  auto const rhs_size = rhs.parts_.size();
98  auto const min_length = (std::min)(lhs_size, rhs_size);
99  for (auto i = 0U; i != min_length; i++) {
100  if (lhs.parts_[i] < rhs.parts_[i]) {
101  return true;
102  }
103  if (lhs.parts_[i] > rhs.parts_[i]) {
104  return false;
105  }
106  }
107  if (lhs_size < rhs_size) {
108  return true;
109  }
110  if (lhs_size > rhs_size) {
111  return false;
112  }
113  return false;
114 }

131 std::vector<std::string> FieldPath::Split(std::string string) {
132  std::vector<std::string> parts;
133  auto index = string.find('.');
134  while (index != std::string::npos) {
135  parts.push_back(string.substr(0, index));
136  string = string.substr(index + 1);
137  index = string.find('.');
138  }
139  parts.push_back(string);
140  return parts;
141 }

143 void FieldPath::ReplaceAll(std::string& string, std::string const& find,
144  std::string const& replace) {
145  auto found = string.find(find);
146  while (found != std::string::npos) {
147  string = string.replace(found, find.size(), replace);
148  found += replace.size();
149  found = string.find(find, found);
150  }
