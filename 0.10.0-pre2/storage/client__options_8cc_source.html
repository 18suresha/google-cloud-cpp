39 std::size_t DefaultConnectionPoolSize() {
40  std::size_t nthreads = std::thread::hardware_concurrency();
41  if (nthreads == 0) {
42  return 4;
43  }
44  return 4 * nthreads;
45 }

47 // There is nothing special about the buffer sizes here. They are relatively
48 // small, because we do not want to consume too much memory from the
49 // application. They are larger than the typical socket buffer size (64KiB), to
50 // be able to read the full buffer into userspace if it happens to be full.
51 #ifndef GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_BUFFER_SIZE
52 #define GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_BUFFER_SIZE (128 * 1024)
53 #endif // GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_BUFFER_SIZE

93 void ClientOptions::SetupFromEnvironment() {
94  auto enable_clog =
95  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_CLOG");
96  if (enable_clog.has_value()) {
97  google::cloud::LogSink::EnableStdClog();
98  }
99  // This is overkill right now, eventually we will have different components
100  // that can be traced (http being the first), so we parse the environment
101  // variable.
102  auto tracing =
103  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_TRACING");
104  if (tracing.has_value()) {
105  std::set<std::string> enabled;
106  std::istringstream is{*tracing};
107  std::string token;
108  while (std::getline(is, token, ',')) {
109  enabled.emplace(token);
110  }
111  if (enabled.end() != enabled.find("http")) {
112  GCP_LOG(INFO) << "Enabling logging for http";
113  set_enable_http_tracing(true);
114  }
115  if (enabled.end() != enabled.find("raw-client")) {
116  GCP_LOG(INFO) << "Enabling logging for RawClient functions";
117  set_enable_raw_client_tracing(true);
118  }
119  }
120 
121  auto project_id = google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT");
122  if (project_id.has_value()) {
123  project_id_ = std::move(*project_id);
124  }
125 }
