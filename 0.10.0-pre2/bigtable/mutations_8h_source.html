63 /**
64  * @name Create mutations to delete a range of cells from a column.
65  *
66  * The following functions create a mutation that deletes all the
67  * cells in the given column family and, column within the given
68  * timestamp in the range.
69  *
70  * The function accepts any instantiation of `std::chrono::duration<>` for the
71  * @p timestamp_begin and @p timestamp_end parameters. For example:
72  *
73  * @code
74  * using namespace std::chrono_literals; // C++14
75  * bigtable::DeleteFromColumn("fam", "col", 0us, 10us)
76  * @endcode
77  *
78  * The ending timestamp is exclusive, while the beginning timestamp is
79  * inclusive. That is, the interval is [@p timestamp_begin, @p timestamp_end).
80  * The value 0 is special and treated as "unbounded" for both the begin and
81  * end endpoints of the time range. The Cloud Bigtable server rejects
82  * invalid and empty ranges, i.e., any range where the endpoint is smaller or
83  * equal than to the initial endpoint unless either endpoint is 0.
84  *
85  * @tparam Rep1 a placeholder to match the Rep tparam for @p timestamp_begin
86  * type. The semantics of this template parameter are documented in
87  * std::chrono::duration<>` (in brief, the underlying arithmetic type
88  * used to store the number of ticks), for our purposes it is simply a
89  * formal parameter.
90  *
91  * @tparam Rep2 similar formal parameter for the type of @p timestamp_end.
92  *
93  * @tparam Period1 a placeholder to match the Period tparam for
94  * @p timestamp_begin type. The semantics of this template parameter are
95  * documented in `std::chrono::duration<>` (in brief, the length of the tick
96  * in seconds,vexpressed as a `std::ratio<>`), for our purposes it is simply
97  * a formal parameter.
98  *
99  * @tparam Period2 similar formal parameter for the type of @p timestamp_end.
100  */
101 template <typename Rep1, typename Period1, typename Rep2, typename Period2>
102 Mutation DeleteFromColumn(std::string family, std::string column,
103  std::chrono::duration<Rep1, Period1> timestamp_begin,
104  std::chrono::duration<Rep2, Period2> timestamp_end) {
105  Mutation m;
106  using namespace std::chrono;
107  auto& d = *m.op.mutable_delete_from_column();
108  d.set_family_name(std::move(family));
109  d.set_column_qualifier(std::move(column));
110  d.mutable_time_range()->set_start_timestamp_micros(
111  duration_cast<microseconds>(timestamp_begin).count());
112  d.mutable_time_range()->set_end_timestamp_micros(
113  duration_cast<microseconds>(timestamp_end).count());
114  return m;
115 }

207 /**
208  * Represent a single row mutation.
209  *
210  * Bigtable can perform multiple changes to a single row atomically.
211  * This class represents 0 or more changes to apply to a single row.
212  * The changes may include setting cells (which implicitly insert the
213  * values), deleting values, etc.
214  */
215 class SingleRowMutation {
216  public:
217  /// Create an empty mutation.
218  explicit SingleRowMutation(std::string row_key)
219  : row_key_(std::move(row_key)) {}
220 
221  /// Create a row mutation from a initializer list.
222  SingleRowMutation(std::string row_key, std::initializer_list<Mutation> list)
223  : row_key_(std::move(row_key)) {
224  for (auto&& i : list) {
225  *ops_.Add() = i.op;
226  }
227  }
228 
229  /// Create a single-row multiple-cell mutation from a variadic list.
230  template <typename... M>
231  explicit SingleRowMutation(std::string row_key, M&&... m)
232  : row_key_(std::move(row_key)) {
233  static_assert(
234  internal::conjunction<std::is_convertible<M, Mutation>...>::value,
235  "The arguments passed to SingleRowMutation(std::string, ...) must be "
236  "convertible to Mutation");
237  emplace_many(std::forward<M>(m)...);
238  }
239 
240  /// Create a row mutation from gRPC proto
241  explicit SingleRowMutation(
242  ::google::bigtable::v2::MutateRowsRequest::Entry entry)
243  : row_key_(std::move(*entry.mutable_row_key())), ops_() {
244  ops_.Swap(entry.mutable_mutations());
245  }
246 
247  /// Create a row mutation from gRPC proto
248  explicit SingleRowMutation(::google::bigtable::v2::MutateRowRequest request)
249  : row_key_(std::move(*request.mutable_row_key())), ops_() {
250  ops_.Swap(request.mutable_mutations());
251  }
252 
253  // Add a mutation at the end.
254  SingleRowMutation& emplace_back(Mutation mut) {
255  ops_.Add()->Swap(&mut.op);
256  return *this;
257  }
258 
259  // Get the row key.
260  std::string const& row_key() const { return row_key_; }
261 
262  friend class Table;
263 
264  SingleRowMutation(SingleRowMutation&& rhs) = default;
265  SingleRowMutation& operator=(SingleRowMutation&& rhs) = default;
266  SingleRowMutation(SingleRowMutation const& rhs) = default;
267  SingleRowMutation& operator=(SingleRowMutation const& rhs) = default;
268 
269  /// Move the contents into a bigtable::v2::MutateRowsRequest::Entry.
270  void MoveTo(google::bigtable::v2::MutateRowsRequest::Entry* entry) {
271  entry->set_row_key(std::move(row_key_));
272  entry->mutable_mutations()->Swap(&ops_);
273  }
274 
275  /// Transfer the contents to @p request.
276  void MoveTo(google::bigtable::v2::MutateRowRequest& request) {
277  request.set_row_key(std::move(row_key_));
278  request.mutable_mutations()->Swap(&ops_);
279  }
280 
281  /// Remove the contents of the mutation.
282  void Clear() {
283  row_key_.clear();
284  ops_.Clear();
285  }
286 
287  private:
288  /// Add multiple mutations to single row
289  template <typename... M>
290  void emplace_many(Mutation first, M&&... tail) {
291  emplace_back(std::move(first));
292  emplace_many(std::forward<M>(tail)...);
293  }
294 
295  void emplace_many(Mutation m) { emplace_back(std::move(m)); }
296 
297  private:
298  std::string row_key_;
299  google::protobuf::RepeatedPtrField<google::bigtable::v2::Mutation> ops_;
300 };

381 /**
382  * Represent a set of mutations across multiple rows.
383  *
384  * Cloud Bigtable can batch multiple mutations in a single request.
385  * The mutations are not atomic, but it is more efficient to send them
386  * in a batch than to make multiple smaller requests.
387  */
388 class BulkMutation {
389  public:
390  /// Create an empty set of mutations.
391  BulkMutation() : request_() {}
392 
393  /// Create a multi-row mutation from a range of SingleRowMutations.
394  template <typename iterator>
395  BulkMutation(iterator begin, iterator end) {
396  static_assert(
397  std::is_convertible<decltype(*begin), SingleRowMutation>::value,
398  "The iterator value type must be convertible to SingleRowMutation");
399  for (auto i = begin; i != end; ++i) {
400  push_back(*i);
401  }
402  }
403 
404  /// Create a multi-row mutation from a initializer list.
405  BulkMutation(std::initializer_list<SingleRowMutation> list)
406  : BulkMutation(list.begin(), list.end()) {}
407 
408  /// Create a multi-row mutation from a SingleRowMutation
409  explicit BulkMutation(SingleRowMutation mutation) : BulkMutation() {
410  emplace_back(std::move(mutation));
411  }
412 
413  /// Create a muti-row mutation from two SingleRowMutation
414  BulkMutation(SingleRowMutation m1, SingleRowMutation m2) : BulkMutation() {
415  emplace_back(std::move(m1));
416  emplace_back(std::move(m2));
417  }
418 
419  /// Create a muti-row mutation from a variadic list.
420  template <typename... M,
421  typename std::enable_if<internal::conjunction<std::is_convertible<
422  M, SingleRowMutation>...>::value,
423  int>::type = 0>
424  BulkMutation(M&&... m) : BulkMutation() {
425  emplace_many(std::forward<M>(m)...);
426  }
427 
428  // Add a mutation to the batch.
429  BulkMutation& emplace_back(SingleRowMutation mut) {
430  mut.MoveTo(request_.add_entries());
431  return *this;
432  }
433 
434  // Add a failed mutation to the batch.
435  BulkMutation& emplace_back(FailedMutation fm) {
436  fm.status_ = google::cloud::Status();
437  return *this;
438  }
439 
440  // Add a mutation to the batch.
441  BulkMutation& push_back(SingleRowMutation mut) {
442  mut.MoveTo(request_.add_entries());
443  return *this;
444  }
445 
446  /// Move the contents into a bigtable::v2::MutateRowsRequest
447  void MoveTo(google::bigtable::v2::MutateRowsRequest* request) {
448  request_.Swap(request);
449  request_ = {};
450  }
451 
452  /// Return true if there are no mutations in this set.
453  bool empty() const { return request_.entries().empty(); }
454 
455  /// Return the number of mutations in this set.
456  std::size_t size() const { return request_.entries().size(); }
457 
458  /// Return the estimated size in bytes of all the mutations in this set.
459  std::size_t estimated_size_in_bytes() const {
460  return request_.ByteSizeLong();
461  }
462 
463  private:
464  template <typename... M>
465  void emplace_many(SingleRowMutation first, M&&... tail) {
466  emplace_back(std::move(first));
467  emplace_many(std::forward<M>(tail)...);
468  }
469 
470  void emplace_many(SingleRowMutation m) { emplace_back(std::move(m)); }
471 
472  private:
473  google::bigtable::v2::MutateRowsRequest request_;
474 };
