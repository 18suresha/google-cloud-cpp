28 /**
29  * Represents a container for providing users with a handful of operation to
30  * users which they can use to add and remove members to Binding which is used
31  * for defining IAM Policy for Cloud Platform Resources.
32  *
33  * For more information about a Binding please refer to:
34  * https://cloud.google.com/resource-manager/reference/rest/Shared.Types/Binding
35  */
36 class IamBindings {
37  public:
38  IamBindings() = default;
39 
40  explicit IamBindings(std::vector<IamBinding> bindings) {
41  for (auto& it : bindings) {
42  bindings_.insert({std::move(it.role()), std::move(it.members())});
43  }
44  }
45 
46  IamBindings(std::string const& role, std::set<std::string> const& members) {
47  bindings_.insert({std::move(role), std::move(members)});
48  }

51

54  * container.

56  iterator begin() const { return bindings_.begin(); }

60  * container.
61  */
62  iterator end() const { return bindings_.end(); }
63 
64  /**
65  * Returns whether the Bindings container is empty.
66  *
67  * @return bool whether the container is empty or not.
68  */
69  bool empty() const { return bindings_.empty(); }
70

72  * Return number of Bindings in container.
73  *
74  * @return int the size of the container.
75  */
76  std::size_t size() const { return bindings_.size(); }
77 
78  std::map<std::string, std::set<std::string>> const& bindings() const {
79  return bindings_;
80  }
81 
82  /**
83  * Finds the members for a role.
84  */
85  iterator find(std::string const& role) const { return bindings_.find(role); }
86 
87  /// Returns the members for a role.
88  std::set<std::string> at(std::string const& role) const {
89  auto loc = bindings_.find(role);
90  if (loc == bindings_.end()) {
91  return {};
92  }
93  return loc->second;
94  }
95 
96  /**
97  * Adds a new member if a binding exists with given role otherwise inserts
98  * a new key-value pair of role and member to the container.
99  *
100  * @param role role of the new member.
101  * @param member specifies the identity requesting access for a cloud
102  * platform resource.
103  */
104  void AddMember(std::string const& role, std::string member);
105 
106  /**
107  * Adds a new key-value pair of role and members to the container if there is
108  * none for the role of given binding else appends members of given binding
109  * to the associated role's key-value entry.
110  *
111  * @param iam_binding binding representing a set of members and role for them.
112  */
113  void AddMembers(google::cloud::IamBinding const& iam_binding);
114 
115  /**
116  * Adds a new key-value pair of role and members to the container if there no
117  * existing for given role else appends the given members to the give role's
118  * member set.
119  *
120  * @param role role of the member set to be added.
121  * @param members a set of member which are needed to be added.
122  */
123  void AddMembers(std::string const& role,
124  std::set<std::string> const& members);
125 
126  /**
127  * Removes the given member from the given role's member set if there exists
128  * one in container.
129  *
130  * @param role role of the member to be removed.
131  * @param member specifies the identity requesting access for a cloud
132  * platform resource.
133  */
134  void RemoveMember(std::string const& role, std::string const& member);
135 
136  /**
137  * Removes the given binding's member from the given binding's role's member
138  * set if there exists one in container.
139  *
140  * @param iam_binding binding representing a set of members and role for them.
141  */
142  void RemoveMembers(google::cloud::IamBinding const& iam_binding);
143 
144  /**
145  * Removes the given members from given role's member set if there exists one
146  * in container.
147  *
148  * @param role role of the member set to be removed.
149  * @param members a set of members which are needed to be removed.
150  */
151  void RemoveMembers(std::string const& role,
152  std::set<std::string> const& members);
153 
154  private:
155  std::map<std::string, std::set<std::string>> bindings_;
156 };
