244  google::longrunning::Operation operation) {
245  promise<StatusOr<gcsa::Instance>> pr;
246  auto f = pr.get_future();
247 
248  // TODO(#4038) - use background_threads_->cq() to run this loop.
249  std::thread t(
250  [](std::shared_ptr<spanner_internal::InstanceAdminStub> stub,
251  google::longrunning::Operation operation,
252  std::unique_ptr<PollingPolicy> polling_policy,
253  google::cloud::promise<StatusOr<gcsa::Instance>> promise,
254  char const* location) mutable {
255  auto result = google::cloud::internal::PollingLoop<
256  google::cloud::internal::PollingLoopResponseExtractor<
257  gcsa::Instance>>(
258  std::move(polling_policy),
259  [stub](grpc::ClientContext& context,
260  google::longrunning::GetOperationRequest const& request) {
261  return stub->GetOperation(context, request);
262  },
263  std::move(operation), location);
264 
265  // Drop our reference to stub; ideally we'd have std::moved into the
266  // lambda. Doing this also prevents a false leak from being reported
267  // when using googlemock.
268  stub.reset();
269  promise.set_value(std::move(result));
270  },
271  stub_, std::move(operation), polling_policy_prototype_->clone(),
272  std::move(pr), __func__);
273  t.detach();
274 
275  return f;
276  }
