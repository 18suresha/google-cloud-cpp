31 struct SafeGrpcRetry {
32  static inline bool IsOk(google::cloud::Status const& status) {
33  return status.ok();
34  }
35  static inline bool IsTransientFailure(google::cloud::Status const& status) {
36  if (status.code() == StatusCode::kUnavailable ||
37  status.code() == StatusCode::kResourceExhausted) {
38  return true;
39  }
40  // Treat the unexpected termination of the gRPC connection as retryable.
41  // There is no explicit indication of this, but it will result in an
42  // INTERNAL status with one of the `kTransientFailureMessages`.
43  if (status.code() == StatusCode::kInternal) {
44  constexpr char const* kTransientFailureMessages[] = {
45  "Received unexpected EOS on DATA frame from server",
46  "Connection closed with unknown cause",
47  "HTTP/2 error code: INTERNAL_ERROR", "RST_STREAM"};
48  for (auto const& message : kTransientFailureMessages) {
49  if (absl::StrContains(status.message(), message)) return true;
50  }
51  }
52  return false;
53  }
54  static inline bool IsPermanentFailure(google::cloud::Status const& status) {
55  return !IsOk(status) && !IsTransientFailure(status);
56  }
57 };
