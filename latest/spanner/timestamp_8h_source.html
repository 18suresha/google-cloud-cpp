122  // Conversion to a `std::chrono::time_point` on the system clock. May
123  // produce out-of-range errors, depending on the properties of `Duration`
124  // and the `std::chrono::system_clock` epoch.
125  template <typename Duration>
126  StatusOr<sys_time<Duration>> ConvertTo(sys_time<Duration> const&) const {
127  using Rep = typename Duration::rep;
128  using Period = typename Duration::period;
129  static_assert(std::ratio_greater_equal<Period, std::nano>::value,
130  "Duration must be no more precise than std::nano");
131  auto count =
132  ToRatio(std::numeric_limits<Rep>::min(),
133  std::numeric_limits<Rep>::max(), Period::num, Period::den);
134  if (!count) return std::move(count).status();
135  auto const unix_epoch = std::chrono::time_point_cast<Duration>(
136  sys_time<Duration>::clock::from_time_t(0));
137  return unix_epoch + Duration(static_cast<Rep>(*count));
138  }

162 template <typename Duration>
163 StatusOr<Timestamp> MakeTimestamp(sys_time<Duration> const& tp) {
164  using Period = typename Duration::period;
165  static_assert(std::ratio_greater_equal<Period, std::nano>::value,
166  "Duration must be no more precise than std::nano");
167  auto const unix_epoch = std::chrono::time_point_cast<Duration>(
168  sys_time<Duration>::clock::from_time_t(0));
169  auto const period = absl::Seconds(Period::num) / Period::den;
170  auto const count = (tp - unix_epoch).count();
171  return MakeTimestamp(absl::UnixEpoch() + count * period);
172 }
