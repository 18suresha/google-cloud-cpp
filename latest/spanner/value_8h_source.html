540  template <typename T, typename V>
541  static StatusOr<absl::optional<T>> GetValue(
542  absl::optional<T> const&, V&& pv, google::spanner::v1::Type const& pt) {
543  if (pv.kind_case() == google::protobuf::Value::kNullValue) {
544  return absl::optional<T>{};
545  }
546  auto value = GetValue(T{}, std::forward<V>(pv), pt);
547  if (!value) return std::move(value).status();
548  return absl::optional<T>{*std::move(value)};
549  }
550  template <typename T, typename V>
551  static StatusOr<std::vector<T>> GetValue(
552  std::vector<T> const&, V&& pv, google::spanner::v1::Type const& pt) {
553  if (pv.kind_case() != google::protobuf::Value::kListValue) {
554  return Status(StatusCode::kUnknown, "missing ARRAY");
555  }
556  std::vector<T> v;
557  for (int i = 0; i < pv.list_value().values().size(); ++i) {
558  auto&& e = GetProtoListValueElement(std::forward<V>(pv), i);
559  using ET = decltype(e);
560  auto value = GetValue(T{}, std::forward<ET>(e), pt.array_element_type());
561  if (!value) return std::move(value).status();
562  v.push_back(*std::move(value));
563  }
564  return v;
565  }
566  template <typename V, typename... Ts>
567  static StatusOr<std::tuple<Ts...>> GetValue(
568  std::tuple<Ts...> const&, V&& pv, google::spanner::v1::Type const& pt) {
569  if (pv.kind_case() != google::protobuf::Value::kListValue) {
570  return Status(StatusCode::kUnknown, "missing STRUCT");
571  }
572  std::tuple<Ts...> tup;
573  Status status; // OK
574  ExtractTupleValues<V> f{status, 0, std::forward<V>(pv), pt};
575  internal::ForEach(tup, f);
576  if (!status.ok()) return status;
577  return tup;
578  }

580  // A functor to be used with internal::ForEach to extract C++ types from a
581  // ListValue proto and store then in a tuple.
582  template <typename V>
583  struct ExtractTupleValues {
584  Status& status;
585  int i;
586  V&& pv;
587  google::spanner::v1::Type const& type;
588  template <typename T>
589  void operator()(T& t) {
590  auto&& e = GetProtoListValueElement(std::forward<V>(pv), i);
591  using ET = decltype(e);
592  auto value = GetValue(T{}, std::forward<ET>(e), type);
593  ++i;
594  if (!value) {
595  status = std::move(value).status();

597  t = *std::move(value);
598  }

604  using ET = decltype(e);

606  ++i;
