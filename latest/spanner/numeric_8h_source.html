201  typename std::enable_if<std::numeric_limits<T>::is_integer &&
202  !std::numeric_limits<T>::is_signed,

204 StatusOr<T> ToInteger(Numeric const& n, int exponent = 0) {
205  std::string const& rep = n.ToString();
206  if (exponent != 0) {
207  auto const en = internal::MakeNumeric(rep, exponent);
208  return en ? ToInteger<T>(*en, 0) : en.status();
209  }
210  T v = 0;
211  constexpr auto kDigits = "0123456789";
212  constexpr auto kMax = (std::numeric_limits<T>::max)();
213  enum { kIntPart, kFracPart } state = kIntPart;
214  for (auto const ch : rep) {
215  auto const* dp = std::strchr(kDigits, ch);
216  if (state == kFracPart) {
217  if (dp - kDigits >= 5) { // dp != nullptr
218  if (v == kMax) return internal::DataLoss(rep);
219  v = static_cast<T>(v + 1);
220  }
221  break;
222  }
223  if (dp == nullptr) {
224  if (ch == '-') return internal::DataLoss(rep);
225  state = kFracPart; // ch == '.'
226  } else {
227  if (v > kMax / 10) return internal::DataLoss(rep);
228  v = static_cast<T>(v * 10);
229  auto d = static_cast<T>(dp - kDigits);
230  if (v > kMax - d) return internal::DataLoss(rep);
231  v = static_cast<T>(v + d);
232  }
233  }
234  return v;
235 }

239  std::numeric_limits<T>::is_signed,
240  int>::type = 0>
241 StatusOr<T> ToInteger(Numeric const& n, int exponent = 0) {
242  std::string const& rep = n.ToString();
243  if (exponent != 0) {
244  auto const en = internal::MakeNumeric(rep, exponent);
245  return en ? ToInteger<T>(*en, 0) : en.status();
246  }
247  T v = 0;
248  constexpr auto kDigits = "0123456789";
249  constexpr auto kMin = (std::numeric_limits<T>::min)();
250  bool negate = true;
251  enum { kIntPart, kFracPart } state = kIntPart;
252  for (auto const ch : rep) {
253  auto const* dp = std::strchr(kDigits, ch);
254  if (state == kFracPart) {
255  if (dp - kDigits >= 5) { // dp != nullptr
256  if (v == kMin) return internal::DataLoss(rep);
257  v = static_cast<T>(v - 1);
258  }
259  break;
260  }
261  if (dp == nullptr) {
262  if (ch == '-') {
263  negate = false;
264  } else {
265  state = kFracPart; // ch == '.'
266  }
267  } else {
268  if (v < kMin / 10) return internal::DataLoss(rep);
269  v = static_cast<T>(v * 10);
270  auto d = static_cast<T>(dp - kDigits);
271  if (v < kMin + d) return internal::DataLoss(rep);
272  v = static_cast<T>(v - d);
273  }
274  }
275  if (!negate) return v;
276  constexpr auto kMax = (std::numeric_limits<T>::max)();
277  if (kMin != -kMax && v == kMin) return internal::DataLoss(rep);
278  return static_cast<T>(-v);
279 }
