45 StatusOr<std::int64_t> Timestamp::ToRatio(std::int64_t min, std::int64_t max,
46  std::int64_t num,
47  std::int64_t den) const {
48  constexpr auto kDestType = "std::chrono::time_point";
49  auto const period = absl::Seconds(num) / den;
50  auto const duration = absl::Floor(t_ - absl::UnixEpoch(), period);
51  if (duration > max * period) return PositiveOverflow(kDestType);
52  if (duration < min * period) return NegativeOverflow(kDestType);
53  return duration / period;
54 }

56 StatusOr<Timestamp> MakeTimestamp(absl::Time t) {
57  constexpr auto kDestType = "google::cloud::spanner::Timestamp";
58  // The min/max values that are allowed to in a Timestamp:
59  // ["0001-01-01T00:00:00Z", "9999-12-31T23:59:59.999999999Z"]
60  // Note: These values can be computed with `date +%s --date="YYYY-MM-...Z"`
61  auto constexpr kMinTime = absl::FromUnixSeconds(-62135596800);
62  auto const kMaxTime = // NOLINT(readability-identifier-naming)
63  absl::FromUnixSeconds(253402300799) + absl::Nanoseconds(999999999);
64  if (t > kMaxTime) return PositiveOverflow(kDestType);
65  if (t < kMinTime) return NegativeOverflow(kDestType);
66  return Timestamp(t);
67 }

75 // Timestamp objects are always formatted in UTC, and we always format them
76 // with a trailing 'Z'. However, we're a bit more liberal in the UTC offsets we
77 // accept, thus the use of '%Ez' in kParseSpec.

79 auto constexpr kParseSpec = "%Y-%m-%dT%H:%M:%E*S%Ez";

81 StatusOr<Timestamp> TimestampFromRFC3339(std::string const& s) {
82  absl::Time t;
83  std::string err;
84  if (absl::ParseTime(kParseSpec, s, &t, &err)) return MakeTimestamp(t);
85  return InvalidArgument(s + ": " + err);
86 }
