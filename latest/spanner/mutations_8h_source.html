108 template <typename Op>
109 class WriteMutationBuilder {
110  public:
111  WriteMutationBuilder(std::string table_name,
112  std::vector<std::string> column_names) {
113  auto& field = Op::mutable_field(m_.proto());
114  field.set_table(std::move(table_name));
115  field.mutable_columns()->Reserve(static_cast<int>(column_names.size()));
116  for (auto& name : column_names) {
117  field.add_columns(std::move(name));
118  }
119  }
120 
121  spanner::Mutation Build() const& { return m_; }
122  spanner::Mutation&& Build() && { return std::move(m_); }
123 
124  WriteMutationBuilder& AddRow(std::vector<spanner::Value> values) & {
125  auto& lv = *Op::mutable_field(m_.proto()).add_values();
126  for (auto& v : values) {
127  std::tie(std::ignore, *lv.add_values()) =
128  spanner_internal::ToProto(std::move(v));
129  }
130  return *this;
131  }
132 
133  WriteMutationBuilder&& AddRow(std::vector<spanner::Value> values) && {
134  return std::move(AddRow(std::move(values)));
135  }

137  template <typename... Ts>
138  WriteMutationBuilder& EmplaceRow(Ts&&... values) & {
139  return AddRow({spanner::Value(std::forward<Ts>(values))...});
140  }
141 
142  template <typename... Ts>
143  WriteMutationBuilder&& EmplaceRow(Ts&&... values) && {
144  return std::move(EmplaceRow(std::forward<Ts>(values)...));
145  }

147  private:
148  spanner::Mutation m_;
149 };
