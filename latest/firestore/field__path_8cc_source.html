58 std::string FieldPath::ToApiRepr() const {
59  // gcc-4.8 ships with a broken regex library (sigh), so don't use it.
60  auto is_simple_field_name = [](std::string const& part) {
61  if (part.empty()) {
62  return false;
63  }
64  if (part[0] != '_' && std::isalpha(part[0]) == 0) {
65  return false;
66  }
67  return std::all_of(part.begin(), part.end(),
68  [](char c) { return c == '_' || std::isalnum(c) != 0; });
69  };
70  std::string s;
71  if (valid_) {
72  for (auto part : parts_) {
73  auto const match = is_simple_field_name(part);
74  if (match) {
75  s += part + '.';
76  } else {
77  ReplaceAll(part, "\\", "\\\\");
78  ReplaceAll(part, "`", "\\`");
79  s += '`' + part + "`.";
80  }
81  }
82  s.resize(s.size() - 1); // cannot be empty and remove final period
83  }
84  return s; // let the server catch the empty string error for invalid
85 }

91 bool operator<(FieldPath const& lhs, FieldPath const& rhs) {
92  auto const lhs_size = lhs.parts_.size();
93  auto const rhs_size = rhs.parts_.size();
94  auto const min_length = (std::min)(lhs_size, rhs_size);
95  for (auto i = 0U; i != min_length; i++) {
96  if (lhs.parts_[i] < rhs.parts_[i]) {
97  return true;
98  }
99  if (lhs.parts_[i] > rhs.parts_[i]) {
100  return false;
101  }
102  }
103  if (lhs_size < rhs_size) {
104  return true;
105  }
106  if (lhs_size > rhs_size) {
107  return false;
108  }
109  return false;
110 }

121 std::vector<std::string> FieldPath::Split(std::string string) {
122  std::vector<std::string> parts;
123  auto index = string.find('.');
124  while (index != std::string::npos) {
125  parts.push_back(string.substr(0, index));
126  string = string.substr(index + 1);
127  index = string.find('.');
128  }
129  parts.push_back(string);
130  return parts;

133 void FieldPath::ReplaceAll(std::string& string, std::string const& find,
134  std::string const& replace) {
135  auto found = string.find(find);
136  while (found != std::string::npos) {
137  string = string.replace(found, find.size(), replace);
138  found += replace.size();
139  found = string.find(find, found);
140  }
