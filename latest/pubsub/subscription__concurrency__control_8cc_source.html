51 void SubscriptionConcurrencyControl::Start(pubsub::ApplicationCallback cb) {
52  std::unique_lock<std::mutex> lk(mu_);
53  if (callback_) return;
54  callback_ = std::move(cb);
55  std::weak_ptr<SubscriptionConcurrencyControl> weak = shared_from_this();
56  source_->Start([weak](google::pubsub::v1::ReceivedMessage r) {
57  if (auto self = weak.lock()) self->OnMessage(std::move(r));
58  });
59  if (total_messages() >= max_concurrency_) return;
60  auto const read_count = max_concurrency_ - total_messages();
61  messages_requested_ = read_count;
62  lk.unlock();
63  source_->Read(read_count);
64 }

81 void SubscriptionConcurrencyControl::MessageHandled() {
82  if (shutdown_manager_->FinishedOperation("handler")) return;
83  std::unique_lock<std::mutex> lk(mu_);
