104 std::shared_ptr<pubsub::PublisherConnection> MakePublisherConnection(
105  pubsub::Topic topic, pubsub::PublisherOptions options,
106  pubsub::ConnectionOptions connection_options,
107  std::vector<std::shared_ptr<PublisherStub>> stubs,
108  std::unique_ptr<pubsub::RetryPolicy const> retry_policy,
109  std::unique_ptr<pubsub::BackoffPolicy const> backoff_policy) {
110  if (stubs.empty()) return nullptr;
111  if (!retry_policy) retry_policy = DefaultRetryPolicy();
112  if (!backoff_policy) backoff_policy = DefaultBackoffPolicy();
113  std::shared_ptr<PublisherStub> stub =
114  std::make_shared<PublisherRoundRobin>(std::move(stubs));
115  stub = std::make_shared<PublisherMetadata>(std::move(stub));
116  if (connection_options.tracing_enabled("rpc")) {
117  GCP_LOG(INFO) << "Enabled logging for gRPC calls";
118  stub = std::make_shared<PublisherLogging>(
119  std::move(stub), connection_options.tracing_options());
120  }
121 
122  auto default_thread_pool_size = []() -> std::size_t {
123  auto constexpr kDefaultThreadPoolSize = 4;
124  auto const n = std::thread::hardware_concurrency();
125  return n == 0 ? kDefaultThreadPoolSize : n;
126  };
127  if (connection_options.background_thread_pool_size() == 0) {
128  connection_options.set_background_thread_pool_size(
129  default_thread_pool_size());
130  }
131 
132  auto background = connection_options.background_threads_factory()();
133  auto make_connection = [&]() -> std::shared_ptr<pubsub::PublisherConnection> {
134  auto cq = background->cq();
135  std::shared_ptr<BatchSink> sink = DefaultBatchSink::Create(
136  stub, cq, std::move(retry_policy), std::move(backoff_policy));
137  if (options.message_ordering()) {
138  auto factory = [topic, options, sink, cq](std::string const& key) {
139  return BatchingPublisherConnection::Create(
140  topic, options, key, SequentialBatchSink::Create(std::move(sink)),
141  cq);
142  };
143  return OrderingKeyPublisherConnection::Create(std::move(factory));
144  }
145  return RejectsWithOrderingKey::Create(BatchingPublisherConnection::Create(
146  std::move(topic), std::move(options), {}, sink, std::move(cq)));
147  };
148  return std::make_shared<pubsub::ContainingPublisherConnection>(
149  std::move(background), make_connection());
150 }
