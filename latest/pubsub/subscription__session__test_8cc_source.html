91  std::lock_guard<std::mutex> lk(mu);
92  m.set_ack_id("test-ack-id-" + std::to_string(count));

97  }
98  return cq.MakeRelativeTimer(us(10)).then([response](TimerFuture) {
99  return absl::make_optional(response);
100  });
101  });
102  EXPECT_CALL(*stream, Cancel).Times(1);
103  EXPECT_CALL(*stream, Read).WillRepeatedly([&] {
104  return cq.MakeRelativeTimer(us(10)).then([](TimerFuture) {
105  return absl::optional<google::pubsub::v1::StreamingPullResponse>{};
106  });
107  });
108  EXPECT_CALL(*stream, Finish).WillOnce([&] {
109  return cq.MakeRelativeTimer(us(10)).then([](TimerFuture) {
110  return Status{StatusCode::kCancelled, "cancel"};
111  });
112  });
113 
114  return stream;
115  });

118  std::generate_n(std::back_inserter(tasks), 4,
119  [&] { return std::thread([&cq] { cq.Run(); }); });
120  std::set<std::thread::id> ids;
121  auto const main_id = std::this_thread::get_id();
122  std::transform(tasks.begin(), tasks.end(), std::inserter(ids, ids.end()),
123  [](std::thread const& t) { return t.get_id(); });
124

126  auto handler = [&](pubsub::Message const& m, pubsub::AckHandler h) {
127  EXPECT_EQ(42, h.delivery_attempt());
128  EXPECT_EQ("test-message-id-" + std::to_string(expected_message_id),

131  EXPECT_NE(ids.end(), pos);
132  EXPECT_NE(main_id, std::this_thread::get_id());
133  // Increment the counter before acking, as the ack() may trigger a new call
134  // before this function gets to run.
135  ++expected_message_id;
136  std::move(h).ack();
137  };
138 
139  auto response = CreateTestingSubscriptionSession(
140  mock, cq,
141  {subscription,
142  pubsub::SubscriberOptions{}.set_concurrency_watermarks(0, 1), handler});

144  std::unique_lock<std::mutex> lk(ack_id_mu);
145  ack_id_cv.wait(lk, [&] { return expected_ack_id >= kAckCount; });
146  }
147  response.cancel();
148  EXPECT_STATUS_OK(response.get());
149 
150  cq.Shutdown();
151  for (auto& t : tasks) t.join();
152 }

234 TEST(SubscriptionSessionTest, ShutdownWaitsFutures) {
235  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
236  pubsub::Subscription const subscription("test-project", "test-subscription");
237 
238  EXPECT_CALL(*mock, AsyncStreamingPull)
239  .Times(AtLeast(1))
240  .WillRepeatedly(FakeAsyncStreamingPull);
241 
242  auto constexpr kMaximumAcks = 10;
243 
244  internal::AutomaticallyCreatedBackgroundThreads background;
245  std::atomic<int> handler_counter{0};
246 
247  // Create a scope for the handler and its variables, this makes it easier to
248  // discover bugs under TSAN/ASAN.
249  {
250  // Now unto the handler, basically it counts messages and nacks starting at
251  // kMaximumAcks.
252  promise<void> got_one;
253  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
254  if (handler_counter.load() == 0) got_one.set_value();
255  if (++handler_counter > kMaximumAcks) return;
256  std::move(h).ack();
257  };
258 
259  auto session = CreateSubscriptionSession(
260  mock, background.cq(), "fake-client-id",
261  {subscription, pubsub::SubscriberOptions{}, handler},
262  pubsub_testing::TestRetryPolicy(), pubsub_testing::TestBackoffPolicy());
263  got_one.get_future()
264  .then([&session](future<void>) { session.cancel(); })
265  .get();
266 
267  auto status = session.get();
268  EXPECT_STATUS_OK(status);
269  EXPECT_LE(1, handler_counter.load());
270  }
271  // Schedule at least a few more iterations of the CQ loop. If the shutdown is
272  // buggy, we will see TSAN/ASAN errors because the `handler` defined above
273  // is still called.
274  auto const initial_value = handler_counter.load();
275  for (int i = 0; i != 10; ++i) {
276  SCOPED_TRACE("Wait loop iteration " + std::to_string(i));
277  promise<void> done;
278  background.cq().RunAsync([&done] { done.set_value(); });
279  done.get_future().get();
280  }
281  auto const final_value = handler_counter.load();
282  EXPECT_EQ(initial_value, final_value);
283 }

286 TEST(SubscriptionSessionTest, ShutdownWaitsConditionVars) {
287  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
288  pubsub::Subscription const subscription("test-project", "test-subscription");
289 
290  EXPECT_CALL(*mock, AsyncStreamingPull)
291  .Times(AtLeast(1))
292  .WillRepeatedly(FakeAsyncStreamingPull);
293 
294  // A number of mocks that return futures satisfied a bit after the call is
295  // made. This better simulates the behavior when running against an actual
296  // service.
297  auto constexpr kMaximumAcks = 20;
298 
299  internal::AutomaticallyCreatedBackgroundThreads background;
300  std::atomic<int> handler_counter{0};
301 
302  // Create a scope for the handler and its variables, makes the errors more
303  // obvious under TSAN/ASAN.
304  {
305  // Now unto the handler, basically it counts messages and nacks starting at
306  // kMaximumAcks.
307  std::mutex mu;
308  std::condition_variable cv;
309  int ack_count = 0;
310  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
311  ++handler_counter;
312  std::unique_lock<std::mutex> lk(mu);
313  if (++ack_count > kMaximumAcks) return;
314  lk.unlock();
315  cv.notify_one();
316  std::move(h).ack();
317  };
318 
319  auto session = CreateSubscriptionSession(
320  mock, background.cq(), "fake-client-id", {subscription, {}, handler},
321  pubsub_testing::TestRetryPolicy(), pubsub_testing::TestBackoffPolicy());
322  {
323  std::unique_lock<std::mutex> lk(mu);
324  cv.wait(lk, [&] { return ack_count >= kMaximumAcks; });
325  }
326  session.cancel();
327  auto status = session.get();
328  EXPECT_STATUS_OK(status);
329  }
330  // Schedule at least a few more iterations of the CQ loop. If the shutdown is
331  // buggy, we will see TSAN/ASAN errors because the `handler` defined above
332  // is still called.
333  auto const initial_value = handler_counter.load();
334  for (int i = 0; i != 10; ++i) {
335  SCOPED_TRACE("Wait loop iteration " + std::to_string(i));
336  promise<void> done;
337  background.cq().RunAsync([&done] { done.set_value(); });
338  done.get_future().get();
339  }
340  auto const final_value = handler_counter.load();
341  EXPECT_EQ(initial_value, final_value);
342 }

345 TEST(SubscriptionSessionTest, ShutdownWaitsEarlyAcks) {
346  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
347  pubsub::Subscription const subscription("test-project", "test-subscription");
348 
349  EXPECT_CALL(*mock, AsyncStreamingPull)
350  .Times(AtLeast(1))
351  .WillRepeatedly(FakeAsyncStreamingPull);
352 
353  auto constexpr kMessageCount = 16;
354 
355  internal::AutomaticallyCreatedBackgroundThreads background(kMessageCount);
356  std::atomic<int> handler_counter{0};
357 
358  // Create a scope for the handler and its variables, which makes the errors
359  // more obvious under TSAN/ASAN.
360  {
361  // Now unto the handler. It counts messages, and uses objects after the
362  // AckHandler has been returned. If the session shutdown is not working
363  // correctly using these variables is a problem under TSAN and ASAN. We also
364  // have a more direct detection of problems later in this test.
365  std::mutex mu;
366  std::condition_variable cv;
367  int ack_count = 0;
368  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
369  std::move(h).ack();
370  // Sleep after the `ack()` call to more easily reproduce #5148
371  std::this_thread::sleep_for(std::chrono::microseconds(500));
372  ++handler_counter;
373  {
374  std::unique_lock<std::mutex> lk(mu);
375  ++ack_count;
376  }
377  cv.notify_one();
378  };
379 
380  auto session = CreateSubscriptionSession(
381  mock, background.cq(), "fake-client-id",
382  {subscription,
383  pubsub::SubscriberOptions{}.set_concurrency_watermarks(
384  kMessageCount / 2, 2 * kMessageCount),
385  handler},
386  pubsub_testing::TestRetryPolicy(), pubsub_testing::TestBackoffPolicy());
387  {
388  std::unique_lock<std::mutex> lk(mu);
389  cv.wait(lk, [&] { return ack_count >= kMessageCount; });
390  }
391  session.cancel();
392  auto status = session.get();
393  EXPECT_STATUS_OK(status);
394  }
395  // Schedule at least a few more iterations of the CQ loop. If the shutdown is
396  // buggy, we will see TSAN/ASAN errors because the `handler` defined above
397  // is still called.
398  auto const initial_value = handler_counter.load();
399  for (std::size_t i = 0; i != 10 * background.pool_size(); ++i) {
400  SCOPED_TRACE("Wait loop iteration " + std::to_string(i));
401  promise<void> done;
402  background.cq().RunAsync([&done] { done.set_value(); });
403  done.get_future().get();
404  }
405  auto const final_value = handler_counter.load();
406  EXPECT_EQ(initial_value, final_value);
407 }

410 TEST(SubscriptionSessionTest, FireAndForget) {
411  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
412  pubsub::Subscription const subscription("test-project", "test-subscription");
413 
414  EXPECT_CALL(*mock, AsyncStreamingPull)
415  .Times(AtLeast(1))
416  .WillRepeatedly(FakeAsyncStreamingPull);
417 
418  std::mutex mu;
419  std::condition_variable cv;
420  int ack_count = 0;
421  Status status;
422  auto wait_ack_count = [&](int n) {
423  std::unique_lock<std::mutex> lk(mu);
424  cv.wait(lk, [&] { return ack_count >= n || !status.ok(); });
425  return ack_count;
426  };
427 
428  auto constexpr kMessageCount = 8;
429  // Create a scope for the background completion queues and threads.
430  {
431  internal::AutomaticallyCreatedBackgroundThreads background;
432 
433  // Create a scope so the future and handler get destroyed, but we want the
434  // test to continue afterwards.
435  {
436  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
437  std::move(h).ack();
438  std::unique_lock<std::mutex> lk(mu);
439  if (++ack_count % kMessageCount == 0) cv.notify_one();
440  };
441 
442  (void)CreateSubscriptionSession(
443  mock, background.cq(), "fake-client-id",
444  {subscription,
445  pubsub::SubscriberOptions{}
446  .set_max_outstanding_messages(kMessageCount / 2)
447  .set_concurrency_watermarks(0, kMessageCount / 2)
448  .set_shutdown_polling_period(std::chrono::milliseconds(20)),
449  handler},
450  pubsub_testing::TestRetryPolicy(),
451  pubsub_testing::TestBackoffPolicy())
452  .then([&](future<Status> f) {
453  std::unique_lock<std::mutex> lk(mu);
454  status = f.get();
455  cv.notify_one();
456  });
457  wait_ack_count(kMessageCount);
458  }
459 
460  auto const initial_value = wait_ack_count(2 * kMessageCount);
461  auto const final_value = wait_ack_count(initial_value + 2 * kMessageCount);
462  EXPECT_NE(initial_value, final_value);
463  std::unique_lock<std::mutex> lk(mu);
464  EXPECT_STATUS_OK(status);
465  }
