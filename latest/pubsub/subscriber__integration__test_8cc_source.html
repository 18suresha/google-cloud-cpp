91  google::pubsub::v1::StreamingPullRequest request;
92  request.set_client_id("test-client-0001");
93  request.set_subscription(subscription_.FullName());
94  request.set_max_outstanding_messages(1000);
95  request.set_stream_ack_deadline_seconds(600);

98  return stub->AsyncStreamingPull(
99  cq, absl::make_unique<grpc::ClientContext>(), request);

101 
102  ASSERT_TRUE(stream->Start().get());
103  ASSERT_TRUE(
104  stream->Write(request, grpc::WriteOptions{}.set_write_through()).get());
105 
106  auto constexpr kPublishCount = 1000;
107  std::set<std::string> expected_ids = [&] {
108  std::set<std::string> ids;
109  std::vector<future<StatusOr<std::string>>> message_ids;
110  for (int i = 0; i != kPublishCount; ++i) {
111  message_ids.push_back(publisher.Publish(
112  MessageBuilder{}.SetData("message-" + std::to_string(i)).Build()));
113  }
114  for (auto& id : message_ids) {
115  auto r = id.get();
116  EXPECT_STATUS_OK(r);
117  if (r) ids.insert(*std::move(r));
118  }
119  return ids;
120  }();

122  for (auto r = stream->Read().get(); r.has_value(); r = stream->Read().get()) {
123  google::pubsub::v1::StreamingPullRequest acks;

125  acks.add_ack_ids(m.ack_id());
126  expected_ids.erase(m.message().message_id());
127  }
128  auto write_ok = stream->Write(acks, grpc::WriteOptions{}).get();
129  if (!write_ok) break;

131  }
132  EXPECT_TRUE(expected_ids.empty());
133 
134  stream->Cancel();
135  // Before closing the stream we need to wait for:
136  // Read().get().has_value() == false
137  for (auto r = stream->Read().get(); r.has_value(); r = stream->Read().get()) {
138  }

146  topic_,
147  MakePublisherConnection(
148  pubsub::ConnectionOptions{}.set_background_thread_pool_size(2)));
149 
150  internal::AutomaticallyCreatedBackgroundThreads background(4);

152 
153  auto shutdown = std::make_shared<pubsub_internal::SessionShutdownManager>();
154  auto source =
155  std::make_shared<pubsub_internal::StreamingSubscriptionBatchSource>(
156  background.cq(), shutdown, stub, subscription_.FullName(),
157  "test-client-0001", pubsub::SubscriberOptions{},
158  pubsub_testing::TestRetryPolicy(),
159  pubsub_testing::TestBackoffPolicy());
160 
161  std::mutex callback_mu;
162  std::condition_variable callback_cv;
163  std::vector<std::string> received_ids;
164  int ack_count = 0;
165  int callback_count = 0;
166  auto wait_ack_count = [&](int count) {
167  std::unique_lock<std::mutex> lk(callback_mu);
168  callback_cv.wait(lk, [&] { return ack_count >= count; });
169  };
170  auto callback =
171  [&](StatusOr<google::pubsub::v1::StreamingPullResponse> const& response) {
172  ASSERT_STATUS_OK(response);
173  std::cout << "callback(" << response->received_messages_size() << ")"
174  << std::endl;
175  {
176  std::lock_guard<std::mutex> lk(callback_mu);
177  for (auto const& m : response->received_messages()) {
178  received_ids.push_back(m.message().message_id());
179  }
180  ++callback_count;
181  for (auto const& m : response->received_messages()) {
182  source->AckMessage(m.ack_id());
183  }
184  ack_count += response->received_messages_size();
185  }
186  callback_cv.notify_one();
187  };
188 
189  auto done = shutdown->Start({});
190  source->Start(std::move(callback));
191 
192  auto constexpr kPublishCount = 1000;
193  std::set<std::string> expected_ids = [&] {
194  std::set<std::string> ids;
195  std::vector<future<StatusOr<std::string>>> message_ids;
196  for (int i = 0; i != kPublishCount; ++i) {
197  message_ids.push_back(publisher.Publish(
198  MessageBuilder{}.SetData("message-" + std::to_string(i)).Build()));
199  }
200  for (auto& id : message_ids) {
201  auto r = id.get();
202  EXPECT_STATUS_OK(r);
203  if (r) ids.insert(*std::move(r));
204  }
205  return ids;
206  }();
207 
208  wait_ack_count(kPublishCount);
209  {
210  std::lock_guard<std::mutex> lk(callback_mu);
211  for (auto const& id : received_ids) {
212  expected_ids.erase(id);
213  }
214  EXPECT_TRUE(expected_ids.empty());
215  }
216 
217  shutdown->MarkAsShutdown("test", {});
218  source->Shutdown();
219  EXPECT_STATUS_OK(done.get());

222 TEST_F(SubscriberIntegrationTest, PublishPullAck) {
223  auto publisher = Publisher(topic_);
224  auto subscriber = Subscriber(subscription_);
225 
226  std::mutex mu;
227  std::map<std::string, int> ids;
228  for (auto const* data : {"message-0", "message-1", "message-2"}) {
229  auto response =
230  publisher.Publish(MessageBuilder{}.SetData(data).Build()).get();
231  EXPECT_STATUS_OK(response);
232  if (response) {
233  std::lock_guard<std::mutex> lk(mu);
234  ids.emplace(*std::move(response), 0);
235  }
236  }
237  EXPECT_FALSE(ids.empty());
238 
239  promise<void> ids_empty;
240  auto handler = [&](pubsub::Message const& m, AckHandler h) {
241  SCOPED_TRACE("Search for message " + m.message_id());
242  std::unique_lock<std::mutex> lk(mu);
243  auto i = ids.find(m.message_id());
244  // Remember that Cloud Pub/Sub has "at least once" semantics, so a dup is
245  // perfectly possible, in that case the message would not be in the map of
246  // of pending ids.
247  if (i == ids.end()) return;
248  // The first time just NACK the message to exercise that path, we expect
249  // Cloud Pub/Sub to retry.
250  if (i->second == 0) {
251  std::move(h).nack();
252  ++i->second;
253  return;
254  }
255  ids.erase(i);
256  if (ids.empty()) ids_empty.set_value();
257  lk.unlock();
258  std::move(h).ack();
259  };
260 
261  auto result = subscriber.Subscribe(handler);
262  // Wait until there are no more ids pending, then cancel the subscription and
263  // get its status.
264  ids_empty.get_future().get();
265  result.cancel();
266  EXPECT_STATUS_OK(result.get());
267 }

269 TEST_F(SubscriberIntegrationTest, FireAndForget) {
270  std::mutex mu;
271  std::condition_variable cv;
272  std::set<std::string> received;
273  Status subscription_result;
274  std::set<std::string> published;
275  std::vector<Status> publish_errors;
276  auto constexpr kMinimumMessages = 10;
277 
278  auto publisher = Publisher(topic_);
279  auto subscriber = Subscriber(subscription_);
280  {
281  (void)subscriber
282  .Subscribe([&](Message const& m, AckHandler h) {
283  std::move(h).ack();
284  std::unique_lock<std::mutex> lk(mu);
285  std::cout << "received " << m.message_id() << std::endl;
286  received.insert(m.message_id());
287  lk.unlock();
288  cv.notify_one();
289  })
290  .then([&](future<Status> f) {
291  std::unique_lock<std::mutex> lk(mu);
292  subscription_result = f.get();
293  cv.notify_one();
294  });
295 
296  std::vector<future<void>> pending;
297  for (int i = 0; i != kMinimumMessages; ++i) {
298  pending.push_back(
299  publisher
300  .Publish(MessageBuilder{}
301  .SetAttributes({{"index", std::to_string(i)}})
302  .Build())
303  .then([&](future<StatusOr<std::string>> f) {
304  std::unique_lock<std::mutex> lk(mu);
305  auto s = f.get();
306  if (!s) {
307  publish_errors.push_back(std::move(s).status());
308  return;
309  }
310  published.insert(*std::move(s));
311  }));
312  }
313  publisher.Flush();
314  for (auto& p : pending) p.get();
315  }
316  {
317  std::unique_lock<std::mutex> lk(mu);
318  cv.wait(lk, [&] { return received.size() >= published.size(); });
319  }
320  EXPECT_THAT(publish_errors, IsEmpty());
321  EXPECT_THAT(received, ElementsAreArray(published));
322 }
