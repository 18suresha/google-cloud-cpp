84 TEST_F(SubscriberIntegrationTest, RawStub) {
85  auto publisher = Publisher(MakePublisherConnection(topic_, {}));
86 
87  internal::AutomaticallyCreatedBackgroundThreads background(4);
88  auto stub = pubsub_internal::CreateDefaultSubscriberStub({}, 0);
89  google::pubsub::v1::StreamingPullRequest request;
90  request.set_client_id("test-client-0001");
91  request.set_subscription(subscription_.FullName());
92  request.set_max_outstanding_messages(1000);
93  request.set_stream_ack_deadline_seconds(600);
94 
95  auto stream = [&stub, &request](CompletionQueue cq) {
96  return stub->AsyncStreamingPull(
97  cq, absl::make_unique<grpc::ClientContext>(), request);
98  }(background.cq());
99 
100  ASSERT_TRUE(stream->Start().get());
101  ASSERT_TRUE(
102  stream->Write(request, grpc::WriteOptions{}.set_write_through()).get());
103 
104  auto constexpr kPublishCount = 1000;
105  std::set<std::string> expected_ids = [&] {
106  std::set<std::string> ids;
107  std::vector<future<StatusOr<std::string>>> message_ids;
108  for (int i = 0; i != kPublishCount; ++i) {
109  message_ids.push_back(publisher.Publish(
110  MessageBuilder{}.SetData("message-" + std::to_string(i)).Build()));
111  }
112  for (auto& id : message_ids) {
113  auto r = id.get();
114  EXPECT_STATUS_OK(r);
115  if (r) ids.insert(*std::move(r));
116  }
117  return ids;
118  }();
119 
120  for (auto r = stream->Read().get(); r.has_value(); r = stream->Read().get()) {
121  google::pubsub::v1::StreamingPullRequest acks;
122  for (auto const& m : r->received_messages()) {
123  acks.add_ack_ids(m.ack_id());
124  expected_ids.erase(m.message().message_id());
125  }
126  auto write_ok = stream->Write(acks, grpc::WriteOptions{}).get();
127  if (!write_ok) break;
128  if (expected_ids.empty()) break;
129  }
130  EXPECT_TRUE(expected_ids.empty());
131 
132  stream->Cancel();
133  // Before closing the stream we need to wait for:
134  // Read().get().has_value() == false
135  for (auto r = stream->Read().get(); r.has_value(); r = stream->Read().get()) {
136  }
137 
138  EXPECT_THAT(stream->Finish().get(), AnyOf(StatusIs(StatusCode::kOk),
139  StatusIs(StatusCode::kCancelled)));
140 }

142 TEST_F(SubscriberIntegrationTest, StreamingSubscriptionBatchSource) {
143  auto publisher = Publisher(MakePublisherConnection(
144  topic_, {},
145  pubsub::ConnectionOptions{}.set_background_thread_pool_size(2)));
146 
147  internal::AutomaticallyCreatedBackgroundThreads background(4);
148  auto stub = pubsub_internal::CreateDefaultSubscriberStub({}, 0);
149 
150  auto shutdown = std::make_shared<pubsub_internal::SessionShutdownManager>();
151  auto source =
152  std::make_shared<pubsub_internal::StreamingSubscriptionBatchSource>(
153  background.cq(), shutdown, stub, subscription_.FullName(),
154  "test-client-0001", pubsub::SubscriptionOptions{},
155  pubsub_testing::TestRetryPolicy(),
156  pubsub_testing::TestBackoffPolicy());
157 
158  std::mutex callback_mu;
159  std::condition_variable callback_cv;
160  std::vector<std::string> received_ids;
161  int ack_count = 0;
162  int callback_count = 0;
163  auto wait_ack_count = [&](int count) {
164  std::unique_lock<std::mutex> lk(callback_mu);
165  callback_cv.wait(lk, [&] { return ack_count >= count; });
166  };
167  auto callback =
168  [&](StatusOr<google::pubsub::v1::StreamingPullResponse> const& response) {
169  ASSERT_STATUS_OK(response);
170  std::cout << "callback(" << response->received_messages_size() << ")"
171  << std::endl;
172  {
173  std::lock_guard<std::mutex> lk(callback_mu);
174  for (auto const& m : response->received_messages()) {
175  received_ids.push_back(m.message().message_id());
176  }
177  ++callback_count;
178  }
179  for (auto const& m : response->received_messages()) {
180  source->AckMessage(m.ack_id()).then([&](future<Status>) {
181  std::lock_guard<std::mutex> lk(callback_mu);
182  ++ack_count;
183  callback_cv.notify_one();
184  });
185  }
186  callback_cv.notify_one();
187  };
188 
189  auto done = shutdown->Start({});
190  source->Start(std::move(callback));
191 
192  auto constexpr kPublishCount = 1000;
193  std::set<std::string> expected_ids = [&] {
194  std::set<std::string> ids;
195  std::vector<future<StatusOr<std::string>>> message_ids;
196  for (int i = 0; i != kPublishCount; ++i) {
197  message_ids.push_back(publisher.Publish(
198  MessageBuilder{}.SetData("message-" + std::to_string(i)).Build()));
199  }
200  for (auto& id : message_ids) {
201  auto r = id.get();
202  EXPECT_STATUS_OK(r);
203  if (r) ids.insert(*std::move(r));
204  }
205  return ids;
206  }();
207 
208  wait_ack_count(kPublishCount);
209  {
210  std::lock_guard<std::mutex> lk(callback_mu);
211  for (auto const& id : received_ids) {
212  expected_ids.erase(id);
213  }
214  EXPECT_TRUE(expected_ids.empty());
215  }
216 
217  shutdown->MarkAsShutdown("test", {});
218  source->Shutdown();
219  EXPECT_STATUS_OK(done.get());
220 }
