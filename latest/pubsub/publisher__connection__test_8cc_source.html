35 TEST(PublisherConnectionTest, Basic) {

38 
39  EXPECT_CALL(*mock, AsyncPublish(_, _, _))
40  .WillOnce([&](google::cloud::CompletionQueue&,
41  std::unique_ptr<grpc::ClientContext>,
42  google::pubsub::v1::PublishRequest const& request) {
43  EXPECT_EQ(topic.FullName(), request.topic());
44  EXPECT_EQ(1, request.messages_size());
45  EXPECT_EQ("test-data-0", request.messages(0).data());
46  google::pubsub::v1::PublishResponse response;
47  response.add_message_ids("test-message-id-0");
48  return make_ready_future(make_status_or(response));
49  });

52  pubsub_internal::MakePublisherConnection(topic, {}, {}, mock);
53  auto response =
54  publisher->Publish({MessageBuilder{}.SetData("test-data-0").Build()})
55  .get();

57  EXPECT_EQ("test-message-id-0", *response);
58 }

145

154  auto publisher =
