25 class NotifyWhenMessageHandled : public pubsub::AckHandler::Impl {
26  public:
27  explicit NotifyWhenMessageHandled(
28  std::shared_ptr<SubscriptionSession> session,
29  std::unique_ptr<pubsub::AckHandler::Impl> child, std::size_t message_size)
30  : session_(std::move(session)),
31  child_(std::move(child)),
32  message_size_(message_size) {}
33  ~NotifyWhenMessageHandled() override = default;
34 
35  void ack() override {
36  auto id = ack_id();
37  child_->ack();
38  NotifySession(id);
39  }
40  void nack() override {
41  auto id = ack_id();
42  child_->nack();
43  NotifySession(id);
44  }
45  std::string ack_id() const override { return child_->ack_id(); }
46  std::int32_t delivery_attempt() const override {
47  return child_->delivery_attempt();
48  }

117 future<Status> SubscriptionSession::Start() {
118  // Transfer the result from shutdown_manager_'s future to a new future that
119  // has a proper cancellation callback.
120  class MoveCapture {
121  public:
122  explicit MoveCapture(std::weak_ptr<SubscriptionSession> const& w)
123  : cancel_([w] {
124  auto self = w.lock();
125  if (!self) return;
126  self->Cancel();
127  }) {}
128 
129  void operator()(future<Status> f) { cancel_.set_value(f.get()); }
130  future<Status> get_future() { return cancel_.get_future(); }
131 
132  private:
133  promise<Status> cancel_;
134  };
135 
136  auto self = shared_from_this();
137  MoveCapture chain(self);
138  auto f = chain.get_future();
139  shutdown_manager_.Start().then(std::move(chain));
140  shutdown_manager_.StartAsyncOperation(__func__, "PullOne", executor_,
141  [self] { self->PullOne(); });
142  return f;
143 }

145 void SubscriptionSession::MessageHandled(std::string const& ack_id,
146  std::size_t) {
147  if (shutdown_manager_.FinishedOperation("callback")) return;
148  std::unique_lock<std::mutex> lk(mu_);
149  leases_.erase(ack_id);
150  if (!callback_flow_control_.Release(1)) return;
151  auto self = shared_from_this();
152  // After the callback re-schedule ourselves.
153  shutdown_manager_.StartAsyncOperation(__func__, "HandleQueue", executor_,
154  [self] { self->HandleQueue(); });
155 }

157 void SubscriptionSession::Cancel() {
158  // Stop any more work from being created as soon as possible.
159  shutdown_manager_.MarkAsShutdown(__func__, {});
160  std::unique_lock<std::mutex> lk(mu_);
161  // Cancel any existing timers.
162  if (refresh_timer_.valid()) refresh_timer_.cancel();
163  // Schedule a nack for each pending message, using the existing executor.
164  NackAll(lk, messages_.begin(), messages_.end());
165 }

167 void SubscriptionSession::PullOne() {
168  std::unique_lock<std::mutex> lk(mu_);
169  auto self = shared_from_this();
170  if (shutdown_manager_.FinishedOperation("PullOne")) return;
171  if (!messages_.empty()) {
172  shutdown_manager_.StartAsyncOperation(__func__, "HandleQueue", executor_,
173  [self] { self->HandleQueue(); });
174  return;
175  }
176 
177  lk.unlock();
178 
179  auto context = absl::make_unique<grpc::ClientContext>();
180  google::pubsub::v1::PullRequest request;
181  request.set_subscription(params_.full_subscription_name);
182  request.set_max_messages(1);
183  shutdown_manager_.StartOperation(__func__, "OnPull", [&] {
184  stub_->AsyncPull(executor_, std::move(context), request)
185  .then([self](future<StatusOr<google::pubsub::v1::PullResponse>> f) {
186  self->OnPull(f.get());
187  });
188  });
189 }

191 void SubscriptionSession::OnPull(StatusOr<google::pubsub::v1::PullResponse> r) {
192  if (!r) {
193  shutdown_manager_.MarkAsShutdown(__func__, std::move(r).status());
194  return;
195  }
196  std::unique_lock<std::mutex> lk(mu_);
197  if (shutdown_manager_.FinishedOperation("OnPull")) {
198  NackAll(lk, r->received_messages().begin(), r->received_messages().end());
199  return;
200  }
201 
202  auto const now = std::chrono::system_clock::now();
203  auto const estimated_server_deadline = now + std::chrono::seconds(10);
204  auto const handling_deadline = now + params_.options.max_deadline_time();
205  for (auto const& rm : r->received_messages()) {
206  leases_.emplace(rm.ack_id(),
207  LeaseStatus{estimated_server_deadline, handling_deadline});
208  }
209  std::move(r->mutable_received_messages()->begin(),
210  r->mutable_received_messages()->end(),
211  std::back_inserter(messages_));
212  auto self = shared_from_this();
213  shutdown_manager_.StartAsyncOperation(__func__, "HandleQueue", executor_,
214  [self] { self->HandleQueue(); });
215 
216  // Setup a timer to refresh the message leases. We do not want to immediately
217  // refresh them because there is a good chance they will be handled before
218  // the minimum lease time, and it seems wasteful to refresh the lease just to
219  // quickly turnaround and ack or nack the message.
220  StartRefreshTimer(std::move(lk),
221  std::chrono::system_clock::now() +
222  std::chrono::seconds(kMinimumAckDeadlineSeconds));
223 }

225 void SubscriptionSession::HandleQueue(std::unique_lock<std::mutex> lk) {
226  auto self = shared_from_this();
227  if (shutdown_manager_.FinishedOperation("HandleQueue")) {
228  // We do not wait for this to complete before completing the session
229  // shutdown, NackAll() does not need to callback or otherwise inform the
230  // session of any state changes.
231  NackAll(lk, messages_.begin(), messages_.end());
232  return;
233  }
234  if (messages_.empty()) {
235  shutdown_manager_.StartAsyncOperation(__func__, "PullOne", executor_,
236  [self] { self->PullOne(); });
237  return;
238  }
239 
240  while (!messages_.empty() && callback_flow_control_.MaybeAdmit(1)) {
241  auto m = std::move(messages_.front());
242  messages_.pop_front();
243 
244  struct MoveCapture {
245  std::function<void(pubsub::Message, pubsub::AckHandler)> callback;
246  pubsub::Message m;
247  std::unique_ptr<pubsub::AckHandler::Impl> h;
248  void operator()() {
249  GCP_LOG(DEBUG) << "calling callback(" << h->ack_id() << ")";
250  callback(std::move(m), pubsub::AckHandler(std::move(h)));
251  }
252  };
253  auto message = FromProto(std::move(*m.mutable_message()));
254  auto const message_size = MessageSize(message);
255  std::unique_ptr<pubsub::AckHandler::Impl> handler =
256  absl::make_unique<DefaultAckHandlerImpl>(
257  executor_, stub_, params_.full_subscription_name,
258  std::move(*m.mutable_ack_id()), m.delivery_attempt());
259  handler = absl::make_unique<NotifyWhenMessageHandled>(
260  self, std::move(handler), message_size);
261 
262  shutdown_manager_.StartAsyncOperation(
263  __func__, "callback", executor_,
264  MoveCapture{params_.callback, std::move(message), std::move(handler)});
265  }

268 void SubscriptionSession::RefreshMessageLeases(
269  std::unique_lock<std::mutex> lk) {
270  if (leases_.empty() || refreshing_leases_) return;
271 
272  google::pubsub::v1::ModifyAckDeadlineRequest request;
273  auto const now = std::chrono::system_clock::now();
274  request.set_subscription(params_.full_subscription_name);
275  using seconds = std::chrono::seconds;
276  auto extension = seconds(kMaximumAckDeadlineSeconds);
277  for (auto const& kv : leases_) {
278  // This message lease cannot be extended any further, and we do not want to
279  // send an extension of 0 seconds because that is a nack.
280  if (kv.second.handling_deadline < now + seconds(1)) continue;
281  auto const message_extension =
282  std::chrono::duration_cast<seconds>(kv.second.handling_deadline - now);
283  extension = (std::min)(extension, message_extension);
284  request.add_ack_ids(kv.first);
285  }
286  auto const new_deadline = now + extension;
287  if (request.ack_ids().empty()) {
288  StartRefreshTimer(std::move(lk), new_deadline);
289  return;
290  }
291  request.set_ack_deadline_seconds(
292  static_cast<std::int32_t>(extension.count()));
293  refreshing_leases_ = true;
294  auto context = absl::make_unique<grpc::ClientContext>();
295  auto weak = std::weak_ptr<SubscriptionSession>(shared_from_this());
296  lk.unlock();
297 
298  shutdown_manager_.StartOperation(__func__, "OnRefreshMessageLeases", [&] {
299  stub_->AsyncModifyAckDeadline(executor_, std::move(context), request)
300  .then([weak, request, new_deadline](future<Status>) {
301  auto self = weak.lock();
302  if (!self) return;
303  self->OnRefreshMessageLeases(request, new_deadline);
304  });
305  });
306 }

308 void SubscriptionSession::OnRefreshMessageLeases(
309  google::pubsub::v1::ModifyAckDeadlineRequest const& request,
310  std::chrono::system_clock::time_point new_server_deadline) {
311  if (shutdown_manager_.FinishedOperation(__func__)) return;
312  std::unique_lock<std::mutex> lk(mu_);
313  refreshing_leases_ = false;
314  for (auto const& ack : request.ack_ids()) {
315  auto i = leases_.find(ack);
316  if (i == leases_.end()) continue;
317  i->second.estimated_server_deadline = new_server_deadline;
318  }
319  StartRefreshTimer(std::move(lk), new_server_deadline);
320 }

322 void SubscriptionSession::StartRefreshTimer(
323  std::unique_lock<std::mutex>,
324  std::chrono::system_clock::time_point new_server_deadline) {
325  auto weak = std::weak_ptr<SubscriptionSession>(shared_from_this());
326  auto deadline =
327  new_server_deadline - std::chrono::seconds(kAckDeadlineSlackSeconds);
328  if (test_refresh_period_.count() != 0) {
329  deadline = std::chrono::system_clock::now() + test_refresh_period_;
330  }
331 
332  shutdown_manager_.StartOperation(__func__, "OnRefreshTimer", [&] {
333  if (refresh_timer_.valid()) refresh_timer_.cancel();
334  refresh_timer_ = executor_.MakeDeadlineTimer(deadline).then(
335  [weak](future<StatusOr<std::chrono::system_clock::time_point>> f) {
336  auto self = weak.lock();
337  if (!self) return;
338  self->OnRefreshTimer(!f.get().ok());
339  });
340  });
341 }

349 void SubscriptionSession::NackAll(
350  std::unique_lock<std::mutex> const&,
351  google::pubsub::v1::ModifyAckDeadlineRequest request) {
352  if (request.ack_ids().empty()) return;
353  request.set_ack_deadline_seconds(0);
354  request.set_subscription(params_.full_subscription_name);
355 
356  // This is a fire-and-forget asynchronous operation. If it fails we will
357  // simply let the normal timeouts expire any messages.
358  (void)stub_->AsyncModifyAckDeadline(
359  executor_, absl::make_unique<grpc::ClientContext>(), request);
360 }
