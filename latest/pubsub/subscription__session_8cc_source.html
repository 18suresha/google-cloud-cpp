24 class NotifyWhenMessageHandled : public pubsub::AckHandler::Impl {
25  public:
26  explicit NotifyWhenMessageHandled(
27  std::weak_ptr<SubscriptionSession> session,
28  std::unique_ptr<pubsub::AckHandler::Impl> child, std::size_t message_size)
29  : session_(std::move(session)),
30  child_(std::move(child)),
31  message_size_(message_size) {}
32  ~NotifyWhenMessageHandled() override = default;
33 
34  void ack() override {
35  auto id = ack_id();
36  child_->ack();
37  NotifySession(id);
38  }
39  void nack() override {
40  auto id = ack_id();
41  child_->nack();
42  NotifySession(id);
43  }
44  std::string ack_id() const override { return child_->ack_id(); }
45  std::int32_t delivery_attempt() const override {
46  return child_->delivery_attempt();
47  }
48 
49  private:
50  void NotifySession(std::string const& id) {
51  auto session = session_.lock();
52  if (session) session->MessageHandled(id, message_size_);
53  }
54 
55  std::weak_ptr<SubscriptionSession> session_;
56  std::unique_ptr<pubsub::AckHandler::Impl> child_;
57  std::size_t message_size_;
58 };

64 future<Status> SubscriptionSession::Start() {
65  auto self = shared_from_this();
66  std::weak_ptr<SubscriptionSession> w(self);
67  result_ = promise<Status>{[w] {
68  auto self = w.lock();
69  if (!self) return;
70  self->Cancel();
71  }};
72  executor_.RunAsync([self] { self->PullOne(); });
73  return result_.get_future();

76 void SubscriptionSession::MessageHandled(std::string const& ack_id,
77  std::size_t) {
78  std::unique_lock<std::mutex> lk(mu_);
79  ack_deadlines_.erase(ack_id);

81  // After the callback re-schedule ourselves.

83  executor_.RunAsync([self] { self->HandleQueue(); });

92 void SubscriptionSession::PullOne() {
93  std::unique_lock<std::mutex> lk(mu_);
94  if (!messages_.empty()) {
95  HandleQueue(std::move(lk));
96  return;
97  }
98 
99  if (shutdown_) {
100  result_.set_value(Status{});
101  return;
102  }
103 
104  auto self = shared_from_this();
105  lk.unlock();
106 
107  auto context = absl::make_unique<grpc::ClientContext>();
108  google::pubsub::v1::PullRequest request;
109  request.set_subscription(params_.full_subscription_name);
110  request.set_max_messages(1);
111  stub_->AsyncPull(executor_, std::move(context), request)
112  .then([self](future<StatusOr<google::pubsub::v1::PullResponse>> f) {
113  self->OnPull(f.get());
114  });
115 }

117 void SubscriptionSession::OnPull(StatusOr<google::pubsub::v1::PullResponse> r) {
118  if (!r) {
119  // TODO(#4699) - retry on transient failures
120  result_.set_value(std::move(r).status());
121  return;
122  }
123 
124  std::unique_lock<std::mutex> lk(mu_);
125  auto const now = std::chrono::system_clock::now();
126  auto const estimated_server_deadline = now + std::chrono::seconds(10);
127  auto const handling_deadline = now + params_.options.max_deadline_time();
128  for (auto const& rm : r->received_messages()) {
129  ack_deadlines_.emplace(
130  rm.ack_id(), AckStatus{estimated_server_deadline, handling_deadline});
131  }
132  std::move(r->mutable_received_messages()->begin(),
133  r->mutable_received_messages()->end(),
134  std::back_inserter(messages_));
135  if (refresh_timer_.valid()) {
136  refresh_timer_.cancel();
137  refresh_timer_ = {};
138  }
139  auto self = shared_from_this();
140  executor_.RunAsync([self] { self->HandleQueue(); });
141 
142  StartRefreshAckDeadlinesTimer(
143  std::move(lk), std::chrono::system_clock::now() +
144  std::chrono::seconds(kMinimumAckDeadlineSeconds));
145 }

147 void SubscriptionSession::HandleQueue(std::unique_lock<std::mutex> lk) {
148  auto self = shared_from_this();
149  if (messages_.empty()) {
150  executor_.RunAsync([self] { self->PullOne(); });
151  return;
152  }
153  auto m = std::move(messages_.front());
154  messages_.pop_front();
155  lk.unlock();
156 
157  std::unique_ptr<pubsub::AckHandler::Impl> handler =
158  absl::make_unique<DefaultAckHandlerImpl>(
159  executor_, stub_, params_.full_subscription_name,
160  std::move(*m.mutable_ack_id()), m.delivery_attempt());
161  // TODO(#4645) - use a better estimation for the message size.
162  auto const message_size = m.message().data().size();
163  handler = absl::make_unique<NotifyWhenMessageHandled>(
164  self, std::move(handler), message_size);
165  params_.callback(FromProto(std::move(*m.mutable_message())),
166  pubsub::AckHandler(std::move(handler)));
167 }

169 void SubscriptionSession::RefreshAckDeadlines(std::unique_lock<std::mutex> lk) {
170  if (ack_deadlines_.empty() || refreshing_deadlines_) return;
171  google::pubsub::v1::ModifyAckDeadlineRequest request;
172  auto const now = std::chrono::system_clock::now();
173  request.set_subscription(params_.full_subscription_name);
174  using seconds = std::chrono::seconds;
175  auto extension = seconds(kMaximumAckDeadlineSeconds);
176  for (auto const& kv : ack_deadlines_) {
177  // This message cannot be extended any further, and we do not want to send
178  // an extension of 0 seconds because that is a nack.
179  if (kv.second.handling_deadline < now + seconds(1)) continue;
180  auto const message_extension =
181  std::chrono::duration_cast<seconds>(kv.second.handling_deadline - now);
182  extension = (std::min)(extension, message_extension);
183  request.add_ack_ids(kv.first);
184  }
185  auto const new_deadline = now + extension;
186  if (request.ack_ids().empty()) {
187  StartRefreshAckDeadlinesTimer(std::move(lk), new_deadline);
188  return;
189  }
190  request.set_ack_deadline_seconds(
191  static_cast<std::int32_t>(extension.count()));
192  refreshing_deadlines_ = true;
193  auto context = absl::make_unique<grpc::ClientContext>();
194  auto weak = std::weak_ptr<SubscriptionSession>(shared_from_this());
195  lk.unlock();
196  stub_->AsyncModifyAckDeadline(executor_, std::move(context), request)
197  .then([weak, request, new_deadline](future<Status>) {
198  auto self = weak.lock();
199  if (!self) return;
200  self->OnRefreshAckDeadlines(request, new_deadline);
201  });
202 }

204 void SubscriptionSession::OnRefreshAckDeadlines(
205  google::pubsub::v1::ModifyAckDeadlineRequest const& request,
206  std::chrono::system_clock::time_point new_server_deadline) {
207  std::unique_lock<std::mutex> lk(mu_);
208  for (auto const& ack : request.ack_ids()) {
209  auto i = ack_deadlines_.find(ack);
210  if (i == ack_deadlines_.end()) continue;
211  i->second.estimated_server_deadline = new_server_deadline;
212  }
213  StartRefreshAckDeadlinesTimer(std::move(lk), new_server_deadline);
214 }

216 void SubscriptionSession::StartRefreshAckDeadlinesTimer(
217  std::unique_lock<std::mutex>,
218  std::chrono::system_clock::time_point new_server_deadline) {
219  if (shutdown_) return;
220  auto weak = std::weak_ptr<SubscriptionSession>(shared_from_this());
221  auto deadline =
222  new_server_deadline - std::chrono::seconds(kAckDeadlineSlackSeconds);
223  if (test_refresh_period_.count() != 0) {
224  deadline = std::chrono::system_clock::now() + test_refresh_period_;
225  }
226  refresh_timer_ = executor_.MakeDeadlineTimer(deadline).then(
227  [weak](future<StatusOr<std::chrono::system_clock::time_point>> f) {
228  auto self = weak.lock();
229  if (!self || !f.get()) return;
230  self->OnRefreshAckDeadlinesTimer();
231  });
232 }

234 void SubscriptionSession::OnRefreshAckDeadlinesTimer() {
235  std::unique_lock<std::mutex> lk(mu_);
236  refreshing_deadlines_ = false;
