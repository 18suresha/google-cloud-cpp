35 void SubscriptionLeaseManagement::Shutdown() {

37  // Cancel any existing timers.

71 void SubscriptionLeaseManagement::OnRead(
72  StatusOr<google::pubsub::v1::StreamingPullResponse> const& response) {
73  if (!response) {
74  shutdown_manager_->MarkAsShutdown(__func__, response.status());
75  std::unique_lock<std::mutex> lk(mu_);
76  // Cancel any existing timers.
77  if (refresh_timer_.valid()) refresh_timer_.cancel();
78  return;
79  }
80  std::unique_lock<std::mutex> lk(mu_);
81  auto const now = std::chrono::system_clock::now();
82  auto const estimated_server_deadline = now + std::chrono::seconds(10);
83  auto const handling_deadline = now + max_deadline_time_;
84  for (auto const& rm : response->received_messages()) {
85  leases_.emplace(rm.ack_id(),
86  LeaseStatus{estimated_server_deadline, handling_deadline});
87  }
88  // Setup a timer to refresh the message leases. We do not want to immediately
89  // refresh them because there is a good chance they will be handled before
90  // the minimum lease time, and it seems wasteful to refresh the lease just to
91  // quickly turnaround and ack or nack the message.
92  StartRefreshTimer(std::move(lk),
93  std::chrono::system_clock::now() + kMinimumAckDeadline);
94 }

96 void SubscriptionLeaseManagement::RefreshMessageLeases(
97  std::unique_lock<std::mutex> lk) {
98  using seconds = std::chrono::seconds;
99 
100  if (leases_.empty() || refreshing_leases_) return;
101 
102  std::vector<std::string> ack_ids;
103  ack_ids.reserve(leases_.size());
104  auto extension = kMaximumAckDeadline;
105  auto const now = std::chrono::system_clock::now();
106  for (auto const& kv : leases_) {
107  // This message lease cannot be extended any further, and we do not want to
108  // send an extension of 0 seconds because that is a nack.
109  if (kv.second.handling_deadline < now + seconds(1)) continue;
110  auto const message_extension =
111  std::chrono::duration_cast<seconds>(kv.second.handling_deadline - now);
112  extension = (std::min)(extension, message_extension);
113  ack_ids.push_back(kv.first);
114  }
115  auto const new_deadline = now + extension;
116  if (ack_ids.empty()) {
117  StartRefreshTimer(std::move(lk), new_deadline);
118  return;
119  }
120  lk.unlock();
121  child_->ExtendLeases(ack_ids, extension);
122  lk.lock();
123  for (auto const& ack : ack_ids) {
124  auto i = leases_.find(ack);
125  if (i == leases_.end()) continue;
126  i->second.estimated_server_deadline = new_deadline;
127  }
128  StartRefreshTimer(std::move(lk), new_deadline);
129 }

131 void SubscriptionLeaseManagement::StartRefreshTimer(
132  std::unique_lock<std::mutex>,
133  std::chrono::system_clock::time_point new_server_deadline) {
134  std::weak_ptr<SubscriptionLeaseManagement> weak = shared_from_this();
135  auto deadline = new_server_deadline - kAckDeadlineSlack;
136 
137  shutdown_manager_->StartOperation(__func__, "OnRefreshTimer", [&] {
138  if (refresh_timer_.valid()) refresh_timer_.cancel();
139  refresh_timer_ = timer_factory_(deadline).then([weak](future<Status> f) {
140  if (auto self = weak.lock()) self->OnRefreshTimer(!f.get().ok());
141  });
142  });
143 }

145 void SubscriptionLeaseManagement::OnRefreshTimer(bool cancelled) {
146  if (shutdown_manager_->FinishedOperation(__func__)) return;
147  if (cancelled) return;
148  RefreshMessageLeases(std::unique_lock<std::mutex>(mu_));
149 }

151 void SubscriptionLeaseManagement::NackAll(std::unique_lock<std::mutex> lk) {
152  if (leases_.empty()) return;
153  std::vector<std::string> ack_ids;
154  ack_ids.reserve(leases_.size());
155  for (auto const& kv : leases_) {
156  ack_ids.push_back(kv.first);
157  }
158  lk.unlock();
159  BulkNack(std::move(ack_ids));
160 }
