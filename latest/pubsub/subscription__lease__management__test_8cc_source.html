47 TEST(SubscriptionLeaseManagementTest, NormalLifecycle) {
48  auto mock = std::make_shared<pubsub_testing::MockSubscriptionBatchSource>();
49  BatchCallback batch_callback;
50  EXPECT_CALL(*mock, Start).WillOnce([&](BatchCallback cb) {
51  batch_callback = std::move(cb);
52  });
53 
54  auto constexpr kTestDeadline = std::chrono::seconds(345);
55  {
56  ::testing::InSequence sequence;
57  // We expect this message to be acked.
58  EXPECT_CALL(*mock, AckMessage("ack-0-1")).WillOnce(SimpleAckNack);
59  EXPECT_CALL(*mock, ExtendLeases)
60  // Then a simulated timer refreshes the lease for the remaining
61  // messages.
62  .WillOnce([&](std::vector<std::string> const& ack_ids,
63  std::chrono::seconds extension) {
64  EXPECT_THAT(ack_ids, ElementsAre("ack-0-0", "ack-0-2"));
65  EXPECT_LE(std::abs((kTestDeadline - extension).count()), 2);
66  return make_ready_future(Status{});
67  });
68  // Then a message is nacked.
69  EXPECT_CALL(*mock, NackMessage("ack-0-2")).WillOnce(SimpleAckNack);
70  EXPECT_CALL(*mock, ExtendLeases)
71  // The a simulated timer refreshes the leases.
72  .WillOnce([&](std::vector<std::string> const& ack_ids,
73  std::chrono::seconds extension) {
74  EXPECT_THAT(ack_ids, ElementsAre("ack-0-0"));
75  EXPECT_LE(std::abs((kTestDeadline - extension).count()), 2);
76  return make_ready_future(Status{});
77  });
78  // Then all unhandled messages are nacked on shutdown.
79  EXPECT_CALL(*mock, BulkNack(ElementsAre("ack-0-0")))
80  .WillOnce([](std::vector<std::string> const&) {
81  return make_ready_future(Status{});
82  });
83  EXPECT_CALL(*mock, Shutdown).Times(1);
84  }

86  google::cloud::internal::AutomaticallyCreatedBackgroundThreads background;
87  std::vector<promise<Status>> timers;
88  int cancel_count = 0;
89  auto make_timer = [&](std::chrono::system_clock::time_point) {
90  promise<Status> p([&cancel_count] { ++cancel_count; });
91  auto f = p.get_future().then([](future<Status> f) { return f.get(); });
92  timers.push_back(std::move(p));
93  return f;
94  };
95  auto shutdown_manager = std::make_shared<SessionShutdownManager>();
96  auto uut = SubscriptionLeaseManagement::CreateForTesting(
97  background.cq(), shutdown_manager, make_timer, mock,
98  std::chrono::seconds(kTestDeadline));
99 
100  auto done = shutdown_manager->Start({});
101  uut->Start([](StatusOr<google::pubsub::v1::StreamingPullResponse> const&) {});
102 
103  batch_callback(GenerateMessages("0-", 3));
104  ASSERT_EQ(1, timers.size());
105 
106  // Ack one of the messages and then fire the timer. The expectations set above
107  // will verify that only the remaining messages have their lease extended.
108  uut->AckMessage("ack-0-1");
109  timers[0].set_value({});
110  EXPECT_THAT(2, timers.size());
111 
112  // Ack one more message and trigger the new timer.
113  uut->NackMessage("ack-0-2");
114  timers[1].set_value({});
115  EXPECT_THAT(3, timers.size());
116 
117  shutdown_manager->MarkAsShutdown(__func__, Status{});
118  uut->Shutdown();
119  EXPECT_EQ(3, cancel_count);
120  timers[2].set_value(Status(StatusCode::kCancelled, "test-cancel"));
121  EXPECT_THAT(done.get(), StatusIs(StatusCode::kOk));
122 }
