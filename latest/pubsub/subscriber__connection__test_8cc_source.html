41 TEST(SubscriberConnectionTest, Basic) {
42  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
43  Subscription const subscription("test-project", "test-subscription");
44 
45  EXPECT_CALL(*mock, AsyncStreamingPull)
46  .Times(AtLeast(1))
47  .WillRepeatedly(FakeAsyncStreamingPull);
48 
49  CompletionQueue cq;
50  auto subscriber = pubsub_internal::MakeSubscriberConnection(
51  subscription, pubsub::SubscriberOptions{},
52  ConnectionOptions{grpc::InsecureChannelCredentials()}
53  .DisableBackgroundThreads(cq),
54  mock, pubsub_testing::TestRetryPolicy(),
55  pubsub_testing::TestBackoffPolicy());
56  std::atomic_flag received_one{false};
57  promise<void> waiter;
58  auto handler = [&](Message const& m, AckHandler h) {
59  if (received_one.test_and_set()) return;
60  EXPECT_THAT(m.message_id(), StartsWith("test-message-id-"));
61  EXPECT_THAT(h.ack_id(), StartsWith("test-ack-id-"));
62  ASSERT_NO_FATAL_FAILURE(std::move(h).ack());
63  waiter.set_value();
64  };
65  std::thread t([&cq] { cq.Run(); });
66  auto response = subscriber->Subscribe({handler});
67  waiter.get_future().wait();
68  response.cancel();
69  ASSERT_STATUS_OK(response.get());
70  // We need to explicitly cancel any pending timers (some of which may be quite
71  // long) left by the subscription.
72  cq.CancelAll();
73  cq.Shutdown();
74  t.join();
75 }
