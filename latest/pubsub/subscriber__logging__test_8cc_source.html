139 TEST_F(SubscriberLoggingTest, AsyncStreamingPull) {
140  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
141  EXPECT_CALL(*mock, AsyncStreamingPull)
142  .WillOnce([](google::cloud::CompletionQueue&,
143  std::unique_ptr<grpc::ClientContext>,
144  google::pubsub::v1::StreamingPullRequest const&) {
145  auto stream = absl::make_unique<pubsub_testing::MockAsyncPullStream>();
146  EXPECT_CALL(*stream, Cancel).Times(1);
147  EXPECT_CALL(*stream, Start).WillOnce([&] {
148  return make_ready_future(true);
149  });
150  EXPECT_CALL(*stream, Read)
151  .WillOnce([&] {
152  return make_ready_future(absl::make_optional(
153  google::pubsub::v1::StreamingPullResponse{}));
154  })
155  .WillOnce([&] {
156  return make_ready_future(
157  absl::optional<google::pubsub::v1::StreamingPullResponse>{});
158  });
159  EXPECT_CALL(*stream, Write)
160  .WillOnce([&](google::pubsub::v1::StreamingPullRequest const&,
161  grpc::WriteOptions const&) {
162  return make_ready_future(true);
163  });
164  EXPECT_CALL(*stream, WritesDone).WillOnce([&] {
165  return make_ready_future(true);
166  });
167  EXPECT_CALL(*stream, Finish).WillOnce([&] {
168  return make_ready_future(Status{});
169  });
170  return stream;
171  });
172  SubscriberLogging stub(mock, TracingOptions{}.SetOptions("single_line_mode"));
173  google::cloud::CompletionQueue cq;
174 
175  google::pubsub::v1::StreamingPullRequest request;
176  request.set_subscription("test-subscription-name");
177  auto stream = stub.AsyncStreamingPull(
178  cq, absl::make_unique<grpc::ClientContext>(), request);
179  EXPECT_THAT(backend_->ClearLogLines(),
180  Contains(HasSubstr("AsyncStreamingPull")));
181 
182  EXPECT_TRUE(stream->Start().get());
183  EXPECT_THAT(backend_->ClearLogLines(), Contains(HasSubstr("Start")));
184 
185  EXPECT_TRUE(
186  stream->Write(request, grpc::WriteOptions{}.set_write_through()).get());
187  EXPECT_THAT(
188  backend_->ClearLogLines(),
189  Contains(AllOf(HasSubstr("Write"), HasSubstr("test-subscription-name"))));
190 
191  EXPECT_TRUE(stream->Read().get().has_value());
192  EXPECT_THAT(backend_->ClearLogLines(), Contains(HasSubstr("Read")));
193 
194  EXPECT_FALSE(stream->Read().get().has_value());
195  EXPECT_THAT(backend_->ClearLogLines(), Contains(HasSubstr("Read")));
196 
197  EXPECT_TRUE(stream->WritesDone().get());
198  EXPECT_THAT(backend_->ClearLogLines(), Contains(HasSubstr("WritesDone")));
199 
200  EXPECT_STATUS_OK(stream->Finish().get());
201  EXPECT_THAT(backend_->ClearLogLines(), Contains(HasSubstr("Finish")));
202 
203  stream->Cancel();
204  EXPECT_THAT(backend_->ClearLogLines(), Contains(HasSubstr("Cancel")));
205 }
