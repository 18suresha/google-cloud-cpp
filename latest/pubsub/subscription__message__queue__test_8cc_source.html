80 TEST(SubscriptionMessageQueueTest, Basic) {

82  EXPECT_CALL(*mock, Shutdown).Times(1);

84  EXPECT_CALL(*mock, Start).WillOnce([&](BatchCallback cb) {
85  batch_callback = std::move(cb);
86  });
87 
88  EXPECT_CALL(*mock, AckMessage("ack-m0")).WillOnce([](std::string const&) {
89  return make_ready_future(Status{});
90  });
91  EXPECT_CALL(*mock, NackMessage("ack-m1")).WillOnce([](std::string const&) {
92  return make_ready_future(Status{});
93  });
94  EXPECT_CALL(*mock, BulkNack).Times(1);
95 
96  auto const messages = GenerateMessages();
97  ASSERT_FALSE(messages.empty());
98  auto const response = GenerateResponse(messages);
99 
100  std::vector<google::pubsub::v1::ReceivedMessage> received;
101  auto handler = [&received](google::pubsub::v1::ReceivedMessage m) {
102  received.push_back(std::move(m));
103  };
104 
105  auto shutdown = std::make_shared<SessionShutdownManager>();
106  shutdown->Start({});
107  auto uut = SubscriptionMessageQueue::Create(shutdown, mock);
108  uut->Start(handler);
109 
110  uut->Read(1);
111  EXPECT_THAT(received, ElementsAre());
112 
113  batch_callback(response);
114  uut->Read(1);
115 
116  EXPECT_THAT(received, ElementsAre(IsProtoEqual(messages[0]),
117  IsProtoEqual(messages[1])));
118  uut->AckMessage("ack-m0", 0);
119  uut->NackMessage("ack-m1", 1);
120 
121  received.clear();
122  uut->Shutdown();
