50  promise<bool> WaitForAction() {
51  GCP_LOG(DEBUG) << __func__ << "()";
52  std::unique_lock<std::mutex> lk(mu_);
53  cv_.wait(lk, [this] { return !actions_.empty(); });
54  auto action = std::move(actions_.front());
55  actions_.pop_front();
56  return action;
57  }

59  std::unique_ptr<pubsub_testing::MockAsyncPullStream> MakeWriteFailureStream(
60  google::cloud::CompletionQueue&, std::unique_ptr<grpc::ClientContext>,
61  google::pubsub::v1::StreamingPullRequest const&) {
62  auto start_response = [this] {
63  return AddAction("Start").then([](future<bool> g) { return g.get(); });
64  };
65  auto write_response = [this](
66  google::pubsub::v1::StreamingPullRequest const&,
67  grpc::WriteOptions const&) {
68  return AddAction("Write").then([](future<bool> g) { return g.get(); });
69  };
70  auto read_response = [this] {
71  return AddAction("Read").then([](future<bool> g) {
72  auto ok = g.get();
73  using Response = google::pubsub::v1::StreamingPullResponse;
74  if (!ok) return absl::optional<Response>{};
75  return absl::make_optional(Response{});
76  });
77  };
78  auto finish_response = [this] {
79  auto s = finish_;
80  return AddAction("Finish").then(
81  [s](future<bool>) mutable { return std::move(s); });
82  };
83 
84  auto stream = absl::make_unique<pubsub_testing::MockAsyncPullStream>();
85  EXPECT_CALL(*stream, Start).WillOnce(start_response);
86  EXPECT_CALL(*stream, Write).WillRepeatedly(write_response);
87  EXPECT_CALL(*stream, Cancel).Times(AtMost(1));
88  EXPECT_CALL(*stream, Read).WillRepeatedly(read_response);
89  EXPECT_CALL(*stream, Finish).WillOnce(finish_response);
90 
91  return stream;
92  }
93 
94  future<bool> AddAction(char const* caller) {
95  GCP_LOG(DEBUG) << __func__ << "(" << caller << ")";
96  std::unique_lock<std::mutex> lk(mu_);
97  actions_.emplace_back(promise<bool>{});
98  auto f = actions_.back().get_future();
99  lk.unlock();
100  cv_.notify_one();
101  return f;
102  }
103 
104  private:
105  Status finish_;
106  std::mutex mu_;
107  std::condition_variable cv_;
108  std::deque<promise<bool>> actions_;
109 };
