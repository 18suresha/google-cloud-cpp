258  std::chrono::duration<Rep, Period> d) {
259  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(d);
260  auto nanos =
261  std::chrono::duration_cast<std::chrono::nanoseconds>(d - seconds);
262  google::protobuf::Duration result;
