72 void SubscriptionMessageQueue::OnRead(
73  std::unique_lock<std::mutex> lk,
74  google::pubsub::v1::StreamingPullResponse r) {
75  auto handle_response = [&] {
76  shutdown_manager_->FinishedOperation("OnRead");
77  for (auto& m : *r.mutable_received_messages()) {
78  auto key = m.message().ordering_key();
79  if (key.empty()) {
80  // Empty key, requires no ordering and therefore immediately runnable.
81  runnable_messages_.push_back(std::move(m));
82  continue;
83  }
84  // The message requires ordering, find out if there is an existing queue
85  // for its ordering key, and insert one if necessary.
86  auto loc = queues_.insert({key, {}});
87  // There is no queue for this ordering key, that means no other messages
88  // are present, we can push the message to the runnable queue. We leave
89  // the per-ordering-key queue as a marker for any other incoming messages
90  // with the same ordering key.
91  if (loc.second) {
92  runnable_messages_.push_back(std::move(m));
93  continue;
94  }
95  // Insert the messages into the existing queue.
96  loc.first->second.push_back(std::move(m));
97  }
98  DrainQueue(std::move(lk));
99  };
100  auto bulk_nack = [&] {
101  lk.unlock();
102  std::vector<std::string> ack_ids(r.mutable_received_messages()->size());
103  std::transform(r.mutable_received_messages()->begin(),
104  r.mutable_received_messages()->end(), ack_ids.begin(),
105  [&](google::pubsub::v1::ReceivedMessage& m) {
106  return std::move(*m.mutable_ack_id());
107  });
108  (void)source_->BulkNack(std::move(ack_ids));
109  };
110  if (!shutdown_manager_->StartOperation(__func__, "OnRead", handle_response)) {
111  bulk_nack();
112  }
113 }

115 void SubscriptionMessageQueue::Shutdown(std::unique_lock<std::mutex> lk) {
116  shutdown_ = true;
117  available_slots_ = 0;
118  QueueByOrderingKey queues;
119  queues.swap(queues_);
120  std::deque<google::pubsub::v1::ReceivedMessage> runnable_messages;
121  runnable_messages.swap(runnable_messages_);
122  lk.unlock();
123 
124  std::vector<std::string> ack_ids;
125  for (auto& kv : queues) {
126  for (auto& m : kv.second) {
127  ack_ids.push_back(std::move(*m.mutable_ack_id()));
128  }
129  }
130  for (auto& m : runnable_messages) {
131  ack_ids.push_back(std::move(*m.mutable_ack_id()));
132  }
133 
134  if (ack_ids.empty()) return;
135  source_->BulkNack(std::move(ack_ids));
136 }

138 void SubscriptionMessageQueue::DrainQueue(std::unique_lock<std::mutex> lk) {
139  while (!runnable_messages_.empty() && available_slots_ > 0 && !shutdown_) {
140  auto m = std::move(runnable_messages_.front());
141  runnable_messages_.pop_front();
142  --available_slots_;
143  // No need to track messages without an ordering key, as there is no action
144  // to take in their HandlerDone() member function.
145  if (!m.message().ordering_key().empty()) {
146  ordering_key_by_ack_id_[m.ack_id()] = m.message().ordering_key();
147  }
148  // Don't hold a lock during the callback, as the callee may call `Read()`
149  // or something similar.
150  lk.unlock();
151  callback_(std::move(m));
152  lk.lock();
153  }
154 }

156 void SubscriptionMessageQueue::HandlerDone(std::string const& ack_id) {
157  std::unique_lock<std::mutex> lk(mu_);
158  // Find out the ordering key for this message.
159  auto loc = ordering_key_by_ack_id_.find(ack_id);
160  // Messages without an ordering key are not inserted in the collection (see
161  // `DrainQueue()`), so this happens routinely.
162  if (loc == ordering_key_by_ack_id_.end()) return;
163  auto key = std::move(loc->second);
164  ordering_key_by_ack_id_.erase(loc);
165  auto ql = queues_.find(key);
166  // This is purely defensive, but should not happen.
167  if (ql == queues_.end()) return;
168  if (ql->second.empty()) {
169  // There are no more messages for this ordering key, remove the queue, as it
170  // also serves as a marker to order the next message.
171  queues_.erase(ql);
172  return;
173  }
174  auto m = std::move(ql->second.front());
175  ql->second.pop_front();
176  runnable_messages_.push_back(std::move(m));
177  DrainQueue(std::move(lk));
178 }
