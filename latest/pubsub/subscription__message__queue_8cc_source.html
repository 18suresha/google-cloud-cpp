24 void SubscriptionMessageQueue::Start(MessageCallback cb) {
25  std::unique_lock<std::mutex> lk(mu_);
26  if (callback_) return;
27  callback_ = std::move(cb);
28  lk.unlock();

30  source_->Start([weak](StatusOr<google::pubsub::v1::StreamingPullResponse> r) {

32  });
33 }

57 void SubscriptionMessageQueue::OnRead(
58  StatusOr<google::pubsub::v1::StreamingPullResponse> r) {
59  std::unique_lock<std::mutex> lk(mu_);
60  if (!r) {
61  shutdown_manager_->MarkAsShutdown(__func__, std::move(r).status());
62  Shutdown(std::move(lk));
63  return;
64  }
65  OnRead(std::move(lk), *std::move(r));
66 }

68 void SubscriptionMessageQueue::OnRead(
69  std::unique_lock<std::mutex> lk,
70  google::pubsub::v1::StreamingPullResponse r) {
71  auto handle_response = [&] {
72  shutdown_manager_->FinishedOperation("OnRead");
73  std::move(r.mutable_received_messages()->begin(),
74  r.mutable_received_messages()->end(),
75  std::back_inserter(messages_));
76  DrainQueue(std::move(lk));
77  };
78  auto bulk_nack = [&] {
79  lk.unlock();
80  std::vector<std::string> ack_ids(r.mutable_received_messages()->size());
81  std::transform(r.mutable_received_messages()->begin(),
82  r.mutable_received_messages()->end(), ack_ids.begin(),
83  [&](google::pubsub::v1::ReceivedMessage& m) {
84  return std::move(*m.mutable_ack_id());
85  });
86  (void)source_->BulkNack(std::move(ack_ids));
87  };
88  if (!shutdown_manager_->StartOperation(__func__, "OnRead", handle_response)) {
89  bulk_nack();
90  }
91 }

93 void SubscriptionMessageQueue::Shutdown(std::unique_lock<std::mutex> lk) {
94  auto messages = [&] {
95  shutdown_ = true;
96  auto messages = std::move(messages_);
97  messages_.clear();
98  lk.unlock();
99  return messages;
100  }();
101 
102  if (messages.empty()) return;
103  std::vector<std::string> ack_ids(messages.size());
104  std::transform(messages.begin(), messages.end(), ack_ids.begin(),
105  [&](google::pubsub::v1::ReceivedMessage& m) {
106  return std::move(*m.mutable_ack_id());
107  });
108  source_->BulkNack(std::move(ack_ids));
109 }

111 void SubscriptionMessageQueue::DrainQueue(std::unique_lock<std::mutex> lk) {
112  while (!messages_.empty() && read_count_ > 0 && !shutdown_) {
113  auto m = std::move(messages_.front());
114  messages_.pop_front();
115  --read_count_;
116  // Don't hold a lock during the callback, as the callee may call `Read()`
117  // or something similar.
118  lk.unlock();
119  callback_(std::move(m));
120  lk.lock();
121  }
122 }
