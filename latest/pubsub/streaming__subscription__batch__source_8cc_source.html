73 void StreamingSubscriptionBatchSource::StartStream(
74  std::shared_ptr<pubsub::RetryPolicy> retry_policy,
75  std::shared_ptr<pubsub::BackoffPolicy> backoff_policy) {
76  GCP_LOG(DEBUG) << __func__ << "()";
77  // Starting a stream is a 4 step process.
78  // 1. Create a new SubscriberStub::AsyncPullStream object.
79  // 2. Call Start() on it, which is asynchronous, which might fail, but rarely
80  // does
81  // 3. Call Write() on it, which is also asynchronous and almost always
82  // succeeds, could fail if the endpoint is invalid
83  // 4. Call Read() on it, which is also asynchronous, and it is the first
84  // chance to get errors such as kPermissionDenied, or kNotFound.
85  // Once Read() completes we can repeatedly call Read() and/or Write().
86  // Because steps 2 through 4 may fail with transient errors we need to wrap
87  // these steps in an asynchronous retry loop.
88 
89  auto request = InitialRequest();
90  auto stream = stub_->AsyncStreamingPull(
91  cq_, absl::make_unique<grpc::ClientContext>(), request);
92  if (!stream) {
93  OnRetryFailure(Status(StatusCode::kUnknown, "null stream"));
94  return;
95  }
96 
97  shutdown_manager_->StartOperation(__func__, "InitialStart", [&] {
98  RetryLoopState rs{std::move(stream), std::move(retry_policy),
99  std::move(backoff_policy)};
100  auto weak = WeakFromThis();
101  rs.stream->Start().then([weak, rs, request](future<bool> f) {
102  if (auto s = weak.lock()) s->OnStart(rs, request, f.get());
103  });
104  });
105 }

123  RetryLoopState rs, google::pubsub::v1::StreamingPullRequest const& request,
124  bool ok) {
125  GCP_LOG(DEBUG) << __func__ << "()";
126  shutdown_manager_->FinishedOperation("InitialStart");
127  if (!ok) {
128  OnInitialError(std::move(rs));
129  return;

134  .then([weak, rs](future<bool> f) mutable {
135  if (auto s = weak.lock()) s->OnInitialWrite(std::move(rs), f.get());
136  });
137  });

158 void StreamingSubscriptionBatchSource::OnInitialRead(
159  RetryLoopState rs,
160  absl::optional<google::pubsub::v1::StreamingPullResponse> response) {
161  GCP_LOG(DEBUG) << __func__ << "()";
162  shutdown_manager_->FinishedOperation("InitialRead");
163  if (!response.has_value()) {
164  OnInitialError(std::move(rs));
165  return;
166  }
167 
168  std::unique_lock<std::mutex> lk(mu_);
169  ChangeState(lk, StreamState::kActive, __func__, "success");
170  status_ = Status{};
171  stream_ = std::move(rs.stream);
172  lk.unlock();
173  auto const scheduled =
174  shutdown_manager_->StartOperation(__func__, "read", [&] {
175  OnRead(std::move(response));
176  shutdown_manager_->FinishedOperation("read");
177  });
178  if (!scheduled) {
179  Shutdown();
180  lk.lock();
181  ShutdownStream(std::move(lk), "early-shutdown");
182  }
183 }

185 void StreamingSubscriptionBatchSource::OnInitialError(RetryLoopState rs) {
186  GCP_LOG(DEBUG) << __func__ << "()";
187  auto weak = WeakFromThis();
188  auto const scheduled =
189  shutdown_manager_->StartOperation(__func__, "finish", [&] {
190  rs.stream->Finish().then([weak, rs](future<Status> f) {
191  if (auto s = weak.lock()) s->OnInitialFinish(std::move(rs), f.get());
192  });
193  shutdown_manager_->FinishedOperation("finish");
194  });
195  if (!scheduled) {
196  shutdown_manager_->FinishedOperation("stream");
197  }

200 void StreamingSubscriptionBatchSource::OnInitialFinish(RetryLoopState rs,
201  Status status) {
202  GCP_LOG(DEBUG) << __func__ << "()";
203  if (!rs.retry_policy->OnFailure(status)) {
204  OnRetryFailure(std::move(status));
205  return;
206  }
207  auto weak = WeakFromThis();
208  using F = future<StatusOr<std::chrono::system_clock::time_point>>;
209  cq_.MakeRelativeTimer(rs.backoff_policy->OnCompletion())
210  .then([weak, rs, status](F f) mutable {
211  auto s = weak.lock();
212  if (!s) return;
213  if (f.get().ok()) {
214  s->OnBackoff(std::move(rs), std::move(status));
215  } else {
216  s->shutdown_manager_->FinishedOperation("stream");
217  }
218  });
219 }

221 void StreamingSubscriptionBatchSource::OnBackoff(RetryLoopState rs,
222  Status status) {
223  GCP_LOG(DEBUG) << __func__ << "()";
224  if (rs.retry_policy->IsExhausted()) {
225  OnRetryFailure(std::move(status));
226  return;
227  }
228  auto const scheduled =
229  shutdown_manager_->StartOperation(__func__, "retry", [&] {
230  StartStream(std::move(rs.retry_policy), std::move(rs.backoff_policy));
231  shutdown_manager_->FinishedOperation("retry");
232  });
233  if (!scheduled) {
234  shutdown_manager_->FinishedOperation("stream");
235  }
236 }

245 void StreamingSubscriptionBatchSource::ReadLoop() {
246  std::unique_lock<std::mutex> lk(mu_);
247  if (stream_state_ != StreamState::kActive) return;
248  pending_read_ = true;
249  auto stream = stream_;
250  lk.unlock();
251  auto weak = WeakFromThis();
252  using ResponseType =
253  absl::optional<google::pubsub::v1::StreamingPullResponse>;

255  if (auto self = weak.lock()) self->OnRead(f.get());

259 void StreamingSubscriptionBatchSource::OnRead(
260  absl::optional<google::pubsub::v1::StreamingPullResponse> response) {
261  auto weak = WeakFromThis();
262  std::unique_lock<std::mutex> lk(mu_);
263  pending_read_ = false;
264  if (response && stream_state_ == StreamState::kActive && !shutdown_) {
265  lk.unlock();
266  callback_(*std::move(response));
267  cq_.RunAsync([weak] {
268  if (auto self = weak.lock()) self->ReadLoop();
269  });
270  return;
271  }
272  ShutdownStream(std::move(lk), response ? "state" : "read error");
273 }

275 void StreamingSubscriptionBatchSource::ShutdownStream(
276  std::unique_lock<std::mutex> lk, char const* reason) {
277  if (stream_state_ != StreamState::kActive &&
278  stream_state_ != StreamState::kDisconnecting) {
279  return;
280  }
281  ChangeState(lk, StreamState::kDisconnecting, __func__, reason);
282  if (pending_read_ || pending_write_) return;
283 
284  auto stream = stream_;
285  ChangeState(lk, StreamState::kFinishing, __func__, reason);
286  lk.unlock();
287  auto weak = WeakFromThis();
288  // There are no pending reads or writes, and something (probable a read or
289  // write error) recommends we shutdown the stream
290  stream->Finish().then([weak, stream](future<Status> f) {
291  if (auto self = weak.lock()) self->OnFinish(f.get());
292  });
293 }

295 void StreamingSubscriptionBatchSource::OnFinish(Status status) {
296  std::unique_lock<std::mutex> lk(mu_);
297  status_ = std::move(status);
298  stream_.reset();
299  ChangeState(lk, StreamState::kNull, __func__, "done");
300  if (shutdown_manager_->FinishedOperation("stream")) return;
301  lk.unlock();
302  shutdown_manager_->StartOperation(__func__, "stream", [this] {
303  StartStream(retry_policy_->clone(), backoff_policy_->clone());
304  });
305 }

307 void StreamingSubscriptionBatchSource::DrainQueues(
308  std::unique_lock<std::mutex> lk) {
309  if (ack_queue_.empty() && nack_queue_.empty() && deadlines_queue_.empty()) {
310  return;
311  }
312  if (stream_state_ != StreamState::kActive || pending_write_) return;
313  auto stream = stream_;
314  pending_write_ = true;
315 
316  std::vector<std::string> acks;
317  acks.swap(ack_queue_);
318  std::vector<std::string> nacks;
319  nacks.swap(nack_queue_);
320  std::vector<std::pair<std::string, std::chrono::seconds>> deadlines;
321  deadlines.swap(deadlines_queue_);
322  lk.unlock();
323 
324  google::pubsub::v1::StreamingPullRequest request;
325  for (auto& a : acks) request.add_ack_ids(std::move(a));
326  for (auto& n : nacks) {
327  request.add_modify_deadline_ack_ids(std::move(n));
328  request.add_modify_deadline_seconds(0);
329  }
330  for (auto& d : deadlines) {
331  request.add_modify_deadline_ack_ids(std::move(d.first));
332  request.add_modify_deadline_seconds(
333  static_cast<std::int32_t>(d.second.count()));
334  }
335  // Note that we do not use `AsyncRetryLoop()` here. The ack/nack pipeline is
336  // best-effort anyway, there is no guarantee that the server will act on any
337  // of these.
338  auto weak = WeakFromThis();
339  stream->Write(request, grpc::WriteOptions{})
340  .then([weak, stream](future<bool> f) {
341  if (auto self = weak.lock()) self->OnWrite(f.get());
342  });
343 }

345 void StreamingSubscriptionBatchSource::OnWrite(bool ok) {
346  std::unique_lock<std::mutex> lk(mu_);
347  pending_write_ = false;
348  if (ok && stream_state_ == StreamState::kActive && !shutdown_) {
349  DrainQueues(std::move(lk));
350  return;
351  }
352  ShutdownStream(std::move(lk), ok ? "state" : "write error");
353 }
