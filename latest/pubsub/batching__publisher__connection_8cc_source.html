22 // A helper callable to handle a response, it is a bit large for a lambda, and
23 // we need move-capture anyways.
24 struct Batch {
25  std::vector<promise<StatusOr<std::string>>> waiters;
26  std::weak_ptr<BatchingPublisherConnection> weak;
27 
28  void operator()(future<StatusOr<google::pubsub::v1::PublishResponse>> f) {
29  auto response = f.get();
30  if (!response) {
31  SatisfyAllWaiters(response.status());
32  if (auto batcher = weak.lock()) batcher->HandleError(response.status());
33  return;
34  }
35  if (static_cast<std::size_t>(response->message_ids_size()) !=
36  waiters.size()) {
37  SatisfyAllWaiters(
38  Status(StatusCode::kUnknown, "mismatched message id count"));
39  return;
40  }
41  int idx = 0;
42  for (auto& w : waiters) {
43  w.set_value(std::move(*response->mutable_message_ids(idx++)));
44  }
45  }
46 
47  void SatisfyAllWaiters(Status const& status) {
48  for (auto& w : waiters) w.set_value(status);
49  }
50 };

52 future<StatusOr<std::string>> BatchingPublisherConnection::Publish(
53  PublishParams p) {
54  auto const bytes = pubsub_internal::MessageSize(p.message);
55  std::unique_lock<std::mutex> lk(mu_);
56  do {
57  if (!corked_on_status_.ok()) return CorkedError();
58  // If empty we need to create the batch, even if it would be oversized,
59  // otherwise the message may be dropped.
60  if (waiters_.empty()) break;
61  auto const has_bytes_capacity =
62  current_bytes_ + bytes <= options_.maximum_batch_bytes();
63  auto const has_messages_capacity =
64  waiters_.size() < options_.maximum_batch_message_count();
65  // If there is enough room just add the message below.
66  if (has_bytes_capacity && has_messages_capacity) break;
67  // We need to flush the existing batch, that will release the lock, and then
68  // we try again.
69  FlushImpl(std::move(lk));
70  lk = std::unique_lock<std::mutex>(mu_);
71  } while (true);
72 
73  auto proto = pubsub_internal::ToProto(std::move(p.message));
74  waiters_.emplace_back();
75  auto f = waiters_.back().get_future();
76 
77  // Use RAII to preserve the strong exception guarantee.
78  struct UndoPush {
79  std::vector<promise<StatusOr<std::string>>>* waiters;
80 
81  ~UndoPush() {
82  if (waiters != nullptr) waiters->pop_back();
83  }
84  void release() { waiters = nullptr; }
85  } undo{&waiters_};
86 
87  *pending_.add_messages() = std::move(proto);
88  undo.release(); // no throws after this point, we can rest easy
89  current_bytes_ += bytes;
90  MaybeFlush(std::move(lk));
91  return f;
92 }

107 void BatchingPublisherConnection::HandleError(Status const& status) {
108  std::unique_lock<std::mutex> lk(mu_);
109  // An error should discard pending messages and block future messages only
110  // if ordering is required.
111  if (!RequiresOrdering()) return;
112  corked_on_status_ = status;
113  pending_.Clear();
114  std::vector<promise<StatusOr<std::string>>> waiters;
115  waiters.swap(waiters_);
116  lk.unlock();
117  for (auto& p : waiters) {
118  struct MoveCapture {
119  promise<StatusOr<std::string>> p;
120  Status status;
121  void operator()() { p.set_value(std::move(status)); }
122  };
123  cq_.RunAsync(MoveCapture{std::move(p), status});
124  }
125 }

127 void BatchingPublisherConnection::MaybeFlush(std::unique_lock<std::mutex> lk) {
128  auto const too_many_messages =
129  waiters_.size() >= options_.maximum_batch_message_count();
130  auto const too_many_bytes = current_bytes_ >= options_.maximum_batch_bytes();
131  if (too_many_messages || too_many_bytes) {
132  FlushImpl(std::move(lk));
133  return;
134  }
135  // If the batch is empty obviously we do not need a timer, and if it has more
136  // than one element then we have setup a timer previously and there is no need
137  // to set it again.
138  if (pending_.messages_size() != 1) return;
139  auto const expiration = batch_expiration_ =
140  std::chrono::system_clock::now() + options_.maximum_hold_time();
141  lk.unlock();
142  // We need a weak_ptr<> because this class owns the completion queue,
143  // creating a lambda with a shared_ptr<> owning this class would create a
144  // cycle. Unfortunately some older compiler/libraries lack
145  // `weak_from_this()`.
146  auto weak = std::weak_ptr<BatchingPublisherConnection>(shared_from_this());
147  // Note that at this point the lock is released, so whether the timer
148  // schedules later on schedules in this thread has no effect.
149  cq_.MakeDeadlineTimer(expiration)
150  .then([weak](future<StatusOr<std::chrono::system_clock::time_point>>) {
151  if (auto self = weak.lock()) self->OnTimer();
152  });
153 }

155 void BatchingPublisherConnection::OnTimer() {
156  std::unique_lock<std::mutex> lk(mu_);
157  if (std::chrono::system_clock::now() < batch_expiration_) {
158  // We may get many "old" timers for batches that have already flushed due to
159  // size. Trying to cancel these timers is a bit hopeless, they might trigger
160  // even if we attempt to cancel them. This test is more robust.
161  return;
162  }
163  FlushImpl(std::move(lk));
164 }

166 future<StatusOr<std::string>> BatchingPublisherConnection::CorkedError() {
167  promise<StatusOr<std::string>> p;
168  auto f = p.get_future();
169 
170  struct MoveCapture {
171  promise<StatusOr<std::string>> p;
172  Status status;
173  void operator()() { p.set_value(std::move(status)); }
174  };
175  cq_.RunAsync(MoveCapture{std::move(p), corked_on_status_});
176 
177  return f;
178 }

180 void BatchingPublisherConnection::FlushImpl(std::unique_lock<std::mutex> lk) {
181  if (pending_.messages().empty()) return;
182 
183  Batch batch;
184  batch.waiters.swap(waiters_);
185  google::pubsub::v1::PublishRequest request;
186  request.Swap(&pending_);
187  // Reserve enough capacity for the next batch.
188  pending_.mutable_messages()->Reserve(
189  static_cast<int>(options_.maximum_batch_message_count()));
190  current_bytes_ = 0;
191  lk.unlock();
192 
193  batch.weak = shared_from_this();
194  request.set_topic(topic_full_name_);
195  sink_->AsyncPublish(std::move(request)).then(std::move(batch));
196 }
