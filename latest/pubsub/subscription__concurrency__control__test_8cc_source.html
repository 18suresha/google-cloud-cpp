37 TEST(SubscriptionConcurrentControlTest, ScheduleParallelCallbacks) {
38  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
39  pubsub::Subscription const subscription("test-project", "test-subscription");
40 
41  std::mutex mu;
42  int count = 0;
43  auto constexpr kTestHwm = 4;
44  auto constexpr kBatchSize = 2 * kTestHwm;
45 
46  auto generate = [&](google::cloud::CompletionQueue&,
47  std::unique_ptr<grpc::ClientContext>,
48  google::pubsub::v1::PullRequest const& request) {
49  EXPECT_EQ(subscription.FullName(), request.subscription());
50  google::pubsub::v1::PullResponse response;
51  for (int i = 0; i != kBatchSize; ++i) {
52  auto& m = *response.add_received_messages();
53  std::lock_guard<std::mutex> lk(mu);
54  m.set_ack_id("test-ack-id-" + std::to_string(count));
55  m.mutable_message()->set_message_id("test-message-id-" +
56  std::to_string(count));
57  ++count;
58  }
59  return make_ready_future(make_status_or(response));
60  };
61 
62  EXPECT_CALL(*mock, AsyncPull).Times(AtLeast(1)).WillRepeatedly(generate);
63  EXPECT_CALL(*mock, AsyncAcknowledge)
64  .Times(AtLeast(kBatchSize))
65  .WillRepeatedly([](google::cloud::CompletionQueue&,
66  std::unique_ptr<grpc::ClientContext>,
67  google::pubsub::v1::AcknowledgeRequest const&) {
68  return make_ready_future(Status{});
69  });
70  EXPECT_CALL(*mock, AsyncModifyAckDeadline)
71  .WillRepeatedly([](google::cloud::CompletionQueue&,
72  std::unique_ptr<grpc::ClientContext>,
73  google::pubsub::v1::ModifyAckDeadlineRequest const&) {
74  return make_ready_future(Status{});
75  });
76 
77  google::cloud::CompletionQueue cq;
78  std::vector<std::thread> tasks;
79  std::generate_n(std::back_inserter(tasks), 4,
80  [&] { return std::thread([&cq] { cq.Run(); }); });
81 
82  std::mutex handlers_mu;
83  std::condition_variable handlers_cv;
84  std::deque<pubsub::AckHandler> handlers;
85  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
86  std::lock_guard<std::mutex> lk(handlers_mu);
87  handlers.push_back(std::move(h));
88  auto const m = static_cast<std::size_t>(kTestHwm);
89  if (handlers.size() >= m) handlers_cv.notify_one();
90  };
91 
92  auto session = SubscriptionSession::Create(
93  mock, cq,
94  {subscription.FullName(), handler,
95  pubsub::SubscriptionOptions{}.set_concurrency_watermarks(0, kTestHwm)});
96  auto response = session->Start();
97 
98  auto ack_current = [&](std::unique_lock<std::mutex> lk) {
99  std::deque<pubsub::AckHandler> tmp;
100  tmp.swap(handlers);
101  lk.unlock();
102  for (auto& h : tmp) std::move(h).ack();
103  };
104  for (auto i = 0; i != kBatchSize / kTestHwm; ++i) {
105  std::unique_lock<std::mutex> lk(handlers_mu);
106  auto const m = static_cast<std::size_t>(kTestHwm);
107  handlers_cv.wait(lk, [&] { return handlers.size() >= m; });
108  ack_current(std::move(lk));
109  }
110 
111  response.cancel();
112  ack_current(std::unique_lock<std::mutex>(handlers_mu));
113  EXPECT_STATUS_OK(response.get());
114 
115  cq.Shutdown();
116  for (auto& t : tasks) t.join();
117 }

120 TEST(SubscriptionConcurrentControlTest, ScheduleParallelCallbacksLimits) {
121  auto mock = std::make_shared<pubsub_testing::MockSubscriberStub>();
122  pubsub::Subscription const subscription("test-project", "test-subscription");
123 
124  auto constexpr kThreadCount = 4;
125  auto constexpr kTestHwm = 16;
126  auto constexpr kBatchSize = 32;
127  auto constexpr kTotalMessages = 4 * kBatchSize;
128 
129  auto generate = [&](google::cloud::CompletionQueue&,
130  std::unique_ptr<grpc::ClientContext>,

142  };

145  EXPECT_CALL(*mock, AsyncAcknowledge)
146  .Times(AtLeast(kBatchSize))
147  .WillRepeatedly([](google::cloud::CompletionQueue&,
148  std::unique_ptr<grpc::ClientContext>,
149  google::pubsub::v1::AcknowledgeRequest const&) {
150  return make_ready_future(Status{});
151  });
152  EXPECT_CALL(*mock, AsyncModifyAckDeadline)
153  .WillRepeatedly([](google::cloud::CompletionQueue&,
154  std::unique_ptr<grpc::ClientContext>,
155  google::pubsub::v1::ModifyAckDeadlineRequest const&) {
156  return make_ready_future(Status{});
157  });
158 
159  google::cloud::CompletionQueue cq;
160  std::vector<std::thread> tasks;
161  std::generate_n(std::back_inserter(tasks), kThreadCount,
162  [&] { return std::thread([&cq] { cq.Run(); }); });
163 
164  auto const callback_running_time = std::chrono::microseconds(100);
165  using TimerFuture = future<StatusOr<std::chrono::system_clock::time_point>>;
166  std::mutex handlers_mu;
167  std::condition_variable handlers_cv;
168  int current_callback_count = 0;
169  int maximum_callback_count = 0;
170  int total_callbacks = 0;
171 
172  auto handler_done = [&](pubsub::AckHandler h) {
173  std::unique_lock<std::mutex> lk(handlers_mu);
174  --current_callback_count;
175  std::move(h).ack();
176  if (++total_callbacks < kTotalMessages) return;
177  lk.unlock();
178  handlers_cv.notify_one();
179  };
180 
181  auto handler = [&](pubsub::Message const&, pubsub::AckHandler h) {
182  std::unique_lock<std::mutex> lk(handlers_mu);
183  ++current_callback_count;
184  maximum_callback_count =
185  (std::max)(current_callback_count, maximum_callback_count);
186  lk.unlock();
187  struct MoveCapture {
188  pubsub::AckHandler h;
189  std::function<void(pubsub::AckHandler)> done;
190  void operator()(TimerFuture) { done(std::move(h)); }
191  };
192  cq.MakeRelativeTimer(callback_running_time)
193  .then(MoveCapture{std::move(h), handler_done});
194  };
195 
196  auto session = [&] {
197  return SubscriptionSession::Create(
198  mock, cq,
199  {subscription.FullName(), handler,
200  pubsub::SubscriptionOptions{}.set_concurrency_watermarks(
201  kTestHwm / 2, kTestHwm)})
202  ->Start();
203  }();
204 
205  {
206  std::unique_lock<std::mutex> lk(handlers_mu);
207  handlers_cv.wait(lk, [&] { return total_callbacks >= kTotalMessages; });
208  }
209  session.cancel();
210  auto status = session.get();
211  EXPECT_STATUS_OK(status);
212 
213  EXPECT_LT(0, maximum_callback_count);
214  EXPECT_GE(kTestHwm, maximum_callback_count);
215 
216  cq.Shutdown();
217  for (auto& t : tasks) t.join();
218 }
