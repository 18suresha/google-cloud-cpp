207 int main(int argc, char* argv[]) try {
208  using CommandType = std::function<void(
209  google::cloud::bigtable::InstanceAdmin,
210  google::cloud::bigtable::CompletionQueue, std::vector<std::string>)>;
211 
212  std::map<std::string, CommandType> commands = {
213  {"async-get-instance", &AsyncGetInstance},
214  {"async-get-cluster", &AsyncGetCluster},
215  {"async-list-instances", &AsyncListInstances},
216  {"async-list-clusters", &AsyncListClusters},
217  {"async-list-all-clusters", &AsyncListAllClusters}};
218 
219  google::cloud::bigtable::CompletionQueue cq;
220 
221  {
222  // Force each command to generate its Usage string, so we can provide a good
223  // usage string for the whole program. We need to create an InstanceAdmin
224  // object to do this, but that object is never used, it is passed to the
225  // commands, without any calls made to it.
226  google::cloud::bigtable::InstanceAdmin unused(
227  google::cloud::bigtable::CreateDefaultInstanceAdminClient(
228  "unused-project", google::cloud::bigtable::ClientOptions()));
229  for (auto&& kv : commands) {
230  try {
231  kv.second(unused, cq, {});
232  } catch (Usage const& u) {
233  command_usage += " ";
234  command_usage += u.msg;
235  command_usage += "\n";
236  }
237  }
238  }
239 
240  if (argc < 3) {
241  PrintUsage(argv[0], "Missing command and/or project-id");
242  return 1;
243  }
244 
245  std::vector<std::string> args;
246  args.emplace_back(argv[0]);
247  std::string const command_name = argv[1];
248  std::string const project_id = argv[2];
249  std::transform(argv + 3, argv + argc, std::back_inserter(args),
250  [](char* x) { return std::string(x); });
251 
252  auto command = commands.find(command_name);
253  if (commands.end() == command) {
254  PrintUsage(argv[0], "Unknown command: " + command_name);
255  return 1;
256  }
257 
258  // Start a thread to run the completion queue event loop.
259  std::thread runner([&cq] { cq.Run(); });
260 
261  // Create an instance admin endpoint.
262  //! [connect instance admin]
263  google::cloud::bigtable::InstanceAdmin instance_admin(
264  google::cloud::bigtable::CreateDefaultInstanceAdminClient(
265  project_id, google::cloud::bigtable::ClientOptions()));
266  //! [connect instance admin]
267 
268  command->second(instance_admin, cq, args);
269 
270  // Shutdown the completion queue event loop and join the thread.
271  cq.Shutdown();
272  runner.join();
273 
274  return 0;
275 } catch (Usage const& ex) {
276  PrintUsage(argv[0], ex.msg);
277  return 1;
278 } catch (std::exception const& ex) {
279  std::cerr << "Standard C++ exception raised: " << ex.what() << std::endl;
280  return 1;
281 }
