34  namespace cbt = google::cloud::bigtable;

36  using google::cloud::StatusOr;
37  [](cbt::InstanceAdmin instance_admin, cbt::CompletionQueue cq,
38  std::string const& instance_id, std::string const& zone) {
39  std::string display_name("Put description here");
40  std::string cluster_id = instance_id + "-c1";
41  cbt::ClusterConfig cluster_config =
42  cbt::ClusterConfig(zone, 3, cbt::ClusterConfig::HDD);
43  cbt::InstanceConfig config(instance_id, display_name,
44  {{cluster_id, cluster_config}});
45  config.set_type(cbt::InstanceConfig::PRODUCTION);
46 
47  future<StatusOr<google::bigtable::admin::v2::Instance>> instance_future =
48  instance_admin.AsyncCreateInstance(cq, config);
49  // Show how to perform additional work while the long running operation
50  // completes. The application could use future.then() instead.

52  instance_future.wait_for(std::chrono::seconds(2));
53  std::cout << '.' << std::flush;
54  auto instance = instance_future.get();
55  if (!instance) throw std::runtime_error(instance.status().message());
56  std::cout << "DONE, details=" << instance->name() << "\n";
57  }
59  (std::move(instance_admin), std::move(cq), argv.at(0), argv.at(1));
60 }

571 void AsyncSetIamPolicy(google::cloud::bigtable::InstanceAdmin instance_admin,
572  google::cloud::bigtable::CompletionQueue cq,
573  std::vector<std::string> const& argv) {
575  namespace cbt = google::cloud::bigtable;
576  using google::cloud::future;
577  using google::cloud::IamPolicy;
578  using google::cloud::StatusOr;
579  [](cbt::InstanceAdmin instance_admin, cbt::CompletionQueue cq,
580  std::string const& instance_id, std::string const& role,
581  std::string const& member) {
582  future<StatusOr<IamPolicy>> updated_future =
583  instance_admin.AsyncGetIamPolicy(cq, instance_id)
584  .then([cq, instance_admin, role, member, instance_id](
585  future<StatusOr<IamPolicy>> current_future) mutable {
586  auto current = current_future.get();
587  if (!current) {
588  return google::cloud::make_ready_future<StatusOr<IamPolicy>>(
589  current.status());
590  }
