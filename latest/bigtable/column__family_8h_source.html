33  * Implement a thin wrapper around google::bigtable::admin::v2::GcRule.
34  *
35  * Provides functions to create GcRules in a convenient form.

37 class GcRule {
38  public:
39  /// Create a garbage collection rule that keeps the last @p n versions.
40  static GcRule MaxNumVersions(std::int32_t n) {
41  GcRule tmp;
42  tmp.gc_rule_.set_max_num_versions(n);
43  return tmp;

46  /**
47  * Return a garbage collection rule that deletes cells in a column older than
48  * the given duration.
49  *
50  * The function accepts any instantiation of `std::chrono::duration<>` for the
51  * @p duration parameter. For example:
52  *
53  * @code
54  * auto rule1 = bigtable::GcRule::MaxAge(std::chrono::hours(48));
55  * auto rule2 = bigtable::GcRule::MaxAge(std::chrono::seconds(48 * 3600));
56  * @endcode
57  *
58  * @tparam Rep a placeholder to match the Rep tparam for @p duration type, the
59  * semantics of this template parameter are documented in
60  * `std::chrono::duration<>` (in brief, the underlying arithmetic type
61  * used to store the number of ticks), for our purposes it is simply a
62  * formal parameter.
63  * @tparam Period a placeholder to match the Period tparam for @p duration
64  * type, the semantics of this template parameter are documented in
65  * `std::chrono::duration<>` (in brief, the length of the tick in seconds,
66  * expressed as a `std::ratio<>`), for our purposes it is simply a formal
67  * parameter.
68  *
69  * @see
70  * [std::chrono::duration<>](http://en.cppreference.com/w/cpp/chrono/duration)
71  * for more details.
72  */
73  template <typename Rep, typename Period>
74  static GcRule MaxAge(std::chrono::duration<Rep, Period> duration) {
75  GcRule tmp;
76  auto& max_age = *tmp.gc_rule_.mutable_max_age();
77  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(duration);
78  max_age.set_seconds(seconds.count());
79  std::chrono::nanoseconds nanos =
80  std::chrono::duration_cast<std::chrono::nanoseconds>(duration) -
81  seconds;
82  max_age.set_nanos(static_cast<std::int32_t>(nanos.count()));
83  return tmp;
84  }

86  /**
87  * Return a GcRule that deletes cells if all the rules passed in would delete
88  * the cells.
89  *
90  * @tparam GcRuleTypes the type of the GC rule arguments. They must all be
91  * convertible to GcRule.
92  * @param gc_rules the set of GC rules.
93  */
94  template <typename... GcRuleTypes>
95  static GcRule Intersection(GcRuleTypes&&... gc_rules) {
96  // This ugly thing provides a better compile-time error message than just
97  // letting the compiler figure things out N levels deep as it recurses on
98  // add_intersection().
99  static_assert(
100  internal::conjunction<
101  std::is_convertible<GcRuleTypes, GcRule>...>::value,
102  "The arguments to Intersection must be convertible to GcRule");
103  GcRule tmp;
104  auto& intersection = *tmp.gc_rule_.mutable_intersection();
105  std::initializer_list<GcRule> list{std::forward<GcRuleTypes>(gc_rules)...};
106  for (GcRule const& rule : list) {
107  *intersection.add_rules() = rule.as_proto();
108  }
109  return tmp;
110  }
111 
112  /**
113  * Return a GcRule that deletes cells if any the rules passed in would delete
114  * the cells.
115  *
116  * @tparam GcRuleTypes the type of the GC rule arguments. They must all be
117  * convertible to GcRule.
118  * @param gc_rules the set of GC rules.
119  */
120  template <typename... GcRuleTypes>
121  static GcRule Union(GcRuleTypes&&... gc_rules) {
122  // This ugly thing provides a better compile-time error message than just
123  // letting the compiler figure things out N levels deep as it recurses on
124  // add_intersection().
125  static_assert(internal::conjunction<
126  std::is_convertible<GcRuleTypes, GcRule>...>::value,
127  "The arguments to Union must be convertible to GcRule");
128  GcRule tmp;
129  auto& union_ = *tmp.gc_rule_.mutable_union_();
130  std::initializer_list<GcRule> list{std::forward<GcRuleTypes>(gc_rules)...};
131  for (GcRule const& rule : list) {
132  *union_.add_rules() = rule.as_proto();
133  }
134  return tmp;
135  }

143  google::bigtable::admin::v2::GcRule&& as_proto() && {
144  return std::move(gc_rule_);

162 /**
163  * Define the interfaces to create column family modifications.
164  *
165  * Applications can modify a Cloud Bigtable schema through a set of column
166  * family modifications. These modifications may include creating new column
167  * families, deleting existing column families, or changing the garbage
168  * collection rules for existing column families.
169  *
170  * This class contain helper functions to create the different protos
171  * encapsulating these changes.
172  */
173 class ColumnFamilyModification {
174  public:
175  /// Return a modification that creates a new column family.
176  static ColumnFamilyModification Create(std::string id, GcRule gc) {
177  ColumnFamilyModification tmp;
178  tmp.mod_.set_id(std::move(id));
179  *tmp.mod_.mutable_create()->mutable_gc_rule() = std::move(gc).as_proto();
180  return tmp;
181  }
182 
183  /// Return a modification that creates a new column family.
184  static ColumnFamilyModification Update(std::string id, GcRule gc) {
185  ColumnFamilyModification tmp;
186  tmp.mod_.set_id(std::move(id));
187  *tmp.mod_.mutable_update()->mutable_gc_rule() = std::move(gc).as_proto();
188  return tmp;
189  }
190 
191  /// Return a modification that drops the @p id column family.
