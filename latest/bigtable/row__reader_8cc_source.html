107 void RowReader::MakeRequest() {
108  response_ = {};
109  processed_chunks_count_ = 0;
110 
111  google::bigtable::v2::ReadRowsRequest request;
112  request.set_table_name(table_name_);
113  request.set_app_profile_id(app_profile_id_);
114 
115  auto row_set_proto = row_set_.as_proto();
116  request.mutable_rows()->Swap(&row_set_proto);
117 
118  auto filter_proto = filter_.as_proto();
119  request.mutable_filter()->Swap(&filter_proto);
120 
121  if (rows_limit_ != NO_ROWS_LIMIT) {
122  request.set_rows_limit(rows_limit_ - rows_count_);
123  }
124 
125  context_ = google::cloud::internal::make_unique<grpc::ClientContext>();
126  retry_policy_->Setup(*context_);
127  backoff_policy_->Setup(*context_);
128  metadata_update_policy_.Setup(*context_);
129  stream_ = client_->ReadRows(context_.get(), request);
130  stream_is_open_ = true;
131 
132  parser_ = parser_factory_->Create();
133 }

135 bool RowReader::NextChunk() {
136  ++processed_chunks_count_;
137  while (processed_chunks_count_ >= response_.chunks_size()) {
138  processed_chunks_count_ = 0;
139  bool response_is_valid = stream_->Read(&response_);
140  if (!response_is_valid) {
141  response_ = {};
142  return false;
143  }
144  }
145  return true;

148 StatusOr<internal::OptionalRow> RowReader::Advance() {
149  if (operation_cancelled_) {
150  return Status(StatusCode::kCancelled, "Operation cancelled.");
151  }
152  while (true) {
153  internal::OptionalRow row;
154  grpc::Status status = AdvanceOrFail(row);
155  if (status.ok()) {
156  return row;
157  }
158  row.reset();
159 
160  // In the unlikely case when we have already reached the requested
161  // number of rows and still receive an error (the parser can throw
162  // an error at end of stream for example), there is no need to
163  // retry and we have no good value for rows_limit anyway.
164  if (rows_limit_ != NO_ROWS_LIMIT && rows_limit_ <= rows_count_) {
165  return row;
166  }
167 
168  if (!last_read_row_key_.empty()) {
169  // We've returned some rows and need to make sure we don't
170  // request them again.
171  row_set_ = row_set_.Intersect(RowRange::Open(last_read_row_key_, ""));
172  }
173 
174  // If we receive an error, but the retriable set is empty, stop.
175  if (row_set_.IsEmpty()) {
176  return row;
177  }
178 
179  if (!retry_policy_->OnFailure(status)) {
180  return internal::MakeStatusFromRpcError(status);
181  }
182 
183  auto delay = backoff_policy_->OnCompletion(status);
184  std::this_thread::sleep_for(delay);
185 
186  // If we reach this place, we failed and need to restart the call.
187  MakeRequest();
188  }
189 }

191 grpc::Status RowReader::AdvanceOrFail(internal::OptionalRow& row) {
192  row.reset();
193  grpc::Status status;
194  if (!stream_) {
195  MakeRequest();
196  }
197  while (!parser_->HasNext()) {
198  if (NextChunk()) {
199  parser_->HandleChunk(
200  std::move(*(response_.mutable_chunks(processed_chunks_count_))),
201  status);
202  if (!status.ok()) {
203  return status;
204  }
205  continue;
206  }
207 
208  // Here, there are no more chunks to look at. Close the stream,
209  // finalize the parser and return OK with no rows unless something
210  // fails during cleanup.
211  stream_is_open_ = false;
212  status = stream_->Finish();
213  if (!status.ok()) {
214  return status;
215  }
216  parser_->HandleEndOfStream(status);
217  return status;
218  }
219 
220  // We have a complete row in the parser.
221  Row parsed_row = parser_->Next(status);
222  if (!status.ok()) {
223  return status;
224  }
225  row.emplace(std::move(parsed_row));
226  ++rows_count_;
227  last_read_row_key_ = std::string(row.value().row_key());
228 
229  return status;
230 }

232 void RowReader::Cancel() {
233  operation_cancelled_ = true;
234  if (!stream_is_open_) {
235  return;
236  }
237  context_->TryCancel();
238 
239  // Also drain any data left unread
240  google::bigtable::v2::ReadRowsResponse response;
241  while (stream_->Read(&response)) {
242  }
243 
244  stream_is_open_ = false;
245  (void)stream_->Finish(); // ignore errors
246 }
