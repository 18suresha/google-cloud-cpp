106 void RowReader::MakeRequest() {
107  response_ = {};
108  processed_chunks_count_ = 0;
109 
110  google::bigtable::v2::ReadRowsRequest request;
111  request.set_table_name(table_name_);
112  request.set_app_profile_id(app_profile_id_);
113 
114  auto row_set_proto = row_set_.as_proto();
115  request.mutable_rows()->Swap(&row_set_proto);
116 
117  auto filter_proto = filter_.as_proto();
118  request.mutable_filter()->Swap(&filter_proto);
119 
120  if (rows_limit_ != NO_ROWS_LIMIT) {
121  request.set_rows_limit(rows_limit_ - rows_count_);
122  }
123 
124  context_ = absl::make_unique<grpc::ClientContext>();
125  retry_policy_->Setup(*context_);
126  backoff_policy_->Setup(*context_);
127  metadata_update_policy_.Setup(*context_);
128  stream_ = client_->ReadRows(context_.get(), request);
129  stream_is_open_ = true;
130 
131  parser_ = parser_factory_->Create();
132 }

134 bool RowReader::NextChunk() {
135  ++processed_chunks_count_;
136  while (processed_chunks_count_ >= response_.chunks_size()) {
137  processed_chunks_count_ = 0;
138  bool response_is_valid = stream_->Read(&response_);
139  if (!response_is_valid) {
140  response_ = {};
141  return false;
142  }
143  }
144  return true;
145 }

147 StatusOr<RowReader::OptionalRow> RowReader::Advance() {
148  if (operation_cancelled_) {
149  return Status(StatusCode::kCancelled, "Operation cancelled.");
150  }
151  while (true) {
152  OptionalRow row;
153  grpc::Status status = AdvanceOrFail(row);
154  if (status.ok()) {
155  return row;
156  }
157  row.reset();
158 
159  // In the unlikely case when we have already reached the requested
160  // number of rows and still receive an error (the parser can throw
161  // an error at end of stream for example), there is no need to
162  // retry and we have no good value for rows_limit anyway.
163  if (rows_limit_ != NO_ROWS_LIMIT && rows_limit_ <= rows_count_) {
164  return row;
165  }
166 
167  if (!last_read_row_key_.empty()) {
168  // We've returned some rows and need to make sure we don't
169  // request them again.
170  row_set_ = row_set_.Intersect(RowRange::Open(last_read_row_key_, ""));
171  }
172 
173  // If we receive an error, but the retryable set is empty, stop.
174  if (row_set_.IsEmpty()) {
175  return row;
176  }
177 
178  if (!retry_policy_->OnFailure(status)) {
179  return MakeStatusFromRpcError(status);
180  }
181 
182  auto delay = backoff_policy_->OnCompletion(status);
183  std::this_thread::sleep_for(delay);
184 
185  // If we reach this place, we failed and need to restart the call.
186  MakeRequest();
187  }
188 }

190 grpc::Status RowReader::AdvanceOrFail(OptionalRow& row) {
191  row.reset();
192  grpc::Status status;
193  if (!stream_) {
194  MakeRequest();
195  }
196  while (!parser_->HasNext()) {
197  if (NextChunk()) {
198  parser_->HandleChunk(
199  std::move(*(response_.mutable_chunks(processed_chunks_count_))),
200  status);
201  if (!status.ok()) {
202  return status;
203  }
204  continue;
205  }
206 
207  // Here, there are no more chunks to look at. Close the stream,
208  // finalize the parser and return OK with no rows unless something
209  // fails during cleanup.
210  stream_is_open_ = false;
211  status = stream_->Finish();
212  if (!status.ok()) {
213  return status;
214  }
215  parser_->HandleEndOfStream(status);
216  return status;
217  }
218 
219  // We have a complete row in the parser.
220  Row parsed_row = parser_->Next(status);
221  if (!status.ok()) {
222  return status;
223  }
224  row.emplace(std::move(parsed_row));
225  ++rows_count_;
226  last_read_row_key_ = std::string(row.value().row_key());
227 
228  return status;
229 }

231 void RowReader::Cancel() {
232  operation_cancelled_ = true;
233  if (!stream_is_open_) {
234  return;
235  }
236  context_->TryCancel();
237 
238  // Also drain any data left unread
239  google::bigtable::v2::ReadRowsResponse response;
240  while (stream_->Read(&response)) {
241  }
242 
243  stream_is_open_ = false;
244  (void)stream_->Finish(); // ignore errors
245 }
