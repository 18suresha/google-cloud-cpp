152 void RowReader::MakeRequest() {
153  response_ = {};
154  processed_chunks_count_ = 0;
155 
156  google::bigtable::v2::ReadRowsRequest request;
157 
158  bigtable::internal::SetCommonTableOperationRequest<
159  google::bigtable::v2::ReadRowsRequest>(request, app_profile_id_.get(),
160  table_name_.get());
161  auto row_set_proto = row_set_.as_proto();
162  request.mutable_rows()->Swap(&row_set_proto);
163 
164  auto filter_proto = filter_.as_proto();
165  request.mutable_filter()->Swap(&filter_proto);
166 
167  if (rows_limit_ != NO_ROWS_LIMIT) {
168  request.set_rows_limit(rows_limit_ - rows_count_);
169  }
170 
171  context_ = google::cloud::internal::make_unique<grpc::ClientContext>();
172  retry_policy_->Setup(*context_);
173  backoff_policy_->Setup(*context_);
174  metadata_update_policy_.Setup(*context_);
175  stream_ = client_->ReadRows(context_.get(), request);
176  stream_is_open_ = true;
177 
178  parser_ = parser_factory_->Create();
179 }

181 bool RowReader::NextChunk() {
182  ++processed_chunks_count_;
183  while (processed_chunks_count_ >= response_.chunks_size()) {
184  processed_chunks_count_ = 0;
185  bool response_is_valid = stream_->Read(&response_);
186  if (!response_is_valid) {
187  response_ = {};
188  return false;
189  }
190  }
191  return true;
192 }

194 void RowReader::Advance(internal::OptionalRow& row) {
195  while (true) {
196  grpc::Status status;
197  status_ = status = AdvanceOrFail(row);
198  if (status.ok()) {
199  return;
200  }
201 
202  // In the unlikely case when we have already reached the requested
203  // number of rows and still receive an error (the parser can throw
204  // an error at end of stream for example), there is no need to
205  // retry and we have no good value for rows_limit anyway.
206  if (rows_limit_ != NO_ROWS_LIMIT && rows_limit_ <= rows_count_) {
207  return;
208  }
209 
210  if (!last_read_row_key_.empty()) {
211  // We've returned some rows and need to make sure we don't
212  // request them again.
213  row_set_ = row_set_.Intersect(RowRange::Open(last_read_row_key_, ""));
214  }
215 
216  // If we receive an error, but the retriable set is empty, stop.
217  if (row_set_.IsEmpty()) {
218  return;
219  }
220 
221  if (!status.ok() && !retry_policy_->OnFailure(status)) {
222  if (raise_on_error_) {
223  google::cloud::internal::ThrowRuntimeError("Unretriable error: " +
224  status.error_message());
225  /*NOTREACHED*/
226  }
227  return;
228  }
229 
230  auto delay = backoff_policy_->OnCompletion(status);
231  std::this_thread::sleep_for(delay);
232 
233  // If we reach this place, we failed and need to restart the call.
234  MakeRequest();
235  }
236 }

238 grpc::Status RowReader::AdvanceOrFail(internal::OptionalRow& row) {
239  grpc::Status status;
240  row.reset();
241  while (!parser_->HasNext()) {
242  if (NextChunk()) {
243  parser_->HandleChunk(
244  std::move(*(response_.mutable_chunks(processed_chunks_count_))),
245  status);
246  if (!status.ok()) {
247  return status;
248  }
249  continue;
250  }
251 
252  // Here, there are no more chunks to look at. Close the stream,
253  // finalize the parser and return OK with no rows unless something
254  // fails during cleanup.
255  stream_is_open_ = false;
256  status = stream_->Finish();
257  if (!status.ok()) {
258  return status;
259  }
260  parser_->HandleEndOfStream(status);
261  return status;
262  }
263 
264  // We have a complete row in the parser.
265  Row parsed_row = parser_->Next(status);
266  if (!status.ok()) {
267  return status;
268  }
269  row.emplace(std::move(parsed_row));
270  ++rows_count_;
271  last_read_row_key_ = std::string(row.value().row_key());
272 
273  return status;
274 }

276 void RowReader::Cancel() {
277  operation_cancelled_ = true;
278  if (!stream_is_open_) {
279  return;
280  }
281  context_->TryCancel();
282 
283  // Also drain any data left unread
284  google::bigtable::v2::ReadRowsResponse response;
285  while (stream_->Read(&response)) {
286  }
287 
288  stream_is_open_ = false;
289  (void)stream_->Finish(); // ignore errors
290 }
