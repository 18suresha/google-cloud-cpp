107  StatusOr<google::bigtable::admin::v2::Table> table =
108  admin.GetTable(table_id, cbt::TableAdmin::FULL);
109  if (!table &&

116  // The schema returned by a `CreateTable()` request does not include all
117  // the metadata for a table, we need to explicitly request the rest:
118  table = admin.GetTable(table_id, cbt::TableAdmin::FULL);
119  }
120  if (!table) throw std::runtime_error(table.status().message());
121  std::cout << "Table metadata: " << table->DebugString() << "\n";

299 void GetOrCreateFamily(google::cloud::bigtable::TableAdmin const& admin,
300  std::vector<std::string> const& argv) {
301  // [START bigtable_get_or_create_family]
302  namespace cbt = google::cloud::bigtable;
303  using google::cloud::StatusOr;
304  [](cbt::TableAdmin admin, std::string const& table_id,
305  std::string const& family_name) {
306  StatusOr<google::bigtable::admin::v2::Table> schema =
307  admin.GetTable(table_id, cbt::TableAdmin::FULL);
308 
309  if (!schema) throw std::runtime_error(schema.status().message());
310  auto pos = schema->column_families().find(family_name);
311  if (pos == schema->column_families().end()) {
312  // Try to create the column family instead:
313  auto modified = admin.ModifyColumnFamilies(
314  table_id,
315  {cbt::ColumnFamilyModification::Create(
316  family_name, cbt::GcRule::MaxAge(std::chrono::hours(5 * 24)))});
317 
318  if (!modified) throw std::runtime_error(schema.status().message());
319  schema = *std::move(modified);
320  pos = schema->column_families().find(family_name);
321  }
322 
323  if (pos == schema->column_families().end()) {
324  throw std::runtime_error("GetOrCreateFamily failed");
325  }
326 
327  google::bigtable::admin::v2::ColumnFamily family = pos->second;
328  std::cout << "Column family name: " << pos->first
329  << "\nColumn family details: " << family.DebugString() << "\n";

445 void WaitForConsistencyCheck(google::cloud::bigtable::TableAdmin const& admin,
446  std::vector<std::string> const& argv) {
448  namespace cbt = google::cloud::bigtable;
449  using google::cloud::future;
450  using google::cloud::StatusOr;
451  [](cbt::TableAdmin admin, std::string const& table_id) {
452  StatusOr<std::string> consistency_token =
453  admin.GenerateConsistencyToken(table_id);
454  if (!consistency_token) {
455  throw std::runtime_error(consistency_token.status().message());
456  }
457  future<StatusOr<cbt::Consistency>> consistent_future =
458  admin.WaitForConsistency(table_id, *consistency_token);
459  auto final = consistent_future.then(
460  [&consistency_token](future<StatusOr<cbt::Consistency>> f) {
461  auto is_consistent = f.get();
462  if (!is_consistent) {
463  throw std::runtime_error(is_consistent.status().message());
464  }
465  if (*is_consistent == cbt::Consistency::kConsistent) {
466  std::cout << "Table is consistent with token " << *consistency_token
467  << "\n";
468  } else {
469  std::cout
470  << "Table is not yet consistent, Please try again later with"
471  << " the same token (" << *consistency_token << ")\n";
472  }
473  });
474  final.get(); // simplify example by blocking until operation is done.
475  }
477  (std::move(admin), argv.at(0));
478 }

517 void RunAll(std::vector<std::string> const& argv) {
518  namespace examples = ::google::cloud::bigtable::examples;
519  namespace cbt = google::cloud::bigtable;
520 
521  if (!argv.empty()) throw examples::Usage{"auto"};
522  if (!examples::RunAdminIntegrationTests()) return;
523  examples::CheckEnvironmentVariablesAreSet({
524  "GOOGLE_CLOUD_PROJECT",
525  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
526  });
527  auto const project_id =
528  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
529  auto const instance_id = google::cloud::internal::GetEnv(
530  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
531  .value();
532 
533  cbt::TableAdmin admin(
534  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
535  instance_id);
536 
537  // If a previous run of these samples crashes before cleaning up there may be
538  // old tables left over. As there are quotas on the total number of tables we
539  // remove stale tables after 48 hours.
540  std::cout << "\nCleaning up old tables" << std::endl;
541  google::cloud::bigtable::testing::CleanupStaleTables(admin);
542 
543  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
544  // This table is actually created and used to test the positive case (e.g.
545  // GetTable() and "table does exist")
546  auto table_id_1 = google::cloud::bigtable::testing::RandomTableId(generator);
547  // This table does not exist and used to test the negative case (e.g.
548  // GetTable() but "table does not exist")
549  auto table_id_2 = google::cloud::bigtable::testing::RandomTableId(generator);
550 
551  auto table_1 = admin.CreateTable(
552  table_id_1, cbt::TableConfig(
553  {
554  {"fam", cbt::GcRule::MaxNumVersions(10)},
555  {"foo", cbt::GcRule::MaxNumVersions(3)},
556  },
557  {}));
558  if (!table_1) throw std::runtime_error(table_1.status().message());
559 
560  std::cout << "\nRunning ListTables() example" << std::endl;
561  ListTables(admin, {});
