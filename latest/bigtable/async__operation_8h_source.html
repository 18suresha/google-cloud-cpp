58 /**
59  * Represents a pending asynchronous operation.
60  *
61  * When applications create an asynchronous operations with a `CompletionQueue`
62  * they provide a callback to be invoked when the operation completes
63  * (successfully or not). The completion queue type-erases the callback and
64  * hides it in a class derived from `AsyncOperation`. A shared pointer to the
65  * `AsyncOperation` is returned by the completion queue so library developers
66  * can cancel the operation if needed.
67  */
68 class AsyncOperation {
69  public:
70  virtual ~AsyncOperation() = default;
71 
72  /**
73  * Requests that the operation be canceled.
74  *
75  * The result of canceling the operation is reported via `Notify()`, which
76  * invokes the callback registered when the operation was created.
77  */
78  virtual void Cancel() = 0;
79 
80  enum Disposition {
81  CANCELLED,
82  COMPLETED,
83  };
84 
85  private:
86  friend class internal::CompletionQueueImpl;
87  /**
88  * Notifies the application that the operation completed.
89  *
90  * Derived classes wrap the callbacks provided by the application and invoke
91  * the callback when this virtual member function is called.
92  *
93  * @param cq the completion queue sending the notification, this is useful in
94  * case the callback needs to retry the operation.
95  * @param disposition `COMPLETED` if the operation completed, `CANCELED` if
96  * the operation were canceled. Note that errors are a "normal" completion.
97  * @return Whether the operation is completed (e.g. in case of streaming
98  * response, it would return true only after the stream is finished).
99  */
100  virtual bool Notify(CompletionQueue& cq, Disposition disposition) = 0;
101 };
