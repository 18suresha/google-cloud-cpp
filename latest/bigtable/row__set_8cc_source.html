21 RowSet RowSet::Intersect(bigtable::RowRange const& range) const {
22  // Special case: "all rows", return the argument range.
23  if (row_set_.row_keys().empty() && row_set_.row_ranges().empty()) {
24  return RowSet(range);
25  }
26  // Normal case: find the intersection with
27  // row keys and row ranges in the RowSet.
28  RowSet result;
29  for (auto const& key : row_set_.row_keys()) {
30  if (range.Contains(key)) {
31  *result.row_set_.add_row_keys() = key;
32  }
33  }
34  for (auto const& r : row_set_.row_ranges()) {
35  auto i = range.Intersect(RowRange(r));
36  if (std::get<0>(i)) {
37  *result.row_set_.add_row_ranges() = std::move(std::get<1>(i)).as_proto();
38  }
39  }
40  // Another special case: a RowSet() with no entries
41  // means "all rows", but we want "no rows".
42  if (result.row_set_.row_keys().empty() &&
43  result.row_set_.row_ranges().empty()) {
44  return RowSet(bigtable::RowRange::Empty());
45  }
46  return result;
47 }

49 bool RowSet::IsEmpty() const {
50  if (row_set_.row_keys_size() > 0) {
51  return false;
52  }
53  for (auto const& r : row_set_.row_ranges()) {
54  if (!RowRange(r).IsEmpty()) {
55  return false;
56  }
57  }
58  // We are left with only empty ranges (empty) or nothing at all
59  // (meaning "all rows").
60  return row_set_.row_ranges_size() > 0;
61 }
