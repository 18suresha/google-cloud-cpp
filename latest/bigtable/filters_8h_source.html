27 /**
28  * Define the interfaces to create filter expressions.
29  *
30  * Example:
31  * @code
32  * // Get only data from the "fam" column family, and only the latest value.
33  * auto filter = Filter::Chain(Filter::Family("fam"), Filter::Latest(1));
34  * table->ReadRow("foo", std::move(filter));
35  * @endcode
36  *
37  * Those filters that use regular expressions, expect the patterns to be in
38  * the [RE2](https://github.com/google/re2/wiki/Syntax) syntax.
39  *
40  * @note Special care need be used with the expression used. Some of the
41  * filtered values (column names, row keys, cell values) are byte sequences,
42  * and can contain arbitrary bytes, the `\C` escape sequence must be used
43  * if a true wildcard is desired. The `.` character will not match the new
44  * line character `\n`, because `.` means `[^\n]` in RE2. As new line
45  * characters may be present in a binary value, you may need to explicitly
46  * match it using "\\n". The double escape is necessary because RE2 needs
47  * to get the escape sequence.
48  */

50  public:
51  Filter(Filter&&) noexcept = default;
52  Filter& operator=(Filter&&) noexcept = default;
53  Filter(Filter const&) = default;
54  Filter& operator=(Filter const&) = default;

170  /**
171  * Return a filter that accepts cells with timestamps in the range
172  * [@p start, @p end).
173  *
174  * The function accepts any instantiation of `std::chrono::duration<>` for the
175  * @p start and @p end parameters. For example:
176  *
177  * @code
178  * using namespace std::chrono_literals; // C++14
179  * auto r1 = bigtable::Filter::TimestampRange(10ms, 500ms);

181  * @endcode
182  *
183  * The timestamp range must be non-empty, i.e. @p start must be strictly
184  * smaller than @p end. The server will reject empty ranges with a
185  * `grpc::StatusCode::INVALID_ARGUMENT` error. This function makes no
186  * attempt to validate the timestamp range before sending it to the server.
187  *
188  * @tparam Rep1 a placeholder to match the Rep tparam for @p start type,
189  * the semantics of this template parameter are documented in
190  * `std::chrono::duration<>` (in brief, the underlying arithmetic type
191  * used to store the number of ticks), for our purposes it is simply a
192  * formal parameter.
193  * @tparam Rep2 similar formal parameter for the type of @p end.
194  * @tparam Period1 a placeholder to match the Period tparam for @p start
195  * type, the semantics of this template parameter are documented in
196  * `std::chrono::duration<>` (in brief, the length of the tick in seconds,
197  * expressed as a `std::ratio<>`), for our purposes it is simply a formal
198  * parameter.
199  * @tparam Period2 similar formal parameter for the type of @p end.
200  *
201  * @see
202  * [std::chrono::duration<>](http://en.cppreference.com/w/cpp/chrono/duration)
203  * for more details.
204  */
205  template <typename Rep1, typename Period1, typename Rep2, typename Period2>
206  static Filter TimestampRange(std::chrono::duration<Rep1, Period1> start,
207  std::chrono::duration<Rep2, Period2> end) {
208  using namespace std::chrono;
209  return TimestampRangeMicros(duration_cast<microseconds>(start).count(),
210  duration_cast<microseconds>(end).count());
211  }

515  * @name Compound filters.
516  *
517  * These filters compose several filters to build complex filter expressions.

519  /**
520  * Returns a per-row conditional filter expression.
521  *
522  * For each row the @p predicate filter is evaluated, if it returns any
523  * cells, then the cells returned by @p true_filter are returned, otherwise
524  * the cells from @p false_filter are returned.
525  *
526  * The server validates the tree of filters, and rejects them if any contain
527  * invalid values. The server may impose additional restrictions on the
528  * resulting collection of filters. This function makes no attempt to
529  * validate the input before sending it to the server.
530  */
531  static Filter Condition(Filter predicate, Filter true_filter,
532  Filter false_filter) {
533  Filter tmp;
534  auto& condition = *tmp.filter_.mutable_condition();
535  condition.mutable_predicate_filter()->Swap(&predicate.filter_);
536  condition.mutable_true_filter()->Swap(&true_filter.filter_);
537  condition.mutable_false_filter()->Swap(&false_filter.filter_);
538  return tmp;
539  }

541  /**
542  * Return a chain filter.
543  *
544  * The filter returned by this function acts like a pipeline. The output
545  * row from each stage is passed on as input for the next stage.
546  *
547  * @tparam FilterTypes the type of the filter arguments. They must all be
548  * convertible to Filter.
549  * @param stages the filter stages. The filter must contain at least two
550  * stages. The server validates each stage, and will reject them as
551  * described in their corresponding function. The server may also impose
552  * additional restrictions on the composition of the Chain. This function
553  * makes no attempt at validating the stages locally, the Chain filter is
554  * sent as-is it to the server.
555  */
556  template <typename... FilterTypes>
557  static Filter Chain(FilterTypes&&... stages) {
558  // This ugly thing provides a better compile-time error message than
559  // just letting the compiler figure things out 3 levels deep
560  // as it recurses on append_types().
561  static_assert(
562  internal::conjunction<
563  std::is_convertible<FilterTypes, Filter>...>::value,
564  "The arguments passed to Chain(...) must be convertible to Filter");
565  Filter tmp;
566  auto& chain = *tmp.filter_.mutable_chain();
567  std::initializer_list<Filter> list{std::forward<FilterTypes>(stages)...};
568  for (Filter const& f : list) {
569  *chain.add_filters() = f.as_proto();
570  }
571  return tmp;
572  }

574  /**
575  * Return a filter that interleaves the results of many other filters.
576  *
577  * This filter executes each stream in parallel and then merges the results by
578  * interleaving the output from each stream. The
579  * [proto
580  * file](https://github.com/googleapis/googleapis/blob/master/google/bigtable/v2/data.proto)
581  * has a nice illustration in the documentation of
582  * `google.bigtable.v2.RowFilter.Interleave`.
583  *
584  * In brief, if the input cells are c1, c2, c3, ..., and you have three
585  * subfilters S1, S2, and S3, the output of Interleave(S1, S2, S3) is:
586  * S1(c1), S2(c1), S3(c1), S1(d2), S2(d2), S3(c2), S1(c3), S2(c3), S3(c2), ...
587  * where some of the Si(c_j) values may be empty if the filter discards the
588  * cell altogether.
589  *
590  * @tparam FilterTypes the type of the filter arguments. They must all be
591  * convertible for Filter.
592  * @param streams the filters to interleave. The filter must contain at least
593  * two streams. The server validates each stream, and will reject them as
594  * described in their corresponding function. The server may also impose

608  for (Filter const& f : list) {
