559  template <typename... FilterTypes>
560  static Filter Chain(FilterTypes&&... stages) {
561  // This ugly thing provides a better compile-time error message than
562  // just letting the compiler figure things out 3 levels deep
563  // as it recurses on append_types().
564  static_assert(
565  absl::conjunction<std::is_convertible<FilterTypes, Filter>...>::value,
566  "The arguments passed to Chain(...) must be convertible to Filter");
567  Filter tmp;
568  auto& chain = *tmp.filter_.mutable_chain();
569  std::initializer_list<Filter> list{std::forward<FilterTypes>(stages)...};
570  for (Filter const& f : list) {
571  *chain.add_filters() = f.as_proto();
572  }
573  return tmp;
574  }
