556  template <typename... FilterTypes>
557  static Filter Chain(FilterTypes&&... stages) {
558  // This ugly thing provides a better compile-time error message than
559  // just letting the compiler figure things out 3 levels deep
560  // as it recurses on append_types().
561  static_assert(
562  absl::conjunction<std::is_convertible<FilterTypes, Filter>...>::value,
563  "The arguments passed to Chain(...) must be convertible to Filter");
564  Filter tmp;
565  auto& chain = *tmp.filter_.mutable_chain();
566  std::initializer_list<Filter> list{std::forward<FilterTypes>(stages)...};
567  for (Filter const& f : list) {
568  *chain.add_filters() = f.as_proto();
569  }
570  return tmp;
571  }
