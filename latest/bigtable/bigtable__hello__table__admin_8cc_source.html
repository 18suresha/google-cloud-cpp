30 void HelloWorldTableAdmin(std::vector<std::string> const& argv) {
31  if (argv.size() != 3) {
32  throw Usage{
33  "hello-world-table-admin <project-id> <instance-id> "
34  "<table-id>"};
35  }
36  std::string const project_id = argv[0];
37  std::string const instance_id = argv[1];
38  std::string const table_id = argv[2];
39 
41  namespace cbt = google::cloud::bigtable;
42  using google::cloud::StatusOr;
44 
45  // Connect to the Cloud Bigtable admin endpoint.
47  cbt::TableAdmin admin(
48  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions()),
49  instance_id);
51 
53  std::cout << "Creating a table:\n";
54  StatusOr<google::bigtable::admin::v2::Table> schema = admin.CreateTable(
55  table_id,
56  cbt::TableConfig({{"fam", cbt::GcRule::MaxNumVersions(10)},
57  {"foo", cbt::GcRule::MaxAge(std::chrono::hours(72))}},
58  {}));
59  std::cout << "DONE\n";
61 
63  std::cout << "Listing tables:\n";
64  auto tables = admin.ListTables(cbt::TableAdmin::NAME_ONLY);
65  if (!tables) throw std::runtime_error(tables.status().message());
66  for (auto const& table : *tables) {
67  std::cout << " " << table.name() << "\n";
68  }
69  std::cout << "DONE\n";
71 
73  std::cout << "Get table metadata:\n";
74  StatusOr<google::bigtable::admin::v2::Table> table =
75  admin.GetTable(table_id, cbt::TableAdmin::FULL);
76  if (!table) throw std::runtime_error(table.status().message());
77  std::cout << "Table name : " << table->name() << "\n";
78 
79  std::cout << "List table families and GC rules:\n";
80  for (auto const& kv : table->column_families()) {
81  std::string const& family_name = kv.first;
82  google::bigtable::admin::v2::ColumnFamily const& metadata = kv.second;
83  std::cout << "Column Family :" << family_name << "\t"
84  << metadata.DebugString() << "\n";
85  }
86  std::cout << "DONE\n";
88 
90  std::cout << "Update a column family GC rule:\n";
91  StatusOr<google::bigtable::admin::v2::Table> updated_schema =
92  admin.ModifyColumnFamilies(
93  table_id,
94  {cbt::ColumnFamilyModification::Update(
95  "fam", cbt::GcRule::Union(
96  cbt::GcRule::MaxNumVersions(5),
97  cbt::GcRule::MaxAge(std::chrono::hours(24 * 7))))});
98  if (!updated_schema) {
99  throw std::runtime_error(updated_schema.status().message());
100  }
101  std::cout << "Schema modified to: " << updated_schema->DebugString() << "\n";
103 
105  std::cout << "Deleting all the rows in " << table_id << "\n";
106  google::cloud::Status status = admin.DropAllRows(table_id);
107  if (!status.ok()) throw std::runtime_error(status.message());
108  std::cout << "DONE\n";
110 
112  std::cout << "Deleting table:\n";
113  google::cloud::Status delete_status = admin.DeleteTable(table_id);
114  if (!delete_status.ok()) throw std::runtime_error(delete_status.message());
115  std::cout << "DONE\n";
117 }
