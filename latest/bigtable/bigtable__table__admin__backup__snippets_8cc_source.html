28  std::vector<std::string> const& argv) {
30  namespace cbt = google::cloud::bigtable;
31  using google::cloud::StatusOr;
32  [](cbt::TableAdmin admin, std::string const& table_id,
33  std::string const& cluster_id, std::string const& backup_id,
34  std::string const& expire_time_string) {
35  absl::Time t;
36  std::string err;
37  if (!absl::ParseTime(absl::RFC3339_full, expire_time_string, &t, &err)) {
38  throw std::runtime_error("Unable to parse expire_time:" + err);

40  auto expire_time = absl::ToChronoTime(t);

42  admin.CreateBackup(cbt::TableAdmin::CreateBackupParams(
43  cluster_id, backup_id, table_id, expire_time));
44  if (!backup) throw std::runtime_error(backup.status().message());
45  std::cout << "Backup successfully created: " << backup->DebugString()

147 void RunAll(std::vector<std::string> const& argv) {

149  namespace cbt = google::cloud::bigtable;

151  if (!argv.empty()) throw examples::Usage{"auto"};
152  if (!examples::RunAdminIntegrationTests()) return;
153 
154  examples::CheckEnvironmentVariablesAreSet({
155  "GOOGLE_CLOUD_PROJECT",
156  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
157  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_SERVICE_ACCOUNT",
158  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_CLUSTER_ID",
159  });
160  auto const project_id =
161  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
162  auto const instance_id = google::cloud::internal::GetEnv(
163  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
164  .value();
165  auto const service_account =
166  google::cloud::internal::GetEnv(
167  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_SERVICE_ACCOUNT")
168  .value();
169  auto const cluster_id = google::cloud::internal::GetEnv(
170  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_CLUSTER_ID")
171  .value();
172 
173  cbt::TableAdmin admin(
174  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
175  instance_id);
176 
177  // If a previous run of these samples crashes before cleaning up there may be
178  // old tables left over. As there are quotas on the total number of tables we
179  // remove stale tables after 48 hours.
180  std::cout << "\nCleaning up old tables" << std::endl;
181  std::string const prefix = "table-admin-snippets-";
182  examples::CleanupOldTables(prefix, admin);
183  std::string const backup_prefix = "table-admin-snippets-backup-";
184  examples::CleanupOldBackups(cluster_id, admin);
185 
186  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
187  // This table is actually created and used to test the positive case (e.g.
188  // GetTable() and "table does exist")
189  auto table_id_1 = examples::RandomTableId(prefix, generator);
190 
191  auto table_1 = admin.CreateTable(
192  table_id_1, cbt::TableConfig(
193  {
194  {"fam", cbt::GcRule::MaxNumVersions(10)},
195  {"foo", cbt::GcRule::MaxNumVersions(3)},
196  },
197  {}));
198  if (!table_1) throw std::runtime_error(table_1.status().message());
199 
200  std::cout << "\nRunning CreateBackup() example" << std::endl;
201  auto backup_id_1 = examples::RandomTableId(backup_prefix, generator);
202  CreateBackup(admin, {table_id_1, cluster_id, backup_id_1,
203  absl::FormatTime(absl::Now() + absl::Hours(12))});
204 
205  std::cout << "\nRunning ListBackups() example" << std::endl;
206  ListBackups(admin, {"-", {}, {}});
207 
208  std::cout << "\nRunning GetBackup() example" << std::endl;
209  GetBackup(admin, {cluster_id, backup_id_1});
210 
211  std::cout << "\nRunning UpdateBackup() example" << std::endl;
212  UpdateBackup(admin, {cluster_id, backup_id_1,
213  absl::FormatTime(absl::Now() + absl::Hours(24))});
214 
215  (void)admin.DeleteTable(table_id_1);
216 
217  std::cout << "\nRunning RestoreTable() example" << std::endl;
218  RestoreTable(admin, {table_id_1, cluster_id, backup_id_1});
219
