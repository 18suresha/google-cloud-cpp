649  grpc::Status status;

654  // Generate a better compile time error message than the default one
655  // if the types do not match
656  static_assert(
657  absl::conjunction<
658  std::is_convertible<Args, bigtable::ReadModifyWriteRule>...>::value,
659  "The arguments passed to ReadModifyWriteRow(row_key,...) must be "
660  "convertible to bigtable::ReadModifyWriteRule");
661 
662  *request.add_rules() = std::move(rule).as_proto();
663  AddRules(request, std::forward<Args>(rules)...);
664  return ReadModifyWriteRowImpl(std::move(request));
665  }
