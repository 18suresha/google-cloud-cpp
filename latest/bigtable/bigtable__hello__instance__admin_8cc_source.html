30 void BigtableHelloInstance(std::vector<std::string> const& argv) {
31  if (argv.size() != 4) {
32  throw Usage{
33  "bigtable-hello-instance <project-id> <instance-id> <cluster-id> "
34  "<zone>"};
35  }
36 
37  std::string const project_id = argv[0];
38  std::string const instance_id = argv[1];
39  std::string const cluster_id = argv[2];
40  std::string const zone = argv[3];
41 
43  namespace cbt = google::cloud::bigtable;
44  using google::cloud::future;
45  using google::cloud::StatusOr;
47 
48  // Connect to the Cloud Bigtable admin endpoint.
50  cbt::InstanceAdmin instance_admin(
51  cbt::CreateDefaultInstanceAdminClient(project_id, cbt::ClientOptions()));
53 
55  std::cout << "\nCheck Instance exists:\n";
56  StatusOr<cbt::InstanceList> instances = instance_admin.ListInstances();
57  if (!instances) throw std::runtime_error(instances.status().message());
58  if (!instances->failed_locations.empty()) {
59  std::cerr
60  << "The service tells us it has no information about these locations:";
61  for (auto const& failed_location : instances->failed_locations) {
62  std::cerr << " " << failed_location;
63  }
64  std::cerr << ". Continuing anyway\n";
65  }
66  auto instance_name =
67  instance_admin.project_name() + "/instances/" + instance_id;
68  auto instance_name_it = std::find_if(
69  instances->instances.begin(), instances->instances.end(),
70  [&instance_name](google::bigtable::admin::v2::Instance const& i) {
71  return i.name() == instance_name;
72  });
73  bool instance_exists = instance_name_it != instances->instances.end();
74  std::cout << "The instance " << instance_id
75  << (instance_exists ? " already exists" : " does not exist")
76  << "\n";
78 
79  // Create instance if does not exists
80  if (!instance_exists) {
82  std::cout << "\nCreating a PRODUCTION Instance: ";
83 
84  // production instance needs at least 3 nodes
85  auto cluster_config = cbt::ClusterConfig(zone, 3, cbt::ClusterConfig::HDD);
86  cbt::InstanceConfig config(instance_id, "Sample Instance",
87  {{cluster_id, cluster_config}});
88  config.set_type(cbt::InstanceConfig::PRODUCTION);
89 
90  google::cloud::future<void> creation_done =
91  instance_admin.CreateInstance(config).then(
92  [instance_id](
93  future<StatusOr<google::bigtable::admin::v2::Instance>> f) {
94  auto instance = f.get();
95  if (!instance) {
96  std::cerr << "Could not create instance " << instance_id
97  << "\n";
98  throw std::runtime_error(instance.status().message());
99  }
100  std::cout << "Successfully created instance: "
101  << instance->DebugString() << "\n";
102  });
103  // Note how this blocks until the instance is created, in production code
104  // you may want to perform this task asynchronously.
105  creation_done.get();
106  std::cout << "DONE\n";
108  }
109 
111  std::cout << "\nListing Instances:\n";
112  instances = instance_admin.ListInstances();
113  if (!instances) throw std::runtime_error(instances.status().message());
114  if (!instances->failed_locations.empty()) {
115  std::cerr
116  << "The service tells us it has no information about these locations:";
117  for (auto const& failed_location : instances->failed_locations) {
118  std::cerr << " " << failed_location;
119  }
120  std::cerr << ". Continuing anyway\n";
121  }
122  for (auto const& instance : instances->instances) {
123  std::cout << " " << instance.name() << "\n";
124  }
125  std::cout << "DONE\n";
127 
129  std::cout << "\nGet Instance:\n";
130  auto instance = instance_admin.GetInstance(instance_id);
131  if (!instance) throw std::runtime_error(instance.status().message());
132  std::cout << "Instance details :\n" << instance->DebugString() << "\n";
134 
136  std::cout << "\nListing Clusters:\n";
137  StatusOr<cbt::ClusterList> cluster_list =
138  instance_admin.ListClusters(instance_id);
139  if (!cluster_list) throw std::runtime_error(cluster_list.status().message());
140  if (!cluster_list->failed_locations.empty()) {
141  std::cout << "The Cloud Bigtable service reports that the following "
142  "locations are temporarily unavailable and no information "
143  "about clusters in these locations can be obtained:\n";
144  for (auto const& failed_location : cluster_list->failed_locations) {
145  std::cout << failed_location << "\n";
146  }
147  }
148  std::cout << "Cluster Name List:\n";
149  for (auto const& cluster : cluster_list->clusters) {
150  std::cout << "Cluster Name: " << cluster.name() << "\n";
151  }
152  std::cout << "DONE\n";
154 
156  std::cout << "Deleting instance " << instance_id << "\n";
157  google::cloud::Status delete_status =
158  instance_admin.DeleteInstance(instance_id);
159  if (!delete_status.ok()) throw std::runtime_error(delete_status.message());
160  std::cout << "DONE\n";
162 }

164 void RunAll(std::vector<std::string> const& argv) {
165  namespace examples = ::google::cloud::bigtable::examples;
166  namespace cbt = google::cloud::bigtable;
167 
168  if (!argv.empty()) throw Usage{"auto"};
169  if (!examples::RunAdminIntegrationTests()) return;
170  examples::CheckEnvironmentVariablesAreSet({
171  "GOOGLE_CLOUD_PROJECT",
172  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_ZONE_A",
173  });
174  auto const project_id =
175  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
176  auto const zone_a =
177  google::cloud::internal::GetEnv("GOOGLE_CLOUD_CPP_BIGTABLE_TEST_ZONE_A")
178  .value();
179 
180  cbt::InstanceAdmin admin(
181  cbt::CreateDefaultInstanceAdminClient(project_id, cbt::ClientOptions{}));
182 
183  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
184  examples::CleanupOldInstances("exin-", admin);
185  auto const instance_id = examples::RandomInstanceId("exin-", generator);
186  auto const cluster_id = instance_id + "-c1";
187 
188  std::cout << "\nRunning the BigtableHelloInstance() example" << std::endl;
189  BigtableHelloInstance({project_id, instance_id, cluster_id, zone_a});
190 }
