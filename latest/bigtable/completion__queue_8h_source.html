58  /**
59  * Create a timer that fires after the @p duration.
60  *
61  * @tparam Rep a placeholder to match the Rep tparam for @p duration type,
62  * the semantics of this template parameter are documented in
63  * `std::chrono::duration<>` (in brief, the underlying arithmetic type
64  * used to store the number of ticks), for our purposes it is simply a
65  * formal parameter.
66  * @tparam Period a placeholder to match the Period tparam for @p duration
67  * type, the semantics of this template parameter are documented in
68  * `std::chrono::duration<>` (in brief, the length of the tick in seconds,
69  * expressed as a `std::ratio<>`), for our purposes it is simply a formal
70  * parameter.
71  *
72  * @param duration when should the timer expire relative to the current time.
73  *
74  * @return a future that becomes satisfied after @p duration time has elapsed.
75  */
76  template <typename Rep, typename Period>

82  /**
83  * Make an asynchronous unary RPC.
84  *
85  * @param async_call a callable to start the asynchronous RPC.
86  * @param request the contents of the request.
87  * @param context an initialized request context to make the call.
88  *
89  * @tparam AsyncCallType the type of @a async_call. It must be invocable with
90  * `(grpc::ClientContext*, RequestType const&, grpc::CompletionQueue*)`.
91  * Furthermore, it should return a
92  * `std::unique_ptr<grpc::ClientAsyncResponseReaderInterface<Response>>>`.
93  * These requirements are verified by
94  * `internal::CheckAsyncUnaryRpcSignature<>`, and this function is
95  * excluded from overload resolution if the parameters do not meet these
96  * requirements.
97  * @tparam Request the type of the request parameter in the gRPC.
98  *
99  * @return a future that becomes satisfied when the operation completes.
100  */
101  template <
102  typename AsyncCallType, typename Request,
103  typename Sig = internal::AsyncCallResponseType<AsyncCallType, Request>,
104  typename Response = typename Sig::type,
105  typename std::enable_if<Sig::value, int>::type = 0>
106  future<StatusOr<Response>> MakeUnaryRpc(
107  AsyncCallType async_call, Request const& request,
108  std::unique_ptr<grpc::ClientContext> context) {
109  auto op =
110  std::make_shared<internal::AsyncUnaryRpcFuture<Request, Response>>();
111  void* tag = impl_->RegisterOperation(op);
112  op->Start(async_call, std::move(context), request, &impl_->cq(), tag);
113  return op->GetFuture();
114  }

116  /**
117  * Make an asynchronous streaming read RPC.
118  *
119  * Reading from the stream starts automatically, and the handler is notified
120  * of all interesting events in the stream. Note that then handler is called
121  * by any thread blocked on this object's Run() member function. However, only
122  * one callback in the handler is called at a time.
123  *
124  * @param async_call a callable to start the asynchronous RPC.
125  * @param request the contents of the request.
126  * @param context an initialized request context to make the call.
127  * @param on_read the callback to be invoked on each successful Read().
128  * @param on_finish the callback to be invoked when the stream is closed.
129  *
130  * @tparam AsyncCallType the type of @a async_call. It must be invocable with
131  * parameters
132  * `(grpc::ClientContext*, RequestType const&, grpc::CompletionQueue*)`.
133  * Furthermore, it should return a type convertible to
134  * `std::unique_ptr<grpc::ClientAsyncReaderInterface<Response>>>`.
135  * These requirements are verified by
136  * `internal::AsyncStreamingReadRpcUnwrap<>`, and this function is
137  * excluded from overload resolution if the parameters do not meet these
138  * requirements.
139  * @tparam Request the type of the request in the streaming RPC.
140  * @tparam Response the type of the response in the streaming RPC.
141  * @tparam OnReadHandler the type of the @p on_read callback.
142  * @tparam OnFinishHandler the type of the @p on_finish callback.
143  */
144  template <typename AsyncCallType, typename Request,
145  typename Response = typename internal::
146  AsyncStreamingReadResponseType<AsyncCallType, Request>::type,
147  typename OnReadHandler, typename OnFinishHandler>
148  std::shared_ptr<AsyncOperation> MakeStreamingReadRpc(
149  AsyncCallType&& async_call, Request const& request,
150  std::unique_ptr<grpc::ClientContext> context, OnReadHandler&& on_read,
151  OnFinishHandler&& on_finish) {
152  auto stream = internal::MakeAsyncReadStreamImpl<Response>(
153  std::forward<OnReadHandler>(on_read),
154  std::forward<OnFinishHandler>(on_finish));
155  stream->Start(std::forward<AsyncCallType>(async_call), request,
156  std::move(context), impl_);
157  return stream;
158  }

160  /**
161  * Asynchronously run a functor on a thread `Run()`ning the `CompletionQueue`.
162  *
163  * @tparam Functor the functor to call on the CompletionQueue thread.
164  * It must satisfy the `void(CompletionQueue&)` signature.
165  * @param functor the value of the functor.
166  * @return an asynchronous operation wrapping the functor; it can be used for
167  * cancelling but it makes little sense given that it will be completed
168  * straight away
169  */
170  template <typename Functor,
171  typename std::enable_if<
172  internal::CheckRunAsyncCallback<Functor>::value, int>::type = 0>
173  void RunAsync(Functor&& functor) {
174  MakeRelativeTimer(std::chrono::seconds(0))
175  .then([this, functor](future<std::chrono::system_clock::time_point>) {
176  CompletionQueue cq(impl_);
177  functor(cq);
178  });
179  }
180 
181  private:
182  std::shared_ptr<internal::CompletionQueueImpl> impl_;
183 };
