23 [[noreturn]] void ReportPermanentFailures(
24  char const* msg, grpc::Status const& status,
25  std::vector<google::cloud::bigtable::FailedMutation> failures) {
26 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
27  throw google::cloud::bigtable::PermanentMutationFailure(msg, status,
28  std::move(failures));
29 #else
30  std::cerr << msg << "\n"
31  << "Status: " << status.error_message() << " ["
32  << status.error_code() << "] - " << status.error_details() << "\n";
33  for (auto const& failed : failures) {
34  std::cerr << "Mutation " << failed.original_index() << " failed with"
35  << failed.status().message() << " [" << failed.status().code()
36  << "]\n";
37  }
38  std::cerr << "Aborting because exceptions are disabled.\n";
39  std::abort();
40 #endif // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
41 }

59 future<Status> Table::AsyncApply(SingleRowMutation mut, CompletionQueue& cq) {
60  promise<StatusOr<google::bigtable::v2::MutateRowResponse>> p;
61  future<StatusOr<google::bigtable::v2::MutateRowResponse>> result =
62  p.get_future();
63 
64  impl_.AsyncApply(
65  cq, internal::MakeAsyncFutureFromCallback(std::move(p), "AsyncApply"),
66  std::move(mut));
67 
68  auto final = result.then(
69  [](future<StatusOr<google::bigtable::v2::MutateRowResponse>> f) {
70  auto mutate_row_response = f.get();
71  if (mutate_row_response) {
72  return Status();
73  }
74  return mutate_row_response.status();
75  });
76 
77  return final;
78 }
