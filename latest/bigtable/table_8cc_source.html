32 Status Table::Apply(SingleRowMutation mut) {
33  // Copy the policies in effect for this operation. Many policy classes change
34  // their state as the operation makes progress (or fails to make progress), so
35  // we need fresh instances.
36  auto rpc_policy = impl_.rpc_retry_policy_->clone();
37  auto backoff_policy = impl_.rpc_backoff_policy_->clone();
38  auto idempotent_policy = impl_.idempotent_mutation_policy_->clone();
39 
40  // Build the RPC request, try to minimize copying.
41  btproto::MutateRowRequest request;
42  bigtable::internal::SetCommonTableOperationRequest<btproto::MutateRowRequest>(
43  request, impl_.app_profile_id_.get(), impl_.table_name_.get());
44  mut.MoveTo(request);
45 
46  bool const is_idempotent =
47  std::all_of(request.mutations().begin(), request.mutations().end(),
48  [&idempotent_policy](btproto::Mutation const& m) {
49  return idempotent_policy->is_idempotent(m);
50  });
51 
52  btproto::MutateRowResponse response;
53  grpc::Status status;
54  while (true) {
55  grpc::ClientContext client_context;
56  rpc_policy->Setup(client_context);
57  backoff_policy->Setup(client_context);
58  impl_.metadata_update_policy_.Setup(client_context);
59  status = impl_.client_->MutateRow(&client_context, request, &response);
60 
61  if (status.ok()) {
62  return google::cloud::Status{};
63  }
64  // It is up to the policy to terminate this loop, it could run
65  // forever, but that would be a bad policy (pun intended).
66  if (!rpc_policy->OnFailure(status) || !is_idempotent) {
67  return bigtable::internal::MakeStatusFromRpcError(
68  status.error_code(),
69  "Permanent (or too many transient) errors in Table::Apply()");
70  }
71  auto delay = backoff_policy->OnCompletion(status);
72  std::this_thread::sleep_for(delay);
73  }

166 StatusOr<std::pair<bool, Row>> Table::ReadRow(std::string row_key,
167  Filter filter) {
168  RowSet row_set(std::move(row_key));
169  std::int64_t const rows_limit = 1;
170  RowReader reader =
171  ReadRows(std::move(row_set), rows_limit, std::move(filter));
172 
173  auto it = reader.begin();
174  if (it == reader.end()) {
175  return std::make_pair(false, Row("", {}));
176  }
177  if (!*it) {
178  return it->status();
179  }
180  auto result = std::make_pair(true, std::move(**it));
181  if (++it != reader.end()) {
182  return Status(StatusCode::kInternal,
183  "internal error - RowReader returned 2 rows in ReadRow()");
184  }
185  return result;
186 }
