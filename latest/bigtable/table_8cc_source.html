38  promise<StatusOr<google::bigtable::v2::MutateRowResponse>> p;
39  future<StatusOr<google::bigtable::v2::MutateRowResponse>> result =
40  p.get_future();
41 
42  impl_.AsyncApply(
43  cq, internal::MakeAsyncFutureFromCallback(std::move(p), "AsyncApply"),
44  std::move(mut));
45 
46  auto final = result.then(
47  [](future<StatusOr<google::bigtable::v2::MutateRowResponse>> f) {
48  auto mutate_row_response = f.get();

50  return Status();

52  return mutate_row_response.status();
53  });
54 
55  return final;
56 }
