164  void TryGiveRowToUser() {
165  // The user is likely to ask for more rows immediately after receiving a
166  // row, which means that this function will be called recursively. The depth
167  // of the recursion can be as deep as the size of ready_rows_, which might
168  // be significant and potentially lead to stack overflow. The way to
169  // overcome this is to always switch thread to a CompletionQueue thread.
170  // Switching thread for every row has a non-trivial cost, though. To find a
171  // good balance, we allow for recursion no deeper than 100 and achieve it by
172  // tracking the level in `recursion_level_`.
173  //
174  // The magic value 100 is arbitrary, but back-of-the-envelope calculation
175  // indicates it should cap this stack usage to below 100K. Default stack
176  // size is usually 1MB.
177  struct CountFrame {
178  explicit CountFrame(int& cntr) : cntr(++cntr){};
179  ~CountFrame() { --cntr; }
180  int& cntr;
181  };
182  CountFrame frame(recursion_level_);
183 
184  if (ready_rows_.empty()) {
185  if (whole_op_finished_) {
186  // The scan is finished for good, there will be no more rows.
187  on_finish_(status_);
188  return;
189  }
190  if (!continue_reading_) {
191  // TODO(#1402): replace with GCP_ASSERT(continue_reading_);
192  Terminate(
193  "No rows are ready and we can't continue reading. This is a bug, "
194  "please report it at "
195  "https://github.com/googleapis/google-cloud-cpp/issues/new");
196  }
197  // No rows, but we can fetch some.
198  auto continue_reading = std::move(continue_reading_);
199  continue_reading_.reset();
200  continue_reading->set_value(true);
201  return;
202  }
203 
204  // Yay! We have something to give to the user and they want it.
205  auto row = std::move(ready_rows_.front());
206  ready_rows_.pop();
207 
208  auto self = this->shared_from_this();
209  bool const break_recursion = recursion_level_ >= 100;
210  on_row_(std::move(row)).then([self, break_recursion](future<bool> fut) {
211  bool should_cancel;
212 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
213  try {
214  should_cancel = !fut.get();
215  } catch (std::exception& ex) {
216  self->Cancel(
217  std::string("future<> returned from the user callback threw an "
218  "exception: ") +
219  ex.what());
220  return;
221  } catch (...) {
222  self->Cancel(
223  "future<> returned from the user callback threw an unknown "
224  "exception");
225  return;
226  }
227 #else // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
228  should_cancel = !fut.get();
229 #endif // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
230  if (should_cancel) {
231  self->Cancel("User cancelled");
232  return;
233  }
234  if (break_recursion) {
235  self->cq_.RunAsync([self] { self->UserWantsRows(); });
236  return;
237  }
238  self->UserWantsRows();
239  });
240  }

243  future<bool> OnDataReceived(
244  google::bigtable::v2::ReadRowsResponse const& response) {
245  // assert(!whole_op_finished_);
246  // assert(!continue_reading_);
247  // assert(status_.ok());
248  status_ = ConsumeResponse(std::move(response));
249  // We've processed the response.
250  //
251  // If there were errors (e.g. malformed response from the server), we should
252  // interrupt this stream. Interrupting it will yield lower layers calling
253  // `OnStreamFinished` with a status unrelated to the real reason, so we
254  // store the actual reason in status_ and proceed exactly the
255  // same way as if the stream was broken for other reasons.
256  //
257  // Even if status_ is not OK, we might have consumed some rows,
258  // but, don't give them to the user yet. We want to keep the invariant that
259  // either the user doesn't hold a `future<>` when we're fetching more rows.
260  // Retries (successful or not) will do it. Improving this behavior makes
261  // little sense because parser errors are very unexpected and probably not
262  // retryable anyway.
263 
264  if (status_.ok()) {
265  continue_reading_.emplace(promise<bool>());
266  auto res = continue_reading_->get_future();
267  TryGiveRowToUser();
268  return res;
269  }
270  return make_ready_future<bool>(false);
271  }

274  // NOLINTNEXTLINE(performance-unnecessary-value-param)
275  void OnStreamFinished(Status status) {
276  // assert(!continue_reading_);
277  if (status_.ok()) {
278  status_ = std::move(status);
279  }
280  grpc::Status parser_status;
281  parser_->HandleEndOfStream(parser_status);
282  if (!parser_status.ok() && status_.ok()) {
283  // If there stream finished with an error ignore what the parser says.
284  status_ = MakeStatusFromRpcError(parser_status);
285  }
286 
287  // In the unlikely case when we have already reached the requested
288  // number of rows and still receive an error (the parser can throw
289  // an error at end of stream for example), there is no need to
290  // retry and we have no good value for rows_limit anyway.
291  if (rows_limit_ != NO_ROWS_LIMIT && rows_limit_ <= rows_count_) {
292  status_ = Status();
293  }
294 
295  if (!last_read_row_key_.empty()) {
296  // We've returned some rows and need to make sure we don't
297  // request them again.
298  row_set_ = row_set_.Intersect(RowRange::Open(last_read_row_key_, ""));
299  }
300 
301  // If we receive an error, but the retryable set is empty, consider it a
302  // success.
303  if (row_set_.IsEmpty()) {
304  status_ = Status();
305  }
306 
307  if (status_.ok()) {
308  // We've successfully finished the scan.
309  whole_op_finished_ = true;
310  TryGiveRowToUser();
311  return;
312  }
313 
314  if (!rpc_retry_policy_->OnFailure(status_)) {
315  // Can't retry.
316  whole_op_finished_ = true;
317  TryGiveRowToUser();
318  return;
319  }
320  auto self = this->shared_from_this();
321  cq_.MakeRelativeTimer(rpc_backoff_policy_->OnCompletion(status_))
322  .then([self](future<StatusOr<std::chrono::system_clock::time_point>>
323  result) {
324  if (auto tp = result.get()) {
325  self->MakeRequest();
326  } else {
327  self->whole_op_finished_ = true;
328  self->TryGiveRowToUser();
329  }
330  });
331  }

334  void Cancel(std::string const& reason) {
335  ready_rows_ = std::queue<Row>();
336  auto continue_reading = std::move(continue_reading_);
337  continue_reading_.reset();
338  Status status(StatusCode::kCancelled, reason);
339  if (!continue_reading) {
340  // If we're not in the middle of the stream fire some user callbacks, but
341  // also override the overall status.
342  // assert(whole_op_finished_);
343  status_ = std::move(status);
344  TryGiveRowToUser();
345  return;
346  }
347  // If we are in the middle of the stream, cancel the stream.
348  status_ = std::move(status);
349  continue_reading->set_value(false);
350  }

353  Status DrainParser() {
354  grpc::Status status;
355  while (parser_->HasNext()) {
356  Row parsed_row = parser_->Next(status);
357  if (!status.ok()) {
358  return MakeStatusFromRpcError(status);
359  }
360  ++rows_count_;
361  last_read_row_key_ = std::string(parsed_row.row_key());
362  ready_rows_.emplace(std::move(parsed_row));
363  }
364  return Status();
365  }

368  Status ConsumeResponse(google::bigtable::v2::ReadRowsResponse response) {
369  for (auto& chunk : *response.mutable_chunks()) {
370  grpc::Status status;
371  parser_->HandleChunk(std::move(chunk), status);
372  if (!status.ok()) {
373  return MakeStatusFromRpcError(status);
374  }
375  Status parser_status = DrainParser();
376  if (!parser_status.ok()) {
377  return parser_status;
378  }
379  }
380  return Status();
381  }
