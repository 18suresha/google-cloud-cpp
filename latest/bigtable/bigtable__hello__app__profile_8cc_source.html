42 
43  // Create an object to access the Cloud Bigtable Data API.
44  auto data_client = cbt::CreateDefaultDataClient(project_id, instance_id,
45  cbt::ClientOptions());
46 
47  // Use the default profile to write some data.
48  cbt::Table write(data_client, table_id);

50  // Modify (and create if necessary) a row.
51  std::vector<std::string> greetings{"Hello World!", "Hello Cloud Bigtable!",
52  "Hello C++!"};
53  int i = 0;
54  for (auto const& greeting : greetings) {
55  // Each row has a unique row key.
56  //
57  // Note: This example uses sequential numeric IDs for simplicity, but
58  // this can result in poor performance in a production application.
59  // Since rows are stored in sorted order by key, sequential keys can
60  // result in poor distribution of operations across nodes.
61  //
62  // For more information about how to design a Bigtable schema for the
63  // best performance, see the documentation:
64  //
65  // https://cloud.google.com/bigtable/docs/schema-design
66  std::string row_key = "key-" + std::to_string(i);
67  write.Apply(cbt::SingleRowMutation(std::move(row_key),
68  cbt::SetCell("fam", "c0", greeting)));
69  ++i;
70  }
71 
72  std::cout << "Wrote some greetings to " << table_id << std::endl;
73 
74  // Access Cloud Bigtable using a different profile

78  auto result =
79  read.ReadRow("key-0", cbt::Filter::ColumnRangeClosed("fam", "c0", "c0"));
80  if (!result.first) {
81  std::cout << "Cannot find row 'key-0' in the table: " << table_id
82  << std::endl;
83  return 1;
84  }
85  auto const& cell = result.second.cells().front();
86  std::cout << cell.family_name() << ":" << cell.column_qualifier() << " @ "

88  << '"' << cell.value() << '"' << std::endl;

96  std::cout << row.row_key() << ":\n";
97  for (auto& cell : row.cells()) {

99  << " @ " << cell.timestamp().count() << "us\n"
100  << "\t\"" << cell.value() << '"' << std::endl;
101  }
102  }

104 
105  return 0;
