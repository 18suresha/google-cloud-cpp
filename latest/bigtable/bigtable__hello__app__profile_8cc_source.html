31 void HelloWorldAppProfile(std::vector<std::string> const& argv) {
32  if (argv.size() != 4) {
33  throw Usage{
34  "hello-world-app-profile <project-id> <instance-id>"
35  " <table-id> <profile-id>"};
36  }
37  std::string const project_id = argv[0];
38  std::string const instance_id = argv[1];
39  std::string const table_id = argv[2];
40  std::string const profile_id = argv[3];
41 
42  // Create an object to access the Cloud Bigtable Data API.
43  auto data_client = cbt::CreateDefaultDataClient(project_id, instance_id,
44  cbt::ClientOptions());
45 
46  // Use the default profile to write some data.
47  cbt::Table write(data_client, table_id);
48 
49  // Modify (and create if necessary) a row.
50  std::vector<std::string> greetings{"Hello World!", "Hello Cloud Bigtable!",
51  "Hello C++!"};
52  int i = 0;
53  for (auto const& greeting : greetings) {
54  // Each row has a unique row key.
55  //
56  // Note: This example uses sequential numeric IDs for simplicity, but
57  // this can result in poor performance in a production application.
58  // Since rows are stored in sorted order by key, sequential keys can
59  // result in poor distribution of operations across nodes.
60  //
61  // For more information about how to design a Bigtable schema for the
62  // best performance, see the documentation:
63  //
64  // https://cloud.google.com/bigtable/docs/schema-design
65  std::string row_key = "key-" + std::to_string(i);
66  google::cloud::Status status = write.Apply(cbt::SingleRowMutation(
67  std::move(row_key), cbt::SetCell("fam", "c0", greeting)));
68  if (!status.ok()) throw std::runtime_error(status.message());
69  ++i;
70  }
71 
72  std::cout << "Wrote some greetings to " << table_id << "\n";
73 
74  // Access Cloud Bigtable using a different profile
76  cbt::Table read(data_client, profile_id, table_id);
77 
78  google::cloud::StatusOr<std::pair<bool, cbt::Row>> result =
79  read.ReadRow("key-0", cbt::Filter::ColumnRangeClosed("fam", "c0", "c0"));
80  if (!result) throw std::runtime_error(result.status().message());
81  if (!result->first) throw std::runtime_error("missing row with key = key-0");
82  cbt::Cell const& cell = result->second.cells().front();
83  std::cout << cell.family_name() << ":" << cell.column_qualifier() << " @ "
84  << cell.timestamp().count() << "us\n"
85  << '"' << cell.value() << '"' << "\n";
87 
88  // Read multiple rows.
90  std::cout << "Scanning all the data from " << table_id << "\n";
91  for (google::cloud::StatusOr<cbt::Row>& row : read.ReadRows(
92  cbt::RowRange::InfiniteRange(), cbt::Filter::PassAllFilter())) {
93  if (!row) throw std::runtime_error(row.status().message());
94  std::cout << row->row_key() << ":\n";
95  for (cbt::Cell const& c : row->cells()) {
96  std::cout << "\t" << c.family_name() << ":" << c.column_qualifier()
97  << " @ " << c.timestamp().count() << "us\n"
98  << "\t\"" << c.value() << '"' << "\n";
99  }
100  }
102 }

104 void RunAll(std::vector<std::string> const& argv) {
105  namespace examples = ::google::cloud::bigtable::examples;
106 
107  if (!argv.empty()) throw examples::Usage{"auto"};
108  if (!examples::RunAdminIntegrationTests()) return;
109  examples::CheckEnvironmentVariablesAreSet({
110  "GOOGLE_CLOUD_PROJECT",
111  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
112  });
113  auto const project_id =
114  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
115  auto const instance_id = google::cloud::internal::GetEnv(
116  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
117  .value();
118 
119  cbt::TableAdmin admin(
120  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
121  instance_id);
122 
123  examples::CleanupOldTables("hw-app-profile-tbl-", admin);
124 
125  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
126  auto table_id = examples::RandomTableId("hw-app-profile-tbl-", generator);
127  auto schema = admin.CreateTable(
128  table_id,
129  cbt::TableConfig({{"fam", cbt::GcRule::MaxNumVersions(10)}}, {}));
130  if (!schema) throw std::runtime_error(schema.status().message());
131 
132  auto profile_id = "hw-app-profile-" +
133  google::cloud::internal::Sample(
134  generator, 8, "abcdefghilklmnopqrstuvwxyz0123456789");
135 
136  cbt::InstanceAdmin instance_admin(
137  cbt::CreateDefaultInstanceAdminClient(project_id, cbt::ClientOptions{}));
138  auto profile = instance_admin.CreateAppProfile(
139  instance_id, cbt::AppProfileConfig::MultiClusterUseAny(profile_id));
140  if (!profile) throw std::runtime_error(profile.status().message());
141 
142  std::cout << "\nRunning the AppProfile hello world example" << std::endl;
143  HelloWorldAppProfile({project_id, instance_id, table_id, profile_id});
144 
145  instance_admin.DeleteAppProfile(instance_id, profile_id);
146  admin.DeleteTable(table_id);
147 }
