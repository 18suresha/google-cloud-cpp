34 std::pair<future<void>, future<Status>> MutationBatcher::AsyncApply(
35  CompletionQueue& cq, SingleRowMutation mut) {
36  AdmissionPromise admission_promise;
37  CompletionPromise completion_promise;
38  auto res = std::make_pair(admission_promise.get_future(),
39  completion_promise.get_future());
40  PendingSingleRowMutation pending(std::move(mut),
41  std::move(completion_promise),
42  std::move(admission_promise));
43  std::unique_lock<std::mutex> lk(mu_);
44 
45  grpc::Status mutation_status = IsValid(pending);
46  if (!mutation_status.ok()) {
47  lk.unlock();
48  // Destroy the mutation before satisfying the admission promise so that we
49  // can limit the memory usage.
50  pending.mut.Clear();
51  pending.completion_promise.set_value(
52  internal::MakeStatusFromRpcError(mutation_status));
53  // No need to consider no_more_pending_promises because this operation
54  // didn't lower the number of pending operations.
55  pending.admission_promise.set_value();
56  return res;
57  }
58  ++num_requests_pending_;
59 
60  if (!CanAppendToBatch(pending)) {
61  pending_mutations_.push(std::move(pending));
62  return res;
63  }
64  std::vector<AdmissionPromise> admission_promises_to_satisfy;
65  admission_promises_to_satisfy.emplace_back(
66  std::move(pending.admission_promise));
67  Admit(std::move(pending));
68  FlushIfPossible(cq);
69  SatisfyPromises(std::move(admission_promises_to_satisfy), lk);
70  return res;
71 }

73 future<void> MutationBatcher::AsyncWaitForNoPendingRequests() {
74  std::unique_lock<std::mutex> lk(mu_);
75  if (num_requests_pending_ == 0) {
76  // TODO(#2112): Use make_satisfied_future<> once it's implemented.
77  promise<void> satisfied_promise;
78  satisfied_promise.set_value();
79  return satisfied_promise.get_future();
80  }
81  no_more_pending_promises_.emplace_back();
82  return no_more_pending_promises_.back().get_future();
83 }

99 grpc::Status MutationBatcher::IsValid(PendingSingleRowMutation& mut) const {
100  // Objects of this class need to be aware of the maximum allowed number of
101  // mutations in a batch because it should not pack more. If we have this
102  // knowledge, we might as well simplify everything and not admit larger
103  // mutations.
104  if (mut.num_mutations > options_.max_mutations_per_batch) {
105  std::stringstream stream;
106  stream << "Too many (" << mut.num_mutations
107  << ") mutations in a SingleRowMutations request. "
108  << options_.max_mutations_per_batch << " is the limit.";
109  return grpc::Status(grpc::StatusCode::INVALID_ARGUMENT, stream.str());
110  }
111  if (mut.num_mutations == 0) {
112  return grpc::Status(grpc::StatusCode::INVALID_ARGUMENT,
113  "Supplied SingleRowMutations has no entries");
114  }
115  if (mut.request_size > options_.max_size_per_batch) {
116  std::stringstream stream;
117  stream << "Too large (" << mut.request_size
118  << " bytes) mutation in a SingleRowMutations request. "
119  << options_.max_size_per_batch << " bytes is the limit.";
120  return grpc::Status(grpc::StatusCode::INVALID_ARGUMENT, stream.str());
121  }
122  return grpc::Status();
123 }

125 bool MutationBatcher::HasSpaceFor(PendingSingleRowMutation const& mut) const {
126  return outstanding_size_ + mut.request_size <=
127  options_.max_outstanding_size &&
128  cur_batch_->requests_size + mut.request_size <=
129  options_.max_size_per_batch &&
130  cur_batch_->num_mutations + mut.num_mutations <=
131  options_.max_mutations_per_batch;

134 bool MutationBatcher::FlushIfPossible(CompletionQueue& cq) {
135  if (cur_batch_->num_mutations > 0 &&
136  num_outstanding_batches_ < options_.max_batches) {
137  ++num_outstanding_batches_;
138  auto batch = cur_batch_;
139  table_.impl_.StreamingAsyncBulkApply(
140  cq,
141  [this, batch](CompletionQueue& cq, std::vector<int> succeeded) {
142  OnSuccessfulMutations(cq, *batch, std::move(succeeded));
143  },
144  [this, batch](CompletionQueue& cq, std::vector<FailedMutation> failed) {
145  OnFailedMutations(cq, *batch, std::move(failed));
146  },
147  [this, batch](CompletionQueue& cq, grpc::Status&) {
148  OnBulkApplyAttemptFinished(cq, *batch);
149  },
150  [this, batch](CompletionQueue& cq, std::vector<FailedMutation>& failed,
151  grpc::Status&) {
152  // It means that there are not going to be anymore retries and the
153  // final failed mutations are passed here.
154  OnFailedMutations(cq, *batch, std::move(failed));
155  },
156  std::move(cur_batch_->requests));
157  cur_batch_ = std::make_shared<Batch>();
158  return true;
159  }
160  return false;
161 }

163 void MutationBatcher::OnSuccessfulMutations(CompletionQueue& cq,
164  MutationBatcher::Batch& batch,
165  std::vector<int> indices) {
166  size_t const num_mutations = indices.size();
167  size_t completed_size = 0;
168 
169  for (int idx : indices) {
170  auto it = batch.mutation_data.find(idx);
171  completed_size += it->second.request_size;
172  it->second.completion_promise.set_value(Status());
173  // Release resources as early as possible.
174  batch.mutation_data.erase(it);
175  }
176 
177  std::unique_lock<std::mutex> lk(mu_);
178  outstanding_size_ -= completed_size;
179  num_requests_pending_ -= num_mutations;
180  SatisfyPromises(TryAdmit(cq), lk); // unlocks the lock
181 }

183 void MutationBatcher::OnFailedMutations(CompletionQueue& cq,
184  MutationBatcher::Batch& batch,
185  std::vector<FailedMutation> failed) {
186  size_t const num_mutations = failed.size();
187  size_t completed_size = 0;
188 
189  for (auto const& f : failed) {
190  int const idx = f.original_index();
191  auto it = batch.mutation_data.find(idx);
192  completed_size += it->second.request_size;
193  it->second.completion_promise.set_value(f.status());
194  // Release resources as early as possible.
195  batch.mutation_data.erase(it);
196  }
197  // TODO(#2093): remove once `FailedMutations` are small.
198  failed.clear();
199  failed.shrink_to_fit();
200 
201  std::unique_lock<std::mutex> lk(mu_);
202  outstanding_size_ -= completed_size;
203  num_requests_pending_ -= num_mutations;
204  SatisfyPromises(TryAdmit(cq), lk); // unlocks the lock
205 }

207 void MutationBatcher::OnBulkApplyAttemptFinished(
208  CompletionQueue& cq, MutationBatcher::Batch& batch) {
209  if (batch.attempt_finished) {
210  // We consider a batch finished if the original request finished. If it is
211  // later retried, we don't count it against the limit. The reasoning is that
212  // it would usually be some long tail of mutations and it should not take up
213  // the resources for the incoming requests.
214  return;
215  }
216  batch.attempt_finished = true;
217  std::unique_lock<std::mutex> lk(mu_);
218  num_outstanding_batches_ -= 1;
219  FlushIfPossible(cq);
220  SatisfyPromises(TryAdmit(cq), lk); // unlocks the lock
221 }

223 std::vector<MutationBatcher::AdmissionPromise> MutationBatcher::TryAdmit(
224  CompletionQueue& cq) {
225  // Defer satisfying promises until we release the lock.
226  std::vector<AdmissionPromise> admission_promises;
227 
228  do {
229  while (!pending_mutations_.empty() &&
230  HasSpaceFor(pending_mutations_.front())) {
231  auto& mut(pending_mutations_.front());
232  admission_promises.emplace_back(std::move(mut.admission_promise));
233  Admit(std::move(mut));
234  pending_mutations_.pop();
235  }
236  } while (FlushIfPossible(cq));
237  return admission_promises;
238 }

240 void MutationBatcher::Admit(PendingSingleRowMutation mut) {
241  outstanding_size_ += mut.request_size;
242  cur_batch_->requests_size += mut.request_size;
243  cur_batch_->num_mutations += mut.num_mutations;
244  cur_batch_->requests.emplace_back(std::move(mut.mut));
245  cur_batch_->mutation_data.emplace(cur_batch_->last_idx++,
246  Batch::MutationData(std::move(mut)));
247 }

249 void MutationBatcher::SatisfyPromises(
250  std::vector<AdmissionPromise> admission_promises,
251  std::unique_lock<std::mutex>& lk) {
252  std::vector<NoMorePendingPromise> no_more_pending_promises;
253  if (num_requests_pending_ == 0 && num_outstanding_batches_ == 0) {
254  // We should wait not only on num_requests_pending_ being zero but also on
255  // num_outstanding_batches_ because we want to allow the user to kill the
256  // completion queue after this promise is fulfilled. Otherwise, the user can
257  // destroy the completion queue while the last batch is still being
258  // processed - we've had this bug (#2140).
259  no_more_pending_promises_.swap(no_more_pending_promises);
260  }
261  lk.unlock();
262 
263  // Inform the user that we've admitted these mutations and there might be some
264  // space in the buffer finally.
265  for (auto& promise : admission_promises) {
266  promise.set_value();
267  }
268  for (auto& promise : no_more_pending_promises) {
269  promise.set_value();
270  }
271 }
