32 struct SafeGrpcRetry {
33  static inline bool IsTransientFailure(google::cloud::StatusCode code) {
34  return code == StatusCode::kAborted || code == StatusCode::kUnavailable ||
35  code == StatusCode::kDeadlineExceeded;
36  }

38  static inline bool IsOk(google::cloud::Status const& status) {
39  return status.ok();
40  }
41  static inline bool IsTransientFailure(google::cloud::Status const& status) {
42  return IsTransientFailure(status.code());
43  }
44  static inline bool IsPermanentFailure(google::cloud::Status const& status) {
45  return !IsOk(status) && !IsTransientFailure(status);
