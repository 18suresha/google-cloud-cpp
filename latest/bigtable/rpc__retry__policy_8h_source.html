31 /// An adapter to use `grpc::Status` with the `google::cloud::*Policies`.
32 struct SafeGrpcRetry {
33  static inline bool IsTransientFailure(google::cloud::StatusCode code) {
34  return code == StatusCode::kAborted || code == StatusCode::kUnavailable ||
35  code == StatusCode::kDeadlineExceeded;
36  }
37 
38  static inline bool IsOk(google::cloud::Status const& status) {
39  return status.ok();
40  }
41  static inline bool IsTransientFailure(google::cloud::Status const& status) {
42  return IsTransientFailure(status.code());
43  }
44  static inline bool IsPermanentFailure(google::cloud::Status const& status) {
45  return !IsOk(status) && !IsTransientFailure(status);
46  }
47 
48  // TODO(coryan) - remove ::grpc::Status version.
49  // Sometimes we need to use google::protobuf::rpc::Status, in which case this
50  // is a easier function
51  static inline bool IsTransientFailure(grpc::StatusCode code) {
52  return code == grpc::StatusCode::ABORTED ||
53  code == grpc::StatusCode::UNAVAILABLE ||
54  code == grpc::StatusCode::DEADLINE_EXCEEDED;
55  }
56  static inline bool IsOk(grpc::Status const& status) { return status.ok(); }
57  static inline bool IsTransientFailure(grpc::Status const& status) {
58  return IsTransientFailure(status.error_code());
59  }
60  static inline bool IsPermanentFailure(grpc::Status const& status) {
61  return !IsOk(status) && !IsTransientFailure(status);
62  }
63 };

66 /**
67  * Define the interface for controlling how the Bigtable client
68  * retries RPC operations.
69  *
70  * The C++ client for Bigtable needs to hide partial and temporary
71  * failures from the application. However, we need to give the users
72  * enough flexibility to control how many attempts are made to reissue
73  * operations, how often these attempts are executed, and how to
74  * signal that an error has occurred.
75  *
76  * The application provides an instance of this class when the Table
77  * (or TableAdmin) object is created. This instance serves as a
78  * prototype to create new RPCRetryPolicy objects of the same
79  * (dynamic) type and with the same initial state.
80  *
81  */
82 class RPCRetryPolicy {
83  public:
84  virtual ~RPCRetryPolicy() = default;
85 
86  /**
87  * Return a new copy of this object.
88  *
89  * Typically implemented as
90  * @code
91  * return std::unique_ptr<RPCRetryPolicy>(new Foo(*this));
92  * @endcode
93  */
94  virtual std::unique_ptr<RPCRetryPolicy> clone() const = 0;
95 
96  /**
97  * Update the ClientContext for the next call.
98  */
99  virtual void Setup(grpc::ClientContext& context) const = 0;
100 
101  /**
102  * Handle an RPC failure.
103  *
104  * @return true if the RPC operation should be retried.
105  */
106  virtual bool OnFailure(google::cloud::Status const& status) = 0;
107  // TODO(coryan) - remove ::grpc::Status version.
108  virtual bool OnFailure(grpc::Status const& status) = 0;
109 
110  static bool IsPermanentFailure(google::cloud::Status const& status) {
111  return internal::SafeGrpcRetry::IsPermanentFailure(status);
112  }
113  // TODO(coryan) - remove ::grpc::Status version.
114  static bool IsPermanentFailure(grpc::Status const& status) {
115  return internal::SafeGrpcRetry::IsPermanentFailure(status);
116  }
117 };
