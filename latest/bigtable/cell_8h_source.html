29 /**
30  * The in-memory representation of a Bigtable cell.
31  *
32  * Bigtable stores data in rows, indexes by row keys. Each row may contain
33  * multiple column families, each column family might contain multiple columns,
34  * and each column has multiple cells indexed by timestamp. Notice that the
35  * storage is sparse, column families, columns, and timestamps might contain
36  * zero cells.
37  *
38  * The Cell class owns all its data.
39  */
40 class Cell {
41  public:
42  /// Create a Cell and fill it with data.
43  Cell(std::string row_key, std::string family_name,
44  std::string column_qualifier, std::int64_t timestamp, std::string value,
45  std::vector<std::string> labels)
46  : row_key_(std::move(row_key)),
47  family_name_(std::move(family_name)),
48  column_qualifier_(std::move(column_qualifier)),
49  timestamp_(timestamp),
50  value_(std::move(value)),
51  labels_(std::move(labels)) {}
52 
53  /// Create a Cell and fill it with a 64-bit value encoded as big endian.
54  Cell(std::string row_key, std::string family_name,
55  std::string column_qualifier, std::int64_t timestamp, std::int64_t value,
56  std::vector<std::string> labels)
57  : Cell(std::move(row_key), std::move(family_name),
58  std::move(column_qualifier), timestamp,
59  google::cloud::internal::EncodeBigEndian(value),
60  std::move(labels)) {}
61 
62  /// Create a cell and fill it with data, but with empty labels.
63  Cell(std::string row_key, std::string family_name,
64  std::string column_qualifier, std::int64_t timestamp, std::string value)
65  : Cell(std::move(row_key), std::move(family_name),
66  std::move(column_qualifier), timestamp, std::move(value), {}) {}

70  Cell(std::string row_key, std::string family_name,
71  std::string column_qualifier, std::int64_t timestamp, std::int64_t value)
72  : Cell(std::move(row_key), std::move(family_name),
73  std::move(column_qualifier), timestamp, std::move(value), {}) {}

90  }
91 
92  /// Return the contents of this cell. The returned value is not valid after
93  /// this object is deleted.
94  std::string const& value() const { return value_; }
95 
96  /**
97  * Interpret the value as a big-endian encoded `T` and return it.
98  *
99  * Google Cloud Bigtable stores arbitrary blobs in each cell. Some
100  * applications interpret these blobs as strings, other as encoded protos,
101  * and sometimes as big-endian integers. This is a helper function to convert
102  * the blob into a T value.
103  */
104  template <typename T>
105  StatusOr<T> decode_big_endian_integer() const {
106  return google::cloud::internal::DecodeBigEndian<T>(value_);
107  }
108 
109  /// Return the labels applied to this cell by label transformer read filters.
110  std::vector<std::string> const& labels() const { return labels_; }
111

115  std::string column_qualifier_;
116  std::int64_t timestamp_;
117  std::string value_;
118  std::vector<std::string> labels_;
119 };
