41 void SetIamPolicy(google::cloud::bigtable::TableAdmin const& admin,
42  std::vector<std::string> const& argv) {
44  namespace cbt = google::cloud::bigtable;
45  using google::cloud::StatusOr;
46  [](cbt::TableAdmin admin, std::string const& table_id,
47  std::string const& role, std::string const& member) {
48  StatusOr<google::iam::v1::Policy> current = admin.GetIamPolicy(table_id);
49  if (!current) throw std::runtime_error(current.status().message());
50  // This example adds the member to all existing bindings for that role. If
51  // there are no such bindgs, it adds a new one. This might not be what the
52  // user wants, e.g. in case of conditional bindings.
53  size_t num_added = 0;
54  for (auto& binding : *current->mutable_bindings()) {
55  if (binding.role() == role) {
56  binding.add_members(member);
57  ++num_added;
58  }
59  }
60  if (num_added == 0) {
61  *current->add_bindings() = cbt::IamBinding(role, {member});
62  }
63  StatusOr<google::iam::v1::Policy> policy =
64  admin.SetIamPolicy(table_id, *current);
65  if (!policy) throw std::runtime_error(policy.status().message());
66  std::cout << "The IAM Policy for " << table_id << " is\n"
67  << policy->DebugString() << "\n";
68  }
70  (std::move(admin), argv.at(0), argv.at(1), argv.at(2));
71 }

135 void AsyncSetIamPolicy(google::cloud::bigtable::TableAdmin const& admin,
136  google::cloud::bigtable::CompletionQueue cq,
137  std::vector<std::string> const& argv) {
139  namespace cbt = google::cloud::bigtable;

141  using google::cloud::StatusOr;
142  using google::iam::v1::Policy;
143  [](cbt::TableAdmin admin, cbt::CompletionQueue cq,
144  std::string const& table_id, std::string const& role,
145  std::string const& member) {
146  future<StatusOr<google::iam::v1::Policy>> updated_future =
147  admin.AsyncGetIamPolicy(cq, table_id)
148  .then([cq, admin, role, member,
149  table_id](future<StatusOr<google::iam::v1::Policy>>
150  current_future) mutable {
151  auto current = current_future.get();
152  if (!current) {
153  return google::cloud::make_ready_future<
154  StatusOr<google::iam::v1::Policy>>(current.status());
155  }
156  // This example adds the member to all existing bindings for
157  // that role. If there are no such bindgs, it adds a new one.
158  // This might not be what the user wants, e.g. in case of
159  // conditional bindings.
160  size_t num_added = 0;
161  for (auto& binding : *current->mutable_bindings()) {
162  if (binding.role() == role) {
163  binding.add_members(member);
164  ++num_added;
165  }
166  }
167  if (num_added == 0) {
168  *current->add_bindings() = cbt::IamBinding(role, {member});
169  }
170  return admin.AsyncSetIamPolicy(cq, table_id, *current);
171  });
172  // Show how to perform additional work while the long running operation
173  // completes. The application could use future.then() instead.
174  std::cout << "Waiting for IAM policy update to complete " << std::flush;
175  updated_future.wait_for(std::chrono::seconds(2));
176  auto result = updated_future.get();
177  std::cout << '.' << std::flush;
178  if (!result) throw std::runtime_error(result.status().message());
179  using cbt::operator<<;
180  std::cout << "DONE, the IAM Policy for " << table_id << " is\n"
181  << *result << "\n";
182  }
184  (std::move(admin), std::move(cq), argv.at(0), argv.at(1), argv.at(2));
