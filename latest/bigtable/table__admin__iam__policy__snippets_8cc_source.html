43 void SetIamPolicy(google::cloud::bigtable::TableAdmin const& admin,
44  std::vector<std::string> const& argv) {
46  namespace cbt = google::cloud::bigtable;
47  using google::cloud::StatusOr;
48  [](cbt::TableAdmin admin, std::string const& table_id,
49  std::string const& role, std::string const& member) {
50  StatusOr<google::iam::v1::Policy> current = admin.GetIamPolicy(table_id);
51  if (!current) throw std::runtime_error(current.status().message());
52  // This example adds the member to all existing bindings for that role. If
53  // there are no such bindgs, it adds a new one. This might not be what the
54  // user wants, e.g. in case of conditional bindings.
55  size_t num_added = 0;
56  for (auto& binding : *current->mutable_bindings()) {
57  if (binding.role() == role) {
58  binding.add_members(member);
59  ++num_added;
60  }
61  }
62  if (num_added == 0) {
63  *current->add_bindings() = cbt::IamBinding(role, {member});
64  }
65  StatusOr<google::iam::v1::Policy> policy =
66  admin.SetIamPolicy(table_id, *current);
67  if (!policy) throw std::runtime_error(policy.status().message());
68  std::cout << "The IAM Policy for " << table_id << " is\n"
69  << policy->DebugString() << "\n";
70  }
72  (std::move(admin), argv.at(0), argv.at(1), argv.at(2));
73 }

123  auto const instance_id = google::cloud::internal::GetEnv(

135  google::cloud::CompletionQueue cq;
136  std::thread th([&cq] { cq.Run(); });
137  examples::AutoShutdownCQ shutdown(cq, std::move(th));

139  // If a previous run of these samples crashes before cleaning up there may be
140  // old tables left over. As there are quotas on the total number of tables we
141  // remove stale tables after 48 hours.
142  std::cout << "\nCleaning up old tables" << std::endl;
143  std::string const prefix = "table-admin-snippets-";
144  examples::CleanupOldTables(prefix, admin);
145 
146  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
147  // This table is actually created and used to test the positive case (e.g.
148  // GetTable() and "table does exist")
149  auto table_id = examples::RandomTableId(prefix, generator);
150 
151  auto table = admin.CreateTable(
152  table_id, cbt::TableConfig(
153  {
154  {"fam", cbt::GcRule::MaxNumVersions(10)},
155  {"foo", cbt::GcRule::MaxNumVersions(3)},
156  },
157  {}));
158  if (!table) throw std::runtime_error(table.status().message());
159 
160  std::cout << "\nRunning GetIamPolicy() example" << std::endl;
161  GetIamPolicy(admin, {table_id});
162 
163  std::cout << "\nRunning SetIamPolicy() example" << std::endl;

166

168  TestIamPermissions(
169  {project_id, instance_id, table_id, "bigtable.tables.get"});
170 
171  (void)admin.DeleteTable(table_id);
172 }
