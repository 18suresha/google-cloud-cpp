294  public:
296  template <
297  typename RowKey,
298  typename std::enable_if<std::is_constructible<RowKeyType, RowKey>::value,
299  int>::type = 0>
300  // NOLINTNEXTLINE(bugprone-forwarding-reference-overload)
301  explicit SingleRowMutation(RowKey&& row_key) {
302  request_.set_row_key(RowKeyType(std::forward<RowKey>(row_key)));
303  }
304 
306  template <typename RowKey>
307  SingleRowMutation(RowKey&& row_key, std::initializer_list<Mutation> list) {
308  request_.set_row_key(std::forward<RowKey>(row_key));
309  for (auto&& i : list) {
310  *request_.add_mutations() = i.op;
311  }
312  }
313 
315  template <
316  typename RowKey, typename... M,
317  typename std::enable_if<std::is_constructible<RowKeyType, RowKey>::value,
318  int>::type = 0>
319  explicit SingleRowMutation(RowKey&& row_key, M&&... m) {
320  static_assert(
321  absl::conjunction<std::is_convertible<M, Mutation>...>::value,
322  "The arguments passed to SingleRowMutation(std::string, ...) must be "
323  "convertible to Mutation");
324  request_.set_row_key(std::forward<RowKey>(row_key));
325  emplace_many(std::forward<M>(m)...);
326  }
327 
329  explicit SingleRowMutation(
330  ::google::bigtable::v2::MutateRowsRequest::Entry entry) {
331  using std::swap;
332  swap(*request_.mutable_row_key(), *entry.mutable_row_key());
333  swap(*request_.mutable_mutations(), *entry.mutable_mutations());
334  }
335 
337  explicit SingleRowMutation(::google::bigtable::v2::MutateRowRequest request)
338  : request_(std::move(request)) {}
339 
340  // Add a mutation at the end.
341  SingleRowMutation& emplace_back(Mutation mut) {
342  *request_.add_mutations() = std::move(mut.op);
343  return *this;
344  }

346  // Get the row key.
