59 /**
60  * @name Create mutations to delete a range of cells from a column.
61  *
62  * The following functions create a mutation that deletes all the
63  * cells in the given column family and, column within the given
64  * timestamp in the range.
65  *
66  * The function accepts any instantiation of `std::chrono::duration<>` for the
67  * @p timestamp_begin and @p timestamp_end parameters. For example:
68  *
69  * @code
70  * using namespace std::chrono_literals; // C++14
71  * bigtable::DeleteFromColumn("fam", "col", 0us, 10us)
72  * @endcode
73  *
74  * The ending timestamp is exclusive, while the beginning timestamp is
75  * inclusive. That is, the interval is [@p timestamp_begin, @p timestamp_end).
76  * The value 0 is special and treated as "unbounded" for both the begin and
77  * end endpoints of the time range. The Cloud Bigtable server rejects
78  * invalid and empty ranges, i.e., any range where the endpoint is smaller or
79  * equal than to the initial endpoint unless either endpoint is 0.
80  *
81  * @tparam Rep1 a placeholder to match the Rep tparam for @p timestamp_begin
82  * type. The semantics of this template parameter are documented in
83  * std::chrono::duration<>` (in brief, the underlying arithmetic type
84  * used to store the number of ticks), for our purposes it is simply a
85  * formal parameter.
86  *
87  * @tparam Rep2 similar formal parameter for the type of @p timestamp_end.
88  *
89  * @tparam Period1 a placeholder to match the Period tparam for
90  * @p timestamp_begin type. The semantics of this template parameter are
91  * documented in `std::chrono::duration<>` (in brief, the length of the tick
92  * in seconds,vexpressed as a `std::ratio<>`), for our purposes it is simply
93  * a formal parameter.
94  *
95  * @tparam Period2 similar formal parameter for the type of @p timestamp_end.
96  */
97 template <typename Rep1, typename Period1, typename Rep2, typename Period2>
98 Mutation DeleteFromColumn(std::string family, std::string column,
99  std::chrono::duration<Rep1, Period1> timestamp_begin,
100  std::chrono::duration<Rep2, Period2> timestamp_end) {
101  Mutation m;
102  using namespace std::chrono;
103  auto& d = *m.op.mutable_delete_from_column();
104  d.set_family_name(std::move(family));
105  d.set_column_qualifier(std::move(column));
106  d.mutable_time_range()->set_start_timestamp_micros(
107  duration_cast<microseconds>(timestamp_begin).count());
108  d.mutable_time_range()->set_end_timestamp_micros(
109  duration_cast<microseconds>(timestamp_end).count());
110  return m;
111 }

203 /**
204  * Represent a single row mutation.
205  *
206  * Bigtable can perform multiple changes to a single row atomically.
207  * This class represents 0 or more changes to apply to a single row.
208  * The changes may include setting cells (which implicitly insert the
209  * values), deleting values, etc.
210  */
211 class SingleRowMutation {
212  public:
213  /// Create an empty mutation.
214  explicit SingleRowMutation(std::string row_key)
215  : row_key_(std::move(row_key)) {}
216 
217  /// Create a row mutation from a initializer list.
218  SingleRowMutation(std::string row_key, std::initializer_list<Mutation> list)
219  : row_key_(std::move(row_key)) {
220  for (auto&& i : list) {
221  *ops_.Add() = i.op;
222  }
223  }
224 
225  /// Create a single-row multiple-cell mutation from a variadic list.
226  template <typename... M>
227  explicit SingleRowMutation(std::string row_key, M&&... m)
228  : row_key_(std::move(row_key)) {
229  static_assert(
230  internal::conjunction<std::is_convertible<M, Mutation>...>::value,
231  "The arguments passed to SingleRowMutation(std::string, ...) must be "
232  "convertible to Mutation");
233  emplace_many(std::forward<M>(m)...);
234  }
235 
236  /// Create a row mutation from gRPC proto
237  explicit SingleRowMutation(
238  ::google::bigtable::v2::MutateRowsRequest::Entry entry)
239  : row_key_(std::move(*entry.mutable_row_key())), ops_() {
240  ops_.Swap(entry.mutable_mutations());
241  }
242 
243  /// Create a row mutation from gRPC proto
244  explicit SingleRowMutation(::google::bigtable::v2::MutateRowRequest request)
245  : row_key_(std::move(*request.mutable_row_key())), ops_() {
246  ops_.Swap(request.mutable_mutations());
247  }
248 
249  // Add a mutation at the end.
250  SingleRowMutation& emplace_back(Mutation mut) {
251  ops_.Add()->Swap(&mut.op);
252  return *this;
253  }
254 
255  // Get the row key.
256  std::string const& row_key() const { return row_key_; }
257 
258  friend class Table;
259 
260  SingleRowMutation(SingleRowMutation&& rhs) = default;
261  SingleRowMutation& operator=(SingleRowMutation&& rhs) = default;
262  SingleRowMutation(SingleRowMutation const& rhs) = default;
263  SingleRowMutation& operator=(SingleRowMutation const& rhs) = default;
264 
265  /// Move the contents into a bigtable::v2::MutateRowsRequest::Entry.
266  void MoveTo(google::bigtable::v2::MutateRowsRequest::Entry* entry) {
267  entry->set_row_key(std::move(row_key_));
268  entry->mutable_mutations()->Swap(&ops_);
269  }
270 
271  /// Transfer the contents to @p request.
272  void MoveTo(google::bigtable::v2::MutateRowRequest& request) {
273  request.set_row_key(std::move(row_key_));
274  request.mutable_mutations()->Swap(&ops_);
275  }
276 
277  /// Remove the contents of the mutation.
278  void Clear() {
279  row_key_.clear();
280  ops_.Clear();
281  }
282 
283  private:
284  /// Add multiple mutations to single row
285  template <typename... M>
286  void emplace_many(Mutation first, M&&... tail) {
287  emplace_back(std::move(first));
288  emplace_many(std::forward<M>(tail)...);
289  }
290 
291  void emplace_many(Mutation m) { emplace_back(std::move(m)); }
292 
293  private:
294  std::string row_key_;
295  google::protobuf::RepeatedPtrField<google::bigtable::v2::Mutation> ops_;
296 };

336 /**
337  * Report unrecoverable errors in a partially completed mutation.
338  */
339 class PermanentMutationFailure : public std::runtime_error {
340  public:
341  PermanentMutationFailure(char const* msg,
342  std::vector<FailedMutation> failures)
343  : std::runtime_error(msg), failures_(std::move(failures)) {}
344 
345  PermanentMutationFailure(char const* msg, grpc::Status status,
346  std::vector<FailedMutation> failures)
347  : std::runtime_error(msg),
348  failures_(std::move(failures)),
349  status_(std::move(status)) {}
350 
351  /**
352  * The details of each mutation failure.
353  *
354  * Because BulkApply() and Apply() take ownership of the data in the mutations
355  * the failures are returned with their full contents, in case the application
356  * wants to take further action with them. Any successful mutations are
357  * discarded.
358  *
359  * Any mutations that fail with an unknown state are included with a
360  * grpc::StatusCode::OK.
361  */
362  std::vector<FailedMutation> const& failures() const { return failures_; }
363 
364  /**
365  * The grpc::Status of the request.
366  *
367  * Notice that it can return grpc::Status::OK when there are partial failures
368  * in a BulkApply() operation.
369  */
370  grpc::Status const& status() const { return status_; }
371 
372  private:
373  std::vector<FailedMutation> failures_;
374  grpc::Status status_;
375 };

377 /**
378  * Represent a set of mutations across multiple rows.
379  *
380  * Cloud Bigtable can batch multiple mutations in a single request.
381  * The mutations are not atomic, but it is more efficient to send them
382  * in a batch than to make multiple smaller requests.
383  */
384 class BulkMutation {
385  public:
386  /// Create an empty set of mutations.
387  BulkMutation() : request_() {}
388 
389  /// Create a multi-row mutation from a range of SingleRowMutations.
390  template <typename iterator>
391  BulkMutation(iterator begin, iterator end) {
392  static_assert(
393  std::is_convertible<decltype(*begin), SingleRowMutation>::value,
394  "The iterator value type must be convertible to SingleRowMutation");
395  for (auto i = begin; i != end; ++i) {
396  push_back(*i);
397  }
398  }
399 
400  /// Create a multi-row mutation from a initializer list.
401  BulkMutation(std::initializer_list<SingleRowMutation> list)
402  : BulkMutation(list.begin(), list.end()) {}
403 
404  /// Create a multi-row mutation from a SingleRowMutation
405  explicit BulkMutation(SingleRowMutation mutation) : BulkMutation() {
406  emplace_back(std::move(mutation));
407  }
408 
409  /// Create a muti-row mutation from two SingleRowMutation
410  BulkMutation(SingleRowMutation m1, SingleRowMutation m2) : BulkMutation() {
411  emplace_back(std::move(m1));
412  emplace_back(std::move(m2));
413  }
414 
415  /// Create a muti-row mutation from a variadic list.
416  template <typename... M>
417  BulkMutation(M&&... m) : BulkMutation() {
418  static_assert(
419  internal::conjunction<
420  std::is_convertible<M, SingleRowMutation>...>::value,
421  "The arguments passed to BulkMutation(...) must be convertible"
422  " to SingleRowMutation");
423  emplace_many(std::forward<M>(m)...);
424  }
425 
426  // Add a mutation to the batch.
427  BulkMutation& emplace_back(SingleRowMutation mut) {
428  mut.MoveTo(request_.add_entries());
429  return *this;
430  }
431 
432  // Add a failed mutation to the batch.
433  BulkMutation& emplace_back(FailedMutation fm) {
434  fm.mutation_.MoveTo(request_.add_entries());
435  fm.status_ = grpc::Status::OK;
436  return *this;
437  }
438 
439  // Add a mutation to the batch.
440  BulkMutation& push_back(SingleRowMutation mut) {
441  mut.MoveTo(request_.add_entries());
442  return *this;
443  }
444 
445  /// Move the contents into a bigtable::v2::MutateRowsRequest
446  void MoveTo(google::bigtable::v2::MutateRowsRequest* request) {
447  request_.Swap(request);
448  request_ = {};
449  }
450 
451  /// Return true if there are no mutations in this set.
452  bool empty() const { return request_.entries().empty(); }
453 
454  /// Return the number of mutations in this set.
455  size_t size() const { return request_.entries().size(); }
456 
457  private:
458  template <typename... M>
459  void emplace_many(SingleRowMutation first, M&&... tail) {
460  emplace_back(std::move(first));
461  emplace_many(std::forward<M>(tail)...);
462  }
463 
464  void emplace_many(SingleRowMutation m) { emplace_back(std::move(m)); }
465 
466  private:
467  google::bigtable::v2::MutateRowsRequest request_;
468 };
