293 class SingleRowMutation {
294  public:
296  template <
297  typename RowKey,
298  typename std::enable_if<std::is_constructible<RowKeyType, RowKey>::value,
299  int>::type = 0>
300  // NOLINTNEXTLINE(bugprone-forwarding-reference-overload)
301  explicit SingleRowMutation(RowKey&& row_key) {
302  request_.set_row_key(RowKeyType(std::forward<RowKey>(row_key)));
303  }
304 
306  template <typename RowKey>
307  SingleRowMutation(RowKey&& row_key, std::initializer_list<Mutation> list) {
308  request_.set_row_key(std::forward<RowKey>(row_key));
309  for (auto&& i : list) {
310  *request_.add_mutations() = i.op;
311  }
312  }
313 
315  template <
316  typename RowKey, typename... M,
317  typename std::enable_if<std::is_constructible<RowKeyType, RowKey>::value,
318  int>::type = 0>
319  explicit SingleRowMutation(RowKey&& row_key, M&&... m) {
320  static_assert(
321  absl::conjunction<std::is_convertible<M, Mutation>...>::value,
322  "The arguments passed to SingleRowMutation(std::string, ...) must be "
323  "convertible to Mutation");
324  request_.set_row_key(std::forward<RowKey>(row_key));
325  emplace_many(std::forward<M>(m)...);
326  }
327 
329  explicit SingleRowMutation(
330  ::google::bigtable::v2::MutateRowsRequest::Entry entry) {
331  using std::swap;
332  swap(*request_.mutable_row_key(), *entry.mutable_row_key());
333  swap(*request_.mutable_mutations(), *entry.mutable_mutations());
334  }
335 
337  explicit SingleRowMutation(::google::bigtable::v2::MutateRowRequest request)
338  : request_(std::move(request)) {}
339 
340  // Add a mutation at the end.
341  SingleRowMutation& emplace_back(Mutation mut) {
342  *request_.add_mutations() = std::move(mut.op);
343  return *this;
344  }
345 
346  // Get the row key.
347  RowKeyType const& row_key() const { return request_.row_key(); }
348 
349  friend class Table;
350 
351  SingleRowMutation(SingleRowMutation&&) = default;
352  SingleRowMutation& operator=(SingleRowMutation&&) = default;
353  SingleRowMutation(SingleRowMutation const&) = default;
354  SingleRowMutation& operator=(SingleRowMutation const&) = default;
355 
357  void MoveTo(google::bigtable::v2::MutateRowsRequest::Entry* entry) {
358  entry->set_row_key(std::move(*request_.mutable_row_key()));
359  *entry->mutable_mutations() = std::move(*request_.mutable_mutations());
360  }
361 
363  void MoveTo(google::bigtable::v2::MutateRowRequest& request) {
364  request.set_row_key(std::move(*request_.mutable_row_key()));
365  *request.mutable_mutations() = std::move(*request_.mutable_mutations());
366  }
367 
369  void Clear() { request_.Clear(); }
370 
371  private:
373  template <typename... M>
374  void emplace_many(Mutation first, M&&... tail) {
375  emplace_back(std::move(first));
376  emplace_many(std::forward<M>(tail)...);
377  }
378 
379  void emplace_many(Mutation m) { emplace_back(std::move(m)); }
380 
381  ::google::bigtable::v2::MutateRowRequest request_;
382 };

472  template <typename Iterator>
473  BulkMutation(Iterator begin, Iterator end) {
474  static_assert(
475  std::is_convertible<decltype(*begin), SingleRowMutation>::value,
476  "The iterator value type must be convertible to SingleRowMutation");
477  for (auto i = begin; i != end; ++i) {
478  push_back(*i);
479  }
480  }
481 
483  BulkMutation(std::initializer_list<SingleRowMutation> list)
484  : BulkMutation(list.begin(), list.end()) {}
485 
487  explicit BulkMutation(SingleRowMutation mutation) : BulkMutation() {
488  emplace_back(std::move(mutation));
489  }
490 
492  BulkMutation(SingleRowMutation m1, SingleRowMutation m2) : BulkMutation() {
493  emplace_back(std::move(m1));
494  emplace_back(std::move(m2));
495  }
496 
498  template <typename... M,
499  typename std::enable_if<absl::conjunction<std::is_convertible<
500  M, SingleRowMutation>...>::value,
501  int>::type = 0>
502  // NOLINTNEXTLINE(google-explicit-constructor)
503  BulkMutation(M&&... m) : BulkMutation() {
504  emplace_many(std::forward<M>(m)...);
505  }
506 
507  // Add a mutation to the batch.
508  BulkMutation& emplace_back(SingleRowMutation mut) {
509  mut.MoveTo(request_.add_entries());
510  return *this;
511  }
512 
513  // Add a failed mutation to the batch.
514  BulkMutation& emplace_back(FailedMutation fm) {
515  fm.status_ = google::cloud::Status();
516  return *this;
517  }
518 
519  // Add a mutation to the batch.
520  BulkMutation& push_back(SingleRowMutation mut) {
521  mut.MoveTo(request_.add_entries());
522  return *this;
523  }
524 
526  void MoveTo(google::bigtable::v2::MutateRowsRequest* request) {
527  request_.Swap(request);
528  request_ = {};
529  }
530 
532  bool empty() const { return request_.entries().empty(); }
533 
535  std::size_t size() const { return request_.entries().size(); }
536 
538  std::size_t estimated_size_in_bytes() const {
539  return request_.ByteSizeLong();
540  }
541 
542  private:
543  template <typename... M>
544  void emplace_many(SingleRowMutation first, M&&... tail) {
545  emplace_back(std::move(first));
546  emplace_many(std::forward<M>(tail)...);
547  }
548 
549  void emplace_many(SingleRowMutation m) { emplace_back(std::move(m)); }
550 
551  google::bigtable::v2::MutateRowsRequest request_;
552 };
