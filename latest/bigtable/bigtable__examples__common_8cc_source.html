40 void CleanupOldTables(std::string const& prefix,
41  google::cloud::bigtable::TableAdmin admin) {
42  namespace cbt = google::cloud::bigtable;
43 
44  auto const threshold =
45  std::chrono::system_clock::now() - std::chrono::hours(48);
46  auto const max_table_id = TablePrefix(prefix, threshold);
47  auto const max_table_name = admin.TableName(max_table_id);
48  auto const prefix_name = admin.TableName(prefix);
49 
50  auto tables = admin.ListTables(cbt::TableAdmin::NAME_ONLY);
51  if (!tables) return;
52  for (auto const& t : *tables) {
53  if (t.name().rfind(prefix_name, 0) != 0) continue;
54  // Eventually (I heard from good authority around year 2286) the date
55  // formatted in seconds will gain an extra digit and this will no longer
56  // work. If you are a programmer from the future, I (coryan) am (a) almost
57  // certainly dead, (b) very confused that this code is still being
58  // maintained or used, and (c) a bit sorry that this caused you problems.
59  if (t.name() >= max_table_name) continue;
60  // Failure to cleanup is not an error.
61  auto const table_id = t.name().substr(t.name().find_last_of('/') + 1);
62  std::cout << "Deleting table " << table_id << std::endl;
63  (void)admin.DeleteTable(table_id);
64  }
65 }

67 void CleanupOldBackups(std::string const& cluster_id,
68  google::cloud::bigtable::TableAdmin admin) {
69  std::string expire_time = google::protobuf::util::TimeUtil::ToString(
70  google::protobuf::util::TimeUtil::GetCurrentTime() -
71  google::protobuf::util::TimeUtil::HoursToDuration(24 * 7));
72  std::string filter = "expire_time < " + expire_time;
73  auto params = google::cloud::bigtable::TableAdmin::ListBackupsParams()
74  .set_cluster(cluster_id)
75  .set_filter(filter);
76  auto backups = admin.ListBackups(params);
77  if (!backups) return;
78  for (auto const& backup : *backups) {
79  std::cout << "Deleting backup " << backup.name() << "in cluster "
80  << cluster_id << std::endl;
81  (void)admin.DeleteBackup(backup);
82  }
83 }

94 std::string RandomInstanceId(std::string const& prefix,
95  google::cloud::internal::DefaultPRNG& generator) {
96  // Cloud Bigtable instance ids must have at least 6 characters, and can have
97  // up to 33 characters. But many of the examples append `-c1` or `-c2` to
98  // create cluster ids based on the instance id. So we make the generated ids
99  // even shorter.
100  auto constexpr kMaxInstanceIdLength = kMaxClusterIdLength - 3;
101  auto const timestamped_prefix =
102  InstancePrefix(prefix, std::chrono::system_clock::now());
103  if (timestamped_prefix.size() >= kMaxInstanceIdLength) {
104  throw std::invalid_argument("prefix too long for RandomInstanceId()");
105  }
106  auto const suffix_len = kMaxInstanceIdLength - timestamped_prefix.size();
107  return timestamped_prefix + google::cloud::internal::Sample(
108  generator, static_cast<int>(suffix_len),
109  "abcdefghijklmnopqrstuvwxyz0123456789");
110 }
