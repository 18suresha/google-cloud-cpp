32  if (argc != 4) {
33  std::string const cmd = argv[0];
34  auto last_slash = std::string(cmd).find_last_of('/');
35  std::cerr << "Usage: " << cmd.substr(last_slash + 1)

40  std::string const project_id = argv[1];
41  std::string const instance_id = argv[2];
42  std::string const table_id = argv[3];
43 
44  // Register the OpenCensus gRPC plugin to enable stats and tracing in gRPC.
45  grpc::RegisterOpenCensusPlugin();
46 
47  // The `ProbabilitySampler` used in this example samples out request and
48  // does not trace all the requests. So, if production system requires tracing
49  // of each request then different samplers must be used
50  //
51  // For more details, see the documentation
52  //
53  // https://opencensus.io/core-concepts/tracing/#sampling
54  // https://github.com/census-instrumentation/opencensus-specs/blob/master/trace/Sampling.md#sampling
55  //
56 
57  opencensus::trace::TraceConfig::SetCurrentTraceParams(
58  {128, 128, 128, 128, opencensus::trace::ProbabilitySampler(1.0)});
59 
60  // For debugging, register exporters that just write to stdout.
61  opencensus::exporters::stats::StdoutExporter::Register();
62  opencensus::exporters::trace::StdoutExporter::Register();
63 
64  // Registration of Stackdriver requires couple of parameters,
65  // project_id The Stackdriver Project ID to use
66  // opencensus_task The opencensus_task is used to uniquely identify the task
67  // in Stackdriver. The recommended format is "{LANGUAGE}-{PID}@{HOSTNAME}".
68  // If PID is not available, a random number may be used.
69  //
70  // For more details, see the documentation
71  // https://github.com/census-instrumentation/opencensus-cpp/tree/master/opencensus/exporters/stats/stackdriver#opencensus-stackdriver-stats-exporter
72 
73  opencensus::exporters::stats::StackdriverExporter::Register(
74  project_id, "bigtable-opencensus-0@unspecified-host");
75  opencensus::exporters::trace::StackdriverExporter::Register(project_id);

77  // Connect to the Cloud Bigtable Admin API.
78  //! [connect admin]
79  google::cloud::bigtable::TableAdmin table_admin(
80  google::cloud::bigtable::CreateDefaultAdminClient(
81  project_id, google::cloud::bigtable::ClientOptions()),
82  instance_id);

85  //! [create table]
86  // Define the desired schema for the Table.
87  auto gc_rule = google::cloud::bigtable::GcRule::MaxNumVersions(1);
88  google::cloud::bigtable::TableConfig schema({{"family", gc_rule}}, {});

98  project_id, instance_id, google::cloud::bigtable::ClientOptions()),
99  table_id);
100  //! [connect data]
101 
102  // Modify (and create if necessary) a row.
103  //! [write rows]
104  std::vector<std::string> greetings{"Hello World!", "Hello Cloud Bigtable!",
105  "Hello C++!"};
106  int i = 0;
107  for (auto const& greeting : greetings) {
108  // Each row has a unique row key.
109  //
110  // Note: This example uses sequential numeric IDs for simplicity, but
111  // this can result in poor performance in a production application.
112  // Since rows are stored in sorted order by key, sequential keys can
113  // result in poor distribution of operations across nodes.
114  //
115  // For more information about how to design a Bigtable schema for the
116  // best performance, see the documentation:
117  //
118  // https://cloud.google.com/bigtable/docs/schema-design
119  std::string row_key = "key-" + std::to_string(i);
120  google::cloud::Status status =
121  table.Apply(google::cloud::bigtable::SingleRowMutation(
122  std::move(row_key),
123  google::cloud::bigtable::SetCell("family", "c0", greeting)));
124  if (!status.ok()) {
125  throw std::runtime_error(status.message());
126  }
127  ++i;
128  }
129  //! [write rows]
130 
131  // Read a single row.
132  //! [read row]
133  auto result = table.ReadRow(
134  "key-0",
135  google::cloud::bigtable::Filter::ColumnRangeClosed("family", "c0", "c0"));
136  if (!result) {
137  throw std::runtime_error(result.status().message());
138  }
139  if (!result->first) {
140  std::cout << "Cannot find row 'key-0' in the table: " << table.table_name()
141  << "\n";
142  return 0;
143  }
144  auto const& cell = result->second.cells().front();
145  std::cout << cell.family_name() << ":" << cell.column_qualifier() << " @ "
146  << cell.timestamp().count() << "us\n"
147  << '"' << cell.value() << '"' << "\n";
148  //! [read row]
149 
150  // Read a single row.
151  //! [scan all]
152  for (auto& row :
153  table.ReadRows(google::cloud::bigtable::RowRange::InfiniteRange(),
154  google::cloud::bigtable::Filter::PassAllFilter())) {
155  if (!row) {
156  throw std::runtime_error(row.status().message());
157  }
158  std::cout << row->row_key() << ":\n";
159  for (auto& cell : row->cells()) {
160  std::cout << "\t" << cell.family_name() << ":" << cell.column_qualifier()
161  << " @ " << cell.timestamp().count() << "us\n"
162  << "\t\"" << cell.value() << '"' << "\n";
163  }
164  }
165  //! [scan all]
166 
167  // Delete the table
168  //! [delete table]
169  google::cloud::Status status = table_admin.DeleteTable(table_id);
170  if (!status.ok()) {
171  throw std::runtime_error(status.message());
172  }
173  //! [delete table]
174 
175  // Stop tracing because the remaining RPCs are OpenCensus related.
176  opencensus::trace::TraceConfig::SetCurrentTraceParams(
177  {128, 128, 128, 128, opencensus::trace::ProbabilitySampler(0.0)});
178 
179  std::cout << "Sleeping to give exporters time " << std::flush;
180  for (int i = 0; i != 30; ++i) {
181  std::this_thread::sleep_for(std::chrono::seconds(10));
182  std::cout << '.' << std::flush;
183  }
184  std::cout << " DONE\n";
185 
186  return 0;
187 } catch (std::exception const& ex) {
188  std::cerr << "Standard C++ exception raised: " << ex.what() << "\n";
189  return 1;
190 }
