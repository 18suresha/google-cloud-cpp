293 void UpdateCluster(google::cloud::bigtable::InstanceAdmin instance_admin,
294  std::vector<std::string> const& argv) {
296  namespace cbt = google::cloud::bigtable;
297  using google::cloud::StatusOr;
298  [](cbt::InstanceAdmin instance_admin, std::string const& instance_id,
299  std::string const& cluster_id) {
300  // GetCluster first and then modify it.
301  StatusOr<google::bigtable::admin::v2::Cluster> cluster =
302  instance_admin.GetCluster(instance_id, cluster_id);
303  if (!cluster) throw std::runtime_error(cluster.status().message());
304 
305  // The state cannot be sent on updates, so clear it first.
306  cluster->clear_state();
307  // Set the desired cluster configuration.
308  cluster->set_serve_nodes(4);
309  auto modified_config = cbt::ClusterConfig(std::move(*cluster));
310 
311  StatusOr<google::bigtable::admin::v2::Cluster> modified_cluster =
312  instance_admin.UpdateCluster(modified_config).get();
313  if (!modified_cluster) {
314  throw std::runtime_error(modified_cluster.status().message());
315  }
316  std::cout << "cluster details : " << cluster->DebugString() << "\n";
317  }
319  (std::move(instance_admin), argv.at(0), argv.at(1));
320 }

552 void DeleteAppProfile(std::vector<std::string> const& argv) {
553  std::string basic_usage =
554  "delete-app-profile <project-id> <instance-id> <profile-id>"
555  " [ignore-warnings (default: true)]";
556  if (argv.size() != 3 && argv.size() != 4) {
557  throw Usage{basic_usage};
558  }
559 
560  bool ignore_warnings = true;
561  if (argv.size() == 4) {
562  std::string arg = argv[3];
563  if (arg == "true") {
564  ignore_warnings = true;
565  } else if (arg == "false") {
566  ignore_warnings = false;
567  } else {
568  auto msg = basic_usage;
569  msg +=
570  "\ndelete-app-profile: ignore-warnings parameter must be either"
571  " 'true' or 'false'";
572  throw Usage{msg};
573  }
574  }
575  google::cloud::bigtable::InstanceAdmin instance(
576  google::cloud::bigtable::CreateDefaultInstanceAdminClient(

654 void SetNativeIamPolicy(google::cloud::bigtable::InstanceAdmin instance_admin,
655  std::vector<std::string> const& argv) {
657  namespace cbt = google::cloud::bigtable;
658  using google::cloud::StatusOr;
659  [](cbt::InstanceAdmin instance_admin, std::string const& instance_id,
660  std::string const& role, std::string const& member) {
661  StatusOr<google::iam::v1::Policy> current =
662  instance_admin.GetNativeIamPolicy(instance_id);
663  if (!current) throw std::runtime_error(current.status().message());
664  // This example adds the member to all existing bindings for that role. If
665  // there are no such bindgs, it adds a new one. This might not be what the
666  // user wants, e.g. in case of conditional bindings.
667  size_t num_added = 0;
668  for (auto& binding : *current->mutable_bindings()) {
669  if (binding.role() == role) {
670  binding.add_members(member);
671  ++num_added;
672  }
673  }
674  if (num_added == 0) {
675  *current->add_bindings() = cbt::IamBinding(role, {member});
676  }
677  StatusOr<google::iam::v1::Policy> policy =
678  instance_admin.SetIamPolicy(instance_id, *current);
679  if (!policy) throw std::runtime_error(policy.status().message());
680  std::cout << "The IAM Policy for " << instance_id << " is\n"
681  << policy->DebugString() << "\n";
682  }
684  (std::move(instance_admin), argv.at(0), argv.at(1), argv.at(2));
685 }
