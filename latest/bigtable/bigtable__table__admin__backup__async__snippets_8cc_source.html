28 void AsyncCreateBackup(google::cloud::bigtable::TableAdmin const& admin,
29  google::cloud::bigtable::CompletionQueue cq,
30  std::vector<std::string> const& argv) {
32  namespace cbt = google::cloud::bigtable;
33  using google::cloud::future;
34  using google::cloud::StatusOr;
35  [](cbt::TableAdmin admin, cbt::CompletionQueue cq,
36  std::string const& table_id, std::string const& cluster_id,
37  std::string const& backup_id, std::string const& expire_time_string) {
38  absl::Time t;
39  std::string err;
40  if (!absl::ParseTime(absl::RFC3339_full, expire_time_string, &t, &err)) {
41  throw std::runtime_error("Unable to parse expire_time:" + err);
42  }
43  auto expire_time = absl::ToChronoTime(t);
44 
45  future<StatusOr<google::bigtable::admin::v2::Backup>> backup_future =
46  admin.AsyncCreateBackup(
47  cq, cbt::TableAdmin::CreateBackupParams(cluster_id, backup_id,
48  table_id, expire_time));
49 
50  auto final = backup_future.then(
51  [](future<StatusOr<google::bigtable::admin::v2::Backup>> f) {
52  auto backup = f.get();
53  if (!backup) {
54  throw std::runtime_error(backup.status().message());
55  }
56  std::cout << "Backup successfully created: " << backup->DebugString()
57  << "\n";
58  return google::cloud::Status();
59  });
60  final.get();
61  }
63  (admin, std::move(cq), argv.at(0), argv.at(1), argv.at(2), argv.at(3));
64 }

256 void RunAll(std::vector<std::string> const& argv) {
257  namespace examples = ::google::cloud::bigtable::examples;
258  namespace cbt = google::cloud::bigtable;
259 
260  if (!argv.empty()) throw examples::Usage{"auto"};
261  if (!examples::RunAdminIntegrationTests()) return;
262 
263  examples::CheckEnvironmentVariablesAreSet({
264  "GOOGLE_CLOUD_PROJECT",
265  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID",
266  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_SERVICE_ACCOUNT",
267  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_CLUSTER_ID",
268  });
269  auto const project_id =
270  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
271  auto const instance_id = google::cloud::internal::GetEnv(
272  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_INSTANCE_ID")
273  .value();
274  auto const service_account =
275  google::cloud::internal::GetEnv(
276  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_SERVICE_ACCOUNT")
277  .value();
278  auto const cluster_id = google::cloud::internal::GetEnv(
279  "GOOGLE_CLOUD_CPP_BIGTABLE_TEST_CLUSTER_ID")
280  .value();
281 
282  cbt::TableAdmin admin(
283  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
284  instance_id);
285 
286  google::cloud::CompletionQueue cq;
287  std::thread th([&cq] { cq.Run(); });
288  examples::AutoShutdownCQ shutdown(cq, std::move(th));
289 
290  // If a previous run of these samples crashes before cleaning up there may be
291  // old tables left over. As there are quotas on the total number of tables we
292  // remove stale tables after 48 hours.
293  std::cout << "\nCleaning up old tables" << std::endl;
294  std::string const prefix = "table-admin-snippets-";
295  examples::CleanupOldTables(prefix, admin);
296  std::string const backup_prefix = "table-admin-snippets-backup-";
297  examples::CleanupOldBackups(cluster_id, admin);
298 
299  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
300  auto table_id = examples::RandomTableId(prefix, generator);
301 
302  auto table = admin.CreateTable(
303  table_id, cbt::TableConfig(
304  {
305  {"fam", cbt::GcRule::MaxNumVersions(10)},
306  {"foo", cbt::GcRule::MaxNumVersions(3)},
307  },
308  {}));
309  if (!table) throw std::runtime_error(table.status().message());
310 
311  std::cout << "\nRunning AsyncCreateBackup() example" << std::endl;
312  auto backup_id = examples::RandomTableId(backup_prefix, generator);
313  AsyncCreateBackup(admin, cq,
314  {table_id, cluster_id, backup_id,
315  absl::FormatTime(absl::Now() + absl::Hours(12))});
316 
317  std::cout << "\nRunning AsyncListBackups() example" << std::endl;
318  AsyncListBackups(admin, cq, {"-", {}, {}});
319 
320  std::cout << "\nRunning AsyncGetBackup() example" << std::endl;
321  AsyncGetBackup(admin, cq, {cluster_id, backup_id});
322 
323  std::cout << "\nRunning AsyncUpdateBackup() example" << std::endl;
324  AsyncUpdateBackup(
325  admin, cq,
326  {cluster_id, backup_id, absl::FormatTime(absl::Now() + absl::Hours(24))});
327 
328  (void)admin.DeleteTable(table_id);
329 
330  std::cout << "\nRunning AsyncRestoreTable() example" << std::endl;
331  AsyncRestoreTable(admin, cq, {table_id, cluster_id, backup_id});
332 
333  (void)admin.DeleteTable(table_id);
334 
335  std::cout << "\nRunning AsyncRestoreTableFromInstance() example" << std::endl;
336  AsyncRestoreTableFromInstance(admin, cq,
337  {table_id, instance_id, cluster_id, backup_id});
338 
339  std::cout << "\nRunning AsyncDeleteBackup() example" << std::endl;
340  AsyncDeleteBackup(admin, cq, {cluster_id, backup_id});
341 
342  (void)admin.DeleteTable(table_id);
343 }
