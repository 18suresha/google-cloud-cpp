41 std::string DefaultDataEndpoint() {
42  auto emulator = google::cloud::internal::GetEnv("BIGTABLE_EMULATOR_HOST");
43  if (emulator.has_value()) return *std::move(emulator);
44  auto direct_path =
45  google::cloud::internal::GetEnv("GOOGLE_CLOUD_ENABLE_DIRECT_PATH");
46  if (direct_path.has_value()) {
47  for (auto const& token : absl::StrSplit(*std::move(direct_path), ',')) {
48  if (token == "bigtable") return "directpath-bigtable.googleapis.com";

83 inline std::size_t CalculateDefaultConnectionPoolSize() {
84  // For batter resource utilization and greater throughput, it is recommended
85  // to calculate the default pool size based on cores(CPU) available. However,
86  // as per C++11 documentation `std::thread::hardware_concurrency()` cannot be
87  // fully rely upon, it is only a hint and the value can be 0 if it is not
88  // well defined or not computable. Apart from CPU count, multiple channels can
89  // be opened for each CPU to increase throughput.
90  std::size_t cpu_count = std::thread::hardware_concurrency();
91  return (cpu_count > 0) ? cpu_count * BIGTABLE_CLIENT_DEFAULT_CHANNELS_PER_CPU
92  : BIGTABLE_CLIENT_DEFAULT_CONNECTION_POOL_SIZE;
93 }
