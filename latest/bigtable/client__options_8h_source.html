149  /**
150  * Set the grpclb fallback timeout with the timestamp @p fallback_timeout
151  * for the channel.
152  *
153  * For example:
154  *
155  * @code
156  * bigtable::ClientOptions::SetGrpclbFallbackTimeout(
157  * std::chrono::milliseconds(5000))
158  * bigtable::ClientOptions::SetGrpclbFallbackTimeout(
159  * std::chrono::seconds(5))
160  * @endcode
161  *
162  * @tparam Rep a placeholder to match the Rep tparam for @p fallback_timeout,
163  * the semantics of this template parameter are documented in
164  * `std::chrono::duration<>` (in brief, the underlying arithmetic type
165  * used to store the number of ticks), for our purposes it is simply a
166  * formal parameter.
167  * @tparam Period a placeholder to match the Period tparam for @p
168  * fallback_timeout, the semantics of this template parameter are
169  * documented in `std::chrono::duration<>` (in brief, the length of the
170  * tick in seconds, expressed as a `std::ratio<>`), for our purposes it
171  * is simply a formal parameter.
172  *
173  * @see
174  * [std::chrono::duration<>](http://en.cppreference.com/w/cpp/chrono/duration)
175  * for more details.
176  *
177  * Please see the docs for grpc::ChannelArguments::SetGrpclbFallbackTimeout()
178  * on https://grpc.io/grpc/cpp/classgrpc_1_1_channel_arguments.html
179  * for more details.
180  *
181  */
182  template <typename Rep, typename Period>
183  google::cloud::Status SetGrpclbFallbackTimeout(
184  std::chrono::duration<Rep, Period> fallback_timeout) {
185  std::chrono::milliseconds ft_ms =
186  std::chrono::duration_cast<std::chrono::milliseconds>(fallback_timeout);
187 
188  if (ft_ms.count() > std::numeric_limits<int>::max()) {
189  return google::cloud::Status(google::cloud::StatusCode::kOutOfRange,
190  "The supplied duration is larger than the "
191  "maximum value allowed by gRPC (INT_MAX)");
192  }
193  auto fallback_timeout_ms = static_cast<int>(ft_ms.count());
194  channel_arguments_.SetGrpclbFallbackTimeout(fallback_timeout_ms);
195  return google::cloud::Status();
196  }
