200  std::chrono::duration<Rep, Period> fallback_timeout) {
201  std::chrono::milliseconds ft_ms =
202  std::chrono::duration_cast<std::chrono::milliseconds>(fallback_timeout);

206  "The supplied duration is larger than the "

208  }
209  auto fallback_timeout_ms = static_cast<int>(ft_ms.count());
210  channel_arguments_.SetGrpclbFallbackTimeout(fallback_timeout_ms);
