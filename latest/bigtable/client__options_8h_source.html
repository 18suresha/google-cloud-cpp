203  std::chrono::duration<Rep, Period> fallback_timeout) {
204  std::chrono::milliseconds ft_ms =
205  std::chrono::duration_cast<std::chrono::milliseconds>(fallback_timeout);
206 
207  if (ft_ms.count() > std::numeric_limits<int>::max()) {

209  "The supplied duration is larger than the "
210  "maximum value allowed by gRPC (INT_MAX)");
211  }
212  auto fallback_timeout_ms = static_cast<int>(ft_ms.count());
213  channel_arguments_.SetGrpclbFallbackTimeout(fallback_timeout_ms);
214  return google::cloud::Status();
