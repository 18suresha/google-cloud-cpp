268 template <typename ConnectionTraits>
269 Options MakeOptions(ConnectionOptions<ConnectionTraits> old) {
270  Options opts = std::move(old.opts_);
271  opts.set<UserAgentPrefixOption>({std::move(old.user_agent_prefix_)});
272  opts.set<GrpcBackgroundThreadsFactoryOption>(
273  old.background_threads_factory());
274  if (!old.channel_pool_domain_.empty()) {
275  // To get a different channel pool one just needs to set any channel
276  // parameter to a different value. Newer versions of gRPC include a macro
277  // for this purpose (GRPC_ARG_CHANNEL_POOL_DOMAIN). As we are compiling
278  // against older versions in some cases, we use the actual value.
279  opts.set<GrpcChannelArgumentsOption>(
280  {{"grpc.channel_pooling_domain", std::move(old.channel_pool_domain_)}});
281  }
282  return opts;
283 }
