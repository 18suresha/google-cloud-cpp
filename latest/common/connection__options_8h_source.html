184  grpc::ChannelArguments CreateChannelArguments() const {
185  grpc::ChannelArguments channel_arguments;
186 
187  if (!channel_pool_domain().empty()) {
188  // To get a different channel pool one just needs to set any channel
189  // parameter to a different value. Newer versions of gRPC include a macro
190  // for this purpose (GRPC_ARG_CHANNEL_POOL_DOMAIN). As we are compiling
191  // against older versions in some cases, we use the actual value.
192  channel_arguments.SetString("grpc.channel_pooling_domain",
193  channel_pool_domain());
194  }
195  channel_arguments.SetUserAgentPrefix(user_agent_prefix());
196  return channel_arguments;
