68 int main(int argc, char* argv[]) try {
69  auto const options = ParseArgs(argc, argv);
70 
71  // Create a connection to Cloud Bigtable and an object to manipulate the
72  // specific table used in this demo.
73  cbt::Table table(cbt::CreateDefaultDataClient(
74  options.project_id, options.instance_id,
75  cbt::ClientOptions().set_connection_pool_size(
76  std::thread::hardware_concurrency())),
77  options.table_id);
78  cbt::MutationBatcher batcher(table);
79 
80  // How often do we print a progress marker ('.') in the reader thread.
81  int const report_reader_progress_rate = 500000;
82  // How often do we print a progress marker ('+') in the worker threads.
83  int const report_worker_progress_rate = 500000;
84  // The size of the thread pool pushing data to Cloud Bigtable
85  std::size_t const thread_pool_size = []() -> std::size_t {
86  if (std::thread::hardware_concurrency() != 0) {
87  return std::thread::hardware_concurrency() - 1;
88  }
89  return 1;
90  }();
91 
92  std::cout << "Starting " << thread_pool_size << " workers ..." << std::flush;
93  google::cloud::bigtable::CompletionQueue cq;
94  std::vector<std::thread> thread_pool;
95  for (std::size_t i = 0; i != thread_pool_size; ++i) {
96  thread_pool.emplace_back([&cq] { cq.Run(); });
97  }
98  std::cout << " DONE\n";
99 
100  google::cloud::StatusOr<gcs::ClientOptions> opts =
101  gcs::ClientOptions::CreateDefaultClientOptions();
102  if (!opts) {
103  std::cerr << "Couldn't create gcs::ClientOptions, status=" << opts.status();
104  return 1;
105  }
106  gcs::Client client(opts->set_project_id(options.project_id));
107  // The main thread just reads the file one line at a time.
108  auto is = client.ReadObject(options.bucket, options.object);
109  std::string line;
110  std::getline(is, line, '\n');
111  int lineno = 0;
112  auto headers = ParseLine(++lineno, line, options.separator);
113 
114  std::cout << "# HEADER " << line << "\n";
115 
116  std::cout << "Reading input file " << std::flush;
117  auto start = std::chrono::steady_clock::now();
118 
119  std::atomic<int> apply_finished_count(0);
120  auto report_progress_callback =
121  [=, &apply_finished_count](
122  ::google::cloud::future<::google::cloud::Status> status_future) {
123  if ((apply_finished_count.fetch_add(1) + 1) %
124  report_worker_progress_rate ==
125  0) {
126  std::cout << '+' << std::flush;
127  }
128  auto status = status_future.get();
129  if (!status.ok()) {
130  std::cerr << "Apply failed: " << status;
131  }
132  };
133  while (std::getline(is, line, '\n')) {
134  ++lineno;
135 
136  if (line.empty()) {
137  break;
138  }
139  auto parsed = ParseLine(lineno, line, options.separator);
140 
141  using std::chrono::milliseconds;
142  auto ts = std::chrono::duration_cast<milliseconds>(
143  std::chrono::system_clock::now().time_since_epoch());
144 
145  // Create the row key by concatenating the desired columns.
146  std::string row_key;
147  std::string row_key_separator;
148  for (auto&& index : options.keys) {
149  row_key += row_key_separator;
150  row_key_separator = options.keys_separator;
151  if (index == -1) {
152  // Magic index, use the line number
153  row_key += std::to_string(lineno);
154  } else {
155  row_key += parsed.at(index);
156  }
157  }
158 
159  // Create a mutation that inserts one column per field, the name of the
160  // column is derived from the header.
161  cbt::SingleRowMutation mutation(row_key);
162  auto field_count = std::min(headers.size(), parsed.size());
163  for (std::size_t i = 0; i != field_count; ++i) {
164  mutation.emplace_back(
165  cbt::SetCell(options.family, headers[i], ts, parsed[i]));
166  }
167  auto admission_and_completion = batcher.AsyncApply(cq, std::move(mutation));
168  admission_and_completion.second.then(report_progress_callback);
169 
170  if (lineno % report_reader_progress_rate == 0) {
171  std::cout << '.' << std::flush;
172  }
173 
174  // Wait until there is space in buffers.
175  admission_and_completion.first.get();
176  }
177  batcher.AsyncWaitForNoPendingRequests().get();
178  std::cout << " DONE\n";
179 
180  std::cout << "Waiting for worker threads " << std::flush;
181  // Let the workers know that they can exit.
182  cq.Shutdown();
183  for (auto& thread : thread_pool) {
184  thread.join();
185  }
186  std::cout << " DONE\n";
187 
188  auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(
189  std::chrono::steady_clock::now() - start);
190  std::cout << "Total running time " << elapsed.count() << "s\n";
191 
192  return 0;
193 } catch (std::exception const& ex) {
194  std::cerr << "Standard exception raised: " << ex.what() << "\n";
195  return 1;
196 }

199 std::vector<std::string> ParseLine(long lineno, // NOLINT(google-runtime-int)
200  std::string const& line,
201  char separator) try {
202  std::vector<std::string> result;
203 
204  // Extract the fields one at a time using a std::istringstream.
205  std::istringstream tokens(line);
206  std::string tk;
207  while (std::getline(tokens, tk, separator)) {
208  result.emplace_back(tk);
209  }
210  return result;
211 } catch (std::exception const& ex) {
212  std::ostringstream os;
213  os << ex.what() << " in line #" << lineno << " (" << line << ")";
214  throw std::runtime_error(os.str());
215 }

217 std::string ConsumeArg(Options& options, std::vector<std::string>& argv,
218  char const* arg_name) {
219  std::string const separator_option = "--separator=";
220  std::string const key_option = "--key=";
221  std::string const keys_separator_option = "--key-separator=";
222 
223  std::string const usage = R""(
224 [options] <project> <instance> <table> <family> <bucket> <object>
225 The options are:
226  --help: produce this help.
227  --separator=c: use the 'c' character instead of comma (',') to separate the
228  values in the CSV file.
229  --key=N: use field number N as part of the row key. The fields are numbered
230  starting at one. They are concatenated in the order provided.
231  --key-separator=sep: use 'sep' to separate the fields when forming the row
232  key.
233  project: the Google Cloud Platform project id for your table.
234  instance: the Cloud Bigtable instance hosting your table.
235  table: the table where you want to upload the CSV file.
236  family: the column family where you want to upload the CSV file.
237  bucket: the name of the GCS bucket that contains the data.
238  object: the name of the GCS object that contains the data.
239 )"";
240  while (argv.size() >= 2) {
241  std::string argument(argv[1]);
242  argv.erase(argv.begin() + 1);
243  if (argument == "--help") {
244  } else if (0 == argument.find(separator_option)) {
245  options.separator = argument.substr(separator_option.size())[0];
246  } else if (0 == argument.find(key_option)) {
247  options.keys.push_back(std::stoi(argument.substr(key_option.size())) - 1);
248  } else if (0 == argument.find(keys_separator_option)) {
249  options.keys_separator = argument.substr(keys_separator_option.size());
250  } else {
251  return argument;
252  }
253  }
254  std::string cmd = argv[0];
255  auto last_slash = std::string(cmd).find_last_of('/');
256  cmd = cmd.substr(last_slash);
257 
258  std::ostringstream os;
259  os << "Missing argument " << arg_name << "\n";
260  os << "Usage: " << cmd << usage << "\n";
261  throw std::runtime_error(os.str());
262 }

264 Options ParseArgsNoAutoRun(int argc, char const* const argv[]) {
265  // Parse the command-line arguments.
266  Options options;
267  std::vector<std::string> args(argv, argv + argc);
268  options.project_id = ConsumeArg(options, args, "project_id");
269  options.instance_id = ConsumeArg(options, args, "instance_id");
270  options.table_id = ConsumeArg(options, args, "table_id");
271  options.family = ConsumeArg(options, args, "family");
272  options.bucket = ConsumeArg(options, args, "bucket");
273  options.object = ConsumeArg(options, args, "object");
274  // If the user does not say, use the first column as the row key.
275  if (options.keys.empty()) {
276  options.keys.push_back(0);
277  }
278  return options;
279 }

283 Options AutoRun() {
284  using google::cloud::internal::GetEnv;
285  using google::cloud::internal::Sample;

287  for (auto const& var :

292  std::ostringstream os;

294  throw std::runtime_error(std::move(os).str());
295  }
296  auto const project_id = GetEnv("GOOGLE_CLOUD_PROJECT").value();
297  auto const instance_id =

299  auto const bucket_name =
300  GetEnv("GOOGLE_CLOUD_CPP_STORAGE_TEST_BUCKET_NAME").value();
301  auto const table_id = "gcs2cbt-auto-run";
302  auto const object_name = "gcs2cbt-sample-data.csv";
303  auto gcs_client = gcs::Client::CreateDefaultClient().value();
304  auto const test_data = R"""(RowId,Header1,Header2,Header3
305 1,v1,v2,v3
306 3,v1,v2,v3
307 )""";
308  gcs_client.InsertObject(bucket_name, object_name, test_data).value();
309 
310  cbt::TableAdmin admin(
311  cbt::CreateDefaultAdminClient(project_id, cbt::ClientOptions{}),
312  instance_id);
313  auto schema = admin.CreateTable(
314  table_id,
315  cbt::TableConfig({{"fam", cbt::GcRule::MaxNumVersions(2)}}, {}));
316  // Throw the error unless it is "already exists"
317  if (!schema &&
318  schema.status().code() != google::cloud::StatusCode::kAlreadyExists) {
319  (void)schema.value();
320  }
321 
322  char const* argv[] = {
323  "auto-run", "--key=1", "--separator=,", project_id.c_str(),
324  instance_id.c_str(), table_id, "fam", bucket_name.c_str(),
325  object_name};
326  int argc = sizeof(argv) / sizeof(argv[0]);
327  return ParseArgsNoAutoRun(argc, argv);
328 }
