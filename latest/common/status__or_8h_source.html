26 /**
27  * `StatusOr<T>` is used to return a value or an error status.
28  *
29  * If the library is compiled with exceptions disabled there is no standard C++
30  * mechanism to report errors. In that case we use this class to wrap the values
31  * returned to the application.
32  *
33  * The application typically calls an API in the library and must check the
34  * returned error status:
35  *
36  * @code
37  * namespace gcs = google::cloud::storage;
38  * void AppCode(gcs::noex::Client client) {
39  * gcs::StatusOr<gcs::BucketMetadata> meta_err = client.GetBucketMetadata(
40  * "my-bucket-name");
41  * if (!meta_err) {
42  * std::cerr << "Error in GetBucketMetadata: " << meta_err.status()
43  * << std::endl;
44  * return;
45  * }
46  * gcs::BucketMetadata meta = *meta_err;
47  * // Do useful work here.
48  * }
49  * @endcode
50  *
51  * Note that the storage client retries most requests for you, resending the
52  * request after an error is probably not useful. You should consider changing
53  * the retry policies instead.
54  *
55  * TODO(...) - the current implementation is fairly naive with respect to `T`,
56  * it is unlikely to work correctly for reference types, types without default
57  * constructors, arrays.
58  *
59  * @tparam T the type of the value.
60  */
61 template <typename T>
62 class StatusOr final {
63  public:
64  /**
65  * Initializes with an error status (UNKNOWN).
66  *
67  * TODO(#548) - currently storage::Status does not define the status codes,
68  * they are simply integers, usually HTTP status codes. We need to map to
69  * the well-defined set of status codes.
70  */
71  StatusOr() : StatusOr(Status(StatusCode::kUnknown, "default")) {}
72 
73  /**
74  * Creates a new `StatusOr<T>` holding the error condition @p rhs.
75  *
76  * @par Post-conditions
77  * `ok() == false` and `status() == rhs`.
78  *
79  * @param rhs the status to initialize the object.
80  * @throws std::invalid_argument if `rhs.ok()`. If exceptions are disabled the
81  * program terminates via `google::cloud::Terminate()`
82  */
83  // NOLINTNEXTLINE(google-explicit-constructor)
84  StatusOr(Status rhs) : status_(std::move(rhs)) {
85  if (status_.ok()) {
86  google::cloud::internal::ThrowInvalidArgument(__func__);
87  }
88  }
89 
90  StatusOr(StatusOr&& rhs) : status_(std::move(rhs.status_)) {
91  if (status_.ok()) {
92  new (&value_) T(std::move(*rhs));
93  }
94  }

96  StatusOr& operator=(StatusOr&& rhs) {
97  // There may be shorter ways to express this, but this is fairly readable,
98  // and should be reasonably efficient. Note that we must avoid destructing
99  // the destination and/or default initializing it unless really needed.
100  if (!ok()) {
101  if (!rhs.ok()) {
102  status_ = std::move(rhs.status_);
103  return *this;
104  }
105  new (&value_) T(std::move(*rhs));
106  status_ = Status();
107  return *this;
108  }
109  if (!rhs.ok()) {
110  value_.~T();
111  status_ = std::move(rhs.status_);
112  return *this;
113  }
114  **this = *std::move(rhs);
115  status_ = Status();
116  return *this;
117  }
118 
119  StatusOr(StatusOr const& rhs) : status_(rhs.status_) {
120  if (status_.ok()) {
121  new (&value_) T(*rhs);
122  }
123  }
124 
125  StatusOr& operator=(StatusOr const& rhs) {
126  // There may be shorter ways to express this, but this is fairly readable,
127  // and should be reasonably efficient. Note that we must avoid destructing
128  // the destination and/or default initializing it unless really needed.
129  if (!ok()) {
130  if (!rhs.ok()) {
131  status_ = rhs.status_;
132  return *this;
133  }
134  new (&value_) T(*rhs);
135  status_ = rhs.status_;
136  return *this;
137  }
138  if (!rhs.ok()) {
139  value_.~T();
140  status_ = rhs.status_;
141  return *this;
142  }
143  **this = *rhs;
144  status_ = rhs.status_;
145  return *this;
146  }
147 
148  ~StatusOr() {
149  if (ok()) {
150  value_.~T();
151  }
152  }

154  /**
155  * Assign a `T` (or anything convertible to `T`) into the `StatusOr`.
156  */
157  // Disable this assignment if U==StatusOr<T>. Well, really if U is a
158  // cv-qualified version of StatusOr<T>, so we need to apply std::decay<> to
159  // it first.
160  template <typename U = T>
161  typename std::enable_if<
162  not std::is_same<StatusOr, typename std::decay<U>::type>::value,
163  StatusOr>::type&
164  operator=(U&& rhs) {
165  // There may be shorter ways to express this, but this is fairly readable,
166  // and should be reasonably efficient. Note that we must avoid destructing
167  // the destination and/or default initializing it unless really needed.
168  if (!ok()) {
169  new (&value_) T(std::forward<U>(rhs));
170  status_ = Status();
171  return *this;
172  }
173  **this = std::forward<U>(rhs);
174  status_ = Status();
175  return *this;
176  }
177 
178  /**
179  * Creates a new `StatusOr<T>` holding the value @p rhs.
180  *
181  * @par Post-conditions
182  * `ok() == true` and `value() == rhs`.
183  *
184  * @param rhs the value used to initialize the object.
185  *
186  * @throws only if `T`'s move constructor throws.
187  */
188  // NOLINTNEXTLINE(google-explicit-constructor)
189  StatusOr(T&& rhs) : status_() { new (&value_) T(std::move(rhs)); }
190 
191  // NOLINTNEXTLINE(google-explicit-constructor)
192  StatusOr(T const& rhs) : status_() { new (&value_) T(rhs); }
193 
194  bool ok() const { return status_.ok(); }
195  explicit operator bool() const { return status_.ok(); }
196 
197  //@{
198  /**
199  * @name Deference operators.
200  *
201  * @warning Using these operators when `ok() == false` results in undefined
202  * behavior.
203  *
204  * @return All these return a (properly ref and const-qualified) reference to
205  * the underlying value.
206  */
207  T& operator*() & { return value_; }
208 
209  T const& operator*() const& { return value_; }
210 
211  T&& operator*() && { return std::move(value_); }
212 
213 #if GOOGLE_CLOUD_CPP_HAVE_CONST_REF_REF
214  T const&& operator*() const&& { return std::move(value_); }
215 #endif // GOOGLE_CLOUD_CPP_HAVE_CONST_REF_REF
216  //@}
217 
218  //@{
219  /**
220  * @name Member access operators.
221  *
222  * @warning Using these operators when `ok() == false` results in undefined
223  * behavior.
224  *
225  * @return All these return a (properly ref and const-qualified) pointer to
226  * the underlying value.
227  */
228  T* operator->() & { return &value_; }
229 
230  T const* operator->() const& { return &value_; }
231  //@}
232 
233  //@{
234  /**
235  * @name Value accessors.
236  *
237  * @return All these member functions return a (properly ref and
238  * const-qualified) reference to the underlying value.
239  *
240  * @throws `RuntimeStatusError` with the contents of `status()` if the object
241  * does not contain a value, i.e., if `ok() == false`.
242  */
243  T& value() & {
244  CheckHasValue();
245  return **this;
246  }
247 
248  T const& value() const& {
249  CheckHasValue();
250  return **this;
251  }
252 
253  T&& value() && {
254  CheckHasValue();
255  return std::move(**this);
256  }
257 
258 #if GOOGLE_CLOUD_CPP_HAVE_CONST_REF_REF
259  T const&& value() const&& {
260  CheckHasValue();
261  return std::move(**this);
262  }
263 #endif // GOOGLE_CLOUD_CPP_HAVE_CONST_REF_REF
264  //@}
265 
266  //@{
267  /**
268  * @name Status accessors.
269  *
270  * @return All these member functions return the (properly ref and
271  * const-qualified) status. If the object contains a value then
272  * `status().ok() == true`.
273  */
274  Status& status() & { return status_; }
275  Status const& status() const& { return status_; }
276  Status&& status() && { return std::move(status_); }
277 #if GOOGLE_CLOUD_CPP_HAVE_CONST_REF_REF
278  Status const&& status() const&& { return std::move(status_); }
279 #endif // GOOGLE_CLOUD_CPP_HAVE_CONST_REF_REF
280  //@}

282  private:
283  void CheckHasValue() const& {
284  if (!ok()) {
285  internal::ThrowStatus(status_);
286  }
287  }
288 
289  // When possible, do not copy the status.
290  void CheckHasValue() && {
291  if (!ok()) {
292  internal::ThrowStatus(std::move(status_));
293  }
294  }

296  Status status_;
297  union {
298  T value_;
299  };

302 /**
303  * `StatusOr<void>` is used to return an error or nothing.
304  *
305  * `StatusOr<T>` does not work for `T = void` because some of the member
306  * functions (`StatusOr<T>(T)`) make no sense for `void`. Likewise, the class
307  * cannot contain an object of type `void` because there is no such thing.
308  *
309  * The application typically calls an API in the library and must check the
310  * returned error status:
311  *
312  * @code
313  * namespace gcs = google::cloud::storage;

315  * gcs::StatusOr<void> delete_err = client.DeleteBucket("my-bucket-name");
316  * if (!delete_err.ok()) {
317  * std::cerr << "Error in DeleteBucket: " << meta_err.status()
318  * << std::endl;
319  * return;
320  * }
321  * // Do useful work here.
322  * }
323  * @endcode
324  *
325  * Note that the storage client retries most requests for you, resending the
326  * request after an error is probably not useful. You should consider changing
327  * the retry policies instead.
