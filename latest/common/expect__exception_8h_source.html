26 /**
27  * Verify that a given functor raises `T` and run a test if it does.
28  *
29  * `EXPECT_THROW()` validates the type of an exception, but does not validate
30  * its contents. Most of the time the type is enough, but in some tests need to
31  * be more detailed. The common idiom in that case is to write an explicit
32  * try/catch to validate the values and then rethrow:
33  *
34  * @code
35  * EXPECT_THROW(
36  * try { something_that_throws(); }
37  * catch(ExpectedException const& ex) {
38  * // Validate the contents of ex, e.g. EXPECT_EQ(..., ex.what());
39  * throw;
40  * }, ExpectedException);
41  * @endcode
42  *
43  * That works, but we need to compile (and run) the tests with
44  * `-fno-exceptions`. In that case the test becomes:
45  *
46  * @code
47  * EXPECT_DEATH_IF_SUPPORTED(something_that_throws(), "expected death message");
48  * @endcode
49  *
50  * And, of course, these two cases need a `#%ifdef` to detect if exceptions are
51  * enabled. This was getting tedious, so we refactor to this function. The
52  * expected use is:
53  *
54  * @code
55  * ExpectException<MyException>(
56  * [&] { something_that_throws(); },
57  * [&](MyException const& ex) {
58  * EXPECT_EQ("everything is terrible", ex.what());
59  * },
60  * "terminating program: everything is terrible");
61  * @endcode
62  *
63  *
64  * @param expression the expression (typically a lambda) that is expected to
65  * raise an exception of type `ExpectedException`.
66  * @param validator a functor to validate the exception contents.
67  * @param expected_message if exceptions are disabled the program is expected
68  * to terminate, and @p expected_message is a regular expression matched
69  * against the program output.
70  * @tparam ExpectedException the type of exception.
71  */
72 template <typename ExpectedException>
73 void ExpectException(
74  std::function<void()> const& expression,
75  std::function<void(ExpectedException const& ex)> const& validator,
76  char const* expected_message) {
77 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
78  EXPECT_THROW(try { expression(); } catch (ExpectedException const& ex) {
79  validator(ex);
80  throw;
81  },
82  ExpectedException);
83  (void)expected_message; // suppress clang-tidy warning.
84 #else
85  (void)validator; // suppress clang-tidy warning.
86  EXPECT_DEATH_IF_SUPPORTED(expression(), expected_message);
87 #endif // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
88 }

90 /**
91  * Verify that an expression does not throw.
92  *
93  * Testing `void` expressions is tedious because `EXPECT_NO_THROW` does not
94  * compile when exceptions are disabled. Writing `expression()` in a test does
95  * detect exceptions, but does not express the intent.
96  *
97  * @param expression the expression (typically a lambda) that we want to verify
98  * does not throw.
99  */
100 inline void ExpectNoException(std::function<void()> const& expression) {
101 #if GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
102  EXPECT_NO_THROW(expression());
103 #else
104  EXPECT_NO_FATAL_FAILURE(expression());
105 #endif // GOOGLE_CLOUD_CPP_HAVE_EXCEPTIONS
106 }
