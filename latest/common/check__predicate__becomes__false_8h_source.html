27 /**
28  * Verify that an assertion becomes false after some prescribed time.
29  *
30  * Test that an assertion is initially true, but eventually becomes false after
31  * some prescribed time. Use tolerance to avoid creating flaky tests.
32  */
33 template <typename Predicate>
34 void CheckPredicateBecomesFalse(Predicate&& predicate,
35  std::chrono::system_clock::time_point deadline,
36  std::chrono::milliseconds tolerance) {
37  // This is one of those tests that can get annoyingly flaky, it is based on
38  // time. Basically we want to know that the policy will accept failures
39  // until around its prescribed deadline (50ms in this test). Instead of
40  // measuring for *exactly* 50ms, we pass the test if:
41  // - All calls to OnFailure() in the first 50ms - 10ms pass.
42  // - Calls to OnFailure() after 50ms + 10ms are rejected.
43  // - We do not care about the results from 40ms to 60ms.
44  // - We must handle the case where the test is called before the 40ms mark
45  // and finishes after the 60ms mark. Yes, that can happen on heavily
46  // loaded machines, which CI servers often are.
47  // I know 10ms feels like a long time, but it is not on a loaded VM running
48  // the tests inside some container.
49  auto must_be_true_before = deadline - tolerance;
50  auto must_be_false_after = deadline + tolerance;
51  int true_count = 0;
52  int false_count = 0;
53  auto duration = deadline - std::chrono::system_clock::now();
54  while (std::chrono::system_clock::now() < deadline + duration) {
55  auto iteration_start = std::chrono::system_clock::now();
56  auto actual = predicate();
57  auto iteration_end = std::chrono::system_clock::now();
58  if (iteration_end < must_be_true_before) {
59  EXPECT_TRUE(actual);
60  if (actual) {
61  ++true_count;
62  }
63  } else if (must_be_false_after < iteration_start) {
64  EXPECT_FALSE(actual);
65  if (!actual) {
66  // Terminate the loop early if we can.
67  ++false_count;
68  break;
69  }
70  }
71  std::this_thread::sleep_for(std::chrono::milliseconds(tolerance / 2));
72  }
73  EXPECT_LE(0, true_count);
74  EXPECT_LE(0, false_count);
75 }
