36 char const kDescription[] = R"""(
37 A throughput vs. CPU benchmark for the Google Cloud Storage C++ client library.
38 
39 This program measures the throughput and CPU utilization when uploading
40 and downloading objects using the Google Cloud Storage (GCS) C++ client library.
41 The program repeats the "experiment" of uploading, then downloading, and then
42 removing an object many times, using a randomly selected size in each iteration.
43 An external script presents these results as a series of plots.
44 
45 The program first creates a GCS bucket that will contain all the objects used
46 by that run of the program. The name of this bucket is selected at random, so
47 multiple copies of the program can run simultaneously. The bucket is deleted at
48 the end of the run of this program. The bucket uses the `STANDARD` storage
49 class, in a region set via the command line. Choosing regions close to where the
50 program is running can be used to estimate the latency without any wide-area
51 network effects. Choosing regions far from where the program is running can be
52 used to evaluate the performance of the library when the wide-area network is
53 taken into account.
54 
55 After creating this bucket the program creates a number of threads, configurable
56 via the command line, to obtain more samples in parallel. Configure this value
57 with a small enough number of threads such that you do not saturate the CPU.
58 
59 Each thread creates C++ objects to perform the "upload experiments". Each one
60 of these objects represents the "api" used to perform the upload, that is XML,
61 JSON and/or gRPC (though technically gRPC is just another protocol for the JSON
62 API). Likewise, the thread creates a number of "download experiments", also
63 based on the APIs configured via the command-line.
64 
65 Then the thread repeats the following steps (see below for the conditions to
66 stop the loop):
67 
68 - Select a random size, between two values configured in the command line of the
69  object to upload.
70 - The application buffer sizes for `read()` and `write()` calls are also
71  selected at random. These sizes are quantized, and the quantum can be
72  configured in the command-line.
73 - Select a random uploader, that is, which API will be used to upload the
74  object.
75 - Select a random downloader, that is, which API will be used to download the
76  object.
77 - Select, at random, if the client library will perform CRC32C and/or MD5 hashes
78  on the uploaded and downloaded data.
79 - Upload an object of the selected size, choosing the name of the object at
80  random.
81 - Once the object is fully uploaded, the program captures the object size, the
82  write buffer size, the elapsed time (in microseconds), the CPU time
83  (in microseconds) used during the upload, and the status code for the upload.
84 - Then the program downloads the same object (3 times), and captures the object
85  size, the read buffer size, the elapsed time (in microseconds), the CPU time
86  (in microseconds) used during the download, and the status code for the
87  download.
88 - The program then deletes this object and starts another iteration.
89 
90 The loop stops when any of the following conditions are met:
91 
92 - The test has obtained more than a prescribed "maximum number of samples"
93 - The test has obtained at least a prescribed "minimum number of samples" *and*
94  the test has been running for more than a prescribed "duration".
95 
96 Once the threads finish running their loops the program prints the captured
97 performance data. The bucket is deleted after the program terminates.
98 
99 A helper script in this directory can generate pretty graphs from the output of
100 this program.
101 )""";

234 TestResults RunThread(ThroughputOptions const& options,
235  std::string const& bucket_name, int thread_id) {
236  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
237  google::cloud::StatusOr<gcs::ClientOptions> client_options =
238  gcs::ClientOptions::CreateDefaultClientOptions();
239  if (!client_options) {
240  std::cout << "# Could not create ClientOptions, status="
241  << client_options.status() << "\n";
242  return {};
243  }
244  auto const upload_buffer_size = client_options->upload_buffer_size();
245  auto const download_buffer_size = client_options->download_buffer_size();
246 
247  gcs::Client rest_client(*client_options);
248 
249  auto uploaders =
250  gcs_bm::CreateUploadExperiments(options, *client_options, thread_id);
251  if (uploaders.empty()) {
252  // This is possible if only gRPC is requested but the benchmark was compiled
253  // without gRPC support.
254  std::cout << "# None of the APIs configured are available\n";
255  return {};
256  }
257  auto downloaders =
258  gcs_bm::CreateDownloadExperiments(options, *client_options, thread_id);
259  if (downloaders.empty()) {
260  // This is possible if only gRPC is requested but the benchmark was compiled
261  // without gRPC support.
262  std::cout << "# None of the APIs configured are available\n";
263  return {};
264  }
265 
266  std::uniform_int_distribution<std::size_t> uploader_generator(
267  0, uploaders.size() - 1);
268  std::uniform_int_distribution<std::size_t> downloader_generator(
269  0, downloaders.size() - 1);
270 
271  std::uniform_int_distribution<std::int64_t> size_generator(
272  options.minimum_object_size, options.maximum_object_size);
273  std::uniform_int_distribution<std::size_t> write_size_generator(
274  options.minimum_write_size / options.write_quantum,
275  options.maximum_write_size / options.write_quantum);
276  std::uniform_int_distribution<std::size_t> read_size_generator(
277  options.minimum_read_size / options.read_quantum,
278  options.maximum_read_size / options.read_quantum);
279 
280  std::uniform_int_distribution<std::size_t> crc32c_generator(
281  0, options.enabled_crc32c.size() - 1);
282  std::uniform_int_distribution<std::size_t> md5_generator(
283  0, options.enabled_crc32c.size() - 1);
284  std::bernoulli_distribution use_insert;
285 
286  std::uniform_int_distribution<std::size_t> api_generator(
287  0, options.enabled_apis.size() - 1);
288 
289  auto deadline = std::chrono::steady_clock::now() + options.duration;
290 
291  gcs_bm::Timer timer;
292  TestResults results;
293 
294  std::int32_t iteration_count = 0;
295  for (auto start = std::chrono::steady_clock::now();
296  iteration_count < options.maximum_sample_count &&
297  (iteration_count < options.minimum_sample_count || start < deadline);
298  start = std::chrono::steady_clock::now(), ++iteration_count) {
299  auto object_name = gcs_bm::MakeRandomObjectName(generator);
300  auto object_size = size_generator(generator);
301  auto write_size = options.write_quantum * write_size_generator(generator);
302  auto read_size = options.read_quantum * read_size_generator(generator);
303  bool const enable_crc = options.enabled_crc32c[crc32c_generator(generator)];
304  bool const enable_md5 = options.enabled_md5[md5_generator(generator)];
305  auto const api = options.enabled_apis[api_generator(generator)];
306 
307  auto& uploader = uploaders[uploader_generator(generator)];
308  auto upload_result =
309  uploader->Run(bucket_name, object_name,
310  gcs_bm::ThroughputExperimentConfig{
311  gcs_bm::kOpWrite, object_size, write_size,
312  upload_buffer_size, enable_crc, enable_md5});
313  auto status = upload_result.status;
314  results.emplace_back(std::move(upload_result));
315 
316  if (!status.ok()) {
317  if (options.thread_count == 1) {
318  std::cout << "# status=" << status << ", api=" << gcs_bm::ToString(api)
319  << "\n";
320  }
321  continue;
322  }
323 
324  auto& downloader = downloaders[downloader_generator(generator)];
325  for (auto op : {gcs_bm::kOpRead0, gcs_bm::kOpRead1, gcs_bm::kOpRead2}) {
326  results.emplace_back(downloader->Run(
327  bucket_name, object_name,
328  gcs_bm::ThroughputExperimentConfig{op, object_size, read_size,
329  download_buffer_size, enable_crc,
330  enable_md5}));
331  }
332  if (options.thread_count == 1) {
333  // Immediately print the results, this makes it easier to debug problems.
334  PrintResults(results);
335  results.clear();
336  }
337 
338  (void)rest_client.DeleteObject(bucket_name, object_name);
339  }
340  return results;
341 }
