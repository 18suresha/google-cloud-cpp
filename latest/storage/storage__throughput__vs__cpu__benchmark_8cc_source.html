29 char const kDescription[] = R"""(
30 A throughput vs. CPU benchmark for the Google Cloud Storage C++ client library.
31 
32 This program measures the throughput and CPU utilization when uploading
33 and downloading relatively large (~250 MiB) objects Google Cloud Storage C++
34 client library. The program repeats the "experiment" of uploading, then
35 downloading, and then removing a file many times, using a randomly selected
36 size in each iteration. An external script performs statistical analysis on
37 the results to estimate likely values for the throughput and the CPU cost per
38 byte on both upload and download operations.
39 
40 The program first creates a GCS bucket that will contain all the objects used
41 by that run of the program. The name of this bucket is selected at random, so
42 multiple copies of the program can run simultaneously. The bucket is deleted at
43 the end of the run of this program. The bucket uses the `REGIONAL` storage
44 class, in a region set via the command line. Choosing regions close to where the
45 program is running can be used to estimate the latency without any wide-area
46 network effects. Choosing regions far from where the program is running can be
47 used to evaluate the performance of the library when the WAN is taken into
48 account.
49 
50 After creating this bucket the program creates a number of threads, configurable
51 via the command line, to obtain more samples in parallel. Configure this value
52 with a small enough number of threads such that you do not saturate the CPU.
53 Each thread creates a separate copy of the `storage::Client` object, and repeats
54 this loop until a prescribed *time* has elapsed:
55 
56 - Select a random size, between two values configured in the command line of the
57  object to upload.
58 - Select a random chunk size, between two values configured in the command line,
59  the data is uploaded in chunks of this size.
60 - Upload an object of the selected size, choosing the name of the object at
61  random.
62 - Once the object is fully uploaded, the program captures the object size, the
63  chunk size, the elapsed time (in microseconds), the CPU time (in microseconds)
64  used during the upload, and the status code for the upload.
65 - Then the program downloads the same object, and captures the object size, the
66  chunk size, the elapsed time (in microseconds), the CPU time (in microseconds)
67  used during the download, and the status code for the download.
68 - The program then deletes this object and starts another iteration.
69 
70 The loop runs for a prescribed number of seconds. Once the threads finish
71 running their loops the program prints the captured performance data. The bucket
72 is deleted after the program terminates.
73 
74 A helper script in this directory can generate pretty graphs from the output of
75 this program.
76 )""";

234 TestResults RunThread(Options const& options, std::string const& bucket_name) {
235  google::cloud::internal::DefaultPRNG generator =
236  google::cloud::internal::MakeDefaultPRNG();
237  auto contents =
238  gcs_bm::MakeRandomData(generator, options.maximum_object_size);
239  google::cloud::StatusOr<gcs::ClientOptions> client_options =
240  gcs::ClientOptions::CreateDefaultClientOptions();
241  if (!client_options) {
242  std::cout << "# Could not create ClientOptions, status="
243  << client_options.status() << "\n";
244  return {};
245  }
246  std::uint64_t upload_buffer_size = client_options->upload_buffer_size();
247  std::uint64_t download_buffer_size = client_options->download_buffer_size();
248  gcs::Client client(*std::move(client_options));
249 
250  std::uniform_int_distribution<std::uint64_t> size_generator(
251  options.minimum_object_size, options.maximum_object_size);
252  std::uniform_int_distribution<std::uint64_t> chunk_generator(
253  options.minimum_chunk_size, options.maximum_chunk_size);
254 
255  auto deadline = std::chrono::steady_clock::now() + options.duration;
256 
257  gcs_bm::SimpleTimer timer;
258  // This obviously depends on the size of the objects, but a good estimate for
259  // the upload + download bandwidth is 250MiB/s.
260  constexpr long expected_bandwidth = 250 * gcs_bm::kMiB;
261  auto const median_size =
262  (options.minimum_object_size + options.minimum_object_size) / 2;
263  auto const objects_per_second = median_size / expected_bandwidth;
264  TestResults results;
265  results.reserve(options.duration.count() * objects_per_second);
266 
267  for (auto start = std::chrono::steady_clock::now(); start < deadline;
268  start = std::chrono::steady_clock::now()) {
269  auto object_name = gcs_bm::MakeRandomObjectName(generator);
270  auto object_size = size_generator(generator);
271  auto chunk_size = chunk_generator(generator);
272 
273  timer.Start();
274  auto writer =
275  client.WriteObject(bucket_name, object_name,
276  gcs::DisableCrc32cChecksum(options.disable_crc32c),
277  gcs::DisableMD5Hash(options.disable_md5));
278  for (std::size_t offset = 0; offset < object_size; offset += chunk_size) {
279  auto len = chunk_size;
280  if (offset + len > object_size) {
281  len = object_size - offset;
282  }
283  writer.write(contents.data() + offset, len);
284  }
285  writer.Close();
286  timer.Stop();
287 
288  auto object_metadata = writer.metadata();
289  results.emplace_back(IterationResult{OP_UPLOAD, object_size, chunk_size,
290  download_buffer_size,
291  timer.elapsed_time(), timer.cpu_time(),
292  object_metadata.status().code()});
293 
294  if (!object_metadata) {
295  continue;
296  }
297 
298  timer.Start();
299  auto reader =
300  client.ReadObject(object_metadata->bucket(), object_metadata->name(),
301  gcs::Generation(object_metadata->generation()),
302  gcs::DisableCrc32cChecksum(options.disable_crc32c),
303  gcs::DisableMD5Hash(options.disable_md5));
304  std::vector<char> buffer(chunk_size);
305  while (reader.read(buffer.data(), buffer.size())) {
306  }
307  timer.Stop();
308  results.emplace_back(IterationResult{
309  OP_DOWNLOAD, object_size, chunk_size, upload_buffer_size,
310  timer.elapsed_time(), timer.cpu_time(), reader.status().code()});
311 
312  auto status =
313  client.DeleteObject(object_metadata->bucket(), object_metadata->name(),
314  gcs::Generation(object_metadata->generation()));
315 
316  if (options.thread_count == 1) {
317  // Immediately print the results, this makes it easier to debug problems.
318  PrintResults(results);
319  results.clear();
320  }
321  }
322  return results;
323 }

329  std::vector<gcs_bm::OptionDescriptor> desc{

338  {"--duration", "set the total execution time for the benchmark",

342  {"--thread-count", "set the number of threads in the benchmark",
343  [&options](std::string const& val) {
344  options.thread_count = std::stoi(val);
345  }},
346  {"--minimum-object-size", "configure the minimum object size in the test",
347  [&options](std::string const& val) {
348  options.minimum_object_size = gcs_bm::ParseSize(val);
349  }},
350  {"--maximum-object-size", "configure the maximum object size in the test",
351  [&options](std::string const& val) {
352  options.maximum_object_size = gcs_bm::ParseSize(val);
353  }},
354  {"--minimum-chunk-size", "configure the minimum chunk size in the test",
355  [&options](std::string const& val) {
356  options.minimum_chunk_size = gcs_bm::ParseSize(val);
357  }},
358  {"--maximum-chunk-size", "configure the maximum chunk size in the test",
359  [&options](std::string const& val) {
360  options.maximum_chunk_size = gcs_bm::ParseSize(val);
361  }},
362  {"--disable-crc32", "disable CRC32C checksums",
363  [&options](std::string const& val) {
364  options.disable_crc32c = gcs_bm::ParseBoolean(val, true);
365  }},
366  {"--disable-md5", "disable MD5 hashes",
367  [&options](std::string const& val) {
368  options.disable_md5 = gcs_bm::ParseBoolean(val, true);
369  }},
370  };
371  auto usage = gcs_bm::BuildUsage(desc, argv[0]);
372 
373  auto unparsed = gcs_bm::OptionsParse(desc, {argv, argv + argc});
374  if (wants_help) {
375  std::cout << usage << "\n";
376  }
377 
378  if (wants_description) {
379  std::cout << kDescription << "\n";
380  }
381 
382  if (unparsed.size() > 2) {
383  std::ostringstream os;
384  os << "Unknown arguments or options\n" << usage << "\n";
385  throw std::runtime_error(std::move(os).str());
386  }
387  if (unparsed.size() == 2) {
388  options.region = unparsed[1];
389  }
390  if (options.region.empty()) {
391  std::ostringstream os;
392  os << "Missing value for --region option" << usage << "\n";
393  throw std::runtime_error(std::move(os).str());
394  }
395 
396  if (options.minimum_object_size > options.maximum_object_size) {
397  std::ostringstream os;
398  os << "Invalid range for objects [" << options.minimum_object_size << ','
399  << options.maximum_object_size << "]";
400  throw std::runtime_error(std::move(os).str());
401  }
402  if (options.minimum_chunk_size > options.maximum_chunk_size) {
403  std::ostringstream os;
404  os << "Invalid range for chunks [" << options.minimum_chunk_size << ','

406  throw std::runtime_error(std::move(os).str());
407  }
408 
409  if (!gcs_bm::SimpleTimer::SupportPerThreadUsage() &&

415  }
416 
417  return options;
418 }
