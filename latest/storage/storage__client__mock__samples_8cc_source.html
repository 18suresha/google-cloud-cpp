163 int main(int argc, char* argv[]) try {
164  using CommandType = std::function<void(int&, char*[])>;
165  std::map<std::string, CommandType> commands = {
166  {"mock-read-object", &MockReadObject},
167  {"mock-write-object", &MockWriteObject},
168  };
169  for (auto&& kv : commands) {
170  try {
171  int fake_argc = 0;
172  kv.second(fake_argc, argv);
173  } catch (Usage const& u) {
174  command_usage += " ";
175  command_usage += u.msg;
176  command_usage += "\n";
177  }
178  }
179 
180  if (argc < 2) {
181  PrintUsage(argc, argv, "Missing command");
182  return 1;
183  }
184 
185  std::string const command = ConsumeArg(argc, argv);
186  auto it = commands.find(command);
187  if (commands.end() == it) {
188  PrintUsage(argc, argv, "Unknown command: " + command);
189  return 1;
190  }

192  it->second(argc, argv);
193 
194  return 0;
195 } catch (Usage const& ex) {

198 } catch (std::exception const& ex) {
