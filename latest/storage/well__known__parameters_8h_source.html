27 /**
28  * Refactor definition of well-known query parameters using the CRTP.
29  *
30  * @tparam P the type we will use to represent the query parameter.
31  * @tparam T the C++ type of the query parameter
32  */
33 template <typename P, typename T>
34 class WellKnownParameter {
35  public:
36  WellKnownParameter() : value_{} {}
37  explicit WellKnownParameter(T&& value) : value_(std::forward<T>(value)) {}
38 
39  char const* parameter_name() const { return P::well_known_parameter_name(); }
40  bool has_value() const { return value_.has_value(); }
41  T const& value() const { return value_.value(); }
42 
43  private:
44  google::cloud::internal::optional<T> value_;
45 };
46 
47 template <typename P, typename T>
48 std::ostream& operator<<(std::ostream& os,
49  WellKnownParameter<P, T> const& rhs) {
50  if (rhs.has_value()) {
51  return os << rhs.parameter_name() << "=" << rhs.value();
52  }
53  return os << rhs.parameter_name() << "=<not set>";
54 }
