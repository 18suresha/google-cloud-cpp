30 /**
31  * Defines well-known request headers using the CRTP.
32  *
33  * @tparam P the type we will use to represent the query parameter.
34  * @tparam T the C++ type of the query parameter
35  */
36 template <typename P, typename T>
37 class WellKnownParameter {
38  public:
39  WellKnownParameter() : value_{} {}
40  explicit WellKnownParameter(T&& value) : value_(std::forward<T>(value)) {}
41  explicit WellKnownParameter(T const& value) : value_(value) {}
42 
43  char const* parameter_name() const { return P::well_known_parameter_name(); }
44  bool has_value() const { return value_.has_value(); }
45  T const& value() const { return value_.value(); }
46 
47  private:
48  google::cloud::optional<T> value_;
49 };
50 
51 template <typename P, typename T>
52 std::ostream& operator<<(std::ostream& os,
53  WellKnownParameter<P, T> const& rhs) {
54  if (rhs.has_value()) {
55  return os << rhs.parameter_name() << "=" << rhs.value();
56  }
57  return os << rhs.parameter_name() << "=<not set>";
58 }
59 
60 template <typename P>
61 std::ostream& operator<<(std::ostream& os,
62  WellKnownParameter<P, bool> const& rhs) {
63  if (rhs.has_value()) {
64  google::cloud::internal::IosFlagsSaver saver(os);
65  return os << rhs.parameter_name() << "=" << std::boolalpha << rhs.value();
66  }
67  return os << rhs.parameter_name() << "=<not set>";
68 }
