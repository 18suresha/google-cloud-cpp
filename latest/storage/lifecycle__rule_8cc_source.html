95 void LifecycleRule::MergeConditions(LifecycleRuleCondition& result,
96  LifecycleRuleCondition const& rhs) {

101  auto tmp = *rhs.age;

103  }
104  }
105  if (rhs.created_before.has_value()) {
106  if (result.created_before.has_value()) {
107  *result.created_before =
108  std::max(*result.created_before, *rhs.created_before);
109  } else {
110  auto tmp = *rhs.created_before;
111  result.created_before.emplace(
112  std::forward<std::chrono::system_clock::time_point>(tmp));
113  }
114  }
115  if (rhs.is_live.has_value()) {
116  if (result.is_live.has_value()) {
117  if (result.is_live.value() != rhs.is_live.value()) {
118  google::cloud::internal::ThrowInvalidArgument(

120  "condition");
121  }
122  } else {
123  auto tmp = *rhs.is_live;
124  result.is_live.emplace(std::forward<bool>(tmp));
125  }
126  }
127  if (rhs.matches_storage_class.has_value()) {
128  if (result.matches_storage_class.has_value()) {
129  std::vector<std::string> tmp;
130  std::vector<std::string> a = *std::move(result.matches_storage_class);
131  std::sort(a.begin(), a.end());

133  std::sort(b.begin(), b.end());

135  std::back_inserter(tmp));

137  } else {
