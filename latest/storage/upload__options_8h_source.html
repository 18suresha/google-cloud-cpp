26 /**
27  * Request a resumable upload, restoring a previous session if necessary.
28  *
29  * When this option is used the client library prefers using resumable uploads.
30  *
31  * If the value passed to this option is the empty string, then the library will
32  * create a new resumable session. Otherwise the value should be the id of a
33  * previous upload sesion, the client library will restore that session in
34  * this case.
35  */
36 struct UseResumableUploadSession
37  : public internal::ComplexOption<UseResumableUploadSession, std::string> {
38  using ComplexOption<UseResumableUploadSession, std::string>::ComplexOption;
39  static char const* name() { return "resumable-upload"; }
40 };
