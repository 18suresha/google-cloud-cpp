288  // Only the pre-conditions on the destination matter. If they are not set, it
289  // is possible for the request to succeed more than once, even if the source
290  // pre-conditions are set. If they are set, the operation can only succeed
291  // once, but the results may be different.
292  return request.HasOption<IfGenerationMatch>();
