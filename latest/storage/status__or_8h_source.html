28 /**
29  * `StatusOr<T>` is used to return a value or an error status.
30  *
31  * If the library is compiled with exceptions disabled there is no standard C++
32  * mechanism to report errors. In that case we use this class to wrap the values
33  * returned to the application.
34  *
35  * The application typically calls an API in the library and must check the
36  * returned error status:
37  *
38  * @code
39  * namespace gcs = google::cloud::storage;
40  * void AppCode(gcs::noex::Client client) {
41  * gcs::StatusOr<gcs::BucketMetadata> meta_err = client.GetBucketMetadata(
42  * "my-bucket-name");
43  * if (not meta_err) {
44  * std::cerr << "Error in GetBucketMetadata: " << meta_err.status()
45  * << std::endl;
46  * return;
47  * }
48  * gcs::BucketMetadata meta = *meta_err;
49  * // Do useful work here.
50  * }
51  * @endcode
52  *
53  * Note that the storage client retries most requests for you, resending the
54  * request after an error is probably not useful. You should consider changing
55  * the retry policies instead.
56  *
57  * TODO(...) - the current implementation is fairly naive with respect to `T`,
58  * it is unlikely to work correctly for reference types, types without default
59  * constructors, arrays.
60  *
61  * @tparam T the type of the value.
62  */
63 template <typename T>
64 class StatusOr final {
65  public:
66  /**
67  * Initializes with an error status (UNKNOWN).
68  *
69  * TODO(#548) - currently storage::Status does not define the status codes,
70  * they are simply integers, usually HTTP status codes. We need to map to
71  * the well-defined set of status codes.
72  */
73  StatusOr() : StatusOr(Status(500, "UNKNOWN")) {}
74 
75  /**
76  * Creates a new `StatusOr<T>` holding the error condition @p rhs.
77  *
78  * @par Post-conditions
79  * `ok() == false` and `status() == rhs`.
80  *
81  * @param rhs the status to initialize the object.
82  * @throws std::invalid_argument if `rhs.ok()`. If exceptions are disabled the
83  * program terminates via `google::cloud::Terminate()`
84  */
85  // NOLINTNEXTLINE(google-explicit-constructor)
86  StatusOr(Status rhs) : status_(std::move(rhs)) {
87  if (status_.ok()) {
88  google::cloud::internal::RaiseInvalidArgument(__func__);
89  }
90  }
91 
92  /**
93  * Creates a new `StatusOr<T>` holding the value @p rhs.
94  *
95  * @par Post-conditions
96  * `ok() == true` and `value() == rhs`.
97  *
98  * @param rhs the value used to initialize the object.
99  *
100  * @throws only if `T`'s move constructor throws.
101  */
102  // NOLINTNEXTLINE(google-explicit-constructor)
103  StatusOr(T rhs) : status_(), value_(std::move(rhs)) {}
104 
105  bool ok() const { return status_.ok(); }
106  explicit operator bool() const { return status_.ok(); }
107 
108  //@{
109  /**
110  * @name Deference operators.
111  *
112  * @warning Using these operators when `ok() == false` results in undefined
113  * behavior.
114  *
115  * @return All these return a (properly ref and const-qualified) reference to
116  * the underlying value.
117  */
118  T& operator*() & { return value_; }
119 
120  T const& operator*() const& { return value_; }
121 
122  T&& operator*() && { return std::move(value_); }
123 
124  T const&& operator*() const&& { return std::move(value_); }
125  //@}
126 
127  //@{
128  /**
129  * @name Member access operators.
130  *
131  * @warning Using these operators when `ok() == false` results in undefined
132  * behavior.
133  *
134  * @return All these return a (properly ref and const-qualified) pointer to
135  * the underlying value.
136  */
137  T* operator->() & { return &value_; }
138 
139  T const* operator->() const& { return &value_; }
140  //@}
141 
142  //@{
143  /**
144  * @name Value accessors.
145  *
146  * @return All these member functions return a (properly ref and
147  * const-qualified) reference to the underlying value.
148  *
149  * @throws `RuntimeStatusError` with the contents of `status()` if the object
150  * does not contain a value, i.e., if `ok() == false`.
151  */
152  T& value() & {
153  CheckHasValue();
154  return value_;
155  }
156 
157  T const& value() const& {
158  CheckHasValue();
159  return value_;
160  }
161 
162  T&& value() && {
163  CheckHasValue();
164  return std::move(value_);
165  }
166 
167  T const&& value() const&& {
168  CheckHasValue();
169  return std::move(value_);
170  }
171  //@}
172 
173  //@{
174  /**
175  * @name Status accessors.
176  *
177  * @return All these member functions return the (properly ref and
178  * const-qualified) status. If the object contains a value then
179  * `status().ok() == true`.
180  */
181  Status& status() & { return status_; }
182  Status const& status() const& { return status_; }
183  Status&& status() && { return std::move(status_); }
184  Status const&& status() const&& { return std::move(status_); }
185  //@}
186 
187  private:
188  void CheckHasValue() const& {
189  if (not ok()) {
190  internal::ThrowStatus(status_);
191  }
192  }
193 
194  // When possible, do not copy the status.
195  void CheckHasValue() && {
196  if (not ok()) {
197  internal::ThrowStatus(std::move(status_));
198  }
199  }
200 
201  Status status_;
202  // TODO(...) - use std::aligned_buffer (or something like it) to support types
203  // without a default constructor.
204  T value_;
205 };
