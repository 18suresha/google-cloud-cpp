30 char const kDescription[] = R"""(
31 A throughput benchmark for the Google Cloud Storage C++ client library.
32 
33 This program first creates a bucket that will contain all the GCS objects used
34 in the test. The bucket is deleted at the end of the test. The name of the
35 bucket is selected at random, so multiple instances of this benchmark can run
36 simultaneously. The bucket uses the `STANDARD` storage class, in a region set
37 via the command-line.
38 
39 After creating this bucket the program creates a number of objects, all the
40 objects have the same contents, but the contents are generated at random.
41 
42 The size of the objects can be configured in the command-line, by default they
43 are 240MiB in size. The objects are constructed by writing N identical
44 chunks, the size of these chunks is configured in the command-line.
45 
46 Once the object creation phase is completed, the program starts T worker
47 threads. The worker threads read a queue of "chunks" to be downloaded, these
48 chunks are selected by the main thread at random.
49 
50 The number of chunks selected by the main thread is configurable from the
51 command-line. After the prescribed number of chunks is generated by the main
52 thread, the program shuts down the worker queue, waits for all the worker
53 threads to finish, and reports the effective download throughput.
54 
55 Then the program removes all the objects in the bucket and reports the time
56 taken to delete each one.
57 
58 A helper script in this directory can generate pretty graphs from the output of
59 this program.
60 )""";

209 std::vector<std::string> CreateAllObjects(
210  gcs::Client const& client, google::cloud::internal::DefaultPRNG& gen,
211  std::string const& bucket_name, Options const& options) {
212  using std::chrono::duration_cast;
213  using std::chrono::milliseconds;
214 
215  std::size_t const max_group_size =
216  std::max(options.object_count / options.thread_count, 1);
217  std::cout << "# Creating test objects [" << max_group_size << "]\n";
218 
219  // Generate the list of object names.
220  std::vector<std::string> object_names;
221  object_names.reserve(options.object_count);
222  // NOLINTNEXTLINE(google-runtime-int)
223  for (long c = 0; c != options.object_count; ++c) {
224  object_names.emplace_back(gcs_bm::MakeRandomObjectName(gen));
225  }
226 
227  // Split the objects in more or less equally sized groups, launch a thread
228  // to create the objects in each group.
229  auto start = std::chrono::steady_clock::now();
230  std::vector<std::future<void>> tasks;
231  std::vector<std::string> group;
232  for (auto const& o : object_names) {
233  group.push_back(o);
234  if (group.size() >= max_group_size) {
235  tasks.emplace_back(std::async(std::launch::async, &CreateGroup, client,
236  bucket_name, options, std::move(group)));
237  group = {}; // after a move, must assign to guarantee it is valid.
238  }
239  }
240  if (!group.empty()) {
241  tasks.emplace_back(std::async(std::launch::async, &CreateGroup, client,
242  bucket_name, options, std::move(group)));
243  }
244  // Wait for the threads to finish.
245  for (auto& t : tasks) {
246  t.get();
247  }
248  auto elapsed = std::chrono::steady_clock::now() - start;
249  std::cout << "# Created in " << duration_cast<milliseconds>(elapsed).count()
250  << "ms\n";
251  return object_names;
252 }

254 void WorkerThread(WorkItemQueue& work_queue) {
255  auto client = gcs::Client::CreateDefaultClient();
256  if (!client) return;
257  std::vector<char> buffer;
258  for (auto w = work_queue.Pop(); w.has_value(); w = work_queue.Pop()) {
259  auto const begin = w->begin;
260  auto const end = w->end;
261  auto stream =
262  client->ReadObject(w->bucket, w->object, gcs::ReadRange(begin, end));
263  buffer.resize(w->end - w->begin);
264  stream.read(buffer.data(), buffer.size());
265  stream.Close();
266  }
267 }

269 IterationResult RunOneIteration(google::cloud::internal::DefaultPRNG& generator,
270  Options const& options,
271  std::string const& bucket_name,
272  std::vector<std::string> const& object_names) {
273  using std::chrono::duration_cast;
274  using std::chrono::microseconds;
275 
276  WorkItemQueue work_queue;
277  std::vector<std::future<void>> workers;
278  std::generate_n(std::back_inserter(workers), options.thread_count,
279  [&work_queue] {
280  return std::async(std::launch::async, WorkerThread,
281  std::ref(work_queue));
282  });
283 
284  std::uniform_int_distribution<std::size_t> object_generator(
285  0, object_names.size() - 1);
286  std::uniform_int_distribution<std::size_t> chunk_generator(
287  0, options.chunk_count - 1);
288 
289  auto const download_start = std::chrono::steady_clock::now();
290  std::int64_t total_bytes = 0;
291  // NOLINTNEXTLINE(google-runtime-int)
292  for (long i = 0; i != options.iteration_size; ++i) {
293  auto const object = object_generator(generator);
294  auto const chunk = chunk_generator(generator);
295  work_queue.Push({bucket_name, object_names.at(object),
296  chunk * options.chunk_size,
297  (chunk + 1) * options.chunk_size});
298  total_bytes += options.chunk_size;
299  }
300  work_queue.Shutdown();
301  for (auto& t : workers) {
302  t.get();
303  }
304  auto const elapsed = std::chrono::steady_clock::now() - download_start;
305  return {total_bytes, duration_cast<microseconds>(elapsed)};
306 }

308 google::cloud::StatusOr<Options> ParseArgsDefault(
309  std::vector<std::string> argv) {
310  Options options;
311  bool wants_help = false;
312  bool wants_description = false;
313  std::vector<gcs_bm::OptionDescriptor> desc{
314  {"--help", "print usage information",
315  [&wants_help](std::string const&) { wants_help = true; }},
316  {"--description", "print benchmark description",
317  [&wants_description](std::string const&) { wants_description = true; }},
318  {"--project-id", "use the given project id for the benchmark",
319  [&options](std::string const& val) { options.project_id = val; }},
320  {"--bucket-prefix", "configure the bucket's prefix",
321  [&options](std::string const& val) { options.bucket_prefix = val; }},
322  {"--region", "use the given region for the benchmark",
323  [&options](std::string const& val) { options.region = val; }},
324  {"--object-count", "set the number of objects created by the benchmark",
325  [&options](std::string const& val) {
326  options.object_count = std::stoi(val);
327  }},
328  {"--thread-count", "set the number of threads in the benchmark",
329  [&options](std::string const& val) {
330  options.thread_count = std::stoi(val);
331  }},
332  {"--iteration-size",
333  "set the number of chunk downloaded in each iteration",
334  [&options](std::string const& val) {
335  options.iteration_size = std::stoi(val);
336  }},
337  {"--iteration-count",
338  "set the number of samples captured by the benchmark",
339  [&options](std::string const& val) {
340  options.iteration_count = std::stoi(val);
341  }},
342  {"--chunk-size", "size of the chunks used in the benchmark",
343  [&options](std::string const& val) {
344  options.chunk_size = gcs_bm::ParseBufferSize(val);
345  }},
346  {"--chunk-count", "the number of chunks in each object",
347  [&options](std::string const& val) {
348  options.chunk_count = std::stoi(val);
349  }},
350  };
351  auto usage = gcs_bm::BuildUsage(desc, argv[0]);
352 
353  auto unparsed = gcs_bm::OptionsParse(desc, argv);
354  if (wants_help) {
355  std::cout << usage << "\n";
356  }
357 
358  if (wants_description) {
359  std::cout << kDescription << "\n";
360  }
361 
362  if (unparsed.size() > 2) {
363  std::ostringstream os;
364  os << "Unknown arguments or options\n" << usage << "\n";
365  return google::cloud::Status{google::cloud::StatusCode::kInvalidArgument,
366  std::move(os).str()};
367  }
368  if (unparsed.size() == 2) {
369  options.region = unparsed[1];
370  }
371  if (options.region.empty()) {
372  std::ostringstream os;
373  os << "Missing value for --region option" << usage << "\n";
374  return google::cloud::Status{google::cloud::StatusCode::kInvalidArgument,
375  std::move(os).str()};
376  }
377 
378  return options;
379 }
