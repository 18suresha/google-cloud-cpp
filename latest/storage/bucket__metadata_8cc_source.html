31 std::ostream& operator<<(std::ostream& os, CorsEntry const& rhs) {
32  auto join = [](char const* sep, std::vector<std::string> const& list) {
33  if (list.empty()) {
34  return std::string{};
35  }
36  return std::accumulate(++list.begin(), list.end(), list.front(),
37  [sep](std::string a, std::string const& b) {
38  a += sep;
39  a += b;
40  return a;
41  });
42  };
43  os << "CorsEntry={";
44  char const* sep = "";
45  if (rhs.max_age_seconds.has_value()) {
46  os << "max_age_seconds=" << *rhs.max_age_seconds;
47  sep = ", ";
48  }
49  return os << sep << "method=[" << join(", ", rhs.method) << "], origin=["
50  << join(", ", rhs.origin) << "], response_header=["
51  << join(", ", rhs.response_header) << "]}";
52 }

284  }

341 BucketMetadataPatchBuilder& BucketMetadataPatchBuilder::SetIamConfiguration(
342  BucketIamConfiguration const& v) {
343  internal::PatchBuilder iam_configuration;
344 
345  if (v.uniform_bucket_level_access.has_value()) {
346  internal::PatchBuilder uniform_bucket_level_access;
347  internal::PatchBuilder bucket_policy_only;
348  uniform_bucket_level_access.SetBoolField(
349  "enabled", v.uniform_bucket_level_access->enabled);
350  bucket_policy_only.SetBoolField("enabled",
351  v.uniform_bucket_level_access->enabled);
352  // The lockedTime field should not be set, this is not a mutable field, it
353  // is set by the server when the policy is enabled.
354  iam_configuration.AddSubPatch("uniformBucketLevelAccess",
355  uniform_bucket_level_access);
356  iam_configuration.AddSubPatch("bucketPolicyOnly",
357  uniform_bucket_level_access);
358  impl_.AddSubPatch("iamConfiguration", iam_configuration);
359  return *this;
360  }
361 
362  if (v.bucket_policy_only.has_value()) {
363  internal::PatchBuilder uniform_bucket_level_access;
364  internal::PatchBuilder bucket_policy_only;
365  uniform_bucket_level_access.SetBoolField("enabled",
366  v.bucket_policy_only->enabled);
367  bucket_policy_only.SetBoolField("enabled", v.bucket_policy_only->enabled);
368  // The lockedTime field should not be set, this is not a mutable field, it
369  // is set by the server when the policy is enabled.
370  iam_configuration.AddSubPatch("uniformBucketLevelAccess",
371  bucket_policy_only);
372  iam_configuration.AddSubPatch("bucketPolicyOnly", bucket_policy_only);
373  }
374  impl_.AddSubPatch("iamConfiguration", iam_configuration);
375  return *this;
376 }

404 BucketMetadataPatchBuilder& BucketMetadataPatchBuilder::SetLifecycle(
405  BucketLifecycle const& v) {
406  if (v.rule.empty()) {
407  return ResetLifecycle();
408  }
409  internal::PatchBuilder subpatch;
410  std::vector<nlohmann::json> array;
411  array.reserve(v.rule.size());
412  for (auto const& a : v.rule) {
413  nlohmann::json condition;
414  auto const& c = a.condition();
415  if (c.age.has_value()) {
416  condition["age"] = *c.age;
417  }
418  if (c.created_before.has_value()) {
419  condition["createdBefore"] =
420  absl::StrFormat("%04d-%02d-%02d", c.created_before->year(),
421  c.created_before->month(), c.created_before->day());
422  }
423  if (c.is_live.has_value()) {
424  condition["isLive"] = *c.is_live;
425  }
426  if (c.matches_storage_class.has_value()) {
427  condition["matchesStorageClass"] = *c.matches_storage_class;
428  }
429  if (c.num_newer_versions.has_value()) {
430  condition["numNewerVersions"] = *c.num_newer_versions;
431  }
432  nlohmann::json action;
433  if (!a.action().type.empty()) {
434  action["type"] = a.action().type;
435  }
436  if (!a.action().storage_class.empty()) {
437  action["storageClass"] = a.action().storage_class;
438  }
439  array.emplace_back(nlohmann::json{
440  {"action", action},
441  {"condition", condition},
442  });
443  }
444  subpatch.SetArrayField("rule", array);
445  impl_.AddSubPatch("lifecycle", subpatch);
446  return *this;
447 }
