29 std::ostream& operator<<(std::ostream& os, CorsEntry const& rhs) {
30  auto join = [](char const* sep, std::vector<std::string> const& list) {
31  if (list.empty()) {
32  return std::string{};
33  }
34  return std::accumulate(++list.begin(), list.end(), list.front(),
35  [sep](std::string a, std::string const& b) {
36  a += sep;
37  a += b;
38  return a;
39  });
40  };
41  os << "CorsEntry={";
42  char const* sep = "";
43  if (rhs.max_age_seconds.has_value()) {
44  os << "max_age_seconds=" << *rhs.max_age_seconds;
45  sep = ", ";
46  }
47  return os << sep << "method=[" << join(", ", rhs.method) << "], origin=["
48  << join(", ", rhs.origin) << "], response_header=["
49  << join(", ", rhs.response_header) << "]}";
50 }

249 BucketMetadataPatchBuilder& BucketMetadataPatchBuilder::SetCors(
250  std::vector<CorsEntry> const& v) {
251  if (v.empty()) {
252  return ResetCors();
253  }
254  std::vector<internal::nl::json> array;
255  array.reserve(v.size());
256  for (auto const& a : v) {
257  internal::nl::json entry;
258  if (a.max_age_seconds.has_value()) {
259  entry["maxAgeSeconds"] = *a.max_age_seconds;
260  }
261  if (!a.method.empty()) {
262  entry["method"] = a.method;
263  }
264  if (!a.origin.empty()) {
265  entry["origin"] = a.origin;
266  }
267  if (!a.response_header.empty()) {
268  entry["responseHeader"] = a.response_header;
269  }
270  array.emplace_back(std::move(entry));
271  }
272  impl_.SetArrayField("cors", array);
273  return *this;
274 }

368 BucketMetadataPatchBuilder& BucketMetadataPatchBuilder::SetLifecycle(
369  BucketLifecycle const& v) {
370  if (v.rule.empty()) {
371  return ResetLifecycle();
372  }
373  internal::PatchBuilder subpatch;
374  std::vector<internal::nl::json> array;
375  array.reserve(v.rule.size());
376  for (auto const& a : v.rule) {
377  internal::nl::json condition;
378  auto const& c = a.condition();
379  if (c.age.has_value()) {
380  condition["age"] = *c.age;
381  }
382  if (c.created_before.has_value()) {
383  condition["createdBefore"] = internal::FormatRfc3339(*c.created_before);
384  }
385  if (c.is_live.has_value()) {
386  condition["isLive"] = *c.is_live;
387  }
388  if (c.matches_storage_class.has_value()) {
389  condition["matchesStorageClass"] = *c.matches_storage_class;
390  }
391  if (c.num_newer_versions.has_value()) {
392  condition["numNewerVersions"] = *c.num_newer_versions;
393  }
394  internal::nl::json action;
395  if (!a.action().type.empty()) {
396  action["type"] = a.action().type;
397  }
398  if (!a.action().storage_class.empty()) {
399  action["storageClass"] = a.action().storage_class;
400  }
401  array.emplace_back(internal::nl::json{
402  {"action", action},
403  {"condition", condition},
404  });
405  }
406  subpatch.SetArrayField("rule", array);
407  impl_.AddSubPatch("lifecycle", subpatch);
408  return *this;
409 }
