31 char const kDescription[] = R"""(
32 A thoughput benchmark for the Google Cloud Storage C++ client library.
33 
34 This program benchmarks the Google Cloud Storage (GCS) C++ client library when
35 used to upload and download files. The program creates a file of a prescribed
36 size, and then repeatedly uploads that file to a GCS object, and then downloads
37 the GCS object to a separate file. The program reports the time taken to perform
38 the each operation, as well as the effective bandwidth (in Gbps and MiB/s). The
39 program deletes the target GCS object after each iteration.
40 
41 To perform this benchmark the program creates a new regional bucket, in a region
42 configured via the command line. Other test parameters, such as the project id,
43 the file size, and the buffer sizes are configurable via the command line too.
44 
45 The bucket name, the local file names, and the object names are all randomly
46 generated, so multiple instances of the program can run simultaneously. The
47 output of this program is an annotated CSV file, that can be analyzed by an
48 external script. The annotation lines start with a '#', analysis scripts should
49 skip these lines.
50 )""";

65 int main(int argc, char* argv[]) try {
66  Options options = ParseArgs(argc, argv);
67 
68  google::cloud::StatusOr<gcs::ClientOptions> client_options =
69  gcs::ClientOptions::CreateDefaultClientOptions();
70  if (!client_options) {
71  std::cerr << "Could not create ClientOptions, status="
72  << client_options.status() << "\n";
73  return 1;
74  }
75  client_options->SetUploadBufferSize(options.upload_buffer_size);
76  client_options->SetDownloadBufferSize(options.download_buffer_size);
77  client_options->set_project_id(options.project_id);
78 
79  gcs::Client client(*std::move(client_options));
80 
81  google::cloud::internal::DefaultPRNG generator =
82  google::cloud::internal::MakeDefaultPRNG();
83 
84  auto bucket_name =
85  gcs_bm::MakeRandomBucketName(generator, "gcs-file-transfer-");
86  auto meta =
87  client
88  .CreateBucket(bucket_name,
89  gcs::BucketMetadata()
90  .set_storage_class(gcs::storage_class::Regional())
91  .set_location(options.region),
92  gcs::PredefinedAcl("private"),
93  gcs::PredefinedDefaultObjectAcl("projectPrivate"),
94  gcs::Projection("full"))
95  .value();
96  std::cout << "# Running test on bucket: " << meta.name() << "\n";
97  std::string notes = google::cloud::storage::version_string() + ";" +
98  google::cloud::internal::compiler() + ";" +
99  google::cloud::internal::compiler_flags();
100  std::transform(notes.begin(), notes.end(), notes.begin(),
101  [](char c) { return c == '\n' ? ';' : c; });
102  std::cout << "# Start time: "
103  << google::cloud::internal::FormatRfc3339(
104  std::chrono::system_clock::now())
105  << "\n# Region: " << options.region
106  << "\n# Duration: " << options.duration.count() << "s"
107  << "\n# File Size: " << options.file_size
108  << "\n# File Size (MiB): " << options.file_size / gcs_bm::kMiB
109  << "\n# Download buffer size (KiB): "
110  << options.download_buffer_size / gcs_bm::kKiB
111  << "\n# Upload buffer size (KiB): "
112  << options.upload_buffer_size / gcs_bm::kKiB
113  << "\n# Build info: " << notes << "\n";
114 
115  std::cout << "# Creating file to upload ..." << std::flush;
116  auto filename = gcs_bm::MakeRandomFileName(generator);
117  std::ofstream(filename, std::ios::binary)
118  << gcs_bm::MakeRandomData(generator, options.file_size);
119  std::cout << " DONE\n"
120  << "# File: " << filename << "\n";
121 
122  auto deadline = std::chrono::system_clock::now() + options.duration;
123  for (auto now = std::chrono::system_clock::now(); now < deadline;
124  now = std::chrono::system_clock::now()) {
125  auto object_name = gcs_bm::MakeRandomObjectName(generator);
126 
127  auto upload_start = std::chrono::steady_clock::now();
128  auto object_metadata =
129  client.UploadFile(filename, bucket_name, object_name);
130  auto upload_elapsed = std::chrono::duration_cast<std::chrono::nanoseconds>(
131  std::chrono::steady_clock::now() - upload_start);
132  double gbps = options.file_size * 8.0 / upload_elapsed.count();
133  auto ms =
134  std::chrono::duration_cast<std::chrono::milliseconds>(upload_elapsed);
135  auto MiBs =
136  (double(options.file_size) / gcs_bm::kMiB) / (ms.count() / 1000.0);
137  std::cout << "FileUpload," << options.file_size << ','
138  << upload_elapsed.count() << ',' << gbps << ',' << ms.count()
139  << ',' << MiBs << ',' << object_metadata.status().code() << "\n";
140  if (!object_metadata) {
141  std::cout << "# Error in FileUpload: " << object_metadata.status()
142  << "\n";
143  continue;
144  }
145 
146  auto destination_filename = gcs_bm::MakeRandomFileName(generator);
147  auto download_start = std::chrono::steady_clock::now();
148  client.DownloadToFile(object_metadata->bucket(), object_metadata->name(),
149  destination_filename);
150  auto download_elapsed =
151  std::chrono::duration_cast<std::chrono::nanoseconds>(
152  std::chrono::steady_clock::now() - download_start);
153  gbps = options.file_size * 8.0 / download_elapsed.count();
154  ms =
155  std::chrono::duration_cast<std::chrono::milliseconds>(download_elapsed);
156  MiBs = (double(options.file_size) / gcs_bm::kMiB) / (ms.count() / 1000.0);
157  std::cout << "FileDownload," << options.file_size << ','
158  << download_elapsed.count() << ',' << gbps << ',' << ms.count()
159  << ',' << MiBs << ',' << object_metadata.status().code() << "\n";
160 
161  (void)client.DeleteObject(object_metadata->bucket(),
162  object_metadata->name(),
163  gcs::Generation(object_metadata->generation()));
164  std::remove(destination_filename.c_str());
165  }
166 
167  std::remove(filename.c_str());
168 
169  std::cout << "# Deleting " << bucket_name << "\n";
170  auto status = client.DeleteBucket(bucket_name);
171  if (!status.ok()) {
172  google::cloud::internal::ThrowStatus(status);
173  }
174 
175  return 0;
176 } catch (std::exception const& ex) {
177  std::cerr << "Standard exception raised: " << ex.what() << "\n";
178  return 1;
179 }

183 Options ParseArgs(int& argc, char* argv[]) {
184  Options options;
185 
186  bool wants_help = false;
187  bool wants_description = false;
188  std::vector<gcs_bm::OptionDescriptor> descriptors{
189  {"--help", "print the usage message",
190  [&wants_help](std::string const&) { wants_help = true; }},
191  {"--description", "print a description of the benchmark",
192  [&wants_description](std::string const&) { wants_description = true; }},
193  {"--project-id", "the GCP project to create the bucket",
194  [&options](std::string const& val) { options.project_id = val; }},
195  {"--duration", "how long should the benchmark run (in seconds).",
196  [&options](std::string const& val) {
197  options.duration = gcs_bm::ParseDuration(val);
198  }},
199  {"--file-size", "the size of the file to upload",
200  [&options](std::string const& val) {
201  options.file_size = gcs_bm::ParseSize(val);
202  }},
203  {"--upload-buffer-size", "configure gcs::Client upload buffer size",
204  [&options](std::string const& val) {
205  options.upload_buffer_size = gcs_bm::ParseSize(val);
206  }},
207  {"--download-buffer-size", "configure gcs::Client download buffer size",
208  [&options](std::string const& val) {
209  options.download_buffer_size = gcs_bm::ParseSize(val);
210  }},
211  {"--region", "The GCS region used for the benchmark",
212  [&options](std::string const& val) { options.region = val; }},
213  };
214  auto usage = gcs_bm::BuildUsage(descriptors, argv[0]);
215 
216  auto unparsed = gcs_bm::OptionsParse(descriptors, {argv, argv + argc});
217  if (wants_help) {
218  std::cout << usage << "\n";
219  }
220 
221  if (wants_description) {
222  std::cout << kDescription << "\n";
223  }
224 
225  if (unparsed.size() > 2) {
226  std::ostringstream os;
227  os << "Unknown arguments or options\n" << usage << "\n";
228  throw std::runtime_error(std::move(os).str());
229  }
230  if (unparsed.size() == 2) {
231  options.region = unparsed[1];
232  }
233  if (options.region.empty()) {
234  std::ostringstream os;
235  os << "Missing value for --region option" << usage << "\n";
236  throw std::runtime_error(std::move(os).str());
237  }
238 
239  return options;
240 }
