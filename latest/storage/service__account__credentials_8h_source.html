139 template <typename HttpRequestBuilderType =
140  storage::internal::CurlRequestBuilder,
141  typename ClockType = std::chrono::system_clock>
142 class ServiceAccountCredentials : public Credentials {
143  public:
144  // NOLINTNEXTLINE(performance-unnecessary-value-param) TODO(#4112)
145  explicit ServiceAccountCredentials(ServiceAccountCredentialsInfo info)
146  : ServiceAccountCredentials(std::move(info), {}) {}
147  ServiceAccountCredentials(ServiceAccountCredentialsInfo info,
148  ChannelOptions const& options)
149  : info_(std::move(info)), clock_() {
150  HttpRequestBuilderType request_builder(
151  info_.token_uri,
152  storage::internal::GetDefaultCurlHandleFactory(options));
153  request_builder.AddHeader(
154  "Content-Type: application/x-www-form-urlencoded");
155  // This is the value of grant_type for JSON-formatted service account
156  // keyfiles downloaded from Cloud Console.
157  std::string grant_type("grant_type=");
158  grant_type +=
159  request_builder
160  .MakeEscapedString("urn:ietf:params:oauth:grant-type:jwt-bearer")
161  .get();
162  grant_type_ = std::move(grant_type);
163  request_ = request_builder.BuildRequest();
164  }
165 
166  StatusOr<std::string> AuthorizationHeader() override {
167  std::unique_lock<std::mutex> lock(mu_);
168  return refreshing_creds_.AuthorizationHeader(clock_.now(),
169  [this] { return Refresh(); });
170  }
171 
183  StatusOr<std::vector<std::uint8_t>> SignBlob(
184  SigningAccount const& signing_account,
185  std::string const& blob) const override {
186  if (signing_account.has_value() &&
187  signing_account.value() != info_.client_email) {
188  return Status(StatusCode::kInvalidArgument,
189  "The current_credentials cannot sign blobs for " +
190  signing_account.value());
191  }
192  return internal::SignStringWithPem(blob, info_.private_key,
193  JwtSigningAlgorithms::RS256);
194  }
195 
196  std::string AccountEmail() const override { return info_.client_email; }
197  std::string KeyId() const override { return info_.private_key_id; }
198 
199  private:
200  StatusOr<RefreshingCredentialsWrapper::TemporaryToken> Refresh() {
201  auto payload =
202  CreateServiceAccountRefreshPayload(info_, grant_type_, clock_.now());
203 
204  auto response = request_.MakeRequest(payload);
205  if (!response) {
206  return std::move(response).status();
207  }
208  if (response->status_code >= 300) {
209  return AsStatus(*response);
210  }
211 
212  return ParseServiceAccountRefreshResponse(*response, clock_.now());
213  }

216  std::string grant_type_;
217  ServiceAccountCredentialsInfo info_;
218  mutable std::mutex mu_;
219  RefreshingCredentialsWrapper refreshing_creds_;
220  ClockType clock_;
221 };
