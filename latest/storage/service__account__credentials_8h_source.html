129 template <typename HttpRequestBuilderType =
130  storage::internal::CurlRequestBuilder,
131  typename ClockType = std::chrono::system_clock>
132 class ServiceAccountCredentials : public Credentials {
133  public:
134  // NOLINTNEXTLINE(performance-unnecessary-value-param) TODO(#4112)
135  explicit ServiceAccountCredentials(ServiceAccountCredentialsInfo info)
136  : ServiceAccountCredentials(std::move(info), {}) {}
137  ServiceAccountCredentials(ServiceAccountCredentialsInfo info,
138  ChannelOptions const& options)
139  : info_(std::move(info)), clock_() {
140  HttpRequestBuilderType request_builder(
141  info_.token_uri,
142  storage::internal::GetDefaultCurlHandleFactory(options));
143  request_builder.AddHeader(
144  "Content-Type: application/x-www-form-urlencoded");
145  // This is the value of grant_type for JSON-formatted service account
146  // keyfiles downloaded from Cloud Console.
147  std::string grant_type("grant_type=");
148  grant_type +=
149  request_builder
150  .MakeEscapedString("urn:ietf:params:oauth:grant-type:jwt-bearer")
151  .get();
152  grant_type_ = std::move(grant_type);
153  request_ = request_builder.BuildRequest();
154  }
155 
156  StatusOr<std::string> AuthorizationHeader() override {
157  std::unique_lock<std::mutex> lock(mu_);
158  return refreshing_creds_.AuthorizationHeader(clock_.now(),
159  [this] { return Refresh(); });
160  }
161 
173  StatusOr<std::vector<std::uint8_t>> SignBlob(
174  SigningAccount const& signing_account,
175  std::string const& blob) const override {
176  if (signing_account.has_value() &&
177  signing_account.value() != info_.client_email) {
178  return Status(StatusCode::kInvalidArgument,
179  "The current_credentials cannot sign blobs for " +
180  signing_account.value());
181  }
182  return internal::SignStringWithPem(blob, info_.private_key,
183  JwtSigningAlgorithms::RS256);
184  }
185 
186  std::string AccountEmail() const override { return info_.client_email; }
187  std::string KeyId() const override { return info_.private_key_id; }
188 
189  private:
190  StatusOr<RefreshingCredentialsWrapper::TemporaryToken> Refresh() {
191  auto payload =
192  CreateServiceAccountRefreshPayload(info_, grant_type_, clock_.now());
193 
194  auto response = request_.MakeRequest(payload);
195  if (!response) {
196  return std::move(response).status();
197  }
198  if (response->status_code >= 300) {
199  return AsStatus(*response);
200  }
201 
202  return ParseServiceAccountRefreshResponse(*response, clock_.now());
203  }
204 
205  typename HttpRequestBuilderType::RequestType request_;
206  std::string grant_type_;
207  ServiceAccountCredentialsInfo info_;
208  mutable std::mutex mu_;
209  RefreshingCredentialsWrapper refreshing_creds_;
210  ClockType clock_;
211 };
