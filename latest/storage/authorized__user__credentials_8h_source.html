46 /**
47  * Wrapper class for Google OAuth 2.0 user account credentials.
48  *
49  * Takes a AuthorizedUserCredentialsInfo and obtains access tokens from the
50  * Google Authorization Service as needed. Instances of this class should
51  * usually be created via the convenience methods declared in
52  * google_credentials.h.
53  *
54  * An HTTP Authorization header, with an access token as its value,
55  * can be obtained by calling the AuthorizationHeader() method; if the current
56  * access token is invalid or nearing expiration, this will class will first
57  * obtain a new access token before returning the Authorization header string.
58  *
59  * @see https://developers.google.com/identity/protocols/OAuth2 for an overview
60  * of using user credentials with Google's OAuth 2.0 system.
61  *
62  * @tparam HttpRequestBuilderType a dependency injection point. It makes it
63  * possible to mock internal libcurl wrappers. This should generally not be
64  * overridden except for testing.
65  */
66 template <typename HttpRequestBuilderType =
67  storage::internal::CurlRequestBuilder>
68 class AuthorizedUserCredentials : public Credentials {
69  public:
70  explicit AuthorizedUserCredentials(
71  AuthorizedUserCredentialsInfo const& info) {
72  HttpRequestBuilderType request_builder(
73  info.token_uri, storage::internal::GetDefaultCurlHandleFactory());
74  std::string payload("grant_type=refresh_token");
75  payload += "&client_id=";
76  payload += request_builder.MakeEscapedString(info.client_id).get();
77  payload += "&client_secret=";
78  payload += request_builder.MakeEscapedString(info.client_secret).get();
79  payload += "&refresh_token=";
80  payload += request_builder.MakeEscapedString(info.refresh_token).get();
81  payload_ = std::move(payload);
82  request_ = request_builder.BuildRequest();
83  }
84 
85  StatusOr<std::string> AuthorizationHeader() override {
86  std::unique_lock<std::mutex> lock(mu_);
87  return refreshing_creds_.AuthorizationHeader([this] { return Refresh(); });
88  }
89 
90  private:
91  StatusOr<RefreshingCredentialsWrapper::TemporaryToken> Refresh() {
92  namespace nl = storage::internal::nl;
93 
94  auto response = request_.MakeRequest(payload_);
95  if (!response) {
96  return std::move(response).status();
97  }
98  if (response->status_code >= 300) {
99  return AsStatus(*response);
100  }
101  nl::json access_token = nl::json::parse(response->payload, nullptr, false);
102  if (access_token.is_discarded() ||
103  access_token.count("access_token") == 0U ||
104  access_token.count("expires_in") == 0U ||
105  access_token.count("id_token") == 0U ||
106  access_token.count("token_type") == 0U) {
107  response->payload +=
108  "Could not find all required fields in response (access_token,"
109  " id_token, expires_in, token_type).";
110  return AsStatus(*response);
111  }
112  std::string header = "Authorization: ";
113  header += access_token.value("token_type", "");
114  header += ' ';
115  header += access_token.value("access_token", "");
116  std::string new_id = access_token.value("id_token", "");
117  auto expires_in =
118  std::chrono::seconds(access_token.value("expires_in", int(0)));
119  auto new_expiration = std::chrono::system_clock::now() + expires_in;
120 
121  return RefreshingCredentialsWrapper::TemporaryToken{std::move(header),
122  new_expiration};
123  }
124 
125  typename HttpRequestBuilderType::RequestType request_;
126  std::string payload_;
127  mutable std::mutex mu_;
128  RefreshingCredentialsWrapper refreshing_creds_;
129 };
