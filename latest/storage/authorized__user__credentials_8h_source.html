87  payload += "&client_id=";

91  payload += "&refresh_token=";
92  payload += request_builder.MakeEscapedString(info.refresh_token).get();
93  payload_ = std::move(payload);

95  }

97  StatusOr<std::string> AuthorizationHeader() override {
98  std::unique_lock<std::mutex> lock(mu_);
99  return refreshing_creds_.AuthorizationHeader(clock_.now(),
100  [this] { return Refresh(); });

103  private:
104  StatusOr<RefreshingCredentialsWrapper::TemporaryToken> Refresh() {
105  auto response = request_.MakeRequest(payload_);
106  if (!response) {
107  return std::move(response).status();
108  }
109  if (response->status_code >= 300) {
110  return AsStatus(*response);
111  }
112  return ParseAuthorizedUserRefreshResponse(*response, clock_.now());
113  }
