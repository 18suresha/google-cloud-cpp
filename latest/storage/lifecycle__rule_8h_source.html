123 /**

126  * Object Lifecycle Rules allow to configure a Bucket to automatically delete
127  * or change the storage class of objects as they go through lifecycle events.

242  /**
243  * Combines multiple LifecycleRule conditions using conjunction.
244  *
245  * Create a condition that require all the @p condition parameters to be met
246  * to take effect.
247  *
248  * @par Example
249  *
250  * @code
251  * // Affect objects that are in the MULTI_REGIONAL storage class, have at
252  * // least 2 new versions, are at least 7 days old, and are alive.
253  * LifecycleRuleCondition condition = LifecycleRule::ConditionConjunction(
254  * LifecycleRule::NumNewerVersions(2),
255  * LifecycleRule::MatchesStorageClassMultiRegional(),
256  * LifecycleRule::MaxAge(7), LifecycleRule::IsLive(true));
257  * @endcode
258  *
259  * @param condition a parameter pack of conditions.
260  * @throws std::invalid_argument if the list of parameters is contradictory,
261  * for example, `IsLive(true)` and `IsLive(false)` are in the @p condition
262  * list.
263  * @return a LifecycleRuleCondition that is satisfied when all the
264  * @p condition conditions are satisfied.
265  * @tparam Condition the types of the conditions, they must all be convertible
266  * to `LifecycleRuleCondition`.
267  */
268  template <typename... Condition>
269  static LifecycleRuleCondition ConditionConjunction(Condition&&... condition) {
270  LifecycleRuleCondition result;
271  MergeConditions(result, std::forward<Condition>(condition)...);
272  return result;
273  }
