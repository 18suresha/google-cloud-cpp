122 /**
123  * Defines objects to read, create, and modify Object Lifecycle Rules.
124  *
125  * Object Lifecycle Rules allow to configure a Bucket to automatically delete
126  * or change the storage class of objects as they go through lifecycle events.

240  /**
241  * Combines multiple LifecycleRule conditions using conjunction.
242  *
243  * Create a condition that require all the @p condition parameters to be met
244  * to take effect.
245  *
246  * @par Example
247  *
248  * @code
249  * // Affect objects that are in the MULTI_REGIONAL storage class, have at
250  * // least 2 new versions, are at least 7 days old, and are alive.
251  * LifecycleRuleCondition condition = LifecycleRule::ConditionConjunction(
252  * LifecycleRule::NumNewerVersions(2),
253  * LifecycleRule::MatchesStorageClassMultiRegional(),
254  * LifecycleRule::MaxAge(7), LifecycleRule::IsLive(true));
255  * @endcode
256  *
257  * @param condition a parameter pack of conditions.
258  * @throws std::invalid_argument if the list of parameters is contradictory,
259  * for example, `IsLive(true)` and `IsLive(false)` are in the @p condition
260  * list.
261  * @return a LifecycleRuleCondition that is satisfied when all the
262  * @p condition conditions are satisfied.
263  * @tparam Condition the types of the conditions, they must all be convertible
264  * to `LifecycleRuleCondition`.
265  */
266  template <typename... Condition>
267  static LifecycleRuleCondition ConditionConjunction(Condition&&... condition) {
268  LifecycleRuleCondition result;
269  MergeConditions(result, std::forward<Condition>(condition)...);
270  return result;
271  }
