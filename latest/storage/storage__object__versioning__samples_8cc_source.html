24 void GetObjectVersioning(google::cloud::storage::Client client,
25  std::vector<std::string> const& argv) {
27  namespace gcs = google::cloud::storage;
28  using ::google::cloud::StatusOr;
29  [](gcs::Client client, std::string const& bucket_name) {
30  StatusOr<gcs::BucketMetadata> metadata =
31  client.GetBucketMetadata(bucket_name);
32  if (!metadata) throw std::runtime_error(metadata.status().message());
33 
34  if (metadata->versioning().has_value()) {
35  std::cout << "Object versioning for bucket " << bucket_name << " is "
36  << (metadata->versioning()->enabled ? "enabled" : "disabled")
37  << "\n";
38  } else {
39  std::cout << "Object versioning for bucket " << bucket_name
40  << " is disabled.\n";
41  }
42  }
44  (std::move(client), argv.at(0));
45 }

47 void EnableObjectVersioning(google::cloud::storage::Client client,
48  std::vector<std::string> const& argv) {
50  namespace gcs = google::cloud::storage;
51  using ::google::cloud::StatusOr;
52  [](gcs::Client client, std::string const& bucket_name) {
53  StatusOr<gcs::BucketMetadata> original =
54  client.GetBucketMetadata(bucket_name);
55  if (!original) throw std::runtime_error(original.status().message());
56 
57  StatusOr<gcs::BucketMetadata> patched = client.PatchBucket(
58  bucket_name,
59  gcs::BucketMetadataPatchBuilder().SetVersioning(
60  gcs::BucketVersioning{true}),
61  gcs::IfMetagenerationMatch(original->metageneration()));
62  if (!patched) throw std::runtime_error(patched.status().message());
63 
64  if (patched->versioning().has_value()) {
65  std::cout << "Object versioning for bucket " << bucket_name << " is "
66  << (patched->versioning()->enabled ? "enabled" : "disabled")
67  << "\n";
68  } else {
69  std::cout << "Object versioning for bucket " << bucket_name
70  << " is disabled.\n";
71  }
72  }

107 void CopyVersionedObject(google::cloud::storage::Client client,
108  std::vector<std::string> const& argv) {
109  // [START storage_copy_file_archived_generation]
110  namespace gcs = google::cloud::storage;
111  using ::google::cloud::StatusOr;
112  [](gcs::Client client, std::string const& source_bucket_name,
113  std::string const& source_object_name,
114  std::string const& destination_bucket_name,
115  std::string const& destination_object_name,
116  std::int64_t source_object_generation) {
117  StatusOr<gcs::ObjectMetadata> copy =
118  client.CopyObject(source_bucket_name, source_object_name,
119  destination_bucket_name, destination_object_name,
120  gcs::SourceGeneration{source_object_generation});
121  if (!copy) throw std::runtime_error(copy.status().message());
122 
123  std::cout << "Successfully copied " << source_object_name << " generation "
124  << source_object_generation << " in bucket " << source_bucket_name
125  << " to bucket " << copy->bucket() << " with name "
126  << copy->name()
127  << ".\nThe full metadata after the copy is: " << *copy << "\n";
128  }
129  // [END storage_copy_file_archived_generation]
130  (std::move(client), argv.at(0), argv.at(1), argv.at(2), argv.at(3),
131  std::stoll(argv.at(4)));
132 }

153 void RunAll(std::vector<std::string> const& argv) {
154  namespace examples = ::google::cloud::storage::examples;
155  namespace gcs = ::google::cloud::storage;
156 
157  if (!argv.empty()) throw examples::Usage{"auto"};
158  examples::CheckEnvironmentVariablesAreSet({
159  "GOOGLE_CLOUD_PROJECT",
160  "GOOGLE_CLOUD_CPP_STORAGE_TEST_CMEK_KEY",
161  });
162  auto const project_id =
163  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
164  auto const cmek_key =
165  google::cloud::internal::GetEnv("GOOGLE_CLOUD_CPP_STORAGE_TEST_CMEK_KEY")
166  .value();
167  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
168  auto const bucket_name = examples::MakeRandomBucketName(generator);
169  auto client = gcs::Client::CreateDefaultClient().value();
170 
171  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
172  << std::endl;
173  (void)client
174  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
175  .value();
176  // In GCS a single project cannot create or delete buckets more often than
177  // once every two seconds. We will pause until that time before deleting the
178  // bucket.
179  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
180 
181  std::cout << "\nRunning the GetObjectVersioning() example [1]" << std::endl;
182  GetObjectVersioning(client, {bucket_name});
183 
184  std::cout << "\nRunning the EnableObjectVersioning() example" << std::endl;
185  EnableObjectVersioning(client, {bucket_name});
186 
187  std::cout << "\nRunning the GetObjectVersioning() example [2]" << std::endl;
188  GetObjectVersioning(client, {bucket_name});
189 
190  auto constexpr kText = R"""(Some text to insert into the test objects.)""";
191  auto const src_object_name =
192  examples::MakeRandomObjectName(generator, "object-") + ".txt";
193  auto const dst_object_name =
194  examples::MakeRandomObjectName(generator, "object-") + ".txt";
195 
196  auto meta_1 =
197  client.InsertObject(bucket_name, src_object_name, kText).value();
198  auto meta_2 =
199  client.InsertObject(bucket_name, src_object_name, kText).value();
200 
201  std::cout << "\nRunning the CopyVersionedObject() example" << std::endl;
202  CopyVersionedObject(
203  client, {meta_1.bucket(), meta_1.name(), bucket_name, dst_object_name,
204  std::to_string(meta_1.generation())});
205 
206  std::cout << "\nRunning the Delete VersionedObject() example [1]"
207  << std::endl;
208  DeleteVersionedObject(client, {meta_1.bucket(), meta_1.name(),
209  std::to_string(meta_1.generation())});
210 
211  std::cout << "\nRunning the Delete VersionedObject() example [2]"
212  << std::endl;
213  DeleteVersionedObject(client, {meta_2.bucket(), meta_2.name(),
214  std::to_string(meta_2.generation())});
215 
216  std::cout << "\nRunning the DisableObjectVersioning() example" << std::endl;
217  DisableObjectVersioning(client, {bucket_name});
218 
219  std::cout << "\nRunning the GetObjectVersioning() example [2]" << std::endl;
220  GetObjectVersioning(client, {bucket_name});
221 
222  (void)client.DeleteObject(bucket_name, dst_object_name);
223  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
224  (void)examples::RemoveBucketAndContents(client, bucket_name);
225 }
