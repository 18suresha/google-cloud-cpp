48 /**
49  * Defines a `std::basic_istream<char>` to read from a GCS Object.
50  */
51 class ObjectReadStream : public std::basic_istream<char> {
52  public:
53  /**
54  * Creates a stream not associated with any buffer.
55  *
56  * Attempts to use this stream will result in failures.
57  */
58  ObjectReadStream() : std::basic_istream<char>(nullptr), buf_() {}
59 
60  /**
61  * Creates a stream associated with the given `streambuf`.
62  */
63  explicit ObjectReadStream(std::unique_ptr<internal::ObjectReadStreambuf> buf)
64  : std::basic_istream<char>(nullptr), buf_(std::move(buf)) {
65  // Initialize the basic_ios<> class
66  init(buf_.get());
67  }
68 
69  ObjectReadStream(ObjectReadStream&& rhs) noexcept
70  : ObjectReadStream(std::move(rhs.buf_)) {
71  // We cannot use set_rdbuf() because older versions of libstdc++ do not
72  // implement this function. Unfortunately `move()` resets `rdbuf()`, and
73  // `rdbuf()` resets the state, so we have to manually copy the rest of
74  // the state.
75  setstate(rhs.rdstate());
76  copyfmt(rhs);
77  rhs.rdbuf(nullptr);
78  }
79 
80  ObjectReadStream& operator=(ObjectReadStream&& rhs) noexcept {
81  buf_ = std::move(rhs.buf_);
82  // Use rdbuf() (instead of set_rdbuf()) because older versions of libstdc++
83  // do not implement this function. Unfortunately `rdbuf()` resets the state,
84  // and `move()` resets `rdbuf()`, so we have to manually copy the rest of
85  // the state.
86  rdbuf(buf_.get());
87  setstate(rhs.rdstate());
88  copyfmt(rhs);
89  rhs.rdbuf(nullptr);
90  return *this;
91  }
92 
93  ObjectReadStream(ObjectReadStream const&) = delete;
94  ObjectReadStream& operator=(ObjectReadStream const&) = delete;
95 
96  /// Closes the stream (if necessary).
97  ~ObjectReadStream() override;
98 
99  bool IsOpen() const { return (bool)buf_ && buf_->IsOpen(); }
100 
101  /**
102  * Terminate the download, possibly before completing it.
103  */
104  void Close();
105 
106  //@{
107  /**
108  * Report any download errors.
109  *
110  * Note that errors may go undetected until the download completes.
111  */
112  Status const& status() const& { return buf_->status(); }
113 
114  /**
115  * The received CRC32C checksum and the MD5 hash values as reported by GCS.
116  *
117  * When the download is finalized (via `Close()` or the end of file) the GCS
118  * server reports the CRC32C checksum and, except for composite objects, the
119  * MD5 hash of the data. This class compares the locally computed and received
120  * hashes so applications can detect data download errors.
121  *
122  * The values are reported as comma separated `tag=value` pairs, e.g.
123  * `crc32c=AAAAAA==,md5=1B2M2Y8AsgTpgAmY7PhCfg==`. The format of this string
124  * is subject to change without notice, they are provided for informational
125  * purposes only.
126  *
127  * @see https://cloud.google.com/storage/docs/hashes-etags for more
128  * information on checksums and hashes in GCS.
129  */
130  std::string const& received_hash() const { return buf_->received_hash(); }
131 
132  /**
133  * The locally computed checksum and hashes, as a string.
134  *
135  * This object computes the CRC32C checksum and MD5 hash of the downloaded
136  * data. Note that there are several cases where these values may be empty or
137  * irrelevant, for example:
138  * - When reading only a portion of a blob the hash of that portion is
139  * irrelevant, note that GCS only reports the hashes for the full blob.
140  * - The application may disable the CRC32C and/or the MD5 hash computation.
141  *
142  * The string has the same format as the value returned by `received_hash()`.
143  * Note that the format of this string is also subject to change without
144  * notice.
145  *
146  * @see https://cloud.google.com/storage/docs/hashes-etags for more
147  * information on checksums and hashes in GCS.
148  */
149  std::string const& computed_hash() const { return buf_->computed_hash(); }
150 
151  /// The headers returned by the service, for debugging only.
152  std::multimap<std::string, std::string> const& headers() const {
153  return buf_->headers();
154  }
155  //@}
156 
157  private:
158  std::unique_ptr<internal::ObjectReadStreambuf> buf_;
159 };
