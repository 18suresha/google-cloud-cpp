69 std::size_t DefaultConnectionPoolSize() {
70  std::size_t nthreads = std::thread::hardware_concurrency();
71  if (nthreads == 0) {

136 void ClientOptions::SetupFromEnvironment() {
137  auto enable_clog =
138  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_CLOG");
139  if (enable_clog.has_value()) {
140  google::cloud::LogSink::EnableStdClog();
141  }
142  // This is overkill right now, eventually we will have different components
143  // that can be traced (http being the first), so we parse the environment
144  // variable.
145  auto tracing =
146  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_TRACING");
147  if (tracing.has_value()) {
148  std::set<std::string> const enabled = absl::StrSplit(*tracing, ',');
149  if (enabled.end() != enabled.find("http")) {
150  GCP_LOG(INFO) << "Enabling logging for http";
151  set_enable_http_tracing(true);
152  }
153  if (enabled.end() != enabled.find("raw-client")) {
154  GCP_LOG(INFO) << "Enabling logging for RawClient functions";
155  set_enable_raw_client_tracing(true);
156  }
157  }
158 
159  auto project_id = google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT");
160  if (project_id.has_value()) {
161  project_id_ = std::move(*project_id);
162  }
163 }
