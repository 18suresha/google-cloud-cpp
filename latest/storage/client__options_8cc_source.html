149 Options DefaultOptions(std::shared_ptr<oauth2::Credentials> credentials,
150  Options opts) {
151  auto o =
152  Options{}
153  .set<Oauth2CredentialsOption>(std::move(credentials))
154  .set<RestEndpointOption>("https://storage.googleapis.com")
155  .set<IamEndpointOption>("https://iamcredentials.googleapis.com/v1")
156  .set<TargetApiVersionOption>("v1")
157  .set<ConnectionPoolSizeOption>(DefaultConnectionPoolSize())
158  .set<DownloadBufferSizeOption>(
159  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_DOWNLOAD_BUFFER_SIZE)
160  .set<UploadBufferSizeOption>(
161  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_UPLOAD_BUFFER_SIZE)
162  .set<MaximumSimpleUploadSizeOption>(
163  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_MAXIMUM_SIMPLE_UPLOAD_SIZE)
164  .set<EnableCurlSslLockingOption>(true)
165  .set<EnableCurlSigpipeHandlerOption>(true)
166  .set<MaximumCurlSocketRecvSizeOption>(0)
167  .set<MaximumCurlSocketSendSizeOption>(0)
168  .set<DownloadStallTimeoutOption>(std::chrono::seconds(
169  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_DOWNLOAD_STALL_TIMEOUT))
170  .set<RetryPolicyOption>(
171  LimitedTimeRetryPolicy(
172  STORAGE_CLIENT_DEFAULT_MAXIMUM_RETRY_PERIOD)
173  .clone())
174  .set<BackoffPolicyOption>(
175  ExponentialBackoffPolicy(
176  STORAGE_CLIENT_DEFAULT_INITIAL_BACKOFF_DELAY,
177  STORAGE_CLIENT_DEFAULT_MAXIMUM_BACKOFF_DELAY,
178  STORAGE_CLIENT_DEFAULT_BACKOFF_SCALING)
179  .clone())
180  .set<IdempotencyPolicyOption>(AlwaysRetryIdempotencyPolicy().clone());
181 
182  o = google::cloud::internal::MergeOptions(std::move(opts), std::move(o));
183  auto emulator = GetEmulator();
184  if (emulator.has_value()) {
185  o.set<RestEndpointOption>(*emulator).set<IamEndpointOption>(*emulator +
186  "/iamapi");
187  }
188 
189  auto tracing =
190  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_TRACING");
191  if (tracing.has_value()) {
192  std::set<std::string> const enabled = absl::StrSplit(*tracing, ',');
193  if (enabled.end() != enabled.find("http")) {
194  GCP_LOG(INFO) << "Enabling logging for http";
195  o.lookup<TracingComponentsOption>().insert("http");
196  }
197  if (enabled.end() != enabled.find("raw-client")) {
198  GCP_LOG(INFO) << "Enabling logging for RawClient functions";
199  o.lookup<TracingComponentsOption>().insert("raw-client");
200  }
201  }
202 
203  auto project_id = google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT");
204  if (project_id.has_value()) {
205  o.set<ProjectIdOption>(std::move(*project_id));
206  }
207 
208  return o;
209 }
