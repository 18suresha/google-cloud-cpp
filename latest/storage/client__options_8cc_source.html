52 std::size_t DefaultConnectionPoolSize() {
53  std::size_t nthreads = std::thread::hardware_concurrency();
54  if (nthreads == 0) {
55  return 4;
56  }
57  return 4 * nthreads;

115 Options DefaultOptions(std::shared_ptr<oauth2::Credentials> credentials,
116  Options opts) {
117  auto o =
118  Options{}
119  .set<Oauth2CredentialsOption>(std::move(credentials))
120  .set<GcsRestEndpointOption>("https://storage.googleapis.com")
121  .set<GcsIamEndpointOption>("https://iamcredentials.googleapis.com/v1")
122  .set<TargetApiVersionOption>("v1")
123  .set<ConnectionPoolSizeOption>(DefaultConnectionPoolSize())
124  .set<DownloadBufferSizeOption>(
125  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_DOWNLOAD_BUFFER_SIZE)
126  .set<UploadBufferSizeOption>(
127  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_UPLOAD_BUFFER_SIZE)
128  .set<MaximumSimpleUploadSizeOption>(
129  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_MAXIMUM_SIMPLE_UPLOAD_SIZE)
130  .set<EnableCurlSslLockingOption>(true)
131  .set<EnableCurlSigpipeHandlerOption>(true)
132  .set<MaximumCurlSocketRecvSizeOption>(0)
133  .set<MaximumCurlSocketSendSizeOption>(0)
134  .set<DownloadStallTimeoutOption>(std::chrono::seconds(
135  GOOGLE_CLOUD_CPP_STORAGE_DEFAULT_DOWNLOAD_STALL_TIMEOUT));
136 
137  o = google::cloud::internal::MergeOptions(std::move(opts), std::move(o));
138  auto emulator = GetEmulator();
139  if (emulator.has_value()) {
140  o.set<GcsRestEndpointOption>(*emulator).set<GcsIamEndpointOption>(
141  *emulator + "/iamapi");
142  }
143 
144  auto tracing =
145  google::cloud::internal::GetEnv("CLOUD_STORAGE_ENABLE_TRACING");
146  if (tracing.has_value()) {
147  std::set<std::string> const enabled = absl::StrSplit(*tracing, ',');
148  if (enabled.end() != enabled.find("http")) {
149  GCP_LOG(INFO) << "Enabling logging for http";
150  o.lookup<TracingComponentsOption>().insert("http");
151  }
152  if (enabled.end() != enabled.find("raw-client")) {
153  GCP_LOG(INFO) << "Enabling logging for RawClient functions";
154  o.lookup<TracingComponentsOption>().insert("raw-client");
155  }
156  }
157 
158  auto project_id = google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT");
159  if (project_id.has_value()) {
160  o.set<ProjectIdOption>(std::move(*project_id));
161  }
162 
163  return o;
164 }
