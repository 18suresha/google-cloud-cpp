24 void GetBilling(google::cloud::storage::Client client,
25  std::vector<std::string> const& argv) {
27  namespace gcs = google::cloud::storage;
28  using ::google::cloud::StatusOr;
29  [](gcs::Client client, std::string const& bucket_name,
30  std::string const& user_project) {
31  StatusOr<gcs::BucketMetadata> metadata =
32  client.GetBucketMetadata(bucket_name, gcs::UserProject(user_project));
33  if (!metadata) throw std::runtime_error(metadata.status().message());
34 
35  if (!metadata->has_billing()) {
36  std::cout
37  << "The bucket " << metadata->name() << " does not have a"
38  << " billing configuration. The default applies, i.e., the project"
39  << " that owns the bucket pays for the requests.\n";
40  return;
41  }
42 
43  if (metadata->billing().requester_pays) {
44  std::cout
45  << "The bucket " << metadata->name()
46  << " is configured to charge the calling project for the requests.\n";
47  } else {
48  std::cout << "The bucket " << metadata->name()
49  << " is configured to charge the project that owns the bucket "
50  "for the requests.\n";
51  }
52  }
54  (std::move(client), argv.at(0), argv.at(1));
55 }

57 void EnableRequesterPays(google::cloud::storage::Client client,

63  StatusOr<gcs::BucketMetadata> metadata = client.PatchBucket(
64  bucket_name,

66  if (!metadata) throw std::runtime_error(metadata.status().message());
67 
68  std::cout << "Billing configuration for bucket " << metadata->name()
69  << " is updated. The bucket now";

74  } else {

111 void WriteObjectRequesterPays(google::cloud::storage::Client client,
112  std::vector<std::string> const& argv) {

115  using ::google::cloud::StatusOr;
116  [](gcs::Client client, std::string const& bucket_name,
117  std::string const& object_name, std::string const& billed_project) {
118  gcs::ObjectWriteStream stream = client.WriteObject(
119  bucket_name, object_name, gcs::UserProject(billed_project));
120  for (int lineno = 0; lineno != 10; ++lineno) {
121  // Add 1 to the counter, because it is conventional to number lines
122  // starting at 1.
123  stream << (lineno + 1) << ": I will write better examples\n";
124  }
125  stream.Close();
126 
127  StatusOr<gcs::ObjectMetadata> metadata = std::move(stream).metadata();
128  if (!metadata) throw std::runtime_error(metadata.status().message());
129  std::cout << "Successfully wrote to object " << metadata->name()
130  << " its size is: " << metadata->size()
131  << "\nFull metadata: " << *metadata << "\n";
132  }
134  (std::move(client), argv.at(0), argv.at(1), argv.at(2));
135 }

171 void RunAll(std::vector<std::string> const& argv) {
172  namespace examples = ::google::cloud::storage::examples;
173  namespace gcs = ::google::cloud::storage;
174 
175  if (!argv.empty()) throw examples::Usage{"auto"};
176  examples::CheckEnvironmentVariablesAreSet({
177  "GOOGLE_CLOUD_PROJECT",
178  });
179  auto const project_id =
180  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
181  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
182  auto const bucket_name = examples::MakeRandomBucketName(generator);
183  auto const object_name =
184  examples::MakeRandomObjectName(generator, "object-") + ".txt";
185  auto client = gcs::Client::CreateDefaultClient().value();
186 
187  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
188  << std::endl;
189  (void)client
190  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
191  .value();
192  // In GCS a single project cannot create or delete buckets more often than
193  // once every two seconds. We will pause until that time before deleting the
194  // bucket.
195  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
196 
197  std::cout << "\nRunning GetBilling() example [1]" << std::endl;
198  GetBilling(client, {bucket_name, project_id});
199 
200  std::cout << "\nRunning EnableRequesterPays() example" << std::endl;
201  EnableRequesterPays(client, {bucket_name});
202 
203  std::cout << "\nRunning GetBilling() example [2]" << std::endl;
204  GetBilling(client, {bucket_name, project_id});
205 
206  std::cout << "\nRunning WriteObjectRequesterPays() example" << std::endl;
207  WriteObjectRequesterPays(client, {bucket_name, object_name, project_id});
208 
209  std::cout << "\nRunning ReadObjectRequesterPays() example" << std::endl;
210  ReadObjectRequesterPays(client, {bucket_name, object_name, project_id});
211 
212  std::cout << "\nRunning DeleteObjectRequesterPays() example" << std::endl;
213  DeleteObjectRequesterPays(client, {bucket_name, object_name, project_id});
214 
215  std::cout << "\nRunning DisableRequesterPays() example" << std::endl;
216  DisableRequesterPays(client, {bucket_name, project_id});
217 
218  std::cout << "\nRunning GetBilling() example [3]" << std::endl;
219  GetBilling(client, {bucket_name, project_id});
220 
221  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
222  (void)examples::RemoveBucketAndContents(client, bucket_name);
223 }
