28  // Create aliases to make the code easier to read.

31  // Create a client to communicate with Google Cloud Storage. This client
32  // uses the default configuration for authentication and project id.
33  google::cloud::StatusOr<gcs::Client> client =
34  gcs::Client::CreateDefaultClient();
35  if (!client) {
36  std::cerr << "Failed to create Storage Client, status=" << client.status()
37  << "\n";
38  return 1;
39  }
40 
41  google::cloud::StatusOr<gcs::BucketMetadata> bucket_metadata =
42  client->CreateBucketForProject(
43  bucket_name, project_id,
44  gcs::BucketMetadata()
45  .set_location("us-east1")
46  .set_storage_class(gcs::storage_class::Regional()));
47 
48  if (!bucket_metadata) {

50  << ", status=" << bucket_metadata.status() << "\n";
51  return 1;
52  }
53 
54  std::cout << "Created bucket " << bucket_metadata->name() << "\n";
