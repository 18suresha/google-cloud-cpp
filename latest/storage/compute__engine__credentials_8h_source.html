37 /**
38  * Wrapper class for Google OAuth 2.0 GCE instance service account credentials.
39  *
40  * Takes a service account email address or alias (e.g. "default") and uses the
41  * Google Compute Engine instance's metadata server to obtain service account
42  * metadata and OAuth 2.0 access tokens as needed. Instances of this class
43  * should usually be created via the convenience methods declared in
44  * google_credentials.h.
45  *
46  * An HTTP Authorization header, with an access token as its value, can be
47  * obtained by calling the AuthorizationHeader() method; if the current access
48  * token is invalid or nearing expiration, this will class will first obtain a
49  * new access token before returning the Authorization header string.
50  *
51  * @see https://cloud.google.com/compute/docs/authentication#using for details
52  * on how to get started with Compute Engine service account credentials.
53  *
54  * @tparam HttpRequestBuilderType a dependency injection point. It makes it
55  * possible to mock internal libcurl wrappers. This should generally not
56  * be overridden except for testing.
57  */
58 template <typename HttpRequestBuilderType =
59  storage::internal::CurlRequestBuilder>
60 class ComputeEngineCredentials : public Credentials {
61  public:
62  explicit ComputeEngineCredentials() : ComputeEngineCredentials("default") {}
63 
64  explicit ComputeEngineCredentials(std::string const& service_account_email)
65  : service_account_email_(service_account_email) {}
66 
67  StatusOr<std::string> AuthorizationHeader() override {
68  std::unique_lock<std::mutex> lock(mu_);
69  return refreshing_creds_.AuthorizationHeader([this] { return Refresh(); });
70  }
71 
72  /**
73  * Returns the email or alias of this credential's service account.
74  *
75  * @note This class must query the Compute Engine instance's metadata server
76  * to fetch service account metadata. Because of this, if an alias (e.g.
77  * "default") was supplied in place of an actual email address when
78  * initializing this credential, that alias is returned as this credential's
79  * email address if the credential has not been refreshed yet.
80  */
81  std::string service_account_email() {
82  std::unique_lock<std::mutex> lock(mu_);
83  return service_account_email_;
84  }
85 
86  /**
87  * Returns the set of scopes granted to this credential's service account.
88  *
89  * @note Because this class must query the Compute Engine instance's metadata
90  * server to fetch service account metadata, this method will return an empty
91  * set if the credential has not been refreshed yet.
92  */
93  std::set<std::string> scopes() {
94  std::unique_lock<std::mutex> lock(mu_);
95  return scopes_;
96  }
97 
98  private:
99  /**
100  * Sends an HTTP GET request to the GCE metadata server.
101  *
102  * @see https://cloud.google.com/compute/docs/storing-retrieving-metadata for
103  * an overview of retrieving information from the GCE metadata server.
104  */
105  StatusOr<storage::internal::HttpResponse> DoMetadataServerGetRequest(
106  std::string path, bool recursive) {
107  // Allows mocking the metadata server hostname for testing.
108  std::string metadata_server_hostname =
109  google::cloud::storage::internal::GceMetadataHostname();
110 
111  HttpRequestBuilderType request_builder(
112  std::move("http://" + metadata_server_hostname + path),
113  storage::internal::GetDefaultCurlHandleFactory());
114  request_builder.AddHeader("metadata-flavor: Google");
115  if (recursive) {
116  request_builder.AddQueryParameter("recursive", "true");
117  }
118  return request_builder.BuildRequest().MakeRequest(std::string{});
119  }
120 
121  /**
122  * Fetches metadata for an instance's service account.
123  *
124  * @see
125  * https://cloud.google.com/compute/docs/access/create-enable-service-accounts-for-instances
126  * for more details.
127  */
128  Status RetrieveServiceAccountInfo() {
129  namespace nl = google::cloud::storage::internal::nl;
130  auto response = DoMetadataServerGetRequest(
131  "/computeMetadata/v1/instance/service-accounts/" +
132  service_account_email_ + "/",
133  true);
134  if (!response) {
135  return std::move(response).status();
136  }
137  if (response->status_code >= 300) {
138  return AsStatus(*response);
139  }
140 
141  nl::json response_body = nl::json::parse(response->payload, nullptr, false);
142  // Note that the "scopes" attribute will always be present and contain a
143  // JSON array. At minimum, for the request to succeed, the instance must
144  // have been granted the scope that allows it to retrieve info from the
145  // metadata server.
146  if (response_body.is_discarded() or response_body.count("email") == 0U or
147  response_body.count("scopes") == 0U) {
148  response->payload +=
149  "Could not find all required fields in response (email, scopes).";
150  return AsStatus(*response);
151  }
152 
153  std::string email = response_body.value("email", "");
154  std::set<std::string> scopes_set = response_body["scopes"];
155 
156  // Do not update any state until all potential exceptions are raised.
157  service_account_email_ = email;
158  scopes_ = scopes_set;
159  return Status();
160  }
161 
162  Status Refresh() {
163  namespace nl = storage::internal::nl;
164 
165  auto status = RetrieveServiceAccountInfo();
166  if (!status.ok()) {
167  return status;
168  }
169 
170  auto response = DoMetadataServerGetRequest(
171  "/computeMetadata/v1/instance/service-accounts/" +
172  service_account_email_ + "/token",
173  false);
174  if (!response) {
175  return std::move(response).status();
176  }
177  if (response->status_code >= 300) {
178  return AsStatus(*response);
179  }
180 
181  // Response should have the attributes "access_token", "expires_in", and
182  // "token_type".
183  nl::json access_token = nl::json::parse(response->payload, nullptr, false);
184  if (access_token.is_discarded() ||
185  access_token.count("access_token") == 0U or
186  access_token.count("expires_in") == 0U or
187  access_token.count("token_type") == 0U) {
188  response->payload +=
189  "Could not find all required fields in response (access_token,"
190  " expires_in, token_type).";
191  return AsStatus(*response);
192  }
193  std::string header = "Authorization: ";
194  header += access_token.value("token_type", "");
195  header += ' ';
196  header += access_token.value("access_token", "");
197  auto expires_in =
198  std::chrono::seconds(access_token.value("expires_in", int(0)));
199  auto new_expiration = std::chrono::system_clock::now() + expires_in;
200 
201  // Do not update any state until all potential exceptions are raised.
202  refreshing_creds_.authorization_header = std::move(header);
203  refreshing_creds_.expiration_time = new_expiration;
204  return Status();
205  }
206 
207  mutable std::mutex mu_;
208  RefreshingCredentialsWrapper refreshing_creds_;
209  std::set<std::string> scopes_;
210  std::string service_account_email_;
211 };
