40  template <typename RefreshFunctor>
41  StatusOr<std::string> AuthorizationHeader(
42  std::chrono::system_clock::time_point now,
43  RefreshFunctor refresh_fn) const {
44  if (IsValid(now)) {
45  return temporary_token.token;
46  }
47 
48  StatusOr<TemporaryToken> new_token = refresh_fn();
49  if (new_token) {
50  temporary_token = *std::move(new_token);
51  return temporary_token.token;
52  }
53  return new_token.status();
54  }

56  /**
57  * Returns whether the current access token should be considered expired.
58  *
59  * When determining if a Credentials object needs to be refreshed, the IsValid
60  * method should be used instead; there may be cases where a Credentials is
61  * not expired but should be considered invalid.
62  *
63  * If a Credentials is close to expiration but not quite expired, this method
64  * may still return false. This helps prevent the case where an access token
65  * expires between when it is obtained and when it is used.
66  */
67  bool IsExpired(std::chrono::system_clock::time_point now) const;

69  /**

75  bool IsValid(std::chrono::system_clock::time_point now) const;
