101 StatusOr<ObjectMetadata> Client::UploadFileResumable(
102  std::string const& file_name,
103  google::cloud::storage::internal::ResumableUploadRequest const& request) {
104  auto status = google::cloud::internal::status(file_name);
105  if (!is_regular(status)) {
106  GCP_LOG(WARNING) << "Trying to upload " << file_name
107  << R"""( which is not a regular file.
108 This is often a problem because:
109  - Some non-regular files are infinite sources of data, and the load will
110  never complete.
111  - Some non-regular files can only be read once, and UploadFile() may need to
112  read the file more than once to compute the checksum and hashes needed to
113  preserve data integrity.
114 
115 Consider using Client::WriteObject() instead. You may also need to disable data
116 integrity checks using the DisableMD5Hash() and DisableCrc32cChecksum() options.
117 )""";
118  }
119 
120  std::ifstream source(file_name);
121  if (!source.is_open()) {
122  std::ostringstream os;
123  os << __func__ << "(" << request << ", " << file_name
124  << "): cannot open upload file source";
125  return Status(StatusCode::kNotFound, std::move(os).str());
126  }
127  // This function only works for regular files, and the `storage::Client`
128  // class checks before calling it.
129  std::uint64_t source_size = google::cloud::internal::file_size(file_name);
130 
131  return UploadStreamResumable(source, source_size, request);
132 }

134 StatusOr<ObjectMetadata> Client::UploadStreamResumable(
135  std::istream& source, std::uint64_t source_size,
136  internal::ResumableUploadRequest const& request) {
137  StatusOr<std::unique_ptr<internal::ResumableUploadSession>> session_status =
138  raw_client()->CreateResumableSession(request);
139  if (!session_status) {
140  return std::move(session_status).status();
141  }
142 
143  auto session = std::move(*session_status);
144 
145  // GCS requires chunks to be a multiple of 256KiB.
146  auto chunk_size = internal::UploadChunkRequest::RoundUpToQuantum(
147  raw_client()->client_options().upload_buffer_size());
148 
149  StatusOr<internal::ResumableUploadResponse> upload_response(
150  internal::ResumableUploadResponse{});
151  // We iterate while `source` is good and the retry policy has not been
152  // exhausted.
153  while (!source.eof() && upload_response && upload_response->payload.empty()) {
154  // Read a chunk of data from the source file.
155  std::string buffer(chunk_size, '\0');
156  source.read(&buffer[0], buffer.size());
157  auto gcount = static_cast<std::size_t>(source.gcount());
158  if (gcount < buffer.size()) {
159  source_size = session->next_expected_byte() + gcount;
160  }
161  buffer.resize(gcount);
162 
163  auto expected = session->next_expected_byte() + gcount - 1;
164  upload_response = session->UploadChunk(buffer, source_size);
165  if (!upload_response) {
166  return std::move(upload_response).status();
167  }
168  if (session->next_expected_byte() != expected) {
169  GCP_LOG(WARNING) << "unexpected last committed byte "
170  << " expected=" << expected
171  << " got=" << session->next_expected_byte();
172  source.seekg(session->next_expected_byte(), std::ios::beg);
173  }
174  }
175 
176  if (!upload_response) {
177  return std::move(upload_response).status();
178  }
179 
180  return internal::ObjectMetadataParser::FromString(upload_response->payload);
181 }

183 Status Client::DownloadFileImpl(internal::ReadObjectRangeRequest const& request,
184  std::string const& file_name) {
185  auto streambuf = raw_client_->ReadObject(request);
186  if (!streambuf) {
187  return streambuf.status();
188  }
189  // Open the download stream and immediately raise an exception on failure.
190  ObjectReadStream stream(*std::move(streambuf));
191 
192  auto report_error = [&request, file_name](char const* func, char const* what,
193  Status const& status) {
194  std::ostringstream msg;
195  msg << func << "(" << request << ", " << file_name << "): " << what
196  << " - status.message=" << status.message();
197  return Status(status.code(), std::move(msg).str());
198  };
199  if (!stream.status().ok()) {
200  return report_error(__func__, "cannot open download source object",
201  stream.status());
202  }
203 
204  // Open the destination file, and immediate raise an exception on failure.
205  std::ofstream os(file_name);
206  if (!os.is_open()) {
207  return report_error(
208  __func__, "cannot open download destination file",
209  Status(StatusCode::kInvalidArgument, "ofstream::open()"));
210  }
211 
212  std::string buffer;
213  buffer.resize(raw_client_->client_options().download_buffer_size(), '\0');
214  do {
215  stream.read(&buffer[0], buffer.size());
216  os.write(buffer.data(), stream.gcount());
217  } while (os.good() && stream.good());
218  os.close();
219  if (!os.good()) {
220  return report_error(__func__, "cannot close download destination file",
221  Status(StatusCode::kUnknown, "ofstream::close()"));
222  }
223  if (!stream.status().ok()) {
224  return report_error(__func__, "error reading download source object",
225  stream.status());
226  }
227  return Status();
228 }

250 StatusOr<std::string> Client::SignUrlV2(
251  internal::V2SignUrlRequest const& request) {
252  SigningAccount signing_account; // TODO(#1595) - = request.signing_account()
253  auto signed_blob = SignBlobImpl(signing_account, request.StringToSign());
254  if (!signed_blob) {
255  return signed_blob.status();
256  }
257 
258  internal::CurlHandle curl;
259  auto encoded = internal::Base64Encode(signed_blob->signed_blob);
260  std::string signature = curl.MakeEscapedString(encoded).get();
261 
262  std::ostringstream os;
263  os << "https://storage.googleapis.com/" << request.bucket_name();
264  if (!request.object_name().empty()) {
265  os << '/' << curl.MakeEscapedString(request.object_name()).get();
266  }
267  os << "?GoogleAccessId=" << SigningEmail(signing_account)
268  << "&Expires=" << request.expiration_time_as_seconds().count()
269  << "&Signature=" << signature;
270 
271  return std::move(os).str();
272 }
