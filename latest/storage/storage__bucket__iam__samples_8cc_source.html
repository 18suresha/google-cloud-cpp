216 void NativeRemoveBucketConditionalIamBinding(
217  google::cloud::storage::Client client,
218  std::vector<std::string> const& argv) {
219  // [START storage_remove_bucket_conditional_iam_binding]
221  namespace gcs = google::cloud::storage;
222  using ::google::cloud::StatusOr;
223  [](gcs::Client client, std::string const& bucket_name,
224  std::string const& role, std::string const& condition_title,
225  std::string const& condition_description,
226  std::string const& condition_expression) {
227  auto policy = client.GetNativeBucketIamPolicy(
228  bucket_name, gcs::RequestedPolicyVersion(3));
229  if (!policy) throw std::runtime_error(policy.status().message());
230 
231  policy->set_version(3);
232  auto& bindings = policy->bindings();
233  auto e = std::remove_if(
234  bindings.begin(), bindings.end(),
235  [role, condition_title, condition_description,
236  condition_expression](gcs::NativeIamBinding b) {
237  return (b.role() == role && b.has_condition() &&
238  b.condition().title() == condition_title &&
239  b.condition().description() == condition_description &&
240  b.condition().expression() == condition_expression);
241  });
242  if (e == bindings.end()) {
243  std::cout << "No matching binding group found.\n";
244  return;
245  }
246  bindings.erase(e);
247  auto updated = client.SetNativeBucketIamPolicy(bucket_name, *policy);
248  if (!updated) throw std::runtime_error(updated.status().message());
249 
250  std::cout << "Conditional binding was removed.\n";
251  }
253  // [END storage_remove_bucket_conditional_iam_binding]
254  (std::move(client), argv.at(0), argv.at(1), argv.at(2), argv.at(3),
255  argv.at(4));

292 void SetBucketPublicIam(google::cloud::storage::Client client,
293  std::vector<std::string> const& argv) {
294  // [START storage_set_bucket_public_iam]
295  namespace gcs = google::cloud::storage;
296  using google::cloud::StatusOr;
297  [](gcs::Client client, std::string const& bucket_name) {
298  StatusOr<google::cloud::IamPolicy> current_policy =
299  client.GetBucketIamPolicy(bucket_name);
300 
301  if (!current_policy) {
302  throw std::runtime_error(current_policy.status().message());
303  }
304 
305  current_policy->bindings.AddMember("roles/storage.objectViewer",
306  "allUsers");
307 
308  // Update the policy. Note the use of `gcs::IfMatchEtag` to implement
309  // optimistic concurrency control.
310  StatusOr<google::cloud::IamPolicy> updated_policy =
311  client.SetBucketIamPolicy(bucket_name, *current_policy,
312  gcs::IfMatchEtag(current_policy->etag));
313 
314  if (!updated_policy) {
315  throw std::runtime_error(current_policy.status().message());
316  }
317 
318  auto role = updated_policy->bindings.find("roles/storage.objectViewer");
319  if (role == updated_policy->bindings.end()) {
320  std::cout << "Cannot find 'roles/storage.objectViewer' in the updated"
321  << " policy. This can happen if another application updates"
322  << " the IAM policy at the same time. Please retry the"
323  << " operation.\n";
324  return;
325  }
326  auto member = role->second.find("allUsers");
327  if (member == role->second.end()) {
328  std::cout << "'allUsers' is not a member of the"
329  << " 'roles/storage.objectViewer' role in the updated"
330  << " policy. This can happen if another application updates"
331  << " the IAM policy at the same time. Please retry the"
332  << " operation.\n";
333  return;
334  }
335  std::cout << "IamPolicy successfully updated for bucket " << bucket_name
336  << '\n';
337  }
338  // [END storage_set_bucket_public_iam]
339  (std::move(client), argv.at(0));
340 }
