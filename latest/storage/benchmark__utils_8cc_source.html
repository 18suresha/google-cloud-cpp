42 std::string MakeRandomData(google::cloud::internal::DefaultPRNG& gen,
43  std::size_t desired_size) {
44  std::string result;
45  result.reserve(desired_size);
46 
47  // Create lines of 128 characters to start with, we can fill the remaining
48  // characters at the end.
49  constexpr int kLineSize = 128;
50  auto gen_random_line = [&gen](std::size_t count) {
51  return google::cloud::internal::Sample(gen, static_cast<int>(count - 1),
52  "abcdefghijklmnopqrstuvwxyz"
53  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
54  "012456789"
55  " - _ : /") +
56  "\n";
57  };
58  while (result.size() + kLineSize < desired_size) {
59  result += gen_random_line(kLineSize);
60  }
61  if (result.size() < desired_size) {
62  result += gen_random_line(desired_size - result.size());
63  }
64 
65  return result;
66 }

68 bool EndsWith(std::string const& val, std::string const& suffix) {
69  auto pos = val.rfind(suffix);
70  if (pos == std::string::npos) {
71  return false;
72  }
73  return pos == val.size() - suffix.size();

109 std::chrono::seconds ParseDuration(std::string const& val) {
110  long s = std::stol(val);
111  if (EndsWith(val, "h")) {
112  return s * std::chrono::seconds(std::chrono::hours(1));
113  }
114  if (EndsWith(val, "m")) {
115  return s * std::chrono::seconds(std::chrono::minutes(1));
116  }
117  if (EndsWith(val, "s")) {
118  return s * std::chrono::seconds(std::chrono::seconds(1));
119  }
120  return std::chrono::seconds(s);
121 }

123 bool ParseBoolean(std::string const& val, bool default_value) {
124  if (val.empty()) {
125  return default_value;
126  }
127  auto lower = val;
128  std::transform(lower.begin(), lower.end(), lower.begin(),
129  [](char x) { return static_cast<char>(std::tolower(x)); });
130  if (lower == "true") {
131  return true;
132  } else if (lower == "false") {
133  return false;
134  }
135  google::cloud::internal::ThrowRuntimeError("Cannot parse " + val +
136  " as a boolean");
137 }

159 std::vector<std::string> OptionsParse(std::vector<OptionDescriptor> const& desc,
160  std::vector<std::string> argv) {
161  if (argv.empty()) {
162  return argv;
163  }
164  std::string const usage = BuildUsage(desc, argv[0]);
165 
166  auto next_arg = argv.begin() + 1;
167  while (next_arg != argv.end()) {
168  std::string const& argument = *next_arg;
169 
170  // Try to match `argument` against the options in `desc`
171  bool matched = false;
172  for (auto const& d : desc) {
173  if (argument.rfind(d.option, 0) != 0) {
174  // Not a match, keep searching
175  continue;
176  }
177  std::string val = argument.substr(d.option.size());
178  if (!val.empty() && val[0] != '=') {
179  // Matched a prefix of an option, keep searching.
180  continue;
181  }
182  if (!val.empty()) {
183  // The first character must be '=', remove it too.
184  val.erase(val.begin());
185  }
186  d.parser(val);
187  // This is a match, consume the argument and stop the search.
188  matched = true;
189  break;
190  }
191  // If next_arg is matched against any option erase it, otherwise skip it.
192  next_arg = matched ? argv.erase(next_arg) : next_arg + 1;
193  }
194  return argv;
195 }

204 void SimpleTimer::Stop() {
205  using namespace std::chrono;
206  elapsed_time_ = duration_cast<microseconds>(steady_clock::now() - start_);
207 
208 #if GOOGLE_CLOUD_CPP_HAS_GETRUSAGE
209  auto as_usec = [](timeval const& tv) {
210  return microseconds(seconds(tv.tv_sec)) + microseconds(tv.tv_usec);
211  };
212 
213  struct rusage now {};
214  (void)getrusage(RUSAGE_THREAD, &now);
215  auto utime = as_usec(now.ru_utime) - as_usec(start_usage_.ru_utime);
216  auto stime = as_usec(now.ru_stime) - as_usec(start_usage_.ru_stime);
217  cpu_time_ = utime + stime;
218  double cpu_fraction = 0;
219  if (elapsed_time_.count() != 0) {
220  cpu_fraction =
221  (cpu_time_).count() / static_cast<double>(elapsed_time_.count());
222  }
223  now.ru_minflt -= start_usage_.ru_minflt;
224  now.ru_majflt -= start_usage_.ru_majflt;
225  now.ru_nswap -= start_usage_.ru_nswap;
226  now.ru_inblock -= start_usage_.ru_inblock;
227  now.ru_oublock -= start_usage_.ru_oublock;
228  now.ru_msgsnd -= start_usage_.ru_msgsnd;
229  now.ru_msgrcv -= start_usage_.ru_msgrcv;
230  now.ru_nsignals -= start_usage_.ru_nsignals;
231  now.ru_nvcsw -= start_usage_.ru_nvcsw;
232  now.ru_nivcsw -= start_usage_.ru_nivcsw;
233 
234  std::ostringstream os;
235  os << "# user time =" << utime.count() << " us\n"
236  << "# system time =" << stime.count() << " us\n"
237  << "# CPU fraction =" << cpu_fraction << "\n"
238  << "# maximum resident set size =" << now.ru_maxrss << " KiB\n"
239  << "# integral shared memory size =" << now.ru_ixrss << " KiB\n"
240  << "# integral unshared data size =" << now.ru_idrss << " KiB\n"
241  << "# integral unshared stack size =" << now.ru_isrss << " KiB\n"
242  << "# soft page faults =" << now.ru_minflt << "\n"
243  << "# hard page faults =" << now.ru_majflt << "\n"
244  << "# swaps =" << now.ru_nswap << "\n"
245  << "# block input operations =" << now.ru_inblock << "\n"
246  << "# block output operations =" << now.ru_oublock << "\n"
247  << "# IPC messages sent =" << now.ru_msgsnd << "\n"
248  << "# IPC messages received =" << now.ru_msgrcv << "\n"
249  << "# signals received =" << now.ru_nsignals << "\n"
250  << "# voluntary context switches =" << now.ru_nvcsw << "\n"
251  << "# involuntary context switches =" << now.ru_nivcsw << "\n";
252  annotations_ = std::move(os).str();
253 #endif // GOOGLE_CLOUD_CPP_HAS_GETRUSAGE
254 }
