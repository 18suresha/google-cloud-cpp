32 /**
33  * Defines well-known request headers using the CRTP.
34  *
35  * @tparam H the type we will use to represent the header.
36  * @tparam T the C++ type of the query parameter
37  */
38 template <typename H, typename T>
39 class WellKnownHeader {
40  public:
41  WellKnownHeader() : value_{} {}
42  explicit WellKnownHeader(T value) : value_(std::move(value)) {}
43 
44  char const* header_name() const { return H::header_name(); }
45  bool has_value() const { return value_.has_value(); }
46  T const& value() const { return value_.value(); }
47 
48  private:
49  google::cloud::optional<T> value_;
50 };
51 
52 template <typename H, typename T>
53 std::ostream& operator<<(std::ostream& os, WellKnownHeader<H, T> const& rhs) {
54  if (rhs.has_value()) {
55  return os << rhs.header_name() << ": " << rhs.value();
56  }
57  return os << rhs.header_name() << ": <not set>";
58 }

154 /**
155  * An optional parameter to set the Customer-Supplied Encryption key.
156  *
157  * Application developers can generate their own encryption keys to protect the
158  * data in GCS. This is known as a Customer-Supplied Encryption key (CSEK). If
159  * the application provides a CSEK, GCS does not retain the key. The object
160  * data, the object CRC32 checksum, and its MD5 hash (if applicable) are all
161  * encrypted with this key, and the key is required to read any of these
162  * elements back.
163  *
164  * Care must be taken to save and protect these keys, if lost, the data is not
165  * recoverable. Also, applications should avoid generating predictable keys,
166  * as this weakens the encryption.
167  *
168  * This option is used in read (download), write (upload), copy, and compose
169  * operations. Note that copy and compose operations use the same key for the
170  * source and destination objects.
171  *
172  * @see https://cloud.google.com/storage/docs/encryption/customer-supplied-keys
173  * for a detailed description of how Customer Supplied Encryption keys are
174  * used in GCS.
175  */

245 /**
246  * Creates an encryption key parameter from a pseudo-random number generator.
247  *
248  * @tparam Generator the pseudo-random number generator type, it must meet the
249  * `UniformRandomBitGenerator` requirements.
250  * @param gen the pseudo-random number generator.
251  *
252  * @par Example
253  * @snippet storage_object_samples.cc generate encryption key
254  *
255  * @see https://en.cppreference.com/w/cpp/numeric/random for a general
256  * overview of C++ pseudo-random number support.
257  *
258  * @see
259  * https://en.cppreference.com/w/cpp/numeric/random/UniformRandomBitGenerator
260  * for a more detailed overview of the requirements for generators.
261  */
262 template <typename Generator>
263 static EncryptionKeyData CreateKeyFromGenerator(Generator& gen) {
264  static_assert(
265  std::numeric_limits<unsigned char>::digits == 8,
266  "The Google Cloud Storage C++ library is only supported on platforms\n"
267  "with 8-bit chars. Please file a bug on\n"
268  " https://github.com/googleapis/google-cloud-cpp/issues\n"
269  "describing your platform details to request support for it.");
270  constexpr int kKeySize = 256 / std::numeric_limits<unsigned char>::digits;
271 
272  constexpr auto minchar = (std::numeric_limits<char>::min)();
273  constexpr auto maxchar = (std::numeric_limits<char>::max)();
274  std::uniform_int_distribution<int> uni(minchar, maxchar);
275  std::string key(static_cast<std::size_t>(kKeySize), ' ');
276  std::generate_n(key.begin(), key.size(),
277  [&uni, &gen] { return static_cast<char>(uni(gen)); });
278  return EncryptionDataFromBinaryKey(key);
279 }
