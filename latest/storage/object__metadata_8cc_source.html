112 ObjectMetadataPatchBuilder& ObjectMetadataPatchBuilder::SetAcl(
113  std::vector<ObjectAccessControl> const& v) {
114  if (v.empty()) {
115  return ResetAcl();
116  }
117  std::vector<internal::nl::json> array;
118  array.reserve(v.size());
119  for (auto const& a : v) {
120  array.emplace_back(internal::nl::json{
121  {"entity", a.entity()},
122  {"role", a.role()},
123  });
124  }
125  impl_.SetArrayField("acl", array);
126  return *this;
127 }
