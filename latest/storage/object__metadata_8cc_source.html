118 ObjectMetadataPatchBuilder& ObjectMetadataPatchBuilder::SetAcl(
119  std::vector<ObjectAccessControl> const& v) {
120  if (v.empty()) {
121  return ResetAcl();
122  }
123  std::vector<nlohmann::json> array;
124  array.reserve(v.size());
125  for (auto const& a : v) {
126  array.emplace_back(nlohmann::json{
127  {"entity", a.entity()},
128  {"role", a.role()},
129  });
130  }
131  impl_.SetArrayField("acl", array);
132  return *this;
133 }
