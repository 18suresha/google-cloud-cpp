198 void RemoveObjectOwner(google::cloud::storage::Client client,
199  std::vector<std::string> const& argv) {
201  namespace gcs = google::cloud::storage;
202  using ::google::cloud::StatusOr;
203  [](gcs::Client client, std::string const& bucket_name,
204  std::string const& object_name, std::string const& entity) {
205  StatusOr<gcs::ObjectMetadata> original_metadata = client.GetObjectMetadata(
206  bucket_name, object_name, gcs::Projection::Full());
207 
208  if (!original_metadata) {
209  throw std::runtime_error(original_metadata.status().message());
210  }
211 
212  std::vector<gcs::ObjectAccessControl> original_acl =
213  original_metadata->acl();
214  auto it = std::find_if(original_acl.begin(), original_acl.end(),
215  [entity](const gcs::ObjectAccessControl& entry) {
216  return entry.entity() == entity &&
217  entry.role() ==
218  gcs::ObjectAccessControl::ROLE_OWNER();
219  });
220 
221  if (it == original_acl.end()) {
222  std::cout << "Could not find entity " << entity << " for file "
223  << object_name << " with role OWNER in bucket " << bucket_name
224  << "\n";
225  return;
226  }
227 
228  gcs::ObjectAccessControl owner = *it;
229  google::cloud::Status status =
230  client.DeleteObjectAcl(bucket_name, object_name, owner.entity());
231 
232  if (!status.ok()) throw std::runtime_error(status.message());
233  std::cout << "Deleted ACL entry for " << owner.entity() << " for file "
234  << object_name << " in bucket " << bucket_name << "\n";
235  }
237  (std::move(client), argv.at(0), argv.at(1), argv.at(2));
238 }

240 void RunAll(std::vector<std::string> const& argv) {
241  namespace examples = ::google::cloud::storage::examples;
242  namespace gcs = ::google::cloud::storage;
243 
244  if (!argv.empty()) throw examples::Usage{"auto"};
245  examples::CheckEnvironmentVariablesAreSet({
246  "GOOGLE_CLOUD_PROJECT",
247  "GOOGLE_CLOUD_CPP_STORAGE_TEST_SERVICE_ACCOUNT",
248  });
249  auto const project_id =
250  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
251  auto const service_account =
252  google::cloud::internal::GetEnv(
253  "GOOGLE_CLOUD_CPP_STORAGE_TEST_SERVICE_ACCOUNT")
254  .value();
255  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
256  auto const bucket_name = examples::MakeRandomBucketName(generator);
257  auto const entity = "user-" + service_account;
258  auto client = gcs::Client::CreateDefaultClient().value();
259  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
260  << std::endl;
261  (void)client
262  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
263  .value();
264  // In GCS a single project cannot create or delete buckets more often than
265  // once every two seconds. We will pause until that time before deleting the
266  // bucket.
267  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
268 
269  auto const object_name = examples::MakeRandomObjectName(generator, "object-");
270  auto object_metadata =
271  client
272  .InsertObject(bucket_name, object_name,
273  "some-string-to-serve-as-object-media")
274  .value();
275 
276  auto const reader = gcs::BucketAccessControl::ROLE_READER();
277  auto const owner = gcs::BucketAccessControl::ROLE_OWNER();
278 
279  std::cout << "\nRunning ListObjectAcl() example" << std::endl;
280  ListObjectAcl(client, {bucket_name, object_name});
281 
282  std::cout << "\nRunning CreateObjectAcl() example" << std::endl;
283  CreateObjectAcl(client, {bucket_name, object_name, entity, reader});
284 
285  std::cout << "\nRunning GetObjectAcl() example [1]" << std::endl;
286  GetObjectAcl(client, {bucket_name, object_name, entity});
287 
288  std::cout << "\nRunning UpdateObjectAcl() example" << std::endl;
289  UpdateObjectAcl(client, {bucket_name, object_name, entity, owner});
290 
291  std::cout << "\nRunning PatchObjectAcl() example" << std::endl;
292  PatchObjectAcl(client, {bucket_name, object_name, entity, reader});
293 
294  std::cout << "\nRunning PatchObjectAclNoRead() example" << std::endl;
295  PatchObjectAclNoRead(client, {bucket_name, object_name, entity, owner});
296 
297  std::cout << "\nRunning DeleteObjectAcl() example" << std::endl;
298  DeleteObjectAcl(client, {bucket_name, object_name, entity});
299 
300  std::cout << "\nRunning AddObjectOwner() example" << std::endl;
301  AddObjectOwner(client, {bucket_name, object_name, entity});
302 
303  std::cout << "\nRunning GetObjectAcl() example [2]" << std::endl;
304  GetObjectAcl(client, {bucket_name, object_name, entity});
305 
306  std::cout << "\nRunning RemoveObjectOwner() example" << std::endl;
307  RemoveObjectOwner(client, {bucket_name, object_name, entity});
