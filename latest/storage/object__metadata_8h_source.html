168  /// Delete a metadata entry. This is a no-op if the key does not exist.
169  ObjectMetadata& delete_metadata(std::string const& key) {
170  auto i = metadata_.find(key);
171  if (i == metadata_.end()) {
172  return *this;
173  }
174  metadata_.erase(i);
175  return *this;
176  }
177 
178  /// Insert or update the metadata entry.
179  ObjectMetadata& upsert_metadata(std::string key, std::string value) {
180  auto i = metadata_.lower_bound(key);
181  if (i == metadata_.end() || i->first != key) {
182  metadata_.emplace_hint(i, std::move(key), std::move(value));
183  } else {
184  i->second = std::move(value);
185  }
186  return *this;
187  }
