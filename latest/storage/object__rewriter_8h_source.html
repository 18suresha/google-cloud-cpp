26 /**
27  * Represents the status of a rewrite operation.
28  *
29  * The ObjectRewrite class may require multiple calls to `Iterate()` to finish
30  * the copy. This class represents the progress in a partially completed
31  * rewrite. Applications can use this information to inform users of the
32  * progress and the expected completion time.
33  */
34 struct RewriteProgress {
35  std::uint64_t total_bytes_rewritten;
36  std::uint64_t object_size;
37  bool done;
38 };
39 
40 /**
41  * Complete long running object rewrite operations.
42  *
43  * The `Client::RewriteObject()` operation allows applications to copy objects
44  * across location boundaries, and to rewrite objects with different encryption
45  * keys. In some circumstances it may take multiple calls to the service to
46  * complete a rewrite, this class encapsulates the state of a partial copy.
47  */
48 class ObjectRewriter {
49  public:
50  ObjectRewriter(std::shared_ptr<internal::RawClient> client,
51  internal::RewriteObjectRequest request);
52 
53  /**
54  * Perform one iteration in the rewrite.
55  *
56  * @return The progress after the iteration. If the rewrite has completed the
57  * application can use `Result()` to examine the metadata for the newly
58  * created object.
59  */
60  StatusOr<RewriteProgress> Iterate();
61 
62  /// The current progress on the rewrite operation.
63  StatusOr<RewriteProgress> CurrentProgress() const {
64  if (!last_error_.ok()) {
65  return last_error_;
66  }
67  return progress_;
68  }
69 
70  /**
71  * Iterate until the operation completes using a callback to report progress.
72  *
73  * @note This operation blocks until the copy is finished. For very large
74  * objects that could take substantial time. Applications may need to persist
75  * the rewrite operation. Some applications may want to wrap this call with
76  * `std::async`, and run the copy on a separate thread.
77  *
78  * @return the object metadata once the copy completes.
79  */
80  StatusOr<ObjectMetadata> Result() {
81  return ResultWithProgressCallback([](StatusOr<RewriteProgress> const&) {});
82  }
83 
84  /**
85  * Iterate until the operation completes using a callback to report progress.
86  *
87  * @note This operation blocks until the copy is finished. For very large
88  * objects that could take substantial time. Applications may need to persist
89  * the rewrite operation. Some applications may want to wrap this call with
90  * `std::async`, and run the copy on a separate thread.
91  *
92  * @param cb the callback object.
93  *
94  * @tparam Functor the type of the callback object. It must satisfy:
95  * `std:is_invocable<Functor, StatusOr<RewriteProgress>>:: value == true`.
96  *
97  * @return the object metadata once the copy completes.
98  */
99  template <
100  typename Functor,
101  typename std::enable_if<google::cloud::internal::is_invocable<
102  Functor, StatusOr<RewriteProgress>>::value,
103  int>::type = 0>
104  StatusOr<ObjectMetadata> ResultWithProgressCallback(Functor cb) {
105  while (!progress_.done) {
106  cb(Iterate());
107  }
108  if (!last_error_.ok()) {
109  return last_error_;
110  }
111  return result_;
112  }
113 
114  /**
115  * The current rewrite token.
116  *
117  * Applications can save the token of partially completed rewrites, and
118  * restart those operations using `Client::CopyObjectRestart`, even if the
119  * application has terminated. It is up to the application to preserve
120  * all the other information for the request, including source and destination
121  * buckets, encryption keys, and any preconditions affecting the request.
122  *
123  * @note For rewrites that have not started the token is an empty string.
124  * For rewrites that have completed the token is also an empty string. The
125  * application should preserve other information (such as the
126  * `RewriteProgress`) to avoid repeating a rewrite.
127  */
128  std::string const& token() const { return request_.rewrite_token(); }
129 
130  private:
131  std::shared_ptr<internal::RawClient> client_;
132  internal::RewriteObjectRequest request_;
133  RewriteProgress progress_;
134  ObjectMetadata result_;
135  Status last_error_;
136 };
