184 void RunAll(std::vector<std::string> const& argv) {
185  namespace examples = ::google::cloud::storage::examples;
186  namespace gcs = ::google::cloud::storage;
187 
188  if (!argv.empty()) throw examples::Usage{"auto"};
189  examples::CheckEnvironmentVariablesAreSet({
190  "GOOGLE_CLOUD_PROJECT",
191  "GOOGLE_CLOUD_CPP_STORAGE_TEST_SERVICE_ACCOUNT",
192  });
193  auto const project_id =
194  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
195  auto const service_account =
196  google::cloud::internal::GetEnv(
197  "GOOGLE_CLOUD_CPP_STORAGE_TEST_SERVICE_ACCOUNT")
198  .value();
199  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
200  auto const bucket_name = examples::MakeRandomBucketName(generator);
201  auto const entity = "user-" + service_account;
202  auto client = gcs::Client::CreateDefaultClient().value();
203  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
204  << std::endl;
205  (void)client
206  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
207  .value();
208  // In GCS a single project cannot create or delete buckets more often than
209  // once every two seconds. We will pause until that time before deleting the
210  // bucket.
211  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
212 
213  auto const reader = gcs::BucketAccessControl::ROLE_READER();
214  auto const owner = gcs::BucketAccessControl::ROLE_OWNER();
215 
216  std::cout << "\nRunning ListDefaultObjectAcl() example" << std::endl;
217  ListDefaultObjectAcl(client, {bucket_name});
218 
219  std::cout << "\nRunning CreateDefaultObjectAcl() example" << std::endl;
220  CreateDefaultObjectAcl(client, {bucket_name, entity, reader});
221 
222  std::cout << "\nRunning GetDefaultObjectAcl() example" << std::endl;
223  GetDefaultObjectAcl(client, {bucket_name, entity});
224 
225  std::cout << "\nRunning UpdateDefaultObjectAcl() example" << std::endl;
226  UpdateDefaultObjectAcl(client, {bucket_name, entity, owner});
227 
228  std::cout << "\nRunning PatchDefaultObjectAcl() example" << std::endl;
229  PatchDefaultObjectAcl(client, {bucket_name, entity, reader});
230 
231  std::cout << "\nRunning PatchDefaultObjectAclNoRead() example" << std::endl;
232  PatchDefaultObjectAclNoRead(client, {bucket_name, entity, owner});
233 
234  std::cout << "\nRunning DeleteDefaultObjectAcl() example" << std::endl;
235  DeleteDefaultObjectAcl(client, {bucket_name, entity});
236 
237  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
238  (void)examples::RemoveBucketAndContents(client, bucket_name);
239 }
