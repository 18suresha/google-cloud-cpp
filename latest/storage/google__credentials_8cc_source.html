41 // Parses the JSON or P12 file at `path` and creates the appropriate
42 // Credentials type.
43 //
44 // If `service_account_scopes` or `service_account_subject` are specified, the
45 // file at `path` must be a P12 service account or a JSON service account. If
46 // a different type of credential file is found, this function returns
47 // nullptr to indicate a service account file wasn't found.
48 StatusOr<std::unique_ptr<Credentials>> LoadCredsFromPath(
49  std::string const& path, bool non_service_account_ok,
50  absl::optional<std::set<std::string>> service_account_scopes,
51  absl::optional<std::string> service_account_subject,
52  ChannelOptions const& options) {
53  std::ifstream ifs(path);
54  if (!ifs.is_open()) {
55  // We use kUnknown here because we don't know if the file does not exist, or
56  // if we were unable to open it for some other reason.
57  return Status(StatusCode::kUnknown, "Cannot open credentials file " + path);
58  }
59  std::string contents(std::istreambuf_iterator<char>{ifs}, {});
60  auto cred_json = nlohmann::json::parse(contents, nullptr, false);
61  if (cred_json.is_discarded()) {
62  // This is not a JSON file, try to load it as a P12 service account.
63  auto info = ParseServiceAccountP12File(path);
64  if (!info) {
65  // Ignore the error returned by the P12 parser, because those are too
66  // specific, they typically say "error in PKCS#12" and the application
67  // may not even be trying to load a PKCS#12 file.
68  return Status(StatusCode::kInvalidArgument,
69  "Invalid credentials file " + path);
70  }
71  info->subject = std::move(service_account_subject);
72  info->scopes = std::move(service_account_scopes);
73  auto credentials = absl::make_unique<ServiceAccountCredentials<>>(*info);
74  return std::unique_ptr<Credentials>(std::move(credentials));
75  }
76  std::string cred_type = cred_json.value("type", "no type given");
77  // If non_service_account_ok==false and the cred_type is authorized_user,
78  // we'll return "Unsupported credential type (authorized_user)".
79  if (cred_type == "authorized_user" && non_service_account_ok) {
80  if (service_account_scopes || service_account_subject) {
81  // No ptr indicates that the file we found was not a service account file.
82  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
83  }
84  auto info = ParseAuthorizedUserCredentials(contents, path);
85  if (!info) {
86  return info.status();
87  }
88  std::unique_ptr<Credentials> ptr =
89  absl::make_unique<AuthorizedUserCredentials<>>(*info);
90  return StatusOr<std::unique_ptr<Credentials>>(std::move(ptr));
91  }
92  if (cred_type == "service_account") {
93  auto info = ParseServiceAccountCredentials(contents, path);
94  if (!info) {
95  return info.status();
96  }
97  info->subject = std::move(service_account_subject);
98  info->scopes = std::move(service_account_scopes);
99  std::unique_ptr<Credentials> ptr =
100  absl::make_unique<ServiceAccountCredentials<>>(*info, options);
101  return StatusOr<std::unique_ptr<Credentials>>(std::move(ptr));
102  }
103  return StatusOr<std::unique_ptr<Credentials>>(
104  Status(StatusCode::kInvalidArgument,
105  "Unsupported credential type (" + cred_type +
106  ") when reading Application Default Credentials file from " +
107  path + "."));
108 }

110 // Tries to load the file at the path specified by the value of the Application
111 // Default %Credentials environment variable and to create the appropriate
112 // Credentials type.
113 //
114 // Returns nullptr if the environment variable is not set or the path does not
115 // exist.
116 //
117 // If `service_account_scopes` or `service_account_subject` are specified, the
118 // found file must be a P12 service account or a JSON service account. If a
119 // different type of credential file is found, this function returns nullptr
120 // to indicate a service account file wasn't found.
121 StatusOr<std::unique_ptr<Credentials>> MaybeLoadCredsFromAdcPaths(
122  bool non_service_account_ok,
123  absl::optional<std::set<std::string>> service_account_scopes,
124  absl::optional<std::string> service_account_subject,
125  ChannelOptions const& options = {}) {
126  // 1) Check if the GOOGLE_APPLICATION_CREDENTIALS environment variable is set.
127  auto path = GoogleAdcFilePathFromEnvVarOrEmpty();
128  if (path.empty()) {
129  // 2) If no path was specified via environment variable, check if the
130  // gcloud ADC file exists.
131  path = GoogleAdcFilePathFromWellKnownPathOrEmpty();
132  if (path.empty()) {
133  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
134  }
135  // Just because we had the necessary information to build the path doesn't
136  // mean that a file exists there.
137  std::error_code ec;
138  auto adc_file_status = google::cloud::internal::status(path, ec);
139  if (!google::cloud::internal::exists(adc_file_status)) {
140  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
141  }
142  }
143 
144  // If the path was specified, try to load that file; explicitly fail if it
145  // doesn't exist or can't be read and parsed.
146  return LoadCredsFromPath(path, non_service_account_ok,
147  std::move(service_account_scopes),
148  std::move(service_account_subject), options);
149 }

151 StatusOr<std::shared_ptr<Credentials>> GoogleDefaultCredentials(
152  ChannelOptions const& options) {
153  // 1 and 2) Check if the GOOGLE_APPLICATION_CREDENTIALS environment variable
154  // is set or if the gcloud ADC file exists.
155  auto creds = MaybeLoadCredsFromAdcPaths(true, {}, {}, options);
156  if (!creds) {
157  return StatusOr<std::shared_ptr<Credentials>>(creds.status());
158  }
159  if (*creds) {
160  return StatusOr<std::shared_ptr<Credentials>>(std::move(*creds));
161  }
162 
163  // 3) Check for implicit environment-based credentials (GCE, GAE Flexible,
164  // Cloud Run or GKE Environment).
165  auto gce_creds = std::make_shared<ComputeEngineCredentials<>>();
166  auto override_val =
167  google::cloud::internal::GetEnv(internal::GceCheckOverrideEnvVar());
168  if (override_val.has_value() ? (std::string("1") == *override_val)
169  : gce_creds->AuthorizationHeader().ok()) {
170  return StatusOr<std::shared_ptr<Credentials>>(std::move(gce_creds));
171  }
172 
173  // We've exhausted all search points, thus credentials cannot be constructed.
174  return StatusOr<std::shared_ptr<Credentials>>(
175  Status(StatusCode::kUnknown,
176  "Could not automatically determine credentials. For more "
177  "information, please see " +
178  std::string(kAdcLink)));
179 }

303 StatusOr<std::shared_ptr<Credentials>>
304 CreateServiceAccountCredentialsFromJsonContents(
305  std::string const& contents, absl::optional<std::set<std::string>> scopes,
306  absl::optional<std::string> subject, ChannelOptions const& options) {
307  auto info = ParseServiceAccountCredentials(contents, "memory");
308  if (!info) {
309  return StatusOr<std::shared_ptr<Credentials>>(info.status());
310  }
311  std::chrono::system_clock::time_point now;
312  auto components = AssertionComponentsFromInfo(*info, now);
313  auto jwt_assertion = internal::MakeJWTAssertionNoThrow(
314  components.first, components.second, info->private_key);
315  if (!jwt_assertion) return std::move(jwt_assertion).status();
316 
317  if (!subject.has_value()) {
318  auto constexpr kStorageScope = "https://storage.googleapis.com/";
319  auto const matching_scope = [&] {
320  if (!scopes.has_value()) return true;
321  return scopes->size() == 1 && scopes->count(kStorageScope) == 1;
322  }();
323  if (matching_scope) {
324  return StatusOr<std::shared_ptr<Credentials>>(
325  std::make_shared<internal::SelfSigningServiceAccountCredentials>(
326  internal::SelfSigningServiceAccountCredentialsInfo{
327  /*.client_email*/ std::move(info->client_email),
328  /*.private_key_id*/ std::move(info->private_key_id),
329  /*.private_key*/ std::move(info->private_key),
330  /*.audience*/ kStorageScope}));
331  }
332  }
333  // These are supplied as extra parameters to this method, not in the JSON
334  // file.
335  info->subject = std::move(subject);
336  info->scopes = std::move(scopes);
337  return StatusOr<std::shared_ptr<Credentials>>(
338  std::make_shared<ServiceAccountCredentials<>>(*info, options));
339 }
