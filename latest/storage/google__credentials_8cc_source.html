35 /// Parses the JSON file at `path` and creates the appropriate Credentials type.
36 std::unique_ptr<Credentials> LoadCredsFromPath(std::string const& path) {
37  using google::cloud::internal::ThrowRuntimeError;
38  namespace nl = google::cloud::storage::internal::nl;
39 
40  std::ifstream ifs(path);
41  if (!ifs.is_open()) {
42  ThrowRuntimeError("Cannot open credentials file " + path);
43  }
44  std::string contents(std::istreambuf_iterator<char>{ifs}, {});
45  auto cred_json = nl::json::parse(contents, nullptr, false);
46  if (cred_json.is_discarded()) {
47  ThrowRuntimeError("Invalid contents in credentials file " + path);
48  }
49  std::string cred_type = cred_json.value("type", "no type given");
50  if (cred_type == "authorized_user") {
51  auto info = ParseAuthorizedUserCredentials(contents, path);
52  return google::cloud::internal::make_unique<AuthorizedUserCredentials<>>(
53  info);
54  }
55  if (cred_type == "service_account") {
56  auto info = ParseServiceAccountCredentials(contents, path);
57  return google::cloud::internal::make_unique<ServiceAccountCredentials<>>(
58  info);
59  }
60  ThrowRuntimeError(
61  "Unsupported credential type (" + cred_type +
62  ") when reading Application Default Credentials file from " + path + ".");
63 }

65 std::unique_ptr<Credentials> MaybeLoadCredsFromAdcEnvVar() {
66  auto path = GoogleAdcFilePathFromEnvVarOrEmpty();
67  if (!path.empty()) {
68  // If the path was specified, try to load that file; explicitly fail if it
69  // doesn't exist or can't be read and parsed.
70  return LoadCredsFromPath(path);
71  }
72  return nullptr;
73 }

75 std::unique_ptr<Credentials> MaybeLoadCredsFromGcloudAdcFile() {
76  auto path = GoogleAdcFilePathFromWellKnownPathOrEmpty();
77  if (!path.empty()) {
78  // Just because we had the necessary information to build the path doesn't
79  // mean that a file exists there.
80  std::error_code ec;
81  auto adc_file_status = google::cloud::internal::status(path, ec);
82  if (google::cloud::internal::exists(adc_file_status)) {
83  return LoadCredsFromPath(path);
84  }
85  }
86  // Either we were unable to construct the well known path or no file existed
87  // at that path.
88  return nullptr;
89 }

91 std::shared_ptr<Credentials> GoogleDefaultCredentials() {
92  // 1) Check if the GOOGLE_APPLICATION_CREDENTIALS environment variable is set.
93  auto creds = MaybeLoadCredsFromAdcEnvVar();
94  if (creds) {
95  return std::move(creds);
96  }
97 
98  // 2) If no path was specified via environment variable, check if the gcloud
99  // ADC file exists.
100  creds = MaybeLoadCredsFromGcloudAdcFile();
101  if (creds) {
102  return std::move(creds);
103  }
104 
105  // 3) Check for implicit environment-based credentials (GCE, GAE Flexible
106  // Environment).
107  // Note: GCE credentials *should* also work when running on a VM instance in
108  // the App Engine Flexible Environment, but this has not been explicitly
109  // tested, as it requires a custom GAEF runtime.
110  if (storage::internal::RunningOnComputeEngineVm()) {
111  return std::make_shared<ComputeEngineCredentials<>>();
112  }
113 
114  // We've exhausted all search points, thus credentials cannot be constructed.
115  std::string adc_link =
116  "https://developers.google.com/identity/protocols"
117  "/application-default-credentials";
118  google::cloud::internal::ThrowRuntimeError(
119  "Could not automatically determine credentials. For more information,"
120  " please see " +
121  adc_link);
122 }
