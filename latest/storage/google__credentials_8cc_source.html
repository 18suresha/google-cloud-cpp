35 /// Parses the JSON file at `path` and creates the appropriate Credentials type.
36 StatusOr<std::unique_ptr<Credentials>> LoadCredsFromPath(
37  std::string const& path) {
38  namespace nl = google::cloud::storage::internal::nl;
39 
40  std::ifstream ifs(path);
41  if (!ifs.is_open()) {
42  // We use kUnknown here because we don't know if the file does not exist, or
43  // if we were unable to open it for some other reason.
44  return Status(StatusCode::kUnknown, "Cannot open credentials file " + path);
45  }
46  std::string contents(std::istreambuf_iterator<char>{ifs}, {});
47  auto cred_json = nl::json::parse(contents, nullptr, false);
48  if (cred_json.is_discarded()) {
49  // This is not a JSON file, try to load it as a P12 service account.
50  auto info = ParseServiceAccountP12File(path);
51  if (!info) {
52  // Ignore the error returned by the P12 parser, because those are too
53  // specific, they typically say "error in PKCS#12" and the application
54  // may not even be trying to load a PKCS#12 file.
55  return Status(StatusCode::kInvalidArgument,
56  "Invalid credentials file " + path);
57  }
58  auto credentials =
59  google::cloud::internal::make_unique<ServiceAccountCredentials<>>(
60  *info);
61  return std::unique_ptr<Credentials>(std::move(credentials));
62  }
63  std::string cred_type = cred_json.value("type", "no type given");
64  if (cred_type == "authorized_user") {
65  auto info = ParseAuthorizedUserCredentials(contents, path);
66  if (!info) {
67  return info.status();
68  }
69  std::unique_ptr<Credentials> ptr =
70  google::cloud::internal::make_unique<AuthorizedUserCredentials<>>(
71  *info);
72  return StatusOr<std::unique_ptr<Credentials>>(std::move(ptr));
73  }
74  if (cred_type == "service_account") {
75  auto info = ParseServiceAccountCredentials(contents, path);
76  if (!info) {
77  return info.status();
78  }
79  std::unique_ptr<Credentials> ptr =
80  google::cloud::internal::make_unique<ServiceAccountCredentials<>>(
81  *info);
82  return StatusOr<std::unique_ptr<Credentials>>(std::move(ptr));
83  }
84  return StatusOr<std::unique_ptr<Credentials>>(
85  Status(StatusCode::kInvalidArgument,
86  "Unsupported credential type (" + cred_type +
87  ") when reading Application Default Credentials file from " +
88  path + "."));
89 }

91 StatusOr<std::unique_ptr<Credentials>> MaybeLoadCredsFromAdcEnvVar() {
92  auto path = GoogleAdcFilePathFromEnvVarOrEmpty();
93  if (!path.empty()) {
94  // If the path was specified, try to load that file; explicitly fail if it
95  // doesn't exist or can't be read and parsed.
96  return LoadCredsFromPath(path);
97  }
98  // No ptr indicates that there was no path to attempt loading creds from.
99  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
100 }

102 StatusOr<std::unique_ptr<Credentials>> MaybeLoadCredsFromGcloudAdcFile() {
103  auto path = GoogleAdcFilePathFromWellKnownPathOrEmpty();
104  if (!path.empty()) {
105  // Just because we had the necessary information to build the path doesn't
106  // mean that a file exists there.
107  std::error_code ec;
108  auto adc_file_status = google::cloud::internal::status(path, ec);
109  if (google::cloud::internal::exists(adc_file_status)) {
110  return LoadCredsFromPath(path);
111  }
112  }
113  // Either we were unable to construct the well known path or no file existed
114  // at that path.
115  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
116 }

118 StatusOr<std::shared_ptr<Credentials>> GoogleDefaultCredentials() {
119  // 1) Check if the GOOGLE_APPLICATION_CREDENTIALS environment variable is set.
120  auto creds = MaybeLoadCredsFromAdcEnvVar();
121  if (!creds) {
122  return StatusOr<std::shared_ptr<Credentials>>(creds.status());
123  }
124  if (*creds) {
125  return StatusOr<std::shared_ptr<Credentials>>(std::move(*creds));
126  }
127 
128  // 2) If no path was specified via environment variable, check if the gcloud
129  // ADC file exists.
130  creds = MaybeLoadCredsFromGcloudAdcFile();
131  if (!creds) {
132  return StatusOr<std::shared_ptr<Credentials>>(creds.status());
133  }
134  if (*creds) {
135  return StatusOr<std::shared_ptr<Credentials>>(std::move(*creds));
136  }
137 
138  // 3) Check for implicit environment-based credentials (GCE, GAE Flexible
139  // Environment).
140  // Note: GCE credentials *should* also work when running on a VM instance in
141  // the App Engine Flexible Environment, but this has not been explicitly
142  // tested, as it requires a custom GAEF runtime.
143  if (storage::internal::RunningOnComputeEngineVm()) {
144  return StatusOr<std::shared_ptr<Credentials>>(
145  std::make_shared<ComputeEngineCredentials<>>());
146  }
147 
148  // We've exhausted all search points, thus credentials cannot be constructed.
149  std::string adc_link =
150  "https://developers.google.com/identity/protocols"
151  "/application-default-credentials";
152  return StatusOr<std::shared_ptr<Credentials>>(
153  Status(StatusCode::kUnknown,
154  "Could not automatically determine credentials. For more "
155  "information, please see " +
156  adc_link));
157 }
