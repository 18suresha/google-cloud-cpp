35 /// Parses the JSON file at `path` and creates the appropriate Credentials type.
36 StatusOr<std::unique_ptr<Credentials>> LoadCredsFromPath(
37  std::string const& path) {
38  namespace nl = google::cloud::storage::internal::nl;
39 
40  std::ifstream ifs(path);
41  if (!ifs.is_open()) {
42  // We use kUnknown here because we don't know if the file does not exist, or
43  // if we were unable to open it for some other reason.
44  return Status(StatusCode::kUnknown, "Cannot open credentials file " + path);
45  }
46  std::string contents(std::istreambuf_iterator<char>{ifs}, {});
47  auto cred_json = nl::json::parse(contents, nullptr, false);
48  if (cred_json.is_discarded()) {
49  return Status(StatusCode::kInvalidArgument,
50  "Invalid contents in credentials file " + path);
51  }
52  std::string cred_type = cred_json.value("type", "no type given");
53  if (cred_type == "authorized_user") {
54  auto info = ParseAuthorizedUserCredentials(contents, path);
55  if (!info) {
56  return info.status();
57  }
58  std::unique_ptr<Credentials> ptr =
59  google::cloud::internal::make_unique<AuthorizedUserCredentials<>>(
60  *info);
61  return StatusOr<std::unique_ptr<Credentials>>(std::move(ptr));
62  }
63  if (cred_type == "service_account") {
64  auto info = ParseServiceAccountCredentials(contents, path);
65  if (!info) {
66  return info.status();
67  }
68  std::unique_ptr<Credentials> ptr =
69  google::cloud::internal::make_unique<ServiceAccountCredentials<>>(
70  *info);
71  return StatusOr<std::unique_ptr<Credentials>>(std::move(ptr));
72  }
73  return StatusOr<std::unique_ptr<Credentials>>(
74  Status(StatusCode::kInvalidArgument,
75  "Unsupported credential type (" + cred_type +
76  ") when reading Application Default Credentials file from " +
77  path + "."));
78 }

80 StatusOr<std::unique_ptr<Credentials>> MaybeLoadCredsFromAdcEnvVar() {
81  auto path = GoogleAdcFilePathFromEnvVarOrEmpty();
82  if (!path.empty()) {
83  // If the path was specified, try to load that file; explicitly fail if it
84  // doesn't exist or can't be read and parsed.
85  return LoadCredsFromPath(path);
86  }
87  // No ptr indicates that there was no path to attempt loading creds from.
88  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
89 }

91 StatusOr<std::unique_ptr<Credentials>> MaybeLoadCredsFromGcloudAdcFile() {
92  auto path = GoogleAdcFilePathFromWellKnownPathOrEmpty();
93  if (!path.empty()) {
94  // Just because we had the necessary information to build the path doesn't
95  // mean that a file exists there.
96  std::error_code ec;
97  auto adc_file_status = google::cloud::internal::status(path, ec);
98  if (google::cloud::internal::exists(adc_file_status)) {
99  return LoadCredsFromPath(path);
100  }
101  }
102  // Either we were unable to construct the well known path or no file existed
103  // at that path.
104  return StatusOr<std::unique_ptr<Credentials>>(nullptr);
105 }

107 StatusOr<std::shared_ptr<Credentials>> GoogleDefaultCredentials() {
108  // 1) Check if the GOOGLE_APPLICATION_CREDENTIALS environment variable is set.
109  auto creds = MaybeLoadCredsFromAdcEnvVar();
110  if (!creds) {
111  return StatusOr<std::shared_ptr<Credentials>>(creds.status());
112  }
113  if (*creds) {
114  return StatusOr<std::shared_ptr<Credentials>>(std::move(*creds));
115  }
116 
117  // 2) If no path was specified via environment variable, check if the gcloud
118  // ADC file exists.
119  creds = MaybeLoadCredsFromGcloudAdcFile();
120  if (!creds) {
121  return StatusOr<std::shared_ptr<Credentials>>(creds.status());
122  }
123  if (*creds) {
124  return StatusOr<std::shared_ptr<Credentials>>(std::move(*creds));
125  }
126 
127  // 3) Check for implicit environment-based credentials (GCE, GAE Flexible
128  // Environment).
129  // Note: GCE credentials *should* also work when running on a VM instance in
130  // the App Engine Flexible Environment, but this has not been explicitly
131  // tested, as it requires a custom GAEF runtime.
132  if (storage::internal::RunningOnComputeEngineVm()) {
133  return StatusOr<std::shared_ptr<Credentials>>(
134  std::make_shared<ComputeEngineCredentials<>>());
135  }
136 
137  // We've exhausted all search points, thus credentials cannot be constructed.
138  std::string adc_link =
139  "https://developers.google.com/identity/protocols"
140  "/application-default-credentials";
141  return StatusOr<std::shared_ptr<Credentials>>(
142  Status(StatusCode::kUnknown,
143  "Could not automatically determine credentials. For more "
144  "information, please see " +
145  adc_link));
146 }
