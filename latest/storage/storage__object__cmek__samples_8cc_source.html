24 void WriteObjectWithKmsKey(google::cloud::storage::Client client,
25  std::vector<std::string> const& argv) {
27  namespace gcs = google::cloud::storage;
28  using ::google::cloud::StatusOr;
29  [](gcs::Client client, std::string const& bucket_name,
30  std::string const& object_name, std::string const& kms_key_name) {
31  gcs::ObjectWriteStream stream = client.WriteObject(
32  bucket_name, object_name, gcs::KmsKeyName(kms_key_name));
33 
34  // Line numbers start at 1.
35  for (int lineno = 1; lineno <= 10; ++lineno) {
36  stream << lineno << ": placeholder text for CMEK example.\n";
37  }
38 
39  stream.Close();
40 
41  StatusOr<gcs::ObjectMetadata> metadata = std::move(stream).metadata();
42  if (!metadata) throw std::runtime_error(metadata.status().message());
43 
44  std::cout << "Successfully wrote to object " << metadata->name()
45  << " its size is: " << metadata->size()
46  << "\nFull metadata: " << *metadata << "\n";
47  }
49  (std::move(client), argv.at(0), argv.at(1), argv.at(2));
50 }

52 void ObjectCsekToCmek(google::cloud::storage::Client client,
53  std::vector<std::string> const& argv) {
55  namespace gcs = google::cloud::storage;
56  using ::google::cloud::StatusOr;
57  [](gcs::Client client, std::string const& bucket_name,
58  std::string const& object_name, std::string const& old_csek_key_base64,
59  std::string const& new_cmek_key_name) {
60  StatusOr<gcs::ObjectMetadata> metadata = client.RewriteObjectBlocking(
61  bucket_name, object_name, bucket_name, object_name,
62  gcs::SourceEncryptionKey::FromBase64Key(old_csek_key_base64),
63  gcs::DestinationKmsKeyName(new_cmek_key_name));
64  if (!metadata) throw std::runtime_error(metadata.status().message());
65 
66  std::cout << "Changed object " << metadata->name() << " in bucket "
67  << metadata->bucket()
68  << " from using CSEK to CMEK key.\nFull Metadata: " << *metadata
69  << "\n";
70  }
72  (std::move(client), argv.at(0), argv.at(1), argv.at(2), argv.at(3));
73 }

93 void RunAll(std::vector<std::string> const& argv) {
94  namespace examples = ::google::cloud::storage::examples;
95  namespace gcs = ::google::cloud::storage;
96 
97  if (!argv.empty()) throw examples::Usage{"auto"};
98  examples::CheckEnvironmentVariablesAreSet({
99  "GOOGLE_CLOUD_PROJECT",
100  "GOOGLE_CLOUD_CPP_STORAGE_TEST_CMEK_KEY",
101  });
102  auto const project_id =
103  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
104  auto const cmek_key =
105  google::cloud::internal::GetEnv("GOOGLE_CLOUD_CPP_STORAGE_TEST_CMEK_KEY")
106  .value();
107  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
108  auto const bucket_name = examples::MakeRandomBucketName(generator);
109  auto client = gcs::Client::CreateDefaultClient().value();
110 
111  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
112  << std::endl;
113  (void)client
114  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
115  .value();
116  // In GCS a single project cannot create or delete buckets more often than
117  // once every two seconds. We will pause until that time before deleting the
118  // bucket.
119  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
120 
121  auto const cmek_object_name =
122  examples::MakeRandomObjectName(generator, "cmek-object-") + ".txt";
123  std::cout << "\nRunning the WriteObjectWithKmsKey() example" << std::endl;
124  WriteObjectWithKmsKey(client, {bucket_name, cmek_object_name, cmek_key});
125 
126  std::cout << "\nRunning the GetObjectKmsKey() example" << std::endl;
127  GetObjectKmsKey(client, {bucket_name, cmek_object_name});
128 
129  std::cout << "\nReading back the contents" << std::endl;
130  {
131  auto r = client.ReadObject(bucket_name, cmek_object_name);
132  std::cout << " contents="
133  << std::string{std::istreambuf_iterator<char>{r}, {}} << "\n";
134  }
135 
136  std::cout << "\nDeleting the object" << std::endl;
137  (void)client.DeleteObject(bucket_name, cmek_object_name);
138 
139  std::cout << "\nCreating an object with a CSEK" << std::endl;
140  auto const csek_object_name =
141  examples::MakeRandomObjectName(generator, "csek-object-") + ".txt";
142  auto const csek = gcs::CreateKeyFromGenerator(generator);
143  auto constexpr kText = R"""(Some text to read and write)""";
144  auto meta = client
145  .InsertObject(bucket_name, csek_object_name, kText,
146  gcs::EncryptionKey(csek))
147  .value();
148 
149  std::cout << "\nRunning the ObjectCsekToCmek() example";
150  ObjectCsekToCmek(client, {bucket_name, csek_object_name, csek.key, cmek_key});
151 
152  std::cout << "\nReading back the contents" << std::endl;
153  {
154  auto r = client.ReadObject(bucket_name, cmek_object_name);
155  std::cout << " contents="
156  << std::string{std::istreambuf_iterator<char>{r}, {}} << "\n";
157  }
158  (void)client.DeleteObject(bucket_name, csek_object_name);
159 
160  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
161  (void)examples::RemoveBucketAndContents(client, bucket_name);
162 }
