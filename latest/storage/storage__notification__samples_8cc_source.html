76  if (!notification) {
77  throw std::runtime_error(notification.status().message());
78  }
79 
80  std::cout << "Notification " << notification->id() << " for bucket "

82  if (notification->object_name_prefix().empty()) {
83  std::cout << "This notification is sent for all objects in the bucket\n";
84  } else {
85  std::cout << "This notification is sent only for objects starting with"
86  << " the prefix " << notification->object_name_prefix() << "\n";
87  }
88  std::cout << "Full details for the notification:\n"
89  << *notification << "\n";
90  }
92  (std::move(client), argv.at(0), argv.at(1));

112 void RunAll(std::vector<std::string> const& argv) {
113  namespace examples = ::google::cloud::storage::examples;
114  namespace gcs = ::google::cloud::storage;
115 
116  if (!argv.empty()) throw examples::Usage{"auto"};
117  examples::CheckEnvironmentVariablesAreSet({
118  "GOOGLE_CLOUD_PROJECT",
119  "GOOGLE_CLOUD_CPP_STORAGE_TEST_TOPIC_NAME",
120  });
121  auto const project_id =
122  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
123  auto const topic_name = google::cloud::internal::GetEnv(
124  "GOOGLE_CLOUD_CPP_STORAGE_TEST_TOPIC_NAME")
125  .value();
126  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
127  auto const bucket_name = examples::MakeRandomBucketName(generator);
128  auto client = gcs::Client::CreateDefaultClient().value();
129 
130  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
131  << std::endl;
132  (void)client
133  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
134  .value();
135  // In GCS a single project cannot create or delete buckets more often than
136  // once every two seconds. We will pause until that time before deleting the
137  // bucket.
138  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
139 
140  std::cout << "\nRunning ListNotifications() example [1]" << std::endl;
141  ListNotifications(client, {bucket_name});
142 
143  std::cout << "\nRunning CreateNotification() example" << std::endl;
144  CreateNotification(client, {bucket_name, topic_name});
145 
146  std::cout << "\nRunning ListNotifications() example [1]" << std::endl;
147  ListNotifications(client, {bucket_name});
148 
149  // We need to create notifications directly to get their ids and call the
150  // other examples.
151  std::cout << "\nManually creating a notification [1]" << std::endl;
152  auto n1 = client
153  .CreateNotification(
154  bucket_name, topic_name, gcs::payload_format::JsonApiV1(),
155  gcs::NotificationMetadata().set_object_name_prefix("foo/"))
156  .value();
157 
158  std::cout << "\nManually creating a notification [2]" << std::endl;
159  auto n2 = client
160  .CreateNotification(bucket_name, topic_name,
161  gcs::payload_format::JsonApiV1(),
162  gcs::NotificationMetadata())
163  .value();
164 
165  std::cout << "\nRunning GetNotification() example" << std::endl;
166  GetNotification(client, {bucket_name, n1.id()});
167 
168  std::cout << "\nRunning GetNotification() example" << std::endl;
169  GetNotification(client, {bucket_name, n2.id()});
170 
171  std::cout << "\nRunning ListNotifications() example [3]" << std::endl;
172  ListNotifications(client, {bucket_name});
173 
174  std::cout << "\nRunning DeleteNotification() example [1]" << std::endl;
175  DeleteNotification(client, {bucket_name, n1.id()});
176 
177  std::cout << "\nRunning DeleteNotification() example [2]" << std::endl;
178  DeleteNotification(client, {bucket_name, n2.id()});
179 
180  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
181  (void)examples::RemoveBucketAndContents(client, bucket_name);
182 }
