28 StatusOr<ServiceAccountCredentialsInfo> ParseServiceAccountCredentials(
29  std::string const& content, std::string const& source,
30  std::string const& default_token_uri) {
31  namespace nl = storage::internal::nl;
32  nl::json credentials = nl::json::parse(content, nullptr, false);
33  if (credentials.is_discarded()) {
34  return Status(StatusCode::kInvalidArgument,
35  "Invalid ServiceAccountCredentials,"
36  "parsing failed on data loaded from " +
37  source);
38  }
39  char const private_key_id_key[] = "private_key_id";
40  char const private_key_key[] = "private_key";
41  char const token_uri_key[] = "token_uri";
42  char const client_email_key[] = "client_email";
43  for (auto const& key :
44  {private_key_id_key, private_key_key, client_email_key}) {
45  if (credentials.count(key) == 0U) {
46  return Status(StatusCode::kInvalidArgument,
47  "Invalid ServiceAccountCredentials, the " +
48  std::string(key) +
49  " field is missing on data loaded from " + source);
50  }
51  if (credentials.value(key, "").empty()) {
52  return Status(StatusCode::kInvalidArgument,
53  "Invalid ServiceAccountCredentials, the " +
54  std::string(key) +
55  " field is empty on data loaded from " + source);
56  }
57  }
58  // The token_uri field may be missing, but may not be empty:
59  if (credentials.count(token_uri_key) != 0U &&
60  credentials.value(token_uri_key, "").empty()) {
61  return Status(StatusCode::kInvalidArgument,
62  "Invalid ServiceAccountCredentials, the " +
63  std::string(token_uri_key) +
64  " field is empty on data loaded from " + source);
65  }
66  return ServiceAccountCredentialsInfo{
67  credentials.value(client_email_key, ""),
68  credentials.value(private_key_id_key, ""),
69  credentials.value(private_key_key, ""),
70  // Some credential formats (e.g. gcloud's ADC file) don't contain a
71  // "token_uri" attribute in the JSON object. In this case, we try using
72  // the default value.
73  credentials.value(token_uri_key, default_token_uri),
74  };
75 }

77 StatusOr<ServiceAccountCredentialsInfo> ParseServiceAccountP12File(
78  std::string const& source, std::string const& default_token_uri) {
79  OpenSSL_add_all_algorithms();
80 
81  PKCS12* p12_raw = [](std::string const& source) {
82  FILE* fp = std::fopen(source.c_str(), "rb");
83  if (fp == nullptr) {
84  return static_cast<PKCS12*>(nullptr);
85  }
86  auto result = d2i_PKCS12_fp(fp, nullptr);
87  fclose(fp);
88  return result;
89  }(source);
90 
91  std::unique_ptr<PKCS12, decltype(&PKCS12_free)> p12(p12_raw, &PKCS12_free);
92 
93  auto capture_openssl_errors = []() {
94  std::string msg;
95  while (auto code = ERR_get_error()) {
96  // OpenSSL guarantees that 256 bytes is enough:
97  // https://www.openssl.org/docs/man1.1.1/man3/ERR_error_string_n.html
98  // https://www.openssl.org/docs/man1.0.2/man3/ERR_error_string_n.html
99  // we could not find a macro or constant to replace the 256 literal.
100  char buf[256];
101  ERR_error_string_n(code, buf, sizeof(buf));
102  msg += buf;
103  }
104  return msg;
105  };
106 
107  if (p12 == nullptr) {
108  std::string msg = "Cannot open PKCS#12 file (" + source + "): ";
109  msg += capture_openssl_errors();
110  return Status(StatusCode::kInvalidArgument, msg);
111  }
112 
113  EVP_PKEY* pkey_raw;
114  X509* cert_raw;
115  if (PKCS12_parse(p12.get(), "notasecret", &pkey_raw, &cert_raw, nullptr) !=
116  1) {
117  std::string msg = "Cannot parse PKCS#12 file (" + source + "): ";
118  msg += capture_openssl_errors();
119  return Status(StatusCode::kInvalidArgument, msg);
120  }
121 
122  std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> pkey(pkey_raw,
123  &EVP_PKEY_free);
124  std::unique_ptr<X509, decltype(&X509_free)> cert(cert_raw, &X509_free);
125 
126  if (pkey_raw == nullptr) {
127  return Status(StatusCode::kInvalidArgument,
128  "No private key found in PKCS#12 file (" + source + ")");
129  }
130  if (cert_raw == nullptr) {
131  return Status(StatusCode::kInvalidArgument,
132  "No private key found in PKCS#12 file (" + source + ")");
133  }
134 
135  // This is automatically deleted by `cert`.
136  X509_NAME* name = X509_get_subject_name(cert.get());
137 
138  std::string service_account_id = [&name]() -> std::string {
139  auto openssl_free = [](void* addr) { OPENSSL_free(addr); };
140  std::unique_ptr<char, decltype(openssl_free)> oneline(
141  X509_NAME_oneline(name, nullptr, 0), openssl_free);
142  // We expect the name to be simply CN/ followed by a (small) number of
143  // digits.
144  if (strncmp("/CN=", oneline.get(), 4) != 0) {
145  return "";
146  }
147  return oneline.get() + 4;
148  }();
149 
150  if (service_account_id.find_first_not_of("0123456789") != std::string::npos ||
151  service_account_id.empty()) {
152  return Status(
153  StatusCode::kInvalidArgument,
154  "Invalid PKCS#12 file (" + source +
155  "): service account id missing or not not formatted correctly");
156  }
157 
158  std::unique_ptr<BIO, decltype(&BIO_free)> mem_io(BIO_new(BIO_s_mem()),
159  &BIO_free);
160 
161  if (PEM_write_bio_PKCS8PrivateKey(mem_io.get(), pkey.get(), nullptr, nullptr,
162  0, nullptr, nullptr) == 0) {
163  std::string msg =
164  "Cannot print private key in PKCS#12 file (" + source + "): ";
165  msg += capture_openssl_errors();
166  return Status(StatusCode::kUnknown, msg);
167  }
168 
169  // This buffer belongs to the BIO chain and is freed upon its destruction.
170  BUF_MEM* buf_mem;
171  BIO_get_mem_ptr(mem_io.get(), &buf_mem);
172 
173  std::string private_key(buf_mem->data, buf_mem->length);
174 
175  return ServiceAccountCredentialsInfo{std::move(service_account_id),
176  "--unknown--", std::move(private_key),
177  default_token_uri};
178 }
