30 StatusOr<ServiceAccountCredentialsInfo> ParseServiceAccountCredentials(
31  std::string const& content, std::string const& source,
32  std::string const& default_token_uri) {
33  auto credentials = nlohmann::json::parse(content, nullptr, false);
34  if (credentials.is_discarded()) {
35  return Status(StatusCode::kInvalidArgument,
36  "Invalid ServiceAccountCredentials,"
37  "parsing failed on data loaded from " +
38  source);
39  }
40  std::string const private_key_id_key = "private_key_id";
41  std::string const private_key_key = "private_key";
42  std::string const token_uri_key = "token_uri";
43  std::string const client_email_key = "client_email";
44  for (auto const& key :
45  {private_key_id_key, private_key_key, client_email_key}) {
46  if (credentials.count(key) == 0) {
47  return Status(StatusCode::kInvalidArgument,
48  "Invalid ServiceAccountCredentials, the " +
49  std::string(key) +
50  " field is missing on data loaded from " + source);
51  }
52  if (credentials.value(key, "").empty()) {
53  return Status(StatusCode::kInvalidArgument,
54  "Invalid ServiceAccountCredentials, the " +
55  std::string(key) +
56  " field is empty on data loaded from " + source);
57  }
58  }
59  // The token_uri field may be missing, but may not be empty:
60  if (credentials.count(token_uri_key) != 0 &&
61  credentials.value(token_uri_key, "").empty()) {
62  return Status(StatusCode::kInvalidArgument,
63  "Invalid ServiceAccountCredentials, the " +
64  std::string(token_uri_key) +
65  " field is empty on data loaded from " + source);
66  }
67  return ServiceAccountCredentialsInfo{
68  credentials.value(client_email_key, ""),
69  credentials.value(private_key_id_key, ""),
70  credentials.value(private_key_key, ""),
71  // Some credential formats (e.g. gcloud's ADC file) don't contain a
72  // "token_uri" attribute in the JSON object. In this case, we try using
73  // the default value.
74  credentials.value(token_uri_key, default_token_uri),
75  /*scopes*/ {},
76  /*subject*/ {}};
77 }

80 StatusOr<ServiceAccountCredentialsInfo> ParseServiceAccountP12File(
81  std::string const& source, std::string const& default_token_uri) {
82  OpenSSL_add_all_algorithms();
83 
84  PKCS12* p12_raw = [](std::string const& source) {
85  FILE* fp = std::fopen(source.c_str(), "rb");
86  if (fp == nullptr) {
87  return static_cast<PKCS12*>(nullptr);
88  }
89  auto result = d2i_PKCS12_fp(fp, nullptr);
90  fclose(fp);
91  return result;
92  }(source);
93 
94  std::unique_ptr<PKCS12, decltype(&PKCS12_free)> p12(p12_raw, &PKCS12_free);
95 
96  auto capture_openssl_errors = []() {
97  std::string msg;
98  while (auto code = ERR_get_error()) {
99  // OpenSSL guarantees that 256 bytes is enough:
100  // https://www.openssl.org/docs/man1.1.1/man3/ERR_error_string_n.html
101  // https://www.openssl.org/docs/man1.0.2/man3/ERR_error_string_n.html
102  // we could not find a macro or constant to replace the 256 literal.
103  auto constexpr kMaxOpenSslErrorLength = 256;
104  std::array<char, kMaxOpenSslErrorLength> buf{};
105  ERR_error_string_n(code, buf.data(), buf.size());
106  msg += buf.data();
107  }
108  return msg;
109  };
110 
111  if (p12 == nullptr) {
112  std::string msg = "Cannot open PKCS#12 file (" + source + "): ";
113  msg += capture_openssl_errors();
114  return Status(StatusCode::kInvalidArgument, msg);
115  }
116 
117  EVP_PKEY* pkey_raw;
118  X509* cert_raw;
119  if (PKCS12_parse(p12.get(), "notasecret", &pkey_raw, &cert_raw, nullptr) !=
120  1) {
121  std::string msg = "Cannot parse PKCS#12 file (" + source + "): ";
122  msg += capture_openssl_errors();
123  return Status(StatusCode::kInvalidArgument, msg);
124  }
125 
126  std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> pkey(pkey_raw,
127  &EVP_PKEY_free);
128  std::unique_ptr<X509, decltype(&X509_free)> cert(cert_raw, &X509_free);
129 
130  if (pkey_raw == nullptr) {
131  return Status(StatusCode::kInvalidArgument,
132  "No private key found in PKCS#12 file (" + source + ")");
133  }
134  if (cert_raw == nullptr) {
135  return Status(StatusCode::kInvalidArgument,
136  "No private key found in PKCS#12 file (" + source + ")");
137  }
138 
139  // This is automatically deleted by `cert`.
140  X509_NAME* name = X509_get_subject_name(cert.get());
141 
142  std::string service_account_id = [&name]() -> std::string {
143  auto openssl_free = [](void* addr) { OPENSSL_free(addr); };
144  std::unique_ptr<char, decltype(openssl_free)> oneline(
145  X509_NAME_oneline(name, nullptr, 0), openssl_free);
146  // We expect the name to be simply CN/ followed by a (small) number of
147  // digits.
148  if (strncmp("/CN=", oneline.get(), 4) != 0) {
149  return "";
150  }
151  return oneline.get() + 4;
152  }();
153 
154  if (service_account_id.find_first_not_of("0123456789") != std::string::npos ||
155  service_account_id.empty()) {
156  return Status(
157  StatusCode::kInvalidArgument,
158  "Invalid PKCS#12 file (" + source +
159  "): service account id missing or not not formatted correctly");
160  }
161 
162  std::unique_ptr<BIO, decltype(&BIO_free)> mem_io(BIO_new(BIO_s_mem()),
163  &BIO_free);
164 
165  if (PEM_write_bio_PKCS8PrivateKey(mem_io.get(), pkey.get(), nullptr, nullptr,
166  0, nullptr, nullptr) == 0) {
167  std::string msg =
168  "Cannot print private key in PKCS#12 file (" + source + "): ";
169  msg += capture_openssl_errors();
170  return Status(StatusCode::kUnknown, msg);
171  }
172 
173  // This buffer belongs to the BIO chain and is freed upon its destruction.
174  BUF_MEM* buf_mem;
175  BIO_get_mem_ptr(mem_io.get(), &buf_mem);
176 
177  std::string private_key(buf_mem->data, buf_mem->length);
178 
179  return ServiceAccountCredentialsInfo{std::move(service_account_id),
180  "--unknown--",
181  std::move(private_key),
182  default_token_uri,
183  /*scopes*/ {},
184  /*subject*/ {}};
185 }

188 std::pair<std::string, std::string> AssertionComponentsFromInfo(
189  ServiceAccountCredentialsInfo const& info,
190  std::chrono::system_clock::time_point now) {
191  nlohmann::json assertion_header = {
192  {"alg", "RS256"}, {"kid", info.private_key_id}, {"typ", "JWT"}};
193 
194  // Scopes must be specified in a comma-delimited string.
195  std::string scope_str;
196  if (!info.scopes) {
197  scope_str = GoogleOAuthScopeCloudPlatform();
198  } else {
199  char const* sep = "";
200  for (const auto& scope : *(info.scopes)) {
201  scope_str += sep + scope;
202  sep = ",";
203  }
204  }
205 
206  auto expiration = now + GoogleOAuthAccessTokenLifetime();
207  // As much as possible, do the time arithmetic using the std::chrono types.
208  // Convert to an integer only when we are dealing with timestamps since the
209  // epoch. Note that we cannot use `time_t` directly because that might be a
210  // floating point.
211  auto const now_from_epoch =
212  static_cast<std::intmax_t>(std::chrono::system_clock::to_time_t(now));
213  auto const expiration_from_epoch = static_cast<std::intmax_t>(
214  std::chrono::system_clock::to_time_t(expiration));
215  nlohmann::json assertion_payload = {
216  {"iss", info.client_email},
217  {"scope", scope_str},
218  {"aud", info.token_uri},
219  {"iat", now_from_epoch},
220  // Resulting access token should be expire after one hour.
221  {"exp", expiration_from_epoch}};
222  if (info.subject) {
223  assertion_payload["sub"] = *(info.subject);
224  }
225 
226  // Note: we don't move here as it would prevent copy elision.
227  return std::make_pair(assertion_header.dump(), assertion_payload.dump());
228 }

230 std::string MakeJWTAssertion(std::string const& header,
231  std::string const& payload,
232  std::string const& pem_contents) {
233  std::string encoded_header = internal::UrlsafeBase64Encode(header);
234  std::string encoded_payload = internal::UrlsafeBase64Encode(payload);
235  std::string encoded_signature = internal::UrlsafeBase64Encode(
236  internal::SignStringWithPem(encoded_header + '.' + encoded_payload,
237  pem_contents, JwtSigningAlgorithms::RS256));
238  return encoded_header + '.' + encoded_payload + '.' + encoded_signature;
239 }

241 std::string CreateServiceAccountRefreshPayload(
242  ServiceAccountCredentialsInfo const& info, std::string const& grant_type,
243  std::chrono::system_clock::time_point now) {
244  auto assertion_components = AssertionComponentsFromInfo(info, now);
245  std::string payload = grant_type;
246  payload += "&assertion=";
247  payload += MakeJWTAssertion(assertion_components.first,
248  assertion_components.second, info.private_key);
249  return payload;
250 }

252 StatusOr<RefreshingCredentialsWrapper::TemporaryToken>
253 ParseServiceAccountRefreshResponse(
254  storage::internal::HttpResponse const& response,
255  std::chrono::system_clock::time_point now) {
256  auto access_token = nlohmann::json::parse(response.payload, nullptr, false);
257  if (access_token.is_discarded() || access_token.count("access_token") == 0 or
258  access_token.count("expires_in") == 0 or
259  access_token.count("token_type") == 0) {
260  auto payload =
261  response.payload +
262  "Could not find all required fields in response (access_token,"
263  " expires_in, token_type).";
264  return AsStatus(storage::internal::HttpResponse{response.status_code,
265  payload, response.headers});
266  }
267  // Response should have the attributes "access_token", "expires_in", and
268  // "token_type".
269  std::string header =
270  "Authorization: " + access_token.value("token_type", "") + " " +
271  access_token.value("access_token", "");
272  auto expires_in =
273  std::chrono::seconds(access_token.value("expires_in", int(0)));
274  auto new_expiration = now + expires_in;
275 
276  return RefreshingCredentialsWrapper::TemporaryToken{std::move(header),
277  new_expiration};
278 }
