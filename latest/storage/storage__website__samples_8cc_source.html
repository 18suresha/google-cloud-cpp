24 void GetStaticWebsiteConfiguration(google::cloud::storage::Client client,
25  std::vector<std::string> const& argv) {
27  // [START storage_print_bucket_website_configuration]
28  namespace gcs = google::cloud::storage;
29  using ::google::cloud::StatusOr;
30  [](gcs::Client client, std::string const& bucket_name) {
31  StatusOr<gcs::BucketMetadata> bucket_metadata =
32  client.GetBucketMetadata(bucket_name);
33 
34  if (!bucket_metadata) {
35  throw std::runtime_error(bucket_metadata.status().message());

37

39  std::cout << "Static website configuration is not set for bucket "

43

55 void SetStaticWebsiteConfiguration(google::cloud::storage::Client client,
56  std::vector<std::string> const& argv) {
58  // [START storage_define_bucket_website_configuration]
59  namespace gcs = google::cloud::storage;
60  using ::google::cloud::StatusOr;
61  [](gcs::Client client, std::string const& bucket_name,
62  std::string const& main_page_suffix, std::string const& not_found_page) {
63  StatusOr<gcs::BucketMetadata> original =
64  client.GetBucketMetadata(bucket_name);
65 
66  if (!original) throw std::runtime_error(original.status().message());
67  StatusOr<gcs::BucketMetadata> patched_metadata = client.PatchBucket(
68  bucket_name,
69  gcs::BucketMetadataPatchBuilder().SetWebsite(
70  gcs::BucketWebsite{main_page_suffix, not_found_page}),
71  gcs::IfMetagenerationMatch(original->metageneration()));
72 
73  if (!patched_metadata) {
74  throw std::runtime_error(patched_metadata.status().message());
75  }
76 
77  if (!patched_metadata->has_website()) {
78  std::cout << "Static website configuration is not set for bucket "
79  << patched_metadata->name() << "\n";
80  return;
81  }
82 
83  std::cout << "Static website configuration successfully set for bucket "
84  << patched_metadata->name() << "\nNew main page suffix is: "
85  << patched_metadata->website().main_page_suffix
86  << "\nNew not found page is: "
87  << patched_metadata->website().not_found_page << "\n";
88  }
89  // [END storage_define_bucket_website_configuration]
91  (std::move(client), argv.at(0), argv.at(1), argv.at(2));
92 }

94 void RemoveStaticWebsiteConfiguration(google::cloud::storage::Client client,
95  std::vector<std::string> const& argv) {
97  namespace gcs = google::cloud::storage;
98  using ::google::cloud::StatusOr;
99  [](gcs::Client client, std::string const& bucket_name) {
100  StatusOr<gcs::BucketMetadata> original =
101  client.GetBucketMetadata(bucket_name);
102 
103  if (!original) throw std::runtime_error(original.status().message());
104  StatusOr<gcs::BucketMetadata> patched_metadata = client.PatchBucket(
105  bucket_name, gcs::BucketMetadataPatchBuilder().ResetWebsite(),
106  gcs::IfMetagenerationMatch(original->metageneration()));
107 
108  if (!patched_metadata) {
109  throw std::runtime_error(patched_metadata.status().message());
110  }
111 
112  if (!patched_metadata->has_website()) {
113  std::cout << "Static website configuration removed for bucket "
114  << patched_metadata->name() << "\n";
115  return;
116  }
117 
118  std::cout << "Static website configuration is set for bucket "
119  << patched_metadata->name()
120  << "\nThis is unexpected, and may indicate that another"
121  << " application has modified the bucket concurrently.\n";
122  }
124  (std::move(client), argv.at(0));
125 }

127 void RunAll(std::vector<std::string> const& argv) {
128  namespace examples = ::google::cloud::storage::examples;
129  namespace gcs = ::google::cloud::storage;
130 
131  if (!argv.empty()) throw examples::Usage{"auto"};
132  examples::CheckEnvironmentVariablesAreSet({
133  "GOOGLE_CLOUD_PROJECT",
134  });
135  auto const project_id =
136  google::cloud::internal::GetEnv("GOOGLE_CLOUD_PROJECT").value();
137  auto generator = google::cloud::internal::DefaultPRNG(std::random_device{}());
138  auto const bucket_name = examples::MakeRandomBucketName(generator);
139  auto client = gcs::Client::CreateDefaultClient().value();
140  std::cout << "\nCreating bucket to run the example (" << bucket_name << ")"
141  << std::endl;
142  (void)client
143  .CreateBucketForProject(bucket_name, project_id, gcs::BucketMetadata{})
144  .value();
145  // In GCS a single project cannot create or delete buckets more often than
146  // once every two seconds. We will pause until that time before deleting the
147  // bucket.
148  auto pause = std::chrono::steady_clock::now() + std::chrono::seconds(2);
149 
150  std::cout << "\nRunning SetStaticWebsiteConfiguration() example" << std::endl;
151  SetStaticWebsiteConfiguration(
152  client, {bucket_name, "main-page.html", "not-found.html"});
153 
154  std::cout << "\nRunning GetStaticWebsiteConfiguration() example" << std::endl;
155  GetStaticWebsiteConfiguration(client, {bucket_name});
156 
157  std::cout << "\nRunning RemoveStaticWebsiteConfiguration() example"
158  << std::endl;
159  RemoveStaticWebsiteConfiguration(client, {bucket_name});
160 
161  if (!examples::UsingEmulator()) std::this_thread::sleep_until(pause);
162  (void)examples::RemoveBucketAndContents(client, bucket_name);
163 }
