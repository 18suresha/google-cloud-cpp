67 void ObjectWriteStream::Close() {
68  if (!IsOpen()) {
69  return;
70  }
71  StatusOr<internal::HttpResponse> response = buf_->Close();
72  if (!response.ok()) {
73  metadata_ = StatusOr<ObjectMetadata>(std::move(response).status());
74  setstate(std::ios_base::badbit);
75  return;
76  }
77  if (response->status_code >= 300) {
78  metadata_ = StatusOr<ObjectMetadata>(AsStatus(*response));
79  setstate(std::ios_base::badbit);
80  return;
81  }
82  headers_ = std::move(response->headers);
83  payload_ = std::move(response->payload);
84  if (payload_.empty()) {
85  // With the XML transport the response includes an empty payload, in that
86  // case it cannot be parsed.
87  metadata_ = ObjectMetadata{};
88  } else {
89  metadata_ = internal::ObjectMetadataParser::FromString(payload_);
90  if (!metadata_.ok()) {
91  setstate(std::ios_base::badbit);
92  return;
93  }
94  }
95 
96  if (!buf_->ValidateHash(*metadata_)) {
97  setstate(std::ios_base::badbit);
98  }
99 }
