72 StatusOr<ObjectWriteStream> ParallelUploadStateImpl::CreateStream(
73  RawClient& raw_client, ResumableUploadRequest const& request) {
74  auto session = raw_client.CreateResumableSession(request);
75  std::unique_lock<std::mutex> lk(mu_);
76  if (!session) {
77  // Preserve the first error.
78  res_ = session.status();
79  return std::move(session).status();
80  }
81 
82  auto idx = streams_.size();
83  ++num_unfinished_streams_;
84  streams_.emplace_back(
85  StreamInfo{request.object_name(), (*session)->session_id(), {}, false});
86  assert(idx < streams_.size());
87  lk.unlock();
88  return ObjectWriteStream(absl::make_unique<ParallelObjectWriteStreambuf>(
89  shared_from_this(), idx, *std::move(session),
90  raw_client.client_options().upload_buffer_size(),
91  CreateHashValidator(request)));
92 }

110 StatusOr<ParallelUploadPersistentState>
111 ParallelUploadPersistentState::FromString(std::string const& json_rep) {
112  ParallelUploadPersistentState res;
113 
114  auto json = nlohmann::json::parse(json_rep, nullptr, false);
115  if (json.is_discarded()) {
116  return Status(StatusCode::kInternal,
117  "Parallel upload state is not a valid JSON.");
118  }
119  if (!json.is_object()) {
120  return Status(StatusCode::kInternal,
121  "Parallel upload state is not a JSON object.");
122  }
123  // `nlohmann::json` doesn't allow for multiple keys with the same name, so
124  // there are either 0 or 1 elements with the same key.
125  if (json.count("destination") != 1) {
126  return Status(StatusCode::kInternal,
127  "Parallel upload state doesn't contain a 'destination'.");
128  }
129  auto& destination_json = json["destination"];
130  if (!destination_json.is_string()) {
131  return Status(StatusCode::kInternal,
132  "Parallel upload state's 'destination' is not a string.");
133  }
134  res.destination_object_name = destination_json.get<std::string>();
135  if (json.count("expected_generation") != 1) {
136  return Status(
137  StatusCode::kInternal,
138  "Parallel upload state doesn't contain a 'expected_generation'.");
139  }
140  auto& expected_generation_json = json["expected_generation"];
141  if (!expected_generation_json.is_number()) {
142  return Status(
143  StatusCode::kInternal,
144  "Parallel upload state's 'expected_generation' is not a number.");
145  }
146  res.expected_generation = expected_generation_json;
147  if (json.count("custom_data") > 0) {
148  auto& custom_data_json = json["custom_data"];
149  if (!custom_data_json.is_string()) {
150  return Status(StatusCode::kInternal,
151  "Parallel upload state's 'custom_data' is not a string.");
152  }
153  res.custom_data = custom_data_json.get<std::string>();
154  }
155  if (json.count("streams") != 1) {
156  return Status(StatusCode::kInternal,
157  "Parallel upload state doesn't contain 'streams'.");
158  }
159  auto& streams_json = json["streams"];
160  if (!streams_json.is_array()) {
161  return Status(StatusCode::kInternal,
162  "Parallel upload state's 'streams' is not an array.");
163  }
164  for (auto& stream_json : streams_json) {
165  if (!stream_json.is_object()) {
166  return Status(StatusCode::kInternal,
167  "Parallel upload state's 'stream' is not an object.");
168  }
169  if (stream_json.count("name") != 1) {
170  return Status(StatusCode::kInternal,
171  "Parallel upload state's stream doesn't contain a 'name'.");
172  }
173  auto object_name_json = stream_json["name"];
174  if (!object_name_json.is_string()) {
175  return Status(StatusCode::kInternal,
176  "Parallel upload state's stream 'name' is not a string.");
177  }
178  if (stream_json.count("resumable_session_id") != 1) {
179  return Status(StatusCode::kInternal,
180  "Parallel upload state's stream doesn't contain a "
181  "'resumable_session_id'.");
182  }
183  auto resumable_session_id_json = stream_json["resumable_session_id"];
184  if (!resumable_session_id_json.is_string()) {
185  return Status(StatusCode::kInternal,
186  "Parallel upload state's stream 'resumable_session_id' is "
187  "not a string.");
188  }
189  res.streams.emplace_back(ParallelUploadPersistentState::Stream{
190  object_name_json, resumable_session_id_json});
191  }
192  if (res.streams.empty()) {
193  return Status(StatusCode::kInternal,
194  "Parallel upload state's stream doesn't contain any streams");
195  }
196  return res;
197 }

199 Status ParallelUploadStateImpl::EagerCleanup() {
200  std::unique_lock<std::mutex> lk(mu_);
201  if (!finished_) {
202  return Status(StatusCode::kFailedPrecondition,
203  "Attempted to cleanup parallel upload state while it is "
204  "still in progress");
205  }
206  // Make sure that only one thread actually interacts with the deleter.
207  if (deleter_) {
208  cleanup_status_ = deleter_->ExecuteDelete();
209  deleter_.reset();
210  }
211  return cleanup_status_;
212 }

214 void ParallelUploadStateImpl::Fail(Status status) {
215  std::unique_lock<std::mutex> lk(mu_);
216  assert(!status.ok());
217  if (!res_) {
218  // Preserve the first error.
219  res_ = std::move(status);

221  if (num_unfinished_streams_ == 0) {
222  AllStreamsFinished(lk);

244 void ParallelUploadStateImpl::AllStreamsFinished(
245  std::unique_lock<std::mutex>& lk) {
246  if (!res_) {
247  std::vector<ComposeSourceObject> to_compose(streams_.size());
248  std::transform(
249  streams_.begin(), streams_.end(), to_compose.begin(),
250  [](StreamInfo const& stream) { return *stream.composition_arg; });
251  // only execute ComposeMany if all the streams succeeded.
252  lk.unlock();
253  auto res = composer_(std::move(to_compose));
254  lk.lock();
255  if (res) {
256  deleter_->Enable(true);
257  }
258  res_ = std::move(res);
259  }
260  // All done, wake up whomever is waiting.
261  finished_ = true;
262  auto promises_to_satisfy = std::move(res_promises_);
263  lk.unlock();
264  for (auto& promise : promises_to_satisfy) {
265  promise.set_value(*res_);
266  }
267 }

269 void ParallelUploadStateImpl::StreamFinished(
270  std::size_t stream_idx, StatusOr<ResumableUploadResponse> const& response) {
271  std::unique_lock<std::mutex> lk(mu_);
272  assert(stream_idx < streams_.size());
273  if (streams_[stream_idx].finished) {
274  return;
275  }
276 
277  --num_unfinished_streams_;
278  streams_[stream_idx].finished = true;
279  if (!response) {
280  // The upload failed, we don't event need to clean this up.
281  if (!res_) {
282  // Preserve the first error.
283  res_ = response.status();
284  }
285  } else {
286  ObjectMetadata const& metadata = *response->payload;
287  deleter_->Add(metadata);
288  streams_[stream_idx].composition_arg =
289  ComposeSourceObject{metadata.name(), metadata.generation(), {}};
290  }
291  if (num_unfinished_streams_ > 0) {
292  return;
293  }
294  AllStreamsFinished(lk);
295 }

297 void ParallelUploadStateImpl::StreamDestroyed(std::size_t stream_idx) {
298  std::unique_lock<std::mutex> lk(mu_);
299  if (!streams_[stream_idx].finished) {
300  --num_unfinished_streams_;
301  streams_[stream_idx].finished = true;
302  // A stream which was not `Close`d is being destroyed. This means that
303  // it had been `Suspend`ed, hence this parallel upload will never finish.
304  if (!res_) {
305  // Preserve the first error.
306  res_ = Status(StatusCode::kCancelled, "A stream has been suspended.");
307  }
308  if (num_unfinished_streams_ == 0) {
309  AllStreamsFinished(lk);
310  }
311  }
312 }

315  const {

317 
318  if (finished_) {
319  return make_ready_future(*res_);
320  }
321  res_promises_.emplace_back();
322  auto res = res_promises_.back().get_future();
323  return res;

356 ParallelUploadFileShard::~ParallelUploadFileShard() {
357  // If the object wasn't moved-from (i.e. `state != nullptr`) and
358  // `left_to_upload_ > 0` it means that the object is being destroyed without
359  // actually uploading the file. We should make sure we don't create the
360  // destination object and instead fail the whole operation.
361  if (state_ != nullptr && left_to_upload_ > 0) {
362  state_->Fail(Status(StatusCode::kCancelled,
363  "Shard destroyed before calling "
364  "ParallelUploadFileShard::Upload()."));
365  std::move(ostream_).Suspend();
366  }

369 Status ParallelUploadFileShard::Upload() {
370  std::vector<char> buf(upload_buffer_size_);
371 
372  auto fail = [this](StatusCode error_code, std::string const& reason) {
373  Status status(error_code, "ParallelUploadFileShard::Upload(" + file_name_ +
374  "): " + reason);
375  state_->Fail(status);
376  std::move(ostream_).Suspend();
377  left_to_upload_ = 0;
378  return status;
379  };
380  auto const already_uploaded = ostream_.next_expected_byte();
381  if (already_uploaded > left_to_upload_) {
382  return fail(StatusCode::kInternal, "Corrupted upload state, uploaded " +
383  std::to_string(already_uploaded) +
384  " out of " +
385  std::to_string(left_to_upload_));
386  }
387  left_to_upload_ -= already_uploaded;
388  offset_in_file_ += already_uploaded;
389  std::ifstream istream(file_name_, std::ios::binary);
390  if (!istream.good()) {
391  return fail(StatusCode::kNotFound, "cannot open upload file source");
392  }
393 
394  static_assert(sizeof(std::ifstream::off_type) >= sizeof(std::uintmax_t),
395  "files cannot handle uintmax_t for offsets uploads");
396 
397  // TODO(#...) - this cast should not be necessary.
398  istream.seekg(static_cast<std::ifstream::off_type>(offset_in_file_));
399  if (!istream.good()) {
400  return fail(StatusCode::kInternal, "file changed size during upload?");
401  }
402  while (left_to_upload_ > 0) {
403  auto const to_copy = static_cast<std::ifstream::off_type>(
404  std::min<std::uintmax_t>(left_to_upload_, upload_buffer_size_));
405  istream.read(buf.data(), to_copy);
406  if (!istream.good()) {
407  return fail(StatusCode::kInternal, "cannot read from file source");
408  }
409  ostream_.write(buf.data(), to_copy);
410  if (!ostream_.good()) {
411  return Status(StatusCode::kInternal,
412  "Writing to output stream failed, look into whole parallel "
413  "upload status for more information");
414  }
415  left_to_upload_ -= to_copy;
416  }
417  ostream_.Close();
418  if (ostream_.metadata()) {
419  return Status();
420  }
421  return ostream_.metadata().status();
422 }

424 StatusOr<std::pair<std::string, std::int64_t>> ParseResumableSessionId(
425  std::string const& session_id) {
426  auto starts_with = [](std::string const& s, std::string const& prefix) {
427  return s.substr(0, prefix.size()) == prefix;
428  };
429  Status invalid(StatusCode::kInternal,
430  "Not a valid parallel upload session ID");
431 
432  auto const prefix = ResumableParallelUploadState::session_id_prefix();
433  if (!starts_with(session_id, prefix)) {
434  return invalid;
435  }
436  std::string const object_and_gen = session_id.substr(prefix.size());
437  auto sep_pos = object_and_gen.find(':');
438  if (sep_pos == std::string::npos) {
439  return invalid;
440  }
441  std::string const object = object_and_gen.substr(0, sep_pos);
442  std::string const generation_str = object_and_gen.substr(sep_pos + 1);
443 
444  char* endptr;
445  std::int64_t const generation =
446  std::strtoll(generation_str.c_str(), &endptr, 10);
447  if (generation_str.c_str() == endptr || *endptr != '\0') {
448  return invalid;
449  }
450  return std::make_pair(object, generation);
451 }
